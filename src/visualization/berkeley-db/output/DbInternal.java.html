208       /**
209        * Search mode used with the internal search and searchBoth methods.
210        */
211       public enum Search {
212  
213           /**
214            * Match the smallest value greater than the key or data param.
215            */
216           GT,
217  
218           /**
219            * Match the smallest value greater than or equal to the key or data
220            * param.
221            */
222           GTE,
223  
224           /**
225            * Match the largest value less than the key or data param.
226            */
227           LT,
228  
229           /**
230            * Match the largest value less than or equal to the key or data param.
231            */
232           LTE,
233       }

235       /**
236        * @deprecated use new-style API below instead.
237        */
238       public static OperationStatus search(
239           final Cursor cursor,
240           final DatabaseEntry key,
241           final DatabaseEntry pKey,
242           final DatabaseEntry data,
243           final Search searchMode,
244           final LockMode lockMode) {
245  
246           final OperationResult result = search(
247               cursor, key, pKey, data, searchMode, getReadOptions(lockMode));
248  
249           return result != null ?
250               OperationStatus.SUCCESS : OperationStatus.NOTFOUND;
251       }
252  
253       /**
254        * Finds the key according to the Search param. If dups are configured, GT
255        * and GTE will land on the first dup for the matching key, while LT and
256        * LTE will land on the last dup for the matching key.
257        *
258        * search() and searchBoth() in this class may eventually be exposed as
259        * public JE Cursor methods, but this isn't practical now for the following
260        * reasons:
261        *
262        * + The API design needs more thought. Perhaps Search.EQ should be added.
263        * Perhaps existing Cursor methods should be deprecated.
264        *
265        * + This implementation moves the cursor multiple times and does not
266        * release locks on the intermediate records.
267        *
268        * + This could be implemented more efficiently using lower level cursor
269        * code. For example, an LTE search would actually more efficient than
270        * the existing GTE search (getSearchKeyRange and getSearchBothRange).
271        *
272        * These methods are used by KVStore.
273        */
274       public static OperationResult search(
275           final Cursor cursor,
276           final DatabaseEntry key,
277           final DatabaseEntry pKey,
278           final DatabaseEntry data,
279           final Search searchMode,
280           final ReadOptions options) {
281  
282           final DatabaseImpl dbImpl = cursor.getDatabaseImpl();
283           KeyRange range = new KeyRange(dbImpl.getBtreeComparator());
284           final boolean first;
285  
286           switch (searchMode) {
287           case GT:
288           case GTE:
289               range = range.subRange(
290                   key, searchMode == Search.GTE, null, false);
291               first = true;
292               break;
293           case LT:
294           case LTE:
295               range = range.subRange(
296                   null, false, key, searchMode == Search.LTE);
297               first = false;
298               break;
299           default:
300               throw EnvironmentFailureException.unexpectedState();
301           }
302  
303           final RangeCursor rangeCursor = new RangeCursor(
304               range, null, dbImpl.getSortedDuplicates(), cursor);
305  
306           final OpReadOptions opReadOptions = OpReadOptions.make(options);
307  
308           final OpResult result = (first) ?
309               rangeCursor.getFirst(key, pKey, data, opReadOptions) :
310               rangeCursor.getLast(key, pKey, data, opReadOptions);
311  
312           /* RangeCursor should not have dup'd the cursor. */
313           assert cursor == rangeCursor.getCursor();
314  
315           return result.jeResult;
316       }

336       /**
337        * Searches with the dups for the given key and finds the dup matching the
338        * pKey value, according to the Search param.
339        *
340        * See search() for more discussion.
341        */
342       public static OperationResult searchBoth(
343           final Cursor cursor,
344           final DatabaseEntry key,
345           final DatabaseEntry pKey,
346           final DatabaseEntry data,
347           final Search searchMode,
348           final ReadOptions options) {
349  
350           final LockMode lockMode =
351               options != null ? options.getLockMode() : null;
352  
353           final DatabaseImpl dbImpl = cursor.getDatabaseImpl();
354           KeyRange range = new KeyRange(dbImpl.getBtreeComparator());
355           range = range.subRange(key);
356           KeyRange pKeyRange = new KeyRange(dbImpl.getDuplicateComparator());
357           final boolean first;
358  
359           switch (searchMode) {
360           case GT:
361           case GTE:
362               pKeyRange = pKeyRange.subRange(
363                   pKey, searchMode == Search.GTE, null, false);
364               first = true;
365               break;
366           case LT:
367           case LTE:
368               pKeyRange = pKeyRange.subRange(
369                   null, false, pKey, searchMode == Search.LTE);
370               first = false;
371               break;
372           default:
373               throw EnvironmentFailureException.unexpectedState();
374           }
375  
376           final RangeCursor rangeCursor = new RangeCursor(
377               range, pKeyRange, dbImpl.getSortedDuplicates(), cursor);
378  
379           final OpReadOptions opReadOptions = OpReadOptions.make(options);
380  
381           final OpResult result = (first) ?
382               rangeCursor.getFirst(key, pKey, data, opReadOptions) :
383               rangeCursor.getLast(key, pKey, data, opReadOptions);
384  
385           /* RangeCursor should not have dup'd the cursor. */
386           assert cursor == rangeCursor.getCursor();
387  
388           return result.jeResult;
389       }
