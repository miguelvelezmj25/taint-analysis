288       /**
289        * Log an item, first logging any items on the lazyLogQueue, and finally
290        * flushing and sync'ing (if requested).
291        */
292       public LogItem log(LogParams params)
293           throws DatabaseException {
294  
295           final LogItem item = new LogItem();
296  
297           /*
298            * In a read-only env we return NULL_LSN (the default value for
299            * LogItem.lsn) for all entries. We allow this to proceed, rather
300            * than throwing an exception, to support logging INs for splits that
301            * occur during recovery, for one reason. Logging LNs in a read-only
302            * env is not allowed, and this is checked in the LN class.
303            */
304           if (readOnly) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
305               return item;
306           }
307  
308           try {
309               /* Flush any pending lazy entries. */
310               for (LazyQueueEntry lqe = lazyLogQueue.poll();
311                    lqe != null; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
312                    lqe = lazyLogQueue.poll()) {
313  
314                   LogParams lqeParams = new LogParams();
315                   lqeParams.entry = lqe.entry;
316                   lqeParams.provisional = Provisional.NO;
317                   lqeParams.repContext = lqe.repContext;
318  
319                   logItem(new LogItem(), lqeParams);
320               }
321  
322               final LogEntry logEntry = params.entry;
323  
324               /*
325                * If possible, marshall this entry outside the log write latch to
326                * allow greater concurrency by shortening the write critical
327                * section. Note that the header may only be created during
328                * marshalling because it calls entry.getSize().
329                */
330               if (logEntry.getLogType().marshallOutsideLatch()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
331  
332                   item.header = new LogEntryHeader(
333                       logEntry, params.provisional, params.repContext);
334  
335                   item.buffer = marshallIntoBuffer(item.header, logEntry);
336               }
337  
338               logItem(item, params);
339  
340               if (params.fsyncRequired || params.flushRequired) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
341  
342                   /* Flush log buffers and write queue, and optionally fsync. */
343                   grpManager.flushAndSync(params.fsyncRequired);
344  
345               } else if (params.switchedLogBuffer) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
346                   /*
347                    * The operation does not require writing to the log file, but
348                    * since we switched log buffers, this thread will write the
349                    * previously dirty log buffers (not this thread's log entry
350                    * though). This is done for NOSYNC durability so those types
351                    * of transactions won't fill all the log buffers thus forcing
352                    * to have to write the buffers under the log write latch.
353                    */
354                   logBufferPool.writeDirty(false /*flushWriteQueue*/);
355               }
356  
357               TestHookExecute.doHookIfSet(flushHook);
358  
359               /*
360                * We've logged this log entry from the replication stream. Let the
361                * Replicator know, so this node can create a VLSN->LSN mapping. Do
362                * this before the ckpt so we have a better chance of writing this
363                * mapping to disk.
364                */
365               if (params.repContext.inReplicationStream()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
366  
367                   assert (item.header.getVLSN() != null) : → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
368                       "Unexpected null vlsn: " + item.header + " " +
369                       params.repContext;
370  
371                   /* Block the VLSN registration, used by unit tests. */
372                   TestHookExecute.doHookIfSet(delayVLSNRegisterHook);
373  
374                   envImpl.registerVLSN(item);
375               }
376  
377           } catch (EnvironmentFailureException e) {
378  
379               /*
380                * Final checks are below for unexpected exceptions during the
381                * critical write path. Most should be caught by
382                * serialLogInternal, but the catches here account for other
383                * exceptions above. Note that Errors must be caught here as well
384                * as Exceptions. [#21929]
385                *
386                * If we've already invalidated the environment, rethrow so as not
387                * to excessively wrap the exception.
388                */
389               if (!envImpl.isValid()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
390                   throw e;
391               }
392               throw EnvironmentFailureException.unexpectedException(envImpl, e);
393  
394           } catch (Exception e) {
395               throw EnvironmentFailureException.unexpectedException(envImpl, e);
396  
397           } catch (Error e) {
398               envImpl.invalidate(e);
399               throw e;
400           }
401  
402           /*
403            * Periodically, as a function of how much data is written, ask the
404            * checkpointer or the cleaner to wake up.
405            */
406           envImpl.getCheckpointer().wakeupAfterWrite();
407           envImpl.getCleaner().wakeupAfterWrite(item.size);
408  
409           /* Update background writes. */
410           if (params.backgroundIO) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
411               envImpl.updateBackgroundWrites(
412                   item.size, logBufferPool.getLogBufferSize());
413           }
414  
415           return item;
416       }

418       private void logItem(final LogItem item, final LogParams params)
419           throws IOException, DatabaseException {
420  
421           final UtilizationTracker tracker = envImpl.getUtilizationTracker();
422  
423           final LogWriteInfo lwi = serialLog(item, params, tracker);
424  
425           if (lwi != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
426  
427               /*
428                * Add checksum, prev offset, and VLSN to the entry.
429                * Copy data into the log buffer.
430                */
431               item.buffer = item.header.addPostMarshallingInfo(
432                   item.buffer, lwi.fileOffset, lwi.vlsn);
433  
434               lwi.lbs.put(item.buffer);
435           }
436  
437           /* Update obsolete info under the LWL */
438           updateObsolete(params, tracker);
439  
440           /* Expiration tracking is protected by the Btree latch, not the LWL. */
441           if (params.expirationTrackerToUse != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
442               params.expirationTrackerToUse.track(params.entry, item.size);
443           }
444  
445           /* Queue flushing of expiration tracker after a file flip. */
446           if (params.expirationTrackerCompleted != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
447               envImpl.getExpirationProfile().addCompletedTracker(
448                   params.expirationTrackerCompleted);
449           }
450       }
451  
452       /**
453        * This method handles exceptions to be certain that the Environment is
454        * invalidated when any exception occurs in the critical write path, and it
455        * checks for an invalid environment to be sure that no subsequent write is
456        * allowed. [#21929]
457        *
458        * Invalidation is necessary because a logging operation does not ensure
459        * that the internal state -- correspondence of LSN pointer, log buffer
460        * position and file position, and the integrity of the VLSN index [#20919]
461        * -- is maintained correctly when an exception occurs. Allowing a
462        * subsequent write can cause log corruption.
463        */
464       private LogWriteInfo serialLog(
465           final LogItem item,
466           final LogParams params,
467           final UtilizationTracker tracker)
468           throws IOException {
469  
470           synchronized (logWriteMutex) {
471  
472               /* Do not attempt to write with an invalid environment. */
473               envImpl.checkIfInvalid();
474  
475               try {
476                   return serialLogWork(item, params, tracker);
477  
478               } catch (EnvironmentFailureException e) {
479                   /*
480                    * If we've already invalidated the environment, rethrow so
481                    * as not to excessively wrap the exception.
482                    */
483                   if (!envImpl.isValid()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
484                       throw e;
485                   }
486  
487                   /* Otherwise, invalidate the environment. */
488                   throw EnvironmentFailureException.unexpectedException(
489                       envImpl, e);
490  
491               } catch (Exception e) {
492                   throw EnvironmentFailureException.unexpectedException(
493                       envImpl, e);
494  
495               } catch (Error e) {
496                   /* Errors must be caught here as well as Exceptions.[#21929] */
497                   envImpl.invalidate(e);
498                   throw e;
499               }
500           }
501       }
502  
503       /**
504        * This method is used as part of writing data to the log. Called
505        * under the LogWriteLatch.
506        * Data is either written to the LogBuffer or allocates space in the
507        * LogBuffer. The LogWriteInfo object is used to save information about
508        * the space allocate in the LogBuffer. The caller uses the object to
509        * copy data into the underlying LogBuffer. A null value returned
510        * indicates that the item was written to the log. This occurs when the
511        * data item is too big to fit into an empty LogBuffer.
512        *
513        * @param params log params.
514        * @param tracker utilization.
515        * @return a LogWriteInfo object used to access allocated LogBuffer space.
516        * If null, the data was written to the log.
517        * @throws IOException
518        */
519       private LogWriteInfo serialLogWork(
520           final LogItem item,
521           final LogParams params,
522           final UtilizationTracker tracker)
523           throws IOException {
524  
525           /*
526            * Do obsolete tracking before marshalling a FileSummaryLN into the
527            * log buffer so that a FileSummaryLN counts itself.
528            * countObsoleteNode must be called before computing the entry
529            * size, since it can change the size of a FileSummaryLN entry that
530            * we're logging
531            */
532           final LogEntryType entryType = params.entry.getLogType();
533  
534           if (!DbLsn.isTransientOrNull(params.oldLsn)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
535               if (params.obsoleteDupsAllowed) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
536                   tracker.countObsoleteNodeDupsAllowed(
537                       params.oldLsn, entryType, params.oldSize, params.nodeDb);
538               } else {
539                   tracker.countObsoleteNode(
540                       params.oldLsn, entryType, params.oldSize, params.nodeDb);
541               }
542           }
543  
544           /* Count auxOldLsn for same database; no specified size. */
545           if (!DbLsn.isTransientOrNull(params.auxOldLsn)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
546               if (params.obsoleteDupsAllowed) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
547                   tracker.countObsoleteNodeDupsAllowed(
548                       params.auxOldLsn, entryType, 0, params.nodeDb);
549               } else {
550                   tracker.countObsoleteNode(
551                       params.auxOldLsn, entryType, 0, params.nodeDb);
552               }
553           }
554  
555           /*
556            * Compute the VLSNs and modify the DTVLSN in commit/abort entries
557            * before the entry is marshalled or its size is required. At that
558            * at this point we are committed to writing a log entry with the
559            * computed VLSN.
560            */
561           final VLSN vlsn;
562  
563           if (params.repContext.getClientVLSN() != null || → [ALLOWCREATE, TRANSACTIONAL, READONLY]
564               params.repContext.mustGenerateVLSN()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
565  
566               if (params.repContext.mustGenerateVLSN()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
567                   vlsn = envImpl.assignVLSNs(params.entry);
568               } else {
569                   vlsn = params.repContext.getClientVLSN();
570               }
571           } else {
572               vlsn = null;
573           }
574  
575           /*
576            * If an entry must be protected within the log write latch for
577            * marshalling, take care to also calculate its size in the
578            * protected section. Note that we have to get the size *before*
579            * marshalling so that the currentLsn and size are correct for
580            * utilization tracking.
581            */
582           final boolean marshallOutsideLatch = (item.buffer != null); → [ALLOWCREATE, TRANSACTIONAL, READONLY]
583           final int entrySize;
584  
585           if (marshallOutsideLatch) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
586               entrySize = item.buffer.limit();
587               assert item.header != null; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
588           } else {
589               assert item.header == null; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
590               item.header = new LogEntryHeader(
591                   params.entry, params.provisional, params.repContext);
592               entrySize = item.header.getEntrySize();
593           }
594  
595           /*
596            * Get the next free slot in the log, under the log write latch.
597            * Bump the LSN values, which gives us a valid previous pointer,
598            * which is part of the log entry header.
