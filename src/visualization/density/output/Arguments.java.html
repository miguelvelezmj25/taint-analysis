410           public Arguments build() throws InvalidArgumentException {
411               if (!internalSkipParamValidation) {
412                   ResourceBundle bundle = ResourceBundle.getBundle("bundles.strings", Locale.getDefault());
413  
414                   if (platform.isEmpty()) {
415                       throw new InvalidArgumentException(bundle.getString("error.missing.platforms"));
416                   }
417  
418                   if (src == null || !src.exists()) {
419                       throw new InvalidArgumentException(MessageFormat.format(bundle.getString("error.missing.src"), src));
420                   }
421  
422                   if (dst == null) {
423                       if (src.isDirectory()) {
424                           dst = src;
425                       } else {
426                           dst = src.getParentFile();
427                       }
428                   }
429  
430                   if (compressionQuality < 0 || compressionQuality > 1.0) {
431                       throw new InvalidArgumentException(MessageFormat.format(bundle.getString("error.invalid.compressionQ"), compressionQuality));
432                   }
433  
434                   if (threadCount < 1 || threadCount > MAX_THREAD_COUNT) {
435                       throw new InvalidArgumentException(MessageFormat.format(bundle.getString("error.invalid.thread"), threadCount, MAX_THREAD_COUNT));
436                   }
437  
438                   switch (scaleType) {
439                       case FACTOR:
440                           if (srcScale <= 0 || srcScale >= 100) {
441                               throw new InvalidArgumentException(MessageFormat.format(bundle.getString("error.invalid.factorscale"), srcScale));
442                           }
443                           break;
444                       case DP_WIDTH:
445                       case DP_HEIGHT:
446                           if (srcScale <= 0 || srcScale >= 9999) {
447                               throw new InvalidArgumentException(MessageFormat.format(bundle.getString("error.invalid.dp"), srcScale));
448                           }
449                           break;
450                   }
451  
452                   if (downScalingAlgorithm == null || upScalingAlgorithm == null) {
453                       throw new InvalidArgumentException(bundle.getString("error.missing.scalealgorithm"));
454                   }
455               }
456               return new Arguments(src, dst, srcScale, platform, outputCompressionMode, scaleType, downScalingAlgorithm, upScalingAlgorithm, compressionQuality, threadCount,
457                       skipExistingFiles, skipUpscaling, verboseLog, includeAndroidLdpiTvdpi, haltOnError, createMipMapInsteadOfDrawableDir,
458                       iosCreateImagesetFolders, enablePngCrush, enableMozJpeg, postConvertWebp, enableAntiAliasing, dryRun, keepUnoptimizedFilesPostProcessor, roundingStrategy, guiAdvancedOptions, clearDirBeforeConvert);
459           }
