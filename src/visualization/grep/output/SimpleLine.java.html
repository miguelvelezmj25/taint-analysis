64       /**
65        * Returns a sub-line of the given {@code line}, similar to a substring or
66        * the method {@link #subSequence(int, int)}. The difference is that the
67        * result is again a proper line, with or without line ending depending on
68        * the cutting point and the {@code preserveLineEnding} flag.
69        *
70        * @param line the line to cut
71        * @param start the start index (inclusive), valid from 0 to {@code end}
72        * @param end the end index (exclusive), valid from 0 to
73        * {@code line.length()} if {@code preserveLineEnding=false},
74        * and from 0 to {@code line.getContentLength()} if
75        * {@code preserveLineEnding=true}
76        * @param preserveLineEnding if true, the line's ending is preserved and
77        * copied to the returned line (only the
78        * content of the line is cut in this case)
79        * @return a new line representing a sub-line of the input line
80        */
81       public static Line subLine(Line line, int start, int end, boolean preserveLineEnding) {
82           if(start == 0 && (end == line.length() || (preserveLineEnding && end == line.getContentLength()))) {
83               return line;
84           }
85           if(start < 0) {
86               throw new IllegalArgumentException("start cannot be negative: " + start);
87           }
88           if(end < 0) {
89               throw new IllegalArgumentException("end cannot be negative: " + end);
90           }
91           if(start > end) {
92               throw new IllegalArgumentException("start cannot be after end: " + start + " > " + end);
93           }
94           if(preserveLineEnding) {
95               if(end > line.getContentLength()) {
96                   throw new IllegalArgumentException("start cannot be after line content end: " + end + " > " + line.getContentLength());
97               }
98               return new SimpleLine(line.subSequence(start, end), line.getLineEnding());
99           }
100           else {
101               final int clen = line.getContentLength();
102               if(end > line.length()) {
103                   throw new IllegalArgumentException("start cannot be after line end: " + end + " > " + line.length());
104               }
105               if(end <= clen) {
106                   return new SimpleLine(line.subSequence(start, end), "");
107               }
108               else {
109                   if(start < clen) {
110                       return new SimpleLine(line.subSequence(start, clen), line.subSequence(line.getContentLength(), end));
111                   }
112                   else {
113                       return new SimpleLine("", line.subSequence(start, end));
114                   }
115               }
116           }
117       }

131           final int clen = content.length();
132           final int elen = lineEnding.length();
133           if(start < clen && end <= clen) {
134               return content.subSequence(start, end);
135           }
136           final int estart = start - clen;
137           final int eend = end - clen;
138           if(estart >= 0 && estart <= elen && eend >= 0 && eend <= elen) {
139               return lineEnding.subSequence(estart, eend);
140           }
141           // overlaps the two strings

143           return sb.append(content).append(lineEnding).subSequence(start, end);
144       }
