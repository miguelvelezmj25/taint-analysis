34  /**
35    * This class encapsulates the mining schema from
36    * a PMML xml file. Specifically, it contains the
37    * fields used in the PMML model as an Instances
38    * object (just the header). It also contains meta
39    * information such as value ranges and how to handle
40    * missing values, outliers etc.
41    *
42    * We also store various other PMML elements here, such as
43    * the TransformationDictionary, DerivedFields and Targets
44    * (if defined). They are not part of the mining schema per se, but
45    * relate to inputs used by the model and it is convenient to
46    * store them here.
47    *
48    * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
49    * @version $Revision: 8034 $
50    */

105     /**
106      * Constructor for MiningSchema.
107      *
108      * @param model the <code>Element</code> encapsulating the pmml model
109      * @param dataDictionary the data dictionary as an Instances object
110      * @throws Exception if something goes wrong during construction of the
111      * mining schema
112      */
113     public MiningSchema(Element model,
114                         Instances dataDictionary,
115                         TransformationDictionary transDict) throws Exception {
116      
117       /*// First check for transformation dictionary/local transformations and derived fields.
118       // These are not supported yet.
119       NodeList temp = model.getElementsByTagName("LocalTransformations");
120       if (temp.getLength() > 0) {
121         throw new Exception("[MiningSchema] LocalTransformations "
122             + "are not supported yet.");
123       }*/
124  
125       ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
126       NodeList fieldList = model.getElementsByTagName("MiningField");
127       int classIndex = -1;
128       int addedCount = 0;
129       for (int i = 0; i < fieldList.getLength(); i++) {
130         Node miningField = fieldList.item(i);
131         if (miningField.getNodeType() == Node.ELEMENT_NODE) {
132           Element miningFieldEl = (Element)miningField;
133  
134           MiningFieldMetaInfo mfi = new MiningFieldMetaInfo(miningFieldEl);
135  
136           if (mfi.getUsageType() == MiningFieldMetaInfo.Usage.ACTIVE ||
137               mfi.getUsageType() == MiningFieldMetaInfo.Usage.PREDICTED) {
138  
139             // find this attribute in the dataDictionary
140             Attribute miningAtt = dataDictionary.attribute(mfi.getName());
141             if (miningAtt != null) {
142               mfi.setIndex(addedCount);
143               attInfo.add(miningAtt);
144               addedCount++;
145  
146               if (mfi.getUsageType() == MiningFieldMetaInfo.Usage.PREDICTED) {
147                 classIndex = addedCount - 1;
148               }
149  
150               // add to the array list
151               m_miningMeta.add(mfi);
152             } else {
153             throw new Exception("Can't find mining field: " + mfi.getName()
154                                 + " in the data dictionary.");
155             }
156           }
157         }
158       }
159  
160       m_miningSchemaInstancesStructure = new Instances("miningSchema", attInfo, 0);
161      
162       // set these instances on the MiningFieldMetaInfos so that the
163       // toString() method can operate correctly
164       for (MiningFieldMetaInfo m : m_miningMeta) {
165         m.setMiningSchemaInstances(m_miningSchemaInstancesStructure);
166       }
167      
168       m_transformationDictionary = transDict;
169      
170       // Handle transformation dictionary and any local transformations
171       if (m_transformationDictionary != null) {
172         ArrayList<DerivedFieldMetaInfo> transDerived = transDict.getDerivedFields();
173         m_derivedMeta.addAll(transDerived);
174       }
175      
176       // Get any local transformations
177       getLocalTransformations(model);
178      
179       // Set up the full instances structure: combo of mining schema fields and
180       // all derived fields
181       ArrayList<Attribute> newStructure = new ArrayList<Attribute>();
182       for (MiningFieldMetaInfo m : m_miningMeta) {
183         newStructure.add(m.getFieldAsAttribute());
184       }
185      
186       for (DerivedFieldMetaInfo d : m_derivedMeta) {
187         newStructure.add(d.getFieldAsAttribute());
188       }
189       m_fieldInstancesStructure = new Instances("FieldStructure", newStructure, 0);
190      
191       if (m_transformationDictionary != null) {
192         // first update the field defs for any derived fields in the transformation dictionary
193         // and our complete list of derived fields, now that we have a fixed
194         // ordering for the mining schema + derived attributes (i.e. could
195         // be different from the order of attributes in the data dictionary that was
196         // used when the transformation dictionary was initially constructed
197         m_transformationDictionary.setFieldDefsForDerivedFields(m_fieldInstancesStructure);
198       }
199  
200       // update the field defs for all our derived fields.
201       for (DerivedFieldMetaInfo d : m_derivedMeta) {
202         d.setFieldDefs(m_fieldInstancesStructure);
203       }
204      
205       if (classIndex != -1) {
206         m_fieldInstancesStructure.setClassIndex(classIndex);
207         m_miningSchemaInstancesStructure.setClassIndex(classIndex);
208       }
209  
210       // do Targets (if any)
211       NodeList targetsList = model.getElementsByTagName("Targets");
212       if (targetsList.getLength() > 0) {
213         if (targetsList.getLength() > 1) {
214           throw new Exception("[MiningSchema] Can only handle a single Target");
215         } else {
216           Node te = targetsList.item(0);
217           if (te.getNodeType() == Node.ELEMENT_NODE) {
218             m_targetMetaInfo = new TargetMetaInfo((Element)te);
219  
220             // fill in any necessary categorical values in the mining schema
221             // class attribute
222             if (m_fieldInstancesStructure.classIndex() >= 0 &&
223                 m_fieldInstancesStructure.classAttribute().isString()) {
224               ArrayList<String> targetVals = m_targetMetaInfo.getValues();
225               if (targetVals.size() > 0) {
226                 Attribute classAtt = m_fieldInstancesStructure.classAttribute();
227                 for (int i = 0; i < targetVals.size(); i++) {
228                   classAtt.addStringValue(targetVals.get(i));
229                 }
230               }
231             }
232           }
233         }
234       }
235     }

237     /**
238      * Apply the missing value treatments (if any) to an incoming instance.
239      *
240      * @param values an array of doubles in order of the fields in the mining schema
241      * that represents the incoming instance (note: use PMMLUtils.instanceToSchema()
242      * to generate this).
243      * @throws Exception if something goes wrong during missing value handling
244      */
245     public void applyMissingValuesTreatment(double[] values) throws Exception {
246       for (int i = 0; i < m_miningMeta.size(); i++) {
247         MiningFieldMetaInfo mfi = m_miningMeta.get(i);
248         values[i] = mfi.applyMissingValueTreatment(values[i]);
249       }
250     }

331     /**
332      * Method to convert any string attributes in the mining schema
333      * Instances to nominal attributes. This may be necessary if there are
334      * no Value elements defined for categorical fields in the data dictionary.
335      * In this case, elements in the actual model definition will probably reveal
336      * the valid values for categorical fields.
337      */
338     public void convertStringAttsToNominal() {
339       Instances miningSchemaI = getFieldsAsInstances();
340       if (miningSchemaI.checkForStringAttributes()) {
341         ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
342         for (int i = 0; i < miningSchemaI.numAttributes(); i++) {
343           Attribute tempA = miningSchemaI.attribute(i);
344           if (tempA.isString()) {
345             ArrayList<String> valueVector = new ArrayList<String>();
346             for (int j = 0; j < tempA.numValues(); j++) {
347               valueVector.add(tempA.value(j));
348             }
349             Attribute newAtt = new Attribute(tempA.name(), valueVector);
350             attInfo.add(newAtt);
351           } else {
352             attInfo.add(tempA);
353           }
354         }
355         Instances newI = new Instances("miningSchema", attInfo, 0);
356         if (m_fieldInstancesStructure.classIndex() >= 0) {
357           newI.setClassIndex(m_fieldInstancesStructure.classIndex());
358         }
359         m_fieldInstancesStructure = newI;

362         stn.setInputFormat(miningSchemaI);

368     /**
369      * Convert a numeric attribute in the mining schema to nominal.
370      *
371      * @param index the index of the attribute to convert
372      * @param newVals an ArrayList of the values of the nominal attribute
373      */
374     public void convertNumericAttToNominal(int index,
375                                            ArrayList<String> newVals) {
376       Instances miningSchemaI = getFieldsAsInstances();
377       if (miningSchemaI.attribute(index).isNominal()) {
378         throw new IllegalArgumentException("[MiningSchema] convertNumericAttToNominal: attribute is "
379                                            + "already nominal!");
380       }
381  
382       ArrayList<String> newValues = new ArrayList<String>();
383       for (int i = 0; i < newVals.size(); i++) {
384         newValues.add(newVals.get(i));
385       }
386  
387       ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
388       for (int i = 0; i < miningSchemaI.numAttributes(); i++) {
389         Attribute tempA = miningSchemaI.attribute(i);
390         if (i == index) {
391           Attribute newAtt = new Attribute(tempA.name(), newValues);
392           attInfo.add(newAtt);
393         } else {
394           attInfo.add(tempA);
395         }
396       }
397  
398       Instances newI = new Instances("miningSchema", attInfo, 0);
399       if (m_fieldInstancesStructure.classIndex() >= 0) {
400         newI.setClassIndex(m_fieldInstancesStructure.classIndex());
401       }
402       m_fieldInstancesStructure = newI;
403     }
