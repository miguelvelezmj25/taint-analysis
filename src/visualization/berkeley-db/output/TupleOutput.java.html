34    * @author Mark Hayes

526       /**
527        * Writes a {@code BigInteger}.
528        *
529        * @param val is the value to write to the buffer.
530        *
531        * @return this tuple output object.
532        *
533        * @throws NullPointerException if val is null.
534        *
535        * @throws IllegalArgumentException if the byte array representation of val
536        * is larger than 0x7fff bytes.
537        *
538        * @see <a href="package-summary.html#integerFormats">Integer Formats</a>
539        */
540       public final TupleOutput writeBigInteger(BigInteger val) {
541      
542           byte[] a = val.toByteArray();
543           if (a.length > Short.MAX_VALUE) {
544               throw new IllegalArgumentException
545                   ("BigInteger byte array is larger than 0x7fff bytes");
546           }
547           int firstByte = a[0];
548           writeShort((firstByte < 0) ? (- a.length) : a.length);
549           writeByte(firstByte);
550           writeFast(a, 1, a.length - 1);
551           return this;
552       }

554       /**
555        * Returns the exact byte length that would would be output for a given
556        * {@code BigInteger} value if {@link TupleOutput#writeBigInteger} were
557        * called.
558        *
559        * @param val the BigInteger
560        *
561        * @return the byte length.
562        *
563        * @see <a href="package-summary.html#integerFormats">Integer Formats</a>
564        */
565       public static int getBigIntegerByteLength(BigInteger val) {
566           return 2 /* length bytes */ +
567                  (val.bitLength() + 1 /* sign bit */ + 7 /* round up */) / 8;

570       /**
571        * Writes an unsorted {@code BigDecimal}.
572        *
573        * @param val is the value to write to the buffer.
574        *
575        * @return this tuple output object.
576        *
577        * @throws NullPointerException if val is null.
578        *
579        * @see <a href="package-summary.html#bigDecimalFormats">BigDecimal
580        * Formats</a>
581        */
582       public final TupleOutput writeBigDecimal(BigDecimal val) {
583      
584           /*
585            * The byte format for a BigDecimal value is:
586            * Byte 0 ~ L: The scale part written as a PackedInteger.
587            * Byte L+1 ~ M: The length of the unscaled value written as a
588            * PackedInteger.
589            * Byte M+1 ~ N: The BigDecimal.toByteArray array, written
590            * without modification.
591            *
592            * Get the scale and the unscaled value of this BigDecimal.
593            */
594           int scale = val.scale();
595           BigInteger unscaledVal = val.unscaledValue();
596          
597           /* Store the scale. */
598           writePackedInt(scale);
599           byte[] a = unscaledVal.toByteArray();
600           int len = a.length;
601          
602           /* Store the length of the following bytes. */
603           writePackedInt(len);
604          
605           /* Store the bytes of the BigDecimal, without modification. */
606           writeFast(a, 0, len);
607           return this;
608       }

615        *

628       /**
629        * Writes a sorted {@code BigDecimal}.
630        *
631        * @param val is the value to write to the buffer.
632        *
633        * @return this tuple output object.
634        *
635        * @see <a href="package-summary.html#bigDecimalFormats">BigDecimal
636        * Formats</a>
637        */
638       public final TupleOutput writeSortedBigDecimal(BigDecimal val) {
639  
640           /*
641            * We have several options for the serialization of sorted BigDecimal.
642            * The reason for choosing this method is that it is simpler and more
643            * compact, and in some cases, comparison time will be less. For other
644            * methods and detailed discussion, please refer to [#18379].
645            *
646            * First, we need to do the normalization, which means we normalize a
647            * given BigDecimal into two parts: decimal part and the exponent part.
648            * The decimal part contains one integer (non zero). For example,
649            * 1234.56 will be normalized to 1.23456E3;
650            * 123.4E100 will be normalized to 1.234E102;
651            * -123.4E-100 will be normalized to -1.234E-98.
652            *
653            * After the normalization, the byte format is:
654            * Byte 0: sign (-1 represents negative, 0 represents zero, and 1
655            * represents positive).
656            * Byte 1 ~ 5: the exponent with sign, and written as a
657            * SortedPackedInteger value.
658            * Byte 6 ~ N: the normalized decimal part with sign.
659            *
660            * Get the scale and the unscaled value of this BigDecimal..
661            */
662           BigDecimal valNoTrailZeros = val.stripTrailingZeros();
663           int scale = valNoTrailZeros.scale();
664           BigInteger unscaledVal = valNoTrailZeros.unscaledValue();
665           int sign = valNoTrailZeros.signum();
666          
667           /* Then do the normalization. */
668           String unscaledValStr = unscaledVal.abs().toString();
669           int normalizedScale = unscaledValStr.length() - 1;
670           BigDecimal normalizedVal = new BigDecimal(unscaledVal,
671                                                     normalizedScale);
672           int exponent = (normalizedScale - scale) * sign;
673          
674           /* Start serializing each part. */
675           writeByte(sign);
676           writeSortedPackedInt(exponent);
677           writeSortedNormalizedBigDecimal(normalizedVal);
678           return this;
679       }
680      
681       /**
682        * Writes a normalized {@code BigDecimal}.
683        */
684       private final TupleOutput writeSortedNormalizedBigDecimal(BigDecimal val) {
685  
686           /*
687            * The byte format for a sorted normalized {@code BigDecimal} value is:
688            * Byte 0 ~ N: Store all digits with sign. Each 9 digits is
689            * regarded as one integer, and written as a
690            * SortedPackedInteger value. If there are not enough
691            * 9 digits, pad trailing zeros. Since we may pad
692            * trailing zeros for serialization, when doing
693            * de-serialization, we need to delete the trailing
694            * zeros. In order to designate a special value as the
695            * terminator byte, we set
696            * val = (val < 0) ? (val - 1) : val.
697            * Byte N + 1: Terminator byte. The terminator byte is -1, and
698            * written as a SortedPackedInteger value.
699            */
700  
701           /* get the precision, scale and sign of the BigDecimal. */
702           int precision = val.precision();
703           int scale = val.scale();
704           int sign = val.signum();
705          
706           /* Start the serialization of the whole digits. */
707           String digitsStr = val.abs().toPlainString();
708          
709           /*
710            * The default capacity of a StringBuilder is 16 chars, which is
711            * enough to hold a group of digits having 9 digits.
712            */
713           StringBuilder groupDigits = new StringBuilder();
714           for (int i = 0; i < digitsStr.length();) {
715               char digit = digitsStr.charAt(i++);
716              
717               /* Ignore the decimal. */
718               if (digit != '.') {
719                   groupDigits.append(digit);
720               }
721              
722               /*
723                * For the last group of the digits, if there are not 9 digits, pad
724                * trailing zeros.
725                */
726               if (i == digitsStr.length() && groupDigits.length() < 9) {
727                   final int insertLen = 9 - groupDigits.length();
728                   for (int k = 0; k < insertLen; k++) {
729                       groupDigits.append("0");
730                   }
731               }
732              
733               /* Group every 9 digits as an Integer. */
734               if (groupDigits.length() == 9) {
735                   int subVal = Integer.valueOf(groupDigits.toString());
736                   if (sign < 0) {
737                       subVal = -subVal;
738                   }
739                  
740                   /*
741                    * Reset the sub-value, so the value -1 will be designated as
742                    * the terminator byte.
743                    */
744                   subVal = subVal < 0 ? subVal - 1 : subVal;
745                   writeSortedPackedInt(subVal);
746                   groupDigits.setLength(0);
747               }
748           }
749          
750           /* Write the terminator byte. */
751           writeSortedPackedInt(-1);
752           return this;
753       }

755       /**
756        * Returns the maximum byte length that would be output for a given {@code
757        * BigDecimal} value if {@link TupleOutput#writeSortedBigDecimal} were
758        * called.
759        *
760        * @param val the BigDecimal.
761        *
762        * @return the byte length.
763        *
764        * @see <a href="package-summary.html#bigDecimalFormats">BigDecimal
765        * Formats</a>
766        */
767       public static int getSortedBigDecimalMaxByteLength(BigDecimal val) {
768      
769           String digitsStr = val.stripTrailingZeros().unscaledValue().abs().
770                              toString();
771          
772           int numOfGroups = (digitsStr.length() + 8 /* round up */) / 9;
773          
774           return 1 /* sign */ +
775                  PackedInteger.MAX_LENGTH /* exponent */ +
776                  PackedInteger.MAX_LENGTH * numOfGroups /* all the digits */ +
777                  1; /* terminator byte */
778       }
