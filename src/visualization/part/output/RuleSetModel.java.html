157       /**
158        * Constructor for a simple rule
159        *
160        * @param ruleE the XML element holding the simple rule
161        * @param miningSchema the mining schema to use
162        * @throws Exception if something goes wrong
163        */
164       public SimpleRule(Element ruleE, MiningSchema miningSchema) throws Exception {
165         super(ruleE, miningSchema);
166        
167         String id = ruleE.getAttribute("id");
168         if (id != null && id.length() > 0) {
169           m_ID = id;
170         }
171        
172         m_scoreString = ruleE.getAttribute("score");
173         Attribute classAtt = miningSchema.getFieldsAsInstances().classAttribute();
174         if (classAtt.isNumeric()) {
175           m_score = Double.parseDouble(m_scoreString);
176         } else {
177           if (classAtt.indexOfValue(m_scoreString) < 0) {
178             throw new Exception("[SimpleRule] class value " + m_scoreString +
179                 "does not exist in class attribute " + classAtt.name());
180           }
181           m_score = classAtt.indexOfValue(m_scoreString);
182         }
183        
184         String recordCount = ruleE.getAttribute("recordCount");
185         if (recordCount != null && recordCount.length() > 0) {
186           m_recordCount = Double.parseDouble(recordCount);
187         }
188        
189         String nbCorrect = ruleE.getAttribute("nbCorrect");
190         if (nbCorrect != null && nbCorrect.length() > 0) {
191           m_nbCorrect = Double.parseDouble(nbCorrect);
192         }
193        
194         String confidence = ruleE.getAttribute("confidence");
195         if (confidence != null && confidence.length() > 0) {
196           m_confidence = Double.parseDouble(confidence);
197         }
198        
199         String weight = ruleE.getAttribute("weight");
200         if (weight != null && weight.length() > 0) {
201           m_weight = Double.parseDouble(weight);
202         }
203        
204         // get the ScoreDistributions (if any)
205         if (miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
206           // see if we have any ScoreDistribution entries
207           NodeList scoreChildren = ruleE.getChildNodes();
208                  
209           for (int i = 0; i < scoreChildren.getLength(); i++) {
210             Node child = scoreChildren.item(i);
211             if (child.getNodeType() == Node.ELEMENT_NODE) {
212               String tagName = ((Element)child).getTagName();
213               if (tagName.equals("ScoreDistribution")) {
214                 TreeModel.ScoreDistribution newDist =
215                   new TreeModel.ScoreDistribution((Element)child,
216                     miningSchema, m_recordCount);
217                 m_scoreDistributions.add(newDist);
218               }
219             }
220           }
221          
222           // check that we have as many score distribution elements as there
223           // are class labels in the data
224           if (m_scoreDistributions.size() > 0 &&
225               m_scoreDistributions.size() !=
226                 miningSchema.getFieldsAsInstances().classAttribute().numValues()) {
227             throw new Exception("[SimpleRule] Number of score distribution elements is "
228                 + " different than the number of class labels!");
229           }
230          
231           //backfit the confidence values (if necessary)
232           if (Utils.isMissingValue(m_recordCount)) {
233             double baseCount = 0;
234             for (TreeModel.ScoreDistribution s : m_scoreDistributions) {
235               baseCount += s.getRecordCount();
236             }
237            
238             for (TreeModel.ScoreDistribution s : m_scoreDistributions) {
239               s.deriveConfidenceValue(baseCount);
240             }
241           }
242         }
243       }
244      
245       /**
246        * Collect the rule(s) that fire for the supplied incoming instance
247        *
248        * @param input a vector of independent and derived independent variables
249        * @param ruleCollection the array list to add any firing rules into
250        */
251       public void fires(double[] input, ArrayList<SimpleRule> ruleCollection) {
252         if (m_predicate.evaluate(input) == TreeModel.Predicate.Eval.TRUE) {
253           ruleCollection.add(this);
254         }
255       }
256      
257       /**
258        * Score the incoming instance
259        *
260        * @param instance a vector containing the incoming independent and
261        * derived independent variables
262        * @param classAtt the class attribute
263        * @param rsm the rule selection method (ignored by simple rules)
264        * @return a probability distribution over the class labels or
265        * the predicted value (in element zero of the array if the class is numeric)
266        * @throws Exception if something goes wrong
267        */
268       public double[] score(double[] instance, Attribute classAtt)
269         throws Exception {
270        
271         double[] preds;
272         if (classAtt.isNumeric()) {
273           preds = new double[1];
274           preds[0] = m_score;
275         } else {
276           preds = new double[classAtt.numValues()];
277           if (m_scoreDistributions.size() > 0) {
278             for (TreeModel.ScoreDistribution s : m_scoreDistributions) {
279               preds[s.getClassLabelIndex()] = s.getConfidence();
280             }
281           } else if (!Utils.isMissingValue(m_confidence)) {
282             preds[classAtt.indexOfValue(m_scoreString)] = m_confidence;
283           } else {
284             preds[classAtt.indexOfValue(m_scoreString)] = 1.0;
285           }
286         }
287        
288         return preds;
289       }

292        * Get the weight of the rule
293        *

324     static class CompoundRule extends Rule {
325      
326       /** For serialization */
327       private static final long serialVersionUID = -2853658811459970718L;
328      
329       /** The child rules of this compound rule */
330       ArrayList<Rule> m_childRules = new ArrayList<Rule>();
331      
332       public String toString(String prefix, int indent) {
333         StringBuffer temp = new StringBuffer();
334  
335         for (int i = 0; i < indent; i++) {
336           prefix += " ";
337         }
338        
339         temp.append(prefix + "Compound rule: " + m_predicate + "\n");
340        
341         for (Rule r : m_childRules) {
342           temp.append(r.toString(prefix, indent + 1));
343         }
344  
345         return temp.toString();
346       }
347      
348       public String toString() {
349         return toString("", 0);
350       }
351      
352       /**
353        * Constructor.
354        *
355        * @param ruleE XML node holding the rule
356        * @param miningSchema the mining schema to use
357        * @throws Exception if something goes wrong
358        */
359       public CompoundRule(Element ruleE, MiningSchema miningSchema) throws Exception {
360        
361         // get the Predicate
362         super(ruleE, miningSchema);
363        
364         // get the nested rules
365         NodeList ruleChildren = ruleE.getChildNodes();
366         for (int i = 0; i < ruleChildren.getLength(); i++) {
367           Node child = ruleChildren.item(i);
368           if (child.getNodeType() == Node.ELEMENT_NODE) {
369             String tagName = ((Element)child).getTagName();
370             if (tagName.equals("SimpleRule")) {
371               Rule childRule = new SimpleRule(((Element)child), miningSchema);
372               m_childRules.add(childRule);
373             } else if (tagName.equals("CompoundRule")) {
374               Rule childRule = new CompoundRule(((Element)child), miningSchema);
375               m_childRules.add(childRule);
376             }
377           }
378         }
379       }
380      
381       /**
382        * Collect the rule(s) that fire for the supplied incoming instance
383        *
384        * @param input a vector of independent and derived independent variables
385        * @param ruleCollection the array list to add any firing rules into
386        */
387       public void fires(double[] input, ArrayList<SimpleRule> ruleCollection) {
388        
389         // evaluate our predicate first
390         if (m_predicate.evaluate(input) == TreeModel.Predicate.Eval.TRUE) {
391           // now check the child rules
392           for (Rule r : m_childRules) {
393             r.fires(input, ruleCollection);
394           }
395         }
396       }
397     }

501       /**
502        * Constructor for a RuleSet.
503        *
504        * @param ruleSetNode the XML node holding the RuleSet
505        * @param miningSchema the mining schema to use
506        * @throws Exception if something goes wrong
507        */
508       public RuleSet(Element ruleSetNode, MiningSchema miningSchema)
509         throws Exception {
510        
511         String recordCount = ruleSetNode.getAttribute("recordCount");
512         if (recordCount != null && recordCount.length() > 0) {
513           m_recordCount = Double.parseDouble(recordCount);
514         }
515        
516         String nbCorrect = ruleSetNode.getAttribute("nbCorrect");
517         if (nbCorrect != null & nbCorrect.length() > 0) {
518           m_nbCorrect = Double.parseDouble(nbCorrect);
519         }
520        
521         String defaultScore = ruleSetNode.getAttribute("defaultScore");
522         if (defaultScore != null && defaultScore.length() > 0) {
523           m_defaultScore = defaultScore;
524          
525           Attribute classAtt = miningSchema.getFieldsAsInstances().classAttribute();
526           if (classAtt == null) {
527             throw new Exception("[RuleSet] class attribute not set!");
528           }
529          
530           if (classAtt.isNumeric()) {
531             m_defaultPrediction = Double.parseDouble(defaultScore);
532           } else {
533             if (classAtt.indexOfValue(defaultScore) < 0) {
534               throw new Exception("[RuleSet] class value " + defaultScore +
535                   " not found!");
536             }
537             m_defaultPrediction = classAtt.indexOfValue(defaultScore);
538           }
539         }
540        
541         String defaultConfidence = ruleSetNode.getAttribute("defaultConfidence");
542         if (defaultConfidence != null && defaultConfidence.length() > 0) {
543           m_defaultConfidence = Double.parseDouble(defaultConfidence);
544         }
545        
546         // get the rule selection methods
547         NodeList selectionNL = ruleSetNode.getElementsByTagName("RuleSelectionMethod");
548         for (int i = 0; i < selectionNL.getLength(); i++) {
549           Node selectN = selectionNL.item(i);
550           if (selectN.getNodeType() == Node.ELEMENT_NODE) {
551             Element sN = (Element)selectN;
552             String criterion = sN.getAttribute("criterion");
553             for (RuleSelectionMethod m : RuleSelectionMethod.values()) {
554               if (m.toString().equals(criterion)) {
555                 m_availableRuleSelectionMethods.add(m);
556                 if (i == 0) {
557                   // set the default (first specified one)
558                   m_currentMethod = m;
559                 }
560               }
561             }
562           }
563         }
564        
565         if (miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
566           // see if we have any ScoreDistribution entries
567           NodeList scoreChildren = ruleSetNode.getChildNodes();
568           for (int i = 0; i < scoreChildren.getLength(); i++) {
569             Node child = scoreChildren.item(i);
570             if (child.getNodeType() == Node.ELEMENT_NODE) {
571               String tagName = ((Element)child).getTagName();
572               if (tagName.equals("ScoreDistribution")) {
573                 TreeModel.ScoreDistribution newDist =
574                   new TreeModel.ScoreDistribution((Element)child,
575                     miningSchema, m_recordCount);
576                 m_scoreDistributions.add(newDist);
577               }
578             }
579           }
580          
581           //backfit the confidence values (if necessary)
582           if (Utils.isMissingValue(m_recordCount)) {
583             double baseCount = 0;
584             for (TreeModel.ScoreDistribution s : m_scoreDistributions) {
585               baseCount += s.getRecordCount();
586             }
587            
588             for (TreeModel.ScoreDistribution s : m_scoreDistributions) {
589               s.deriveConfidenceValue(baseCount);
590             }
591           }
592         }
593        
594         // Get the rules in this rule set
595         NodeList ruleChildren = ruleSetNode.getChildNodes();
596         for (int i = 0; i < ruleChildren.getLength(); i++) {
597           Node child = ruleChildren.item(i);
598           if (child.getNodeType() == Node.ELEMENT_NODE) {
599             String tagName = ((Element)child).getTagName();
600             if (tagName.equals("SimpleRule")) {
601               Rule tempRule = new SimpleRule(((Element)child), miningSchema);
602               m_rules.add(tempRule);
603             } else if (tagName.equals("CompoundRule")) {
604               Rule tempRule = new CompoundRule(((Element)child), miningSchema);
605               m_rules.add(tempRule);
606             }
607           }
608         }
609       }

611       /**
612        * Score an incoming instance by collecting all rules that fire.
613        *
614        * @param instance a vector of incoming attribte and derived field values
615        * @param classAtt the class attribute
616        * @return a predicted probability distribution
