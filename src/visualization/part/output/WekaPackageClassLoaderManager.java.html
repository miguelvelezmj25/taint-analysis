32  /**
33    * Class that manages classloaders from individual Weka plugin packages.
34    * Maintains a collection of {@code WekaPackageLibIsolatingClassLoader}s - one
35    * for each package. {@code Utils.forName()} and {@code weka.Run} use this
36    * classloader to find/instantiate schemes exposed in top-level package jar
37    * files. Client code in a package should do the same, unless directly referring
38    * to classes in other packages, in which case the other packages should be
39    * explicit dependencies. This classloader will not find classes in third-party
40    * libraries inside a package's lib directory.<br>
41    * <br>
42    * Classes are searched for first in the parent classloader and then in the
43    * top-level package jar files.
44    *
45    * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
46    * @version $Revision: $
47    * @see WekaPackageLibIsolatingClassLoader
48    */

219       /**

221        *

223        * can be read from the stream
224        * @param classJarPaths an optional list of paths to classes in the jar file
225        * to inject (if null then all classes in the jar file are injected)
226        * @param classes an optional list of fully qualified class names. This should
227        * correspond to the paths in classJarPaths, but contain just the
228        * class name along with slashes replaced by "." and the .class
229        * extension removed.
230        * @param injectToRootClassLoader true if the classes are to be injected into
231        * the root classloader rather than the Weka classloader
232        * @throws Exception if a problem occurs
233        */
234       protected static void injectClasses(InputStream jarStream,
235                                           List<String> classJarPaths, List<String> classes,
236                                           boolean injectToRootClassLoader) throws Exception {
237           String debugS = System.getProperty("weka.core.classloader.debug", "false");
238           boolean debug = debugS.equalsIgnoreCase("true");
239  
240           boolean processAllClasses = classes == null || classJarPaths == null;
241           if(processAllClasses) {
242               classes = new ArrayList<>();
243               classJarPaths = new ArrayList<>();
244           }
245           List<byte[]> preloadClassByteCode = new ArrayList<>();
246  
247           ZipInputStream zi = new ZipInputStream(jarStream);
248           ZipEntry zipEntry = null;
249           while((zipEntry = zi.getNextEntry()) != null) {
250               if(!zipEntry.isDirectory() && zipEntry.getName().endsWith(".class")) {
251                   String zipPart = zipEntry.getName().replace("\\", "/");
252                   if(classJarPaths.contains(zipPart) || processAllClasses) {
253                       preloadClassByteCode.add(getByteCode(zi, false));
254                       zi.closeEntry(); // move to next entry in zip
255                       if(processAllClasses) {
256                           classes.add(zipEntry.getName().replace(".class", "")
257                                   .replace("\\", "/").replace("/", "."));
258                       }
259                   }
260               }
261           }
262  
263           zi.close();
264  
265           List<byte[]> okBytes = new ArrayList<>();
266           List<String> okClasses = new ArrayList<>();
267           for(int i = 0; i < classes.size(); i++) {
268               if(!classExists(classes.get(i))) {
269                   okClasses.add(classes.get(i));
270                   okBytes.add(preloadClassByteCode.get(i));
271               }
272           }
273           preloadClassByteCode = okBytes;
274           classes = okClasses;
275  
276           if(preloadClassByteCode.size() > 0) {
277               ClassLoader rootClassloader =
278                       injectToRootClassLoader ? getRootClassLoader()
279                               : getWekaLevelClassloader();
280  
281               Class<?> classLoader = Class.forName("java.lang.ClassLoader");
282               Method defineClass =
283                       classLoader.getDeclaredMethod("defineClass", String.class,
284                               byte[].class, int.class, int.class, ProtectionDomain.class);
285  
286               ProtectionDomain pd = System.class.getProtectionDomain();
287  
288               // ClassLoader.defineClass is a protected method, so we have to make it
289               // accessible
290               defineClass.setAccessible(true);
291               List<byte[]> failedToInject = new ArrayList<>();
292               List<String> classesF = new ArrayList<>();
293               boolean cont = true;
294               int numLeft = classes.size();
295               try {
296                   do {
297                       if(debug) {
298                           System.out
299                                   .println("[WekaPackageClassLoaderManager] Injecting classes "
300                                           + "into the "
301                                           + (injectToRootClassLoader ? "root classloader..."
302                                           : "weka-level classloader..."));
303                       }
304                       for(int i = 0; i < classes.size(); i++) {
305                           if(debug) {
306                               System.out.println("** Injecting " + classes.get(i));
307                           }
308                           byte[] b = preloadClassByteCode.get(i);
309                           try {
310                               defineClass.invoke(rootClassloader, classes.get(i), b, 0,
311                                       b.length, pd);
312                           } catch(Exception ex) {
313                               failedToInject.add(b);
314                               classesF.add(classes.get(i));
315                           }
316                       }
317  
318                       cont = failedToInject.size() < numLeft;
319                       preloadClassByteCode = failedToInject;
320                       classes = classesF;
321                       numLeft = failedToInject.size();
322                       failedToInject = new ArrayList<>();
323                       classesF = new ArrayList<>();
324                   } while(classes.size() > 0 && cont);
325               } finally {
326                   defineClass.setAccessible(false);
327               }
328           }
329       }

335       private static byte[] getByteCode(InputStream in, boolean closeInput)
336               throws IOException {
337           byte[] buf = new byte[1024];
338           ByteArrayOutputStream byteCodeBuf = new ByteArrayOutputStream();
339  
340           for(int readLength; (readLength = in.read(buf)) != -1; ) {
341               byteCodeBuf.write(buf, 0, readLength);
342           }
343  
344           if(closeInput) {
345               in.close();
346           }
347  
348           return byteCodeBuf.toByteArray();
349       }
