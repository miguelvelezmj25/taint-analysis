40  /**
41    * LogBufferPool manages a circular pool of LogBuffers.
42    * The currentWriteBuffer is the buffer that is currently
43    * used to add data. When the buffer is full, the next (adjacent)
44    * buffer is made available for writing. The buffer pool has a dirty
45    * list of buffers. A buffer becomes a member of the dirty list when the
46    * currentWriteBuffer is moved to another buffer. Buffers are removed
47    * from the dirty list when they are written.
48    * The dirtyStart/dirtyEnd variables indicate the list of dirty buffers.
49    * A value of -1 for either variables indicates that there are no dirty
50    * buffers. These variable are synchronized via the
51    * LogBufferPool.bufferPoolLatch. The LogManager.logWriteLatch (aka LWL)
52    * is used to serialize access to the currentWriteBuffer, so that entries are
53    * added in write/LSN order.
54    *
55    * A buffer used for writing is accessed by the getWriteBuffer method.
56    * This method must be called while holding the LWL. The buffer returned
57    * is not latched.
58    *
59    * A LogBuffer has a pin count (LogBuffer.writePinCount) associated with
60    * it. The pin count is incremented when space is allocated in the buffer.
61    * The allocation of space is serialized under the LWL. Threads will add
62    * data to the buffer by latching the buffer, but without holding the LWL.
63    * After the data is added, the pin count is decremented. A buffer cannot be
64    * used for reading unless the pin count is zero. It should be noted that the
65    * increment of the pin count is done with the buffer latched. The decrement
66    * does not latch the buffer.
67    *
68    * Read access to a log buffer is allowed only if the buffer is latched
69    * and the pin count is zero. A thread that attempt to access a log
70    * buffer for reading will latch and check the pin count. If the pin
71    * count is not zero, the latch is released and the process is retried.
72    * The thread attempting to access the log buffer for reading may be delayed.
73    * The worst case is when the reader has to wait until the buffer
74    * is filled (the pin count would be zero).
75    *
76    * @see LogBuffer
77    */

177       /**
178        * Initialize the pool at construction time and when the cache is resized.
179        * This method is called after the memory budget has been calculated.

183       void reset(DbConfigManager configManager)
184           throws DatabaseException {
185  
186           /*
187            * When running in memory, we can't clear the existing pool and
188            * changing the buffer size is not very useful, so just return.
189            */
190           if (runInMemory && bufferPool != null) {
191               return;
192           }
193  
194           /*
195            * Write the currentWriteBuffer to the file and reset
196            * currentWriteBuffer.
197            */
198           if (currentWriteBuffer != null) {
199               bumpAndWriteDirty(0, true);
200           }
201  
202           /*
203            * Based on the log budget, figure the number and size of
204            * log buffers to use.
205            */
206           int numBuffers =
207               configManager.getInt(EnvironmentParams.NUM_LOG_BUFFERS);
208           long logBufferBudget = envImpl.getMemoryBudget().getLogBufferBudget();
209  
210           long logFileSize =
211               configManager.getLong(EnvironmentParams.LOG_FILE_MAX);
212           /* Buffers must be int sized. */
213           int newBufferSize = (int) logBufferBudget / numBuffers;
214           /* Limit log buffer size to size of a log file. */
215           newBufferSize = Math.min(newBufferSize, (int) logFileSize);
216           /* list of buffers that are available for log writing */
217           LinkedList<LogBuffer> newPool = new LinkedList<LogBuffer>();
218  
219           /*
220            * If we're running in memory only, don't pre-allocate all the buffers.
221            * This case only occurs when called from the constructor.
222            */
223           if (runInMemory) {
224               numBuffers = 1;
225           }
226  
227           for (int i = 0; i < numBuffers; i++) {
228               newPool.add(new LogBuffer(newBufferSize, envImpl));
229           }
230  
231           /*
232            * The following applies when this method is called to reset the pool
233            * when an existing pool is in use:
234            * - The old pool will no longer be referenced.
235            * - Buffers being read in the old pool will be no longer referenced
236            * after the read operation is complete.
237            * - The currentWriteBuffer field is not changed here; it will be no
238            * longer referenced after it is written to the file and a new
239            * currentWriteBuffer is assigned.
240            * - The logBufferSize can be changed now because it is only used for
241            * allocating new buffers; it is not used as the size of the
242            * currentWriteBuffer.
243            */
244           bufferPoolLatch.acquireExclusive();
245           bufferPool = newPool;
246           nLogBuffers = numBuffers;
247           logBufferSize = newBufferSize;
248           /* Current buffer is the active buffer that writes go into. */
249           currentWriteBuffer = bufferPool.getFirst();
250           currentWriteBufferIndex = 0;
251           bufferPoolLatch.release();
252       }

254       /**
255        * Get a log buffer for writing an entry of sizeNeeded bytes.
256        *
257        * If sizeNeeded will fit in currentWriteBuffer, currentWriteBuffer is
258        * returned without requiring any flushing. The caller can allocate the
259        * entry in the buffer returned.
260        *
261        * If sizeNeeded won't fit in currentWriteBuffer, but is LTE the LogBuffer
262        * capacity, we bump the buffer to get an empty currentWriteBuffer. If
263        * there are no free write buffers, then all dirty buffers must be flushed.
264        * The caller can allocate the entry in the buffer returned.
265        *
266        * If sizeNeeded is greater than the LogBuffer capacity, flush all dirty
267        * buffers and return an empty (but too small) currentWriteBuffer. The
268        * caller must then write the entry to the file directly.
269        *
270        * The LWL must be held.
271        *
272        * @param sizeNeeded size of the entry to be written.
273        *
274        * @param flippedFile if true, always flush all dirty buffers and get an
275        * empty currentWriteBuffer, and fsync/finish the log file.
276        *
277        * @return currentWriteBuffer, which may or may not have enough room for
278        * sizeNeeded.
279        */
280       LogBuffer getWriteBuffer(int sizeNeeded, boolean flippedFile)
281           throws IOException, DatabaseException {
282  
283           /*
284            * We need a new log buffer either because this log buffer is full, or
285            * the LSN has marched along to the next file. Each log buffer only
286            * holds entries that belong to a single file. If we've flipped over
287            * into the next file, we'll need to get a new log buffer even if the
288            * current one has room.
289            */
290           if (flippedFile) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
291  
292               /*
293                * Write the dirty buffers to the file and get an empty
294                * currentWriteBuffer.
295                *
296                * TODO: Why pass true for flushWriteQueue before doing an fsync?
297                */
298               bumpAndWriteDirty(sizeNeeded, true /*flushWriteQueue*/);
299  
300               /* Now that the buffers have been written, fsync. */
301               if (!runInMemory) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
302                   fileManager.syncLogEndAndFinishFile();
303               }
304           } else if (!currentWriteBuffer.hasRoom(sizeNeeded)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
305  
306               /*
307                * Try to bump the current write buffer since there
308                * was not enough space in the current write buffer.
309                */
310  
311               if (!bumpCurrent(sizeNeeded) || → [ALLOWCREATE, TRANSACTIONAL, READONLY]
312                   !currentWriteBuffer.hasRoom(sizeNeeded) ) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
313  
314                   /*
315                    * We could not bump because there was no free
316                    * buffer, or the item is larger than the buffer size.
317                    * Write the dirties to free a buffer up, or to flush
318                    * in preparation for writing a temporary buffer.
319                    */
320                   bumpAndWriteDirty(sizeNeeded, false /*flushWriteQueue*/);
321               }
322           }
323  
324           return currentWriteBuffer;
325       }

327       /**
328        * Bump current write buffer and write the dirty buffers.
329        *
330        * The LWL must be held to ensure that, if there are no free buffers, we
331        * can write the dirty buffers and have a free one, which is required to
332        * bump the current write buffer.
333        *
334        * @param sizeNeeded used only if running in memory. Size of the log buffer
335        * buffer that is needed.
336        *
337        * @param flushWriteQueue true if data is written to log, otherwise data
338        * may be placed on the write queue.
339        */
340       void bumpAndWriteDirty(int sizeNeeded, boolean flushWriteQueue) {
341  
342           /*
343            * Write the currentWriteBuffer to the file and reset
344            * currentWriteBuffer.
345            */
346           if (!bumpCurrent(sizeNeeded)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
347  
348               /*
349                * Could not bump the current write buffer; no clean buffers.
350                * Write the current dirty buffers so we can bump.
351                */
352               writeDirty(flushWriteQueue);
353  
354               if (bumpCurrent(sizeNeeded)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
355  
356                   /*
357                    * Since we have the log write latch we should be
358                    * able to bump the current buffer.
359                    */
360                   writeDirty(flushWriteQueue);
361               } else {
362                   /* should not ever get here */
363                   throw EnvironmentFailureException.unexpectedState(
364                       envImpl, "No free log buffers.");
365               }
366  
367           } else {
368  
369               /*
370                * Since we have the log write latch we should be
371                * able to bump the current buffer.
372                */
373               writeDirty(flushWriteQueue);
374           }
375       }

385       private int getNextSlot(int slotNumber) {
386           assert bufferPoolLatch.isExclusiveOwner(); → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
387           return (slotNumber < (bufferPool.size() -1)) ? ++slotNumber : 0; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
388       }

390       /**
391        * Writes the dirty log buffers.
392        *
393        * No latches need be held.
394        *
395        * Note that if no buffers are dirty, nothing will be written, even when
396        * data is buffered in the write queue and flushWriteQueue is true.
397        * If we were to allow a condition where 1) the write queue is non-empty,
398        * 2) there are no dirty log buffers, and 3) the current write buffer is
399        * empty, then a flushNoSync at that time won't flush the write queue.
400        * This should never happen because: a) flushNoSync leaves the write queue
401        * empty, b) LogManager.serialLogWork leaves the current write buffer
402        * non-empty, and c) writeDirty(false) doesn't change the state of the
403        * current write buffer. TODO: Confirm this with a more detailed analysis.
404        *
405        * @param flushWriteQueue true then data is written to file otherwise
406        * the data may be placed on the FileManager WriteQueue.
407        */
408       void writeDirty(boolean flushWriteQueue) {
409           bufferPoolLatch.acquireExclusive();
410           try {
411               if (dirtyStart < 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
412                   return;
413               }
414               boolean process = true;
415               do {
416                   LogBuffer lb = bufferPool.get(dirtyStart);
417                   lb.waitForZeroAndLatch();
418                   try {
419                       writeBufferToFile(lb, flushWriteQueue);
420                   } finally {
421                       lb.release();
422                   }
423                   if (dirtyStart == dirtyEnd) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
424                       process = false;
425                   } else {
426                       dirtyStart = getNextSlot(dirtyStart);
427                   }
428               } while (process); → [ALLOWCREATE, TRANSACTIONAL, READONLY]
429               dirtyStart = -1;
430               dirtyEnd = -1;
431           } finally {
432               bufferPoolLatch.releaseIfOwner();
433           }
434       }

436       /**
437        * Writes a log buffer.
438        *
439        * The LWL or buffer latch must be held.
440        *
441        * @param latchedBuffer buffer to write
442        *
443        * @param flushWriteQueue true then data is written to file otherwise
444        * the data may be placed on the FileManager WriteQueue.
445        */
446       private void writeBufferToFile(LogBuffer latchedBuffer,
447                                      boolean flushWriteQueue) {
448  
449           if (runInMemory) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
450               return;
451           }
452  
453           /*
454            * Check for an invalid env while buffer is latched and before writing.
455            * This is necessary to prevent writing a buffer corruption that was
456            * detected during a read from the buffer. The read will invalidate the
457            * env while holding the buffer latch.
458            */
459           envImpl.checkIfInvalid();
460  
461           try {
462               ByteBuffer currentByteBuffer = latchedBuffer.getDataBuffer();
463               int savePosition = currentByteBuffer.position();
464               int saveLimit = currentByteBuffer.limit();
465               currentByteBuffer.flip();
466  
467               /*
468                * If we're configured for writing (not memory-only situation),
469                * write this buffer to disk and find a new buffer to use.
470                */
471               try {
472                   fileManager.writeLogBuffer(latchedBuffer, flushWriteQueue);
473               } catch (Throwable t) {
474                   currentByteBuffer.position(savePosition);
475                   currentByteBuffer.limit(saveLimit);
476  
477                   /*
478                    * Exceptions thrown during logging are expected to be
479                    * fatal. Ensure that the environment is invalidated
480                    * when a non-fatal exception is unexpectedly thrown.
481                    */
482                    if (t instanceof EnvironmentFailureException) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
483  
484                       /*
485                        * If we've already invalidated the environment,
486                        * re-throw so as not to excessively wrap the
487                        * exception.
488                        */
489                       if (!envImpl.isValid()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
490                           throw (EnvironmentFailureException)t;
491                       }
492                       /* Otherwise, invalidate the environment. */
493                       throw EnvironmentFailureException.unexpectedException(
494                           envImpl, (EnvironmentFailureException)t);
495                   } else if (t instanceof Error) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
496                       envImpl.invalidate((Error)t);
497                       throw (Error)t;
498                   } else if (t instanceof Exception) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
499                       throw EnvironmentFailureException.unexpectedException(
500                           envImpl, (Exception)t);
501                   } else {
502                       throw EnvironmentFailureException.unexpectedException(
503                           envImpl, t.getMessage(), null);
504                   }
505               }
506           } finally {
507               latchedBuffer.release();
508           }
509       }
510  
511       /**
512        * Move the current write buffer to the next. Will not bump the current
513        * write buffer if the buffer is empty.
514        *
515        * The LWL must be held.
516        *
517        * @param sizeNeeded used only if running in memory. Size of the log
518        * buffer that is needed.
519        *
520        * @return false when the buffer needs flushing, but there are no free
521        * buffers. Returns true when when running in memory, when the buffer is
522        * empty, or when the buffer is non-empty and is bumped.
523        */
524        boolean bumpCurrent(int sizeNeeded) {
525  
526           /* We're done with the buffer, flip to make it readable. */
527           bufferPoolLatch.acquireExclusive();
528           currentWriteBuffer.latchForWrite();
529  
530           LogBuffer latchedBuffer = currentWriteBuffer;
531           try {
532  
533               /*
534                * Is there anything in this write buffer?
535                */
536               if (currentWriteBuffer.getFirstLsn() == DbLsn.NULL_LSN) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
537                   return true;
538               }
539  
540               if (runInMemory) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
541                   int bufferSize = → [ALLOWCREATE, TRANSACTIONAL, READONLY]
542                       ((logBufferSize > sizeNeeded) ?
543                           logBufferSize : sizeNeeded);
544                   /* We're supposed to run in-memory, allocate another buffer. */
545                   currentWriteBuffer = new LogBuffer(bufferSize, envImpl);
546                   bufferPool.add(currentWriteBuffer);
547                   currentWriteBufferIndex = bufferPool.size() - 1;
548                   return true;
549               }
550  
551               if (dirtyStart < 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
552                   dirtyStart = currentWriteBufferIndex;
553               } else {
554                   /* Check to see if there is an undirty buffer to use. */
555                   if (getNextSlot(currentWriteBufferIndex) == dirtyStart) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
556                       nNoFreeBuffer.increment();
557                       return false;
558                   }
559               }
560  
561               dirtyEnd = currentWriteBufferIndex;
562               currentWriteBufferIndex = getNextSlot(currentWriteBufferIndex);
563               LogBuffer nextToUse = bufferPool.get(currentWriteBufferIndex);
564               LogBuffer newInitialBuffer =
565                   bufferPool.get(getNextSlot(currentWriteBufferIndex));
566               nextToUse.reinit();
567  
568               /* Assign currentWriteBuffer with the latch held. */
569               currentWriteBuffer = nextToUse;
570  
571               /* Paranoia: do this after transition to new buffer. */
572               updateMinBufferLsn(newInitialBuffer);
573               return true;
574           } finally {
575               latchedBuffer.release();
576               bufferPoolLatch.releaseIfOwner();
577           }
578       }

580       /**
581        * Set minBufferLsn to start of new initial buffer. The update occurs only
582        * after cycling once through the buffers in the pool. This is a simple
583        * implementation, and waiting until we've filled the buffer pool to
584        * initialize it is sufficient for reducing read contention in
