12  class RawCoder {
13       static void validate(FilterCoder[] filters)
14               throws UnsupportedOptionsException {
15           for(int i = 0; i < filters.length - 1; ++i) → [GZIPCOMPRESSOR]
16               if(!filters[i].nonLastOK()) { → [GZIPCOMPRESSOR]
17                   throw new UnsupportedOptionsException(
18                           "Unsupported XZ filter chain");
19               }
20  
21           if(!filters[filters.length - 1].lastOK()) { → [GZIPCOMPRESSOR]
22               throw new UnsupportedOptionsException(
23                       "Unsupported XZ filter chain");
24           }
25  
26           int changesSizeCount = 0;
27           for(int i = 0; i < filters.length; ++i) → [GZIPCOMPRESSOR]
28               if(filters[i].changesSize()) { → [GZIPCOMPRESSOR]
29                   ++changesSizeCount;
30               }
31  
32           if(changesSizeCount > 3) { → [GZIPCOMPRESSOR]
33               throw new UnsupportedOptionsException(
34                       "Unsupported XZ filter chain");
35           }
36       }
37  }
