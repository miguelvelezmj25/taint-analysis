105  /**
106    * The Cleaner is responsible for effectively garbage collecting the JE log.
107    * It selects the least utilized log file for cleaning (see FileSelector),
108    * reads through the log file (FileProcessor) and determines whether each entry
109    * is obsolete (no longer relevant) or active (referenced by the Btree).
110    * Entries that are active are migrated (copied) to the end of the log, and
111    * finally the cleaned file is deleted.
112    *
113    * The migration of active entries is a multi-step process that can be
114    * configured to operate in different ways. Eviction and checkpointing, as
115    * well as the cleaner threads (FileProcessor instances) are participants in
116    * this process. Migration may be immediate or lazy.
117    *
118    * Active INs are always migrated lazily, which means that they are marked
119    * dirty by the FileProcessor, and then logged later by an eviction or
120    * checkpoint. Active LNs are always migrated immediately by the FileProcessor
121    * by logging them.
122    *
123    * When the FileProcessor is finished with a file, all lazy migration for that
124    * file is normally completed by the end of the next checkpoint, if not sooner
125    * via eviction. The checkpoint/recovery mechanism will ensure that obsolete
126    * entries will not be referenced by the Btree. At the end of the checkpoint,
127    * it is therefore safe to delete the log file.
128    *
129    * There is one exception to the above paragraph. When attempting to migrate
130    * an LN, if the LN cannot be locked then we must retry the migration at a
131    * later time. Also, if a database removal is in progress, we consider all
132    * entries in the database obsolete but cannot delete the log file until
133    * database removal is complete. Such "pending" LNs and databases are queued
134    * and processed periodically during file processing and at the start of a
135    * checkpoint; see processPending(). In this case, we may have to wait for
136    * more than one checkpoint to occur before the log file can be deleted. See
137    * FileSelector and the use of the pendingLNs and pendingDBs collections.
138    */

362           lockTimeout = cm.getDuration(EnvironmentParams.CLEANER_LOCK_TIMEOUT);
363  
364           readBufferSize = cm.getInt(EnvironmentParams.CLEANER_READ_SIZE);

369

371               cm.getInt(EnvironmentParams.CLEANER_LOOK_AHEAD_CACHE_SIZE);
372  
373           nDeadlockRetries = cm.getInt(EnvironmentParams.CLEANER_DEADLOCK_RETRY);
374  
375           expunge = cm.getBoolean(EnvironmentParams.CLEANER_REMOVE);
376  
377           useDeletedDir =
378               cm.getBoolean(EnvironmentParams.CLEANER_USE_DELETED_DIR);
379  
380           twoPassGap =
381               cm.getInt(EnvironmentParams.CLEANER_TWO_PASS_GAP);
382  
383           twoPassThreshold =
384               cm.getInt(EnvironmentParams.CLEANER_TWO_PASS_THRESHOLD);
385  
386           if (twoPassThreshold == 0) {
387               twoPassThreshold =
388                   cm.getInt(EnvironmentParams.CLEANER_MIN_UTILIZATION) - 5;
389           }
390  
391           gradualExpiration =
392               cm.getBoolean(EnvironmentParams.CLEANER_GRADUAL_EXPIRATION);
393  
394           dbCacheClearCount =
395               cm.getInt(EnvironmentParams.ENV_DB_CACHE_CLEAR_COUNT);
396  
397           int nThreads = cm.getInt(EnvironmentParams.CLEANER_THREADS);
398           assert nThreads > 0;
399  
400           if (nThreads != threads.length) {
401  
402               /* Shutdown threads when reducing their number. */
403               for (int i = nThreads; i < threads.length; i += 1) {
404                   if (threads[i] == null) {
405                       continue;
406                   }
407                   threads[i].shutdown();
408                   threads[i] = null;
409               }
410  
411               /* Copy existing threads that are still used. */
412               FileProcessor[] newThreads = new FileProcessor[nThreads];
413               for (int i = 0; i < nThreads && i < threads.length; i += 1) {
414                   newThreads[i] = threads[i];
415               }
416  
417               /* Don't lose track of new threads if an exception occurs. */
418               threads = newThreads;
419  
420               /* Start new threads when increasing their number. */
421               for (int i = 0; i < nThreads; i += 1) {
422                   if (threads[i] != null) {
423                       continue;
424                   }
425                   threads[i] = new FileProcessor(
426                       name + '-' + (i + 1),
427                       i == 0 /*firstThread*/,
428                       env, this, profile, calculator, fileSelector);
429               }
430           }
431  
432           cleanerBytesInterval = cm.getLong(
433               EnvironmentParams.CLEANER_BYTES_INTERVAL);
434  
435           if (cleanerBytesInterval == 0) {
436               cleanerBytesInterval =
437                   cm.getLong(EnvironmentParams.LOG_FILE_MAX) / 4;
438           }
439  
440           final int wakeupInterval =
441               cm.getDuration(EnvironmentParams.CLEANER_WAKEUP_INTERVAL);
442  
443           for (FileProcessor thread : threads) {
444               if (thread == null) {
445                   continue;
446               }
447               thread.setWaitTime(wakeupInterval);
448           }
449  
450           fetchObsoleteSize =
451               cm.getBoolean(EnvironmentParams.CLEANER_FETCH_OBSOLETE_SIZE);
452  
453           minAge = cm.getInt(EnvironmentParams.CLEANER_MIN_AGE);
454           minUtilization = cm.getInt(EnvironmentParams.CLEANER_MIN_UTILIZATION);
455           minFileUtilization =
456               cm.getInt(EnvironmentParams.CLEANER_MIN_FILE_UTILIZATION);
457       }

635       /**
636        * Deletes all files that are safe-to-delete and which are not protected by
637        * a DbBackup or replication. Files are deleted only if there are no
638        * read-only processes.
639        *
640        * Log file deletion is coordinated by the use of three mechanisms:
641        *
642        * 1) To guard against read/only processes, the would-be deleter tries to
643        * get an exclusive lock on the environment. This will not be possible if a
644        * read/only process exists. File locks must be used for inter-process
645        * coordination. But note that file locks are not supported intra-process.
646        *
647        * 2) Synchronization on the protectedFileRanges field. Elements are added
648        * to and removed from the protectedFileRanges collection by DbBackup.
649        * More than one backup may be occurring at once, hence a collection of
650        * protectedFileRanges is maintained, and the files protected are the range
651        * starting with the minimum value returned by the objects in that
652        * collection.
653        *
654        * 3) In a replicated environment, files are protected from deletion by the
655        * CBVLSN (CleanerBarrier VLSN). No file greater or equal to the CBVLSN
656        * file may be deleted.
657        *
658        * For case (2) and (3), all coordinated activities -- replication, backup
659        * and file deletion -- can only be carried out by a read-write process, so
660        * we know that all activities are occurring in the same process because
661        * there can only be one JE read-write process per environment.
662        *
663        * This method is synchronized to prevent multiple threads from requesting
664        * the environment lock or deleting the same files.
665        */
666       synchronized void deleteSafeToDeleteFiles() {
667  
668           /* Fail loudly if the environment is invalid. */
669           env.checkIfInvalid();
670  
671           if (env.mayNotWrite() || !fileDeletionEnabled) {
672               return;
673           }
674  
675           final NavigableSet<Long> safeToDeleteFiles =
676               fileSelector.getSafeToDeleteFiles();
677  
678           if (safeToDeleteFiles.isEmpty()) {
679               return; /* Nothing to do. */
680           }
681  
682           final StringBuilder haMsg = new StringBuilder();
683  
684           /*
685            * Ask HA to filter the "safe to delete" file set to determine which
686            * are needed for HA purposes, and are protected. We can safely assume
687            * that if a file is declared to be unprotected by HA, and eligible to
688            * delete, it will never be deemed on a later call to be protected.
689            * This lets us avoid any synchronization between cleaning and HA.
690            */
691           NavigableSet<Long> unprotectedFiles =
692               env.getUnprotectedFileSet(safeToDeleteFiles, haMsg);
693  
694           if (unprotectedFiles == null) {
695  
696               /*
697                * The replicated node is not available, so the cleaner barrier can
698                * not be read. Don't delete any files.
699                */
700               return;
701           }
702  
703           if (unprotectedFiles.isEmpty()) {
704               /* Leave a clue for analyzing log file deletion problems. */
705               logProtectedFiles(
706                   "they are protected by replication", safeToDeleteFiles, haMsg);
707               return; /* Nothing to do. */
708           }
709  
710           /*
711            * Truncate the entries in the VLSNIndex that reference VLSNs in the
712            * files to be deleted. [#16566]
713            *
714            * This is done prior to deleting the files to ensure that the
715            * replicator removes the files from the VLSNIndex. If we were to
716            * truncate after deleting a file, we may crash before the truncation
717            * and would have to "replay" the truncation later in
718            * UtilizationProfile.populateCache. This would be more complex and
719            * the lastVLSN for the files would not be available.
720            *
721            * OTOH, if we crash after the truncation and before deleting a file,
722            * it is very likely that we will re-clean the zero utilization file
723            * and delete it later. This will only cause a redundant truncation.
724            *
725            * This is done before locking the environment to minimize the interval
726            * during which the environment is locked and read-only processes are
727            * blocked. We may unnecessarily truncate the VLSNIndex if we can't
728            * lock the environment, but that is a lesser priority.
729            *
730            * We intentionally do not honor the protected file ranges specified by
731            * DbBackups when truncating, because the VLSNIndex is protected only
732            * by the CBVLSN. Luckily, this also means we do not need to
733            * synchronize on protectedFileRanges while truncating, and DbBackups
734            * will not be blocked by this potentially expensive operation.
735            */

737               new Long[unprotectedFiles.size()]);

739           for (int i = unprotectedFilesArray.length - 1; i >= 0; i -= 1) {

744                * search from high to low because some files may not contain a
745                * VLSN. If the truncate does have to do work, the VLSNIndex will
746                * ensure that the change is fsynced to disk. [#20702]
747                */

749               if ((lastVlsn != null) && !lastVlsn.isNull()) {
750                   env.vlsnHeadTruncate(lastVlsn, fileNum);

757            * with the environment open read-only and we can't delete any files.
758            */
759           final FileManager fileManager = env.getFileManager();

761               logProtectedFiles(
762                   "of read-only processes", safeToDeleteFiles, haMsg);
