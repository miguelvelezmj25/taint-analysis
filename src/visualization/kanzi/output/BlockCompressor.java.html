56       public BlockCompressor(Map<String, Object> map, ExecutorService threadPool) {

58           this.verbosity = (Integer) map.remove("verbose");
59           Boolean bForce = (Boolean) map.remove("overwrite");
60           this.overwrite = (bForce == null) ? false : bForce;
61           this.inputName = (String) map.remove("inputName");
62           this.outputName = (String) map.remove("outputName");
63           String strTransf;
64           String strCodec;
65  
66           if(this.level >= 0) {
67               String tranformAndCodec = getTransformAndCodec(this.level);
68               String[] tokens = tranformAndCodec.split("&");
69               strTransf = tokens[0];
70               strCodec = tokens[1];
71           }
72           else {
73               strTransf = (String) map.remove("transform");
74               strCodec = (String) map.remove("entropy");
75           }
76  
77           this.codec = (strCodec == null) ? "HUFFMAN" : strCodec;

85           this.checksum = (bChecksum == null) ? false : bChecksum;
86           this.jobs = (Integer) map.remove("jobs");
87           this.pool = (this.jobs == 1) ? null :
88                   ((threadPool == null) ? Executors.newCachedThreadPool() : threadPool);
89           this.ownPool = (threadPool == null) && (this.pool != null);
90           this.listeners = new ArrayList<>(10);
91  
92           if(this.verbosity > 2) {
93               this.addListener(new InfoPrinter(this.verbosity, InfoPrinter.Type.ENCODING, System.out));
94           }

102       public BlockCompressor(int verbose, boolean force, int level, int block, String entropy, String transform, boolean checksum, int jobs, Map<String, Object> map, ExecutorService threadPool) {
103           this.level = level;
104           this.verbosity = verbose;
105           Boolean bForce = force;
106           this.overwrite = (bForce == null) ? false : bForce; → [FORCE]
107           this.inputName = (String) map.remove("inputName");
108           this.outputName = (String) map.remove("outputName");
109           String strTransf;
110           String strCodec;
111  
112           if(this.level >= 0) { → [LEVEL]
113               String tranformAndCodec = getTransformAndCodec(this.level);
114               String[] tokens = tranformAndCodec.split("&");
115               strTransf = tokens[0];
116               strCodec = tokens[1];
117           }
118           else {
119               strTransf = transform;
120               strCodec = entropy;
121           }
122  
123           this.codec = (strCodec == null) ? "HUFFMAN" : strCodec; → [LEVEL, ENTROPY]
124           Integer iBlockSize = block;
125           this.blockSize = (iBlockSize == null) ? 1024 * 1024 : iBlockSize; → [BLOCK]
126  
127           // Extract transform names. Curate input (EG. NONE+NONE+xxxx => xxxx)
128           ByteFunctionFactory bff = new ByteFunctionFactory();
129           this.transform = (strTransf == null) ? "BWT+MTFT+ZRLT" : bff.getName(bff.getType(strTransf)); → [TRANSFORM, LEVEL]
130           Boolean bChecksum = checksum;
131           this.checksum = bChecksum == null ? false : bChecksum; → [CHECKSUM]
132           this.jobs = jobs;

138           if(this.verbosity > 2) { → [VERBOSE]
139               this.addListener(new InfoPrinter(this.verbosity, InfoPrinter.Type.ENCODING, System.out));

142           if((this.verbosity > 0) && (map.size() > 0)) { → [VERBOSE] & [VERBOSE]

164       private static String getTransformAndCodec(int level) {
165           switch (level) { → [LEVEL]
166               case 0:
167                   return "NONE&NONE";
168  
169               case 1:
170                   return "TEXT+LZ4&HUFFMAN";
171  
172               case 2:
173                   return "BWT+RANK+ZRLT&RANGE";
174  
175               case 3:
176                   return "BWT+RANK+ZRLT&FPAQ";
177  
178               case 4:
179                   return "BWT&CM";
180  
181               case 5:
182                   return "RLT+TEXT&TPAQ";
183  
184               default:
185                   return "Unknown&Unknown";
186           }

189       public void dispose() {
190           try {
191               if(this.is != null) { → [FORCE]
192                   this.is.close();
193               }
194           } catch (IOException ioe) {
195            /* ignore */
196           }
197  
198           try {
199               if(this.cos != null) { → [CHECKSUM, FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
200                   this.cos.close();
201               }
202           } catch (IOException ioe) {
203               System.err.println("Compression failure: " + ioe.getMessage());
204               System.exit(Error.ERR_WRITE_FILE);
205           }
206  
207           if((this.pool != null) && (this.ownPool == true)) { → [FORCE] & [FORCE]
208               this.pool.shutdown();
209           }
210       }

217       // Return status (success = 0, error < 0)
218       @Override
219       public Integer call() {
220           boolean printFlag = this.verbosity > 2; → [VERBOSE]
221           printOut("Kanzi 1.1 (C) 2017, Frederic Langlet", this.verbosity >= 1); → [VERBOSE]
222           printOut("Input file name set to '" + this.inputName + "'", printFlag);
223           printOut("Output file name set to '" + this.outputName + "'", printFlag);
224           printOut("Block size set to " + this.blockSize + " bytes", printFlag);
225           printOut("Verbosity set to " + this.verbosity, printFlag);
226           printOut("Overwrite set to " + this.overwrite, printFlag);
227           printOut("Checksum set to " + this.checksum, printFlag);
228  
229           if(this.level < 0) { → [LEVEL]
230               String etransform = ("NONE".equals(this.transform)) ? "no" : this.transform; → [TRANSFORM, LEVEL]
231               printOut("Using " + etransform + " transform (stage 1)", printFlag);
232               String ecodec = ("NONE".equals(this.codec)) ? "no" : this.codec; → [LEVEL, ENTROPY]
233               printOut("Using " + ecodec + " entropy codec (stage 2)", printFlag);
234           }
235           else {
236               printOut("Compression level set to " + this.level, printFlag);
237           }
238  
239           printOut("Using " + this.jobs + " job" + ((this.jobs > 1) ? "s" : ""), printFlag);
240  
241           OutputStream os;
242  
243           try {
244               if(this.outputName.equalsIgnoreCase("NONE")) {
245                   os = new NullOutputStream();
246               }
247               else if(this.outputName.equalsIgnoreCase("STDOUT")) {
248                   os = System.out;
249               }
250               else {
251                   File output = new File(this.outputName);
252  
253                   if(output.exists()) {
254                       if(output.isDirectory()) {
255                           System.err.println("The output file is a directory");
256                           return Error.ERR_OUTPUT_IS_DIR;
257                       }
258  
259                       if(this.overwrite == false) { → [FORCE]
260                           System.err.println("The output file exists and the 'overwrite' command "
261                                   + "line option has not been provided");
262                           return Error.ERR_OVERWRITE_FILE;
263                       }
264  
265                       Path path1 = FileSystems.getDefault().getPath(this.inputName).toAbsolutePath();
266                       Path path2 = FileSystems.getDefault().getPath(this.outputName).toAbsolutePath();
267  
268                       if(path1.equals(path2)) { → [FORCE]
269                           System.err.println("The input and output files must be different");
270                           return Error.ERR_CREATE_FILE;
271                       }
272                   }
273  
274                   os = new FileOutputStream(output);
275               }
276  
277               try {
278                   this.cos = new CompressedOutputStream(this.codec, this.transform,
279                           os, this.blockSize, this.checksum,
280                           this.pool, this.jobs);
281  
282                   for(Listener bl : this.listeners) → [FORCE]
283                       this.cos.addListener(bl);
284               } catch (Exception e) {
285                   System.err.println("Cannot create compressed stream: " + e.getMessage());
286                   return Error.ERR_CREATE_COMPRESSOR;
287               }
288           } catch (Exception e) {
289               System.err.println("Cannot open output file '" + this.outputName + "' for writing: " + e.getMessage());
290               return Error.ERR_CREATE_FILE;
291           }
292  
293           try {
294               this.is = (this.inputName.equalsIgnoreCase("STDIN")) ? System.in : new FileInputStream(this.inputName); → [FORCE]
295           } catch (Exception e) {
296               System.err.println("Cannot open input file '" + this.inputName + "': " + e.getMessage());
297               return Error.ERR_OPEN_FILE;
298           }
299  
300           // Encode
301           printFlag = this.verbosity > 1; → [FORCE, VERBOSE]
302           printOut("Encoding ...", printFlag);
303           int read = 0;
304           SliceByteArray sa = new SliceByteArray(new byte[DEFAULT_BUFFER_SIZE], 0);
305           int len;
306  
307           if(this.listeners.size() > 0) { → [FORCE]
308               Event evt = new Event(Event.Type.COMPRESSION_START, -1, 0);
309               Listener[] array = this.listeners.toArray(new Listener[this.listeners.size()]);
310               notifyListeners(array, evt);
311           }
312  
313           long before = System.nanoTime();
314  
315           try {
316               while (true) {
317                   try {
318                       len = this.is.read(sa.array, 0, sa.length);
319                   } catch (Exception e) {
320                       System.err.print("Failed to read block from file '" + this.inputName + "': ");
321                       System.err.println(e.getMessage());
322                       return Error.ERR_READ_FILE;
323                   }
324  
325                   if(len <= 0) { → [FORCE]
326                       break;
327                   }
328  
329                   // Just write block to the compressed output stream !
330                   read += len;
331                   this.cos.write(sa.array, 0, len);
332               }
333           } catch (kanzi.io.IOException e) {
334               System.err.println(e.getMessage());
335               return e.getErrorCode();
336           } catch (Exception e) {
337               System.err.println("An unexpected condition happened. Exiting ...");
338               System.err.println(e.getMessage());
339               return Error.ERR_UNKNOWN;
340           } finally {
341               // Close streams to ensure all data are flushed
342               this.dispose();
343  
344               try {
345                   os.close();
346               } catch (IOException e) {
347                   // Ignore
348               }
349           }
350  
351           if(read == 0) { → [FORCE]
352               System.out.println("Empty input file ... nothing to do");
353               return WARN_EMPTY_INPUT;
354           }
355  
356           long after = System.nanoTime();
357           long delta = (after - before) / 1000000L; // convert to ms
358           printOut("", this.verbosity >= 1); → [FORCE, VERBOSE]
359           printOut("Encoding: " + delta + " ms", printFlag);
360           printOut("Input size: " + read, printFlag);
361           printOut("Output size: " + this.cos.getWritten(), printFlag);
362           float f = this.cos.getWritten() / (float) read;
363           printOut("Ratio: " + String.format("%1$.6f", f), printFlag);
364           printOut("Encoding: " + read + " => " + this.cos.getWritten() + → [FORCE, VERBOSE]
365                   " bytes in " + delta + " ms", this.verbosity == 1);
366  
367           if(delta > 0) { → [FORCE]
368               printOut("Throughput (KB/s): " + (((read * 1000L) >> 10) / delta), printFlag);
369           }
370  
371           printOut("", this.verbosity >= 1); → [FORCE, VERBOSE]
372  
373           if(this.listeners.size() > 0) { → [FORCE]
374               Event evt = new Event(Event.Type.COMPRESSION_END, -1, this.cos.getWritten());
375               Listener[] array = this.listeners.toArray(new Listener[this.listeners.size()]);
376               notifyListeners(array, evt);
377           }
378  
379           return 0;
380       }
