21  package weka.core;
22  
23  import java.util.Properties;
24  
25  /**
26    * Extends Properties to allow the value of a system property (if set) to
27    * override that which has been loaded/set. This allows the user to override
28    * from the command line any property that is loaded into a property file via
29    * weka.core.Utils.readProperties().
30    *
31    * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
32    * @version $Revision: 12232 $
33    */
34  public class EnvironmentProperties extends Properties {
35  
36       protected transient Environment m_env = Environment.getSystemWide();
37  
38       public EnvironmentProperties() {
39           super();
40       }
41  
42       public EnvironmentProperties(Properties props) {
43           super(props);
44       }
45  
46       @Override
47       public String getProperty(String key) {
48           // allow system-wide properties to override
49           if(m_env == null) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
50               m_env = Environment.getSystemWide();
51           }
52           String result = m_env.getVariableValue(key);
53  
54           if(result == null) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
55               result = super.getProperty(key);
56           }
57  
58           return result;
59       }
60  }
