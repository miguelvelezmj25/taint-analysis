7  import org.prevayler.Clock;

9  import java.util.Date;
10  
11  public class PausableClock implements Clock {
12  
13       private final Clock _realClock;
14       private final BrokenClock _brokenClock = new BrokenClock();
15       private Clock _activeClock;
16  
17  
18       public PausableClock(Clock realClock) {
19           _realClock = realClock;  → 6
20           resume();
21       }  → 7
22  
23       public synchronized Date time() {
24           return _activeClock.time();
25       }
26  
27       public synchronized void pause() {
28           advanceTo(_realClock.time());  → 28
29           _activeClock = _brokenClock;
30       }
31  
32       public void advanceTo(Date time) {
33           _brokenClock.advanceTo(time);  → 29 → 30
34       }
35  
36       public synchronized void resume() {
37           _activeClock = _realClock;
38       }
39  
40       public Date realTime() {
41           return _realClock.time();
42       }
43  }
