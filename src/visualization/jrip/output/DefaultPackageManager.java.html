55  /**
56    * A concrete implementation of PackageManager that uses Java properties
57    * files/class to manage package meta data. Assumes that meta data for
58    * individual packages is stored on the central repository (accessible via http)
59    * in properties files that live in a subdirectory with the same name as the
60    * package. Furthermore, each property file is assumed to be named as the
61    * version number of the package in question with a ".props" extension. A
62    * "Latest.props" file should exist for each package and should always hold meta
63    * data on the latest version of a package.
64    *
65    * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
66    * @version $Revision: 52515 $

183     protected File downloadArchive(URL packageURL, String fileExtension,
184       PrintStream... progress) throws Exception {
185       String packageArchiveName = packageURL.toString();
186  
187       packageArchiveName =
188         packageArchiveName.substring(0,
189           packageArchiveName.lastIndexOf("." + fileExtension) + 3);
190  
191       packageArchiveName =
192         packageArchiveName.substring(0, packageArchiveName.lastIndexOf('.'));
193  
194       packageArchiveName =
195         packageArchiveName.substring(packageArchiveName.lastIndexOf('/'),
196           packageArchiveName.length());
197  
198       // make a temp file to hold the downloaded archive
199       File tmpDownload =
200         File.createTempFile(packageArchiveName, "." + fileExtension);
201  
202       for (PrintStream progres : progress) {
203         progres.println("[DefaultPackageManager] Tmp file: "
204           + tmpDownload.toString());
205       }
206  
207       URLConnection conn = getConnection(packageURL);
208  
209       BufferedInputStream bi = new BufferedInputStream(conn.getInputStream());
210  
211       BufferedOutputStream bo =
212         new BufferedOutputStream(new FileOutputStream(tmpDownload));
213  
214       // download the archive
215       int totalBytesRead = 0;
216       while (true) {
217         synchronized (m_buffer) {
218           int amountRead = bi.read(m_buffer);
219           if (amountRead == -1) {
220             for (PrintStream progres : progress) {
221               progres.print("[DefaultPackageManager] downloaded "
222                 + (totalBytesRead / 1000) + " KB\r");
223             }
224             break;
225           }
226           bo.write(m_buffer, 0, amountRead);
227           totalBytesRead += amountRead;
228           for (PrintStream progres : progress) {
229             progres.print("%%[DefaultPackageManager] downloaded "
230               + (totalBytesRead / 1000) + " KB\r");
231           }
232         }
233       }
234  
235       bi.close();
236       bo.close();
237  
238       return tmpDownload;
239     }

609     /**
610      * Checks a given package's list of dependencies for any conflicts with the
611      * packages in the supplied Map. Any packages from this packages dependency
612      * list that are not in the Map are simply added and checkDependencies is
613      * called recursively for each.
614      *
615      * @param toCheck the package to check.
616      * @param lookup a Map of package name, Dependency pairs to check against.
617      * @param conflicts a list of Dependency objects for any conflicts that are
618      * detected.
619      * @return true if no conflicts are found.
620      * @throws Exception if a problem occurs when checking for conflicts.
621      */
622     protected static boolean checkDependencies(PackageConstraint toCheck,
623       Map<String, Dependency> lookup, Map<String, List<Dependency>> conflicts)
624       throws Exception {
625       boolean ok = true;
626  
627       // get the dependencies for the package to check
628       List<Dependency> deps = toCheck.getPackage().getDependencies();
629  
630       for (Dependency p : deps) {
631         String depName =
632           p.getTarget().getPackage().getPackageMetaDataElement("PackageName")
633             .toString();
634         if (!lookup.containsKey(depName)) {
635           // just add this package to the lookup
636           lookup.put(depName, p);
637  
638           // check its dependencies
639           ok = checkDependencies(p.getTarget(), lookup, conflicts);
640         } else {
641           // we have to see if the version number for this package is compatible
642           // with the one already in the lookup
643           Dependency checkAgainst = lookup.get(depName);
644           PackageConstraint result =
645             checkAgainst.getTarget().checkConstraint(p.getTarget());
646           if (result != null) {
647             checkAgainst.setTarget(result);
648             lookup.put(depName, checkAgainst);
649           } else {
650             // there is a conflict here
651             List<Dependency> conflictList = conflicts.get(depName);
652             conflictList.add(p);
653             ok = false;
654           }
655         }
656       }
657  
658       return ok;
659     }
