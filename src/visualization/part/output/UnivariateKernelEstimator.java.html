103     /**

109      * than m_MinWidth, the width is set to the value of m_MinWidth.

134     /**
135      * Returns the interval for the given confidence value.
136      *
137      * @param conf the confidence value in the interval [0, 1]
138      * @return the interval
139      */
140     @Override
141     public double[][] predictIntervals(double conf) {
142  
143       // Update the bandwidth
144       updateWidth();
145  
146       // Compute minimum and maximum value, and delta
147       double val = Statistics.normalInverse(1.0 - (1.0 - conf) / 2);
148       double min = m_TM.firstKey() - val * m_Width;
149       double max = m_TM.lastKey() + val * m_Width;
150       double delta = (max - min) / m_NumIntervals;
151  
152       // Create array with estimated probabilities
153       double[] probabilities = new double[m_NumIntervals];
154       double leftVal = Math.exp(logDensity(min));
155       for (int i = 0; i < m_NumIntervals; i++) {
156         double rightVal = Math.exp(logDensity(min + (i + 1) * delta));
157         probabilities[i] = 0.5 * (leftVal + rightVal) * delta;
158         leftVal = rightVal;
159       }
160  
161       // Sort array based on area of bin estimates
162       int[] sortedIndices = Utils.sort(probabilities);
163  
164       // Mark the intervals to use
165       double sum = 0;
166       boolean[] toUse = new boolean[probabilities.length];
167       int k = 0;
168       while ((sum < conf) && (k < toUse.length)) {
169         toUse[sortedIndices[toUse.length - (k + 1)]] = true;
170         sum += probabilities[sortedIndices[toUse.length - (k + 1)]];
171         k++;
172       }
173  
174       // Don't need probabilities anymore
175       probabilities = null;
176  
177       // Create final list of intervals
178       ArrayList<double[]> intervals = new ArrayList<double[]>();
179  
180       // The current interval
181       double[] interval = null;
182  
183       // Iterate through kernels
184       boolean haveStartedInterval = false;
185       for (int i = 0; i < m_NumIntervals; i++) {
186  
187         // Should the current bin be used?
188         if (toUse[i]) {
189  
190           // Do we need to create a new interval?
191           if (haveStartedInterval == false) {
192             haveStartedInterval = true;
193             interval = new double[2];
194             interval[0] = min + i * delta;
195           }
196  
197           // Regardless, we should update the upper boundary
198           interval[1] = min + (i + 1) * delta;
199         } else {
200  
201           // We need to finalize and store the last interval
202           // if necessary.
203           if (haveStartedInterval) {
204             haveStartedInterval = false;
205             intervals.add(interval);
206           }
207         }
208       }
209  
210       // Add last interval if there is one
211       if (haveStartedInterval) {
212         intervals.add(interval);
213       }
214  
215       return intervals.toArray(new double[0][0]);
216     }

218     /**
219      * Returns the quantile for the given percentage.
220      *
221      * @param percentage the percentage
222      * @return the quantile
223      */
224     @Override
225     public double predictQuantile(double percentage) {
226  
227       // Update the bandwidth
228       updateWidth();
229  
230       // Compute minimum and maximum value, and delta
231       double val = Statistics.normalInverse(1.0 - (1.0 - 0.95) / 2);
232       double min = m_TM.firstKey() - val * m_Width;
233       double max = m_TM.lastKey() + val * m_Width;
234       double delta = (max - min) / m_NumIntervals;
235  
236       double sum = 0;
237       double leftVal = Math.exp(logDensity(min));
238       for (int i = 0; i < m_NumIntervals; i++) {
239         if (sum >= percentage) {
240           return min + i * delta;
241         }
242         double rightVal = Math.exp(logDensity(min + (i + 1) * delta));
243         sum += 0.5 * (leftVal + rightVal) * delta;
244         leftVal = rightVal;
245       }
246       return max;
247     }

306     /**
