40  /**
41    * INFileReader supports recovery by scanning log files during the IN rebuild
42    * pass. It looks for internal nodes (all types), segregated by whether they
43    * belong to the main tree or the duplicate trees.
44    *
45    * <p>This file reader can also be run in tracking mode to keep track of the
46    * maximum node ID, database ID and txn ID seen so those sequences can be
47    * updated properly at recovery. In this mode it also performs utilization
48    * counting. It is only run once in tracking mode per recovery, in the first
49    * phase of recovery.</p>
50    */
51  public class INFileReader extends FileReader {
52  
53       /* Information about the last entry seen. */
54       private boolean lastEntryWasDelete;
55       private boolean lastEntryWasDupDelete;
56       private LogEntryType fromLogType;
57       private boolean isProvisional;
58  
59       /*
60        * targetEntryMap maps DbLogEntryTypes to log entries. We use this
61        * collection to find the right LogEntry instance to read in the current
62        * entry.
63        */
64       private Map<LogEntryType, LogEntry> targetEntryMap;
65       private LogEntry targetLogEntry;
66  
67       /* Set of non-target log entry types for ID tracking. */
68       private Set<LogEntryType> idTrackingSet;
69       /* Cache of non-target log entries for ID tracking. */
70       private Map<LogEntryType, LogEntry> idTrackingMap;
71  
72       private boolean trackIds;
73       private long minReplicatedNodeId;
74       private long maxNodeId;
75       private long minReplicatedDbId;
76       private long maxDbId;
77       private long minReplicatedTxnId;
78       private long maxTxnId;

80

83       private RecoveryUtilizationTracker tracker;
84

110        * Create with logVersion8UpgradeDbs and logVersion8UpgradeDeltas params.
111        */
112       public INFileReader(EnvironmentImpl env,
113                           int readBufferSize,
114                           long startLsn,
115                           long finishLsn,
116                           boolean trackIds,
117                           long partialCkptStart,
118                           long ckptEnd,
119                           RecoveryUtilizationTracker tracker,
120                           Set<DatabaseId> logVersion8UpgradeDbs,
121                           AtomicBoolean logVersion8UpgradeDeltas)
122           throws DatabaseException {
123  
124           super(env, readBufferSize, true, startLsn, null,
125                 DbLsn.NULL_LSN, finishLsn);
126  
127           this.trackIds = trackIds;
128           this.ckptEnd = ckptEnd;
129           targetEntryMap = new HashMap<LogEntryType, LogEntry>();
130  
131           if (trackIds) {
132               maxNodeId = 0;
133               maxDbId = 0;
134               maxTxnId = 0;
135               minReplicatedNodeId = 0;
136               minReplicatedDbId = DbTree.NEG_DB_ID_START;
137               minReplicatedTxnId = 0;
138               this.tracker = tracker;
139               this.partialCkptStart = partialCkptStart;
140  
141               idTrackingSet = new HashSet<LogEntryType>();
142               idTrackingMap = new HashMap<LogEntryType, LogEntry>();
143  
144               /*
145                * Need all nodes for tracking:
146                * - Need all INs for node ID tracking.
147                * - Need all LNs for obsolete tracking.
148                * - Need txnal LNs for txn ID tracking.
149                * - Need FileSummaryLN for obsolete tracking.
150                * - Need MapLN for obsolete and DB ID tracking.
151                * - Need BIN-delta for obsolete tracking.
152                */
153               for (LogEntryType entryType : LogEntryType.getAllTypes()) {
154                   if (entryType.isNodeType()) {
155                       idTrackingSet.add(entryType);
156                   }
157               }
158               idTrackingSet.add(LogEntryType.LOG_BIN_DELTA);
159               idTrackingSet.add(LogEntryType.LOG_OLD_BIN_DELTA);
160  
161               /* For tracking VLSNs. */
162               vlsnProxy = envImpl.getVLSNProxy();
163               idTrackingSet.add(LogEntryType.LOG_ROLLBACK_START);
164  
165               /* For checking for log version 8 upgrade errors. */
166               this.logVersion8UpgradeDbs = logVersion8UpgradeDbs;
167               this.logVersion8UpgradeDeltas = logVersion8UpgradeDeltas;
168           }

180       /*
181        * Utilization Tracking
182        * --------------------
183        * This class counts all new log entries and obsolete INs. Obsolete LNs,
184        * on the other hand, are counted by RecoveryManager undo/redo.
185        *
186        * Utilization counting is done in the first recovery pass where IDs are
187        * tracked (trackIds=true). Processing is split between isTargetEntry
188        * and processEntry as follows.
189        *
190        * isTargetEntry counts only new non-node entries; this can be done very
191        * efficiently using only the LSN and entry type, without reading and
192        * unmarshalling the entry.
193        *
194        * processEntry counts new node entries and obsolete INs.
195        *
196        * processEntry also resets (sets all counters to zero and clears obsolete
197        * offsets) the tracked summary for a file or database when a FileSummaryLN
198        * or MapLN is encountered. This clears the totals that have accumulated
199        * during this recovery pass for entries prior to that point. We only want
200        * to count utilization for entries after that point.
201        *
202        * In addition, when processEntry encounters a FileSummaryLN or MapLN, its
203        * LSN is recorded in the tracker. This information is used during IN and
204        * LN utilization counting. For each file, knowing the LSN of the last
205        * logged FileSummaryLN for that file allows the undo/redo code to know
206        * whether to perform obsolete countng. If the LSN of the FileSummaryLN is
207        * less than (to the left of) the LN's LSN, obsolete counting should be
208        * performed. If it is greater, obsolete counting is already included in
209        * the logged FileSummaryLN and should not be repeated to prevent double
210        * counting. The same thing is true of counting per-database utilization
211        * relative to the LSN of the last logged MapLN.
212        */
213  
214       /**
215        * If we're tracking node, database and txn IDs, we want to see all node
216        * log entries. If not, we only want to see IN entries.
217        */
218       @Override
219       protected boolean isTargetEntry()
220           throws DatabaseException {
221  
222           lastEntryWasDelete = false;
223           lastEntryWasDupDelete = false;
224           targetLogEntry = null;
225           isProvisional = currentEntryHeader.getProvisional().isProvisional
226               (getLastLsn(), ckptEnd);
227  
228           /* Get the log entry type instance we need to read the entry. */
229           fromLogType = LogEntryType.findType(currentEntryHeader.getType());
230           LogEntry possibleTarget = targetEntryMap.get(fromLogType);
231  
232           /* Always select a non-provisional target entry. */
233           if (!isProvisional) {
234               targetLogEntry = possibleTarget;
235           }
236  
237           /* Recognize IN deletion. */
238           if (LogEntryType.LOG_IN_DELETE_INFO.equals(fromLogType)) {
239               lastEntryWasDelete = true;
240           }
241           if (LogEntryType.LOG_IN_DUPDELETE_INFO.equals(fromLogType)) {
242               lastEntryWasDupDelete = true;
243           }
244  
245           /* If we're not tracking IDs, select only the targeted entry. */
246           if (!trackIds) {
247               return (targetLogEntry != null);
248           }
249  
250           /*
251            * Count all non-node non-delta entries except for the file header as
252            * new. UtilizationTracker does not count the file header. Node/delta
253            * entries will be counted in processEntry. Null is passed for the
254            * database ID; it is only needed for node entries.
255            */
256           if (!fromLogType.isNodeType() &&
257               !fromLogType.equals(LogEntryType.LOG_BIN_DELTA) &&
258               !fromLogType.equals(LogEntryType.LOG_OLD_BIN_DELTA) &&
259               !LogEntryType.LOG_FILE_HEADER.equals(fromLogType)) {
260               tracker.countNewLogEntry(getLastLsn(),
261                                        fromLogType,
262                                        currentEntryHeader.getSize() +
263                                        currentEntryHeader.getItemSize(),
264                                        null); // DatabaseId
265           }
266  
267           /*
268            * When we encouter a DbTree log entry, reset the tracked summary for
269            * the ID and Name mapping DBs. This clears what we accumulated
270            * previously for these databases during this recovery pass. Save the
271            * LSN for these databases for use by undo/redo.
272            */
273           if (LogEntryType.LOG_DBTREE.equals(fromLogType)) {
274               tracker.saveLastLoggedMapLN(DbTree.ID_DB_ID, getLastLsn());
275               tracker.saveLastLoggedMapLN(DbTree.NAME_DB_ID, getLastLsn());
276               tracker.resetDbInfo(DbTree.ID_DB_ID);
277               tracker.resetDbInfo(DbTree.NAME_DB_ID);
278           }
279  
280           /* Track VLSNs in the log entry header of all replicated entries. */
281           if (currentEntryHeader.getReplicated()) {
282               vlsnProxy.trackMapping(getLastLsn(),
283                                      currentEntryHeader,
284                                      null /*targetLogEntry*/);
285           }
286  
287           /* Return true if this logrec should be passed on to processEntry. */
288           return (targetLogEntry != null ||
289                   idTrackingSet.contains(fromLogType));
290       }
291  
292       /**
293        * This reader returns non-provisional INs and IN delete entries.
294        * In tracking mode, it may also scan log entries that aren't returned:
295        * -to set the sequences for txn, node, and database ID.
296        * -to update utilization and obsolete offset information.
297        * -for VLSN mappings for recovery
298        */
299       protected boolean processEntry(ByteBuffer entryBuffer)
300           throws DatabaseException {
301  
302           boolean useEntry = false;
303  
304           /* Read targeted entry. */
305           if (targetLogEntry != null) {
306               targetLogEntry.readEntry(envImpl, currentEntryHeader, entryBuffer);
307               useEntry = true;
308           }
309  
310           /* If we're not tracking IDs, we're done. */
311           if (!trackIds) {
312               return useEntry;
313           }
314  
315           /* Read non-target entry. */
316           if (targetLogEntry == null) {
317               assert idTrackingSet.contains(fromLogType);
318  
319               targetLogEntry = idTrackingMap.get(fromLogType);
320               if (targetLogEntry == null) {
321                   targetLogEntry = fromLogType.getNewLogEntry();
322                   idTrackingMap.put(fromLogType, targetLogEntry);
323               }
324  
325               targetLogEntry.readEntry(envImpl, currentEntryHeader, entryBuffer);
326           }
327  
328           /*
329            * Count node and delta entries as new. Non-node/delta entries are
330            * counted in isTargetEntry.
331            */
332           if (fromLogType.isNodeType() ||
333               fromLogType.equals(LogEntryType.LOG_BIN_DELTA) ||
334               fromLogType.equals(LogEntryType.LOG_OLD_BIN_DELTA)) {
335               tracker.countNewLogEntry(getLastLsn(), fromLogType,
336                                        currentEntryHeader.getSize() +
337                                        currentEntryHeader.getItemSize(),
338                                        targetLogEntry.getDbId());
339           }
340  
341           /* Track VLSNs in RollbackStart. */
342           if (fromLogType.equals(LogEntryType.LOG_ROLLBACK_START)) {
343               vlsnProxy.trackMapping(getLastLsn(),
344                                      currentEntryHeader,
345                                      targetLogEntry);
346           }
347  
348           /* Process LN types. */
349           if (fromLogType.isLNType()) {
350  
351               LNLogEntry<?> lnEntry = (LNLogEntry<?>) targetLogEntry;
352  
353               /*
354                * When a MapLN is encountered, reset the tracked info for that
355                * DB. This clears what we have accummulated so far for the DB
356                * during this recovery pass. This is important to eliminate
357                * potential double counting of obsolete logrecs done earlier
358                * in this recovery pass.
359                *
360                * Also, save the LSN of the MapLN for use in utilization counting
361                * during LN undo/redo.
362                */
363               if (fromLogType.equals(LogEntryType.LOG_MAPLN)) { → [TRANSACTIONAL]
364  
365                   MapLN mapLN = (MapLN) lnEntry.getMainItem();
366                   DatabaseId dbId = mapLN.getDatabase().getId();
367  
368                   /* Track latest DB ID. */
369                   long dbIdVal = dbId.getId();
370                   maxDbId = (dbIdVal > maxDbId ? dbIdVal : maxDbId); → [TRANSACTIONAL]
371                   minReplicatedDbId = (dbIdVal < minReplicatedDbId ? → [TRANSACTIONAL]
372                                        dbIdVal : minReplicatedDbId);
373  
374                   tracker.resetDbInfo(dbId);
375  
376                   tracker.saveLastLoggedMapLN(dbId, getLastLsn());
377               }
378  
379               /* Track latest txn ID. */
380               if (fromLogType.isTransactional()) {
381                   long txnId = lnEntry.getTxnId().longValue();
382                   maxTxnId = (txnId > maxTxnId ? txnId : maxTxnId);
383                   minReplicatedTxnId = (txnId < minReplicatedTxnId ?
384                                         txnId : minReplicatedTxnId);
385               }
386  
387               /*
388                * When a FSLN is encountered, reset the tracked summary info for
389                * that file. This clears what we have accummulated so far for the
390                * file during this recovery pass. This is important to eliminate
391                * potential double counting of obsolete logrecs done earlier
392                * in this recovery pass.
393                *
394                * Also, save the LSN of the FSLN for use in utilization counting
395                * during LN undo/redo.
396                */
397               if (LogEntryType.LOG_FILESUMMARYLN.equals(fromLogType)) { → [TRANSACTIONAL]
398  
399                   lnEntry.postFetchInit(false /*isDupDb*/);
400  
401                   long fileNum = FileSummaryLN.getFileNumber(lnEntry.getKey());
402  
403                   tracker.resetFileInfo(fileNum);
404  
405                   tracker.saveLastLoggedFileSummaryLN(fileNum, getLastLsn());
406  
407                   /*
408                    * Do not cache the file summary in the UtilizationProfile here,
409                    * since it may be for a deleted log file. [#10395]
410                    */
411               }
412           }
413  
414           /* Process IN types. */
415           if (fromLogType.isINType()) {
416               INLogEntry<?> inEntry = (INLogEntry<?>) targetLogEntry;
417  
418               /* Keep track of the largest node ID seen. */
419               long nodeId = inEntry.getNodeId();
420               assert (nodeId != Node.NULL_NODE_ID);
421  
422               maxNodeId = (nodeId > maxNodeId ? nodeId : maxNodeId);
423  
424               minReplicatedNodeId =
425                   (nodeId < minReplicatedNodeId ? nodeId : minReplicatedNodeId);
426           }
427  
428           /* Process INContainingEntry types. */
429           if (fromLogType.isINType() ||
430               fromLogType.equals(LogEntryType.LOG_OLD_BIN_DELTA)) {
431  
432               INContainingEntry inEntry = (INContainingEntry) targetLogEntry;
433               DatabaseId dbId = inEntry.getDbId();
434  
435               long newLsn = getLastLsn();
436  
437               /*
438                * Count the previous version of this IN as obsolete. If lsn
439                * (i.e. the current version) is non-provisional, then oldLsn is
440                * indeed obsolete. However, if lsn is provisional, oldLsn is
441                * obsolete only if an ancestor of this IN has been logged non-
442                * provisionally later in the log, and unless lsn < CKPT_END
443                * we cannot know if this is true or not. For this reason, we
444                * conservatively assume that oldLsn is indeed obsolete, but we
445                * use inexact counting, so that the oldLsn value will not be
446                * recorded by the tracker. (another reason is that earlier log
447                * versions did not have a full LSN in oldLsn; they had only the
448                * file number).
449                *
450                * Notice also that there may be an FSLN logrec after lsn that
451                * includes oldLsn as an obsolete IN. If so, we are double counting
452                * here, but the double counting will go away when we later process
453                * that FSLN and as a result wipe-out the utilization info we have
454                * collected so far about the log file containing oldLsn.
455                */
456               long oldLsn = inEntry.getPrevFullLsn();
457  
458               if (oldLsn != DbLsn.NULL_LSN && !inEntry.isBINDelta()) {
459                   tracker.countObsoleteUnconditional(
460                       oldLsn, fromLogType, 0, dbId, false /*countExact*/);
461               }
462  
463               oldLsn = inEntry.getPrevDeltaLsn();
464  
465               if (oldLsn != DbLsn.NULL_LSN) {
466                   tracker.countObsoleteUnconditional(
467                       oldLsn, fromLogType, 0, dbId, false /*countExact*/);
468               }
469  
470               /*
471                * Count the current IN version as obsolete if lsn is
472                * provisional and is after partialCkptStart. In this case, the
473                * crash occurred during a ckpt and it the crash event itself
474                * that may make the current logrec be obsolete. Again, whether
475                * the lsn is indeed obsolete or not depends on whether an
476                * ancestor of this IN has been logged non-provisionally later
477                * in the log. At this point we cannot know if this is true or
478                * not. For this reason, we conservatively assume that lsn
479                * is indeed obsolete, but we use inexact counting, so that the
480                * lsn value will not be recorded by the tracker. As explained
481                * above, we may be double counting here, but this will be fixed
482                * if we later find an FSLN for the same log file as lsn.
483                *
484                * We are too conservative here in assuming that lsn is
485                * obsolete. This is because of the "grouping" behaviour of the
486                * checpointer. Specifically:
487                *
488                * Most of the provisional IN logrecs in this region of the log
489                * (i.e., after the start of an incomplete ckpt) are for BINs
490                * logged by the checkpointer (eviction during a ckpt logs dirty
491                * BINs and UINs provisionally as well). The checkpointer logs
492                * all the dirty BIN siblings and then logs their parent non-
493                * provisionally, before logging any other BINs. So, not taking
494                * eviction into account, there can be at most 128 BIN logrecs
495                * after the partialCkptStart that are trully obsolete.
496                *
497                * Note that older versions of the checkpointer did not use to
498                * group together the logging of siblings BINs and their parent.
499                * Without this grouping, the assumption done here that most
500                * provisional logrecs after partialCkptStart are obsolete is
501                * much more accurate.
502                *
503                * A potential solution: instead of counting these logrecs as
504                * obsolete here, save their LSNs on-the-side inside the tracker.
505                * When, later, a UIN N is replayed and attached to the tree,
506                * remove from the saved LSN set any LSNs that appear in the slots
507                * of N. After all REDO-INs passes are done, count as obsolete any
508                * LSNs remaining in the saved set.
509                */
510               if (isProvisional &&
511                   partialCkptStart != DbLsn.NULL_LSN &&
512                   DbLsn.compareTo(partialCkptStart, newLsn) < 0) {
513                   tracker.countObsoleteUnconditional(
514                       newLsn, fromLogType, 0, inEntry.getDbId(),
515                       false /*countExact*/);
516               }
517           }
518  
519           /*
520            * Add candidate DB IDs and note deltas for possible log version 8
521            * upgrade violations.
522            */
523           if (currentEntryHeader.getVersion() < 8) {
524               if (logVersion8UpgradeDbs != null &&
525                   fromLogType.isNodeType()) {
526                   logVersion8UpgradeDbs.add(targetLogEntry.getDbId());
527               }
528               if (logVersion8UpgradeDeltas != null &&
529                   (fromLogType.equals(LogEntryType.LOG_OLD_BIN_DELTA) ||
530                    fromLogType.equals(LogEntryType.LOG_OLD_DUP_BIN_DELTA))) {
531                   logVersion8UpgradeDeltas.set(true);
532               }
533           }
534  
535           /* Return true if this is a targeted entry. */
536           return useEntry;
537       }
