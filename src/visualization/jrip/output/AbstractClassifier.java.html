160     /**
161      * Classifies the given test instance. The instance has to belong to a dataset
162      * when it's being classified. Note that a classifier MUST implement either
163      * this or distributionForInstance().
164      *
165      * @param instance the instance to be classified
166      * @return the predicted most likely class for the instance or
167      * Utils.missingValue() if no prediction is made
168      * @exception Exception if an error occurred during the prediction
169      */
170     @Override
171     public double classifyInstance(Instance instance) throws Exception {
172  
173       double[] dist = distributionForInstance(instance);
174       if (dist == null) {
175         throw new Exception("Null distribution predicted");
176       }
177       switch (instance.classAttribute().type()) {
178       case Attribute.NOMINAL:
179         double max = 0;
180         int maxIndex = 0;
181  
182         for (int i = 0; i < dist.length; i++) {
183           if (dist[i] > max) {
184             maxIndex = i;
185             max = dist[i];
186           }
187         }
188         if (max > 0) {
189           return maxIndex;
190         } else {
191           return Utils.missingValue();
192         }
193       case Attribute.NUMERIC:
194       case Attribute.DATE:
195         return dist[0];
196       default:
197         return Utils.missingValue();
198       }
199     }

201     /**
202      * Predicts the class memberships for a given instance. If an instance is
203      * unclassified, the returned array elements must be all zero. If the class is
204      * numeric, the array must consist of only one element, which contains the
205      * predicted value. Note that a classifier MUST implement either this or
206      * classifyInstance().
207      *
208      * @param instance the instance to be classified
209      * @return an array containing the estimated membership probabilities of the
210      * test instance in each class or the numeric prediction
211      * @exception Exception if distribution could not be computed successfully
212      */
213     @Override
214     public double[] distributionForInstance(Instance instance) throws Exception {
215  
216       double[] dist = new double[instance.numClasses()];
217       switch (instance.classAttribute().type()) {
218       case Attribute.NOMINAL:
219         double classification = classifyInstance(instance);
220         if (Utils.isMissingValue(classification)) {
221           return dist;
222         } else {
223           dist[(int) classification] = 1.0;
224         }
225         return dist;
226       case Attribute.NUMERIC:
227       case Attribute.DATE:
228         dist[0] = classifyInstance(instance);
229         return dist;
230       default:
231         return dist;
232       }
233     }

266     /**
267      * Gets the current settings of the Classifier.
268      *
269      * @return an array of strings suitable for passing to setOptions
270      */
271     @Override
272     public String[] getOptions() {
273  
274       Vector<String> options = new Vector<String>();
275       for (String s : Option.getOptionsForHierarchy(this,
276         AbstractClassifier.class)) {
277         options.add(s);
278       }
279  
280       if (getDebug()) {
281         options.add("-output-debug-info");
282       }
283       if (getDoNotCheckCapabilities()) {
284         options.add("-do-not-check-capabilities");
285       }
286       if (getNumDecimalPlaces() != NUM_DECIMAL_PLACES_DEFAULT) {
287         options.add("-num-decimal-places");
288         options.add("" + getNumDecimalPlaces());
289       }
290       if (!(getBatchSize().equals(BATCH_SIZE_DEFAULT))) {
291         options.add("-batch-size");
292         options.add("" + getBatchSize());
293       }
294       return options.toArray(new String[0]);
295     }
