24  /**
25    * The Transaction object is the handle for a transaction. Methods off the
26    * transaction handle are used to configure, abort and commit the transaction.
27    * Transaction handles are provided to other Berkeley DB methods in order to
28    * transactionally protect those operations.
29    *
30    * <p>A single Transaction may be used to protect operations for any number of
31    * Databases in a given environment. However, a single Transaction may not be
32    * used for operations in more than one distinct environment.</p>
33    *
34    * <p>Transaction handles are free-threaded; transactions handles may be used
35    * concurrently by multiple threads. Once the {@link Transaction#abort
36    * Transaction.abort} or {@link Transaction#commit Transaction.commit} method
37    * is called, the handle may not be accessed again, regardless of the success
38    * or failure of the method, with one exception: the {@code abort} method may
39    * be called any number of times to simplify error handling.</p>
40    *
41    * <p>To obtain a transaction with default attributes:</p>
42    *
43    * <blockquote><pre>
44    * Transaction txn = myEnvironment.beginTransaction(null, null);
45    * </pre></blockquote>
46    *
47    * <p>To customize the attributes of a transaction:</p>
48    *
49    * <blockquote><pre>
50    * TransactionConfig config = new TransactionConfig();
51    * config.setReadUncommitted(true);
52    * Transaction txn = myEnvironment.beginTransaction(null, config);
53    * </pre></blockquote>
54    */
55  public class Transaction {
56  
57       /**
58        * The current state of the transaction.
59        *
60        * @since 5.0.48
61        */
62       public enum State {
63  
64           /**
65            * The transaction has not been committed or aborted, and can be used
66            * for performing operations. This state is also indicated if {@link
67            * #isValid} returns true. For all other states, {@link #isValid} will
68            * return false.
69            */
70           OPEN,
71  
72           /**
73            * An exception was thrown by the {@code commit} method due to an error
74            * that occurred while attempting to make the transaction durable. The
75            * transaction may or may not be locally durable, according to the
76            * {@link Durability#getLocalSync local SyncPolicy} requested.
77            * <p>
78            * This is an unusual situation and is normally due to a system
79            * failure, storage device failure, disk full condition, thread
80            * interrupt, or a bug of some kind. When a transaction is in this
81            * state, the Environment will have been {@link Environment#isValid()
82            * invalidated} by the error.
83            * <p>
84            * In a replicated environment, a transaction in this state is not
85            * transferred to replicas. If it turns out that the transaction is
86            * indeed durable, it will be transferred to replicas via normal
87            * replication mechanisms when the Environment is re-opened.
88            * <p>
89            * When the {@code commit} method throws an exception and the
90            * transaction is in the {@code POSSIBLY_COMMITTED} state, some
91            * applications may wish to perform a data query to determine whether
92            * the transaction is durable or not. Note that in the event of a
93            * system level failure, the reads themselves may be unreliable, e.g.
94            * the data may be in the file system cache but not on disk. Other
95            * applications may wish to repeat the transaction unconditionally,
96            * after resolving the error condition, particularly when the set of
97            * operations in the transaction is designed to be idempotent.
98            */
99           POSSIBLY_COMMITTED,
100  
101           /**
102            * The transaction has been committed and is locally durable according
103            * to the {@link Durability#getLocalSync local SyncPolicy} requested.
104            * <p>
105            * Note that a transaction may be in this state even when an exception
106            * is thrown by the {@code commit} method. For example, in a
107            * replicated environment, an {@link
108            * com.sleepycat.je.rep.InsufficientAcksException} may be thrown after
109            * the transaction is committed locally.
110            */
111           COMMITTED,
112  
113           /**
114            * The transaction has been invalidated by an exception and cannot be
115            * committed. See {@link OperationFailureException} for a description
116            * of how a transaction can become invalid. The application is
117            * responsible for aborting the transaction.
118            */
119           MUST_ABORT,
120  
121           /**
122            * The transaction has been aborted.
123            */
124           ABORTED,
125       }

167       /**
168        * Cause an abnormal termination of the transaction.
169        *
170        * <p>The log is played backward, and any necessary undo operations are
171        * done. Before Transaction.abort returns, any locks held by the
172        * transaction will have been released.</p>
173        *
174        * <p>In the case of nested transactions, aborting a parent transaction
175        * causes all children (unresolved or not) of the parent transaction to be
176        * aborted.</p>
177        *
178        * <p>All cursors opened within the transaction must be closed before the
179        * transaction is aborted.</p>
180        *
181        * <p>After this method has been called, regardless of its return, the
182        * {@link Transaction} handle may not be accessed again, with one
183        * exception: the {@code abort} method itself may be called any number of
184        * times to simplify error handling.</p>
185        *
186        * <p>WARNING: To guard against memory leaks, the application should
187        * discard all references to the closed handle. While BDB makes an effort
188        * to discard references from closed objects to the allocated memory for an
189        * environment, this behavior is not guaranteed. The safe course of action
190        * for an application is to discard all references to closed BDB
191        * objects.</p>
192        *
193        * @throws EnvironmentFailureException if an unexpected, internal or
194        * environment-wide failure occurs.
195        *
196        * @throws IllegalStateException if the environment has been closed, or
197        * cursors associated with the transaction are still open.
198        */
199       public synchronized void abort()
200           throws DatabaseException {
201  
202           try {
203  
204               /*
205                * If the transaction is already closed, do nothing. Do not call
206                * checkOpen in order to support any number of calls to abort().
207                */
208               if (txn == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
209                   return;
210               }
211  
212               /*
213                * Check env only after checking for closed txn, to mimic close()
214                * behavior for Cursors, etc, and avoid unnecessary exception
215                * handling. [#21264]
216                */
217               checkEnv();
218  
219               env.removeReferringHandle(this);
220               txn.abort();
221  
222               /* Remove reference to internal txn, so we can reclaim memory. */
223               setTxnNull();
224           } catch (Error E) {
225               DbInternal.getNonNullEnvImpl(env).invalidate(E);
226               throw E;
227           }
228       }

239       /**
240        * This method is intended for use with a replicated environment.
241        * <p>
242        * It returns the commitToken associated with a successful replicated
243        * commit. A null value is returned if the txn was not associated with a
244        * replicated environment, or the txn did not result in any changes to the
245        * environment. This method should only be called after the transaction
246        * has finished.
247        * <p>
248        * This method is typically used in conjunction with the <code>
249        * CommitPointConsistencyPolicy</code>.
250        *
251        * @return the token used to identify the replicated commit. Return null if
252        * the transaction has aborted, or has committed without making any
253        * updates.
254        *
255        * @throws IllegalStateException if the method is called before the
256        * transaction has committed or aborted.
257        *
258        * @see com.sleepycat.je.rep.CommitPointConsistencyPolicy
259        */
260       public CommitToken getCommitToken()
261           throws IllegalStateException {
262  
263           if (txn == null) {
264  
265               /*
266                * The commit token is only legitimate after the transaction is
267                * closed. A null txn field means the transaction is closed.
268                */
269               return commitToken;
270           }
271          
272           throw new IllegalStateException
273              ("This transaction is still in progress and a commit token " +
274               "is not available");
275       }

277       /**
278        * End the transaction. If the environment is configured for synchronous
279        * commit, the transaction will be committed synchronously to stable
280        * storage before the call returns. This means the transaction will
281        * exhibit all of the ACID (atomicity, consistency, isolation, and
282        * durability) properties.
283        *
284        * <p>If the environment is not configured for synchronous commit, the
285        * commit will not necessarily have been committed to stable storage before
286        * the call returns. This means the transaction will exhibit the ACI
287        * (atomicity, consistency, and isolation) properties, but not D
288        * (durability); that is, database integrity will be maintained, but it is
289        * possible this transaction may be undone during recovery.</p>
290        *
291        * <p>All cursors opened within the transaction must be closed before the
292        * transaction is committed.</p>
293        *
294        * <p>If the method encounters an error, the transaction <!-- and all child
295        * transactions of the transaction --> will have been aborted when the call
296        * returns.</p>
297        *
298        * <p>After this method has been called, regardless of its return, the
299        * {@link Transaction} handle may not be accessed again, with one
300        * exception: the {@code abort} method may be called any number of times
301        * to simplify error handling.</p>
302        *
303        * <p>WARNING: To guard against memory leaks, the application should
304        * discard all references to the closed handle. While BDB makes an effort
305        * to discard references from closed objects to the allocated memory for an
306        * environment, this behavior is not guaranteed. The safe course of action
307        * for an application is to discard all references to closed BDB
308        * objects.</p>
309        *
310        * @throws com.sleepycat.je.rep.InsufficientReplicasException if the master
311        * in a replicated environment could not contact a quorum of replicas as
312        * determined by the {@link ReplicaAckPolicy}.
313        *
314        * @throws com.sleepycat.je.rep.InsufficientAcksException if the master in
315        * a replicated environment did not receive enough replica acknowledgments,
316        * although the commit succeeded locally.
317        *
318        * @throws com.sleepycat.je.rep.ReplicaWriteException if a write operation
319        * was performed with this transaction, but this node is now a Replica.
320        *
321        * @throws OperationFailureException if this exception occurred earlier and
322        * caused the transaction to be invalidated.
323        *
324        * @throws EnvironmentFailureException if an unexpected, internal or
325        * environment-wide failure occurs.
326        *
327        * @throws IllegalStateException if the transaction or environment has been
328        * closed, or cursors associated with the transaction are still open.
329        */
330       public synchronized void commit()
331           throws DatabaseException {
332  
333           try {
334               checkEnv();
335               checkOpen();
336               env.removeReferringHandle(this);
337               txn.commit();
338               commitToken = txn.getCommitToken();
339               /* Remove reference to internal txn, so we can reclaim memory. */
340               setTxnNull();
341           } catch (Error E) {
342               DbInternal.getNonNullEnvImpl(env).invalidate(E);
343               throw E;
344           }
345       }
