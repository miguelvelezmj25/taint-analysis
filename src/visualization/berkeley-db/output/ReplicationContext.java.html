19  /**
20    * ReplicationContext provides context about high-level operations so that the
21    * logging level can determine which replication related actions are required
22    * for a given Loggable item.
23    *
24    * Those lower level actions are:
25    * - does a log entry need to be logged with a VLSN generated by this
26    * (master) node?
27    * - does the log entry need to be logged with the VLSN which accompanied a
28    * replication message?
29    * - do we need to wait for PERM acks after logging an entry?
30    * - do we need to record the client VLSN that was just written to the log?
31    *
32    * ReplicationContext subclasses may hold additional information about the
33    * logical operation which instigated logging, so that this can be added
34    * to the log entry.
35    *
36    * All LogEntryType(s) have a "replicationPossible" attribute. For example,
37    * INs will never be replicated, but LN_TX's may or may not be replicated,
38    * depending on whether the owning database is replicated.
39    *
40    * If a LogEntryType will never be replicated, it should be logged with
41    * the static ReplicationContext.NO_REPLICATE instance.
42    * If replication is possible, the replication context may be:
43    * - one allocated for this operation, as the result of client apply
44    * - the static instance MASTER, if this node is the replication master
45    * - the static instance NO_REPLICATE, if this is a local operation
46    *
47    */
48  public class ReplicationContext {
49  
50       /*
51        * Convenience static instance used when you know this operation is
52        * executing on a replication master node.
53        */
54       public static final ReplicationContext MASTER =
55           new ReplicationContext(true /* inReplicationStream */);
56  
57       /*
58        * Convenience static instance used when you know this operation will not
59        * be replicated, either because it's executing on a non-replicated node,
60        * it's a local operation for a local database, it's a read only operation,
61        * or because this loggable item is the type that is never replicated.
62        */
63       public static final ReplicationContext NO_REPLICATE =
64           new ReplicationContext(false /* inReplicationStream */);
65  
66       /*
67        * If true, this Loggable item is part of the replication stream, and
68        * needs to be logged with a VLSN.
69        */
70       private final boolean inReplicationStream;
71  
72       /*
73        * The VLSN value passed in from a replication message directed at
74        * this replication client.
75        */
76       private final VLSN clientVLSN;
77  
78       protected ReplicationContext(boolean inReplicationStream) {
79           this.inReplicationStream = inReplicationStream;
80           clientVLSN = null;
81       }
82  
83       /**
84        * Used to pass the VLSN held in an arriving message down to the logging
85        * levels.
86        */
87       public ReplicationContext(VLSN clientVLSN) {
88           this.inReplicationStream = true;
89           this.clientVLSN = clientVLSN;
90       }
91  
92       /**
93        * Used to pass the VLSN held in a migrated LN down to the logging levels.
94        */
95       public ReplicationContext(VLSN clientVLSN, boolean inReplicationStream) {
96           this.clientVLSN = clientVLSN;
97           this.inReplicationStream = inReplicationStream;
98       }
99  
100       /**
101        * @return the VLSN that arrived in the replication message which
102        * instigated this Loggable item.
103        */
104       public VLSN getClientVLSN() {
105           return clientVLSN;
106       }
107  
108       /**
109        * @return true if this loggable item is part of the replication stream
110        */
111       public boolean inReplicationStream() {
112           return inReplicationStream;
113       }
114  
115       /**
116        * @return true if this node is the master, and should
117        * generate a VLSN for this log entry
118        */
119       public boolean mustGenerateVLSN() {
120           return (inReplicationStream && (clientVLSN == null)); → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
121       }
122  
123       /**
124        * @return the type of database operation in progress. For the default
125        * case, we return DbOperationType.NONE.
126        */
127       public DbOperationType getDbOperationType() {
128           return DbOperationType.NONE;
129       }
130  
131       @Override
132       public String toString() {
133           StringBuilder sb = new StringBuilder();
134           sb.append("inRepStream=").append(inReplicationStream);
135           sb.append(" clientVLSN=").append(clientVLSN);
136           return sb.toString();
137       }
138  }
