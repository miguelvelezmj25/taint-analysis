72       /**
73        * This field contains the names of secondary keys that are incorrectly
74        * ordered because, in an earlier version, we failed to set the dup
75        * comparator. This bug applies only when the primary key has a
76        * comparator. The bug was fixed by setting the dup comparator to the
77        * primary key comparator, for all new secondary databases. [#17252]
78        *
79        * A field containing an empty set signifies that no keys are incorrectly
80        * ordered, while a null field signifies that all keys are incorrect (when
81        * the primary key has a comparator). The field is assigned to an empty
82        * set when declared, so that it will be null only when a pre-fix version
83        * of the format is deserialized. (With Java serialization, when a field is
84        * added to a class and a previously serialized instance is deserialized,
85        * the new field will always be null).
86        *
87        * This field is used to determine when a dup comparator should be set. We
88        * cannot set the comparator for secondary databases created prior to the
89        * bug fix, since ordering cannot be changed for existing records. See
90        * isSecKeyIncorrectlyOrdered and setSecKeyCorrectlyOrdered.
91        *
92        * This field does not count in comparisons of formats during evolution.
93        * When the user wants to correct the ordering for an incorrectly ordered
94        * secondary database, she must delete the database but does not need to
95        * increment the class version. In other words, this is information about
96        * the database order but is not considered class metadata.
97        */
98       private Set<String> incorrectlyOrderedSecKeys = new HashSet<String>();

100       /**
101        * In JE 5.0 we changed the format for String fields. Instead of treating
102        * the String as an object with a format ID embedded in the serialized
103        * bytes, we treat it as a primitive and do not include the format ID.
104        * This works well because a field declared to be type String cannot be
105        * used to store any other object, and because the String tuple format
106        * supports null values.
107        *
108        * A field containing false signifies that the old String format was used
109        * when the entity was written, while a true value signifies that the new
110        * String format was used. The field is assigned to true when declared, so
111        * that it will be false only when a pre-JE 5.0 version of the format is
112        * deserialized. (With Java serialization, when a boolean field is added to
113        * a class and a previously serialized instance is deserialized, the new
114        * field will always be false).
115        */
116       private boolean newStringFormat = true;

160                   if (rel == Relationship.ONE_TO_MANY || → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]

163                           throw new IllegalArgumentException

165                                " have an array or Collection type: " +
166                                getClassName() + '.' + fieldName);
167                       }
168                   } else {
169                       if (PersistKeyCreator.isManyType(fieldCls)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
170                           throw new IllegalArgumentException
171                               ("ONE_TO_ONE and MANY_TO_ONE keys must not" +
172                                " have an array or Collection type: " +
173                                getClassName() + '.' + fieldName);
174                       }

178                       throw new IllegalArgumentException

366       @Override
367       void initialize(Catalog catalog, EntityModel model, int initVersion) {
368  
369           Class type = getType();
370           boolean useEnhanced = false;
371           if (type != null) {
372               useEnhanced = EnhancedAccessor.isEnhanced(type);
373           }
374           /* Initialize all fields. */
375           if (priKeyField != null) {
376               priKeyField.initialize(catalog, model, initVersion);
377           }
378           for (FieldInfo field : secKeyFields) {
379               field.initialize(catalog, model, initVersion);
380           }
381           for (FieldInfo field : nonKeyFields) {
382               field.initialize(catalog, model, initVersion);
383           }
384           /* Set the superclass format for a new (never initialized) format. */
385           ComplexFormat superFormat = getComplexSuper();
386           if (type != null && superFormat == null) {
387               Class superCls = type.getSuperclass();
388               if (superCls != Object.class) {
389                   superFormat =
390                       (ComplexFormat) catalog.getFormat(superCls.getName());
391                   setSuperFormat(superFormat);
392               }
393           }
394           /* Initialize the superclass format and validate the super accessor. */
395           if (superFormat != null) {
396               superFormat.initializeIfNeeded(catalog, model);
397               Accessor superAccessor = superFormat.objAccessor;
398               if (type != null && superAccessor != null) {
399                   if (useEnhanced) {
400                       if (!(superAccessor instanceof EnhancedAccessor)) {
401                           throw new IllegalStateException
402                               ("The superclass of an enhanced class must also " +
403                                "be enhanced: " + getClassName() +
404                                " extends " + superFormat.getClassName());
405                       }
406                   } else {
407                       if (!(superAccessor instanceof ReflectionAccessor)) {
408                           throw new IllegalStateException
409                               ("The superclass of an unenhanced class must " +
410                                "not be enhanced: " + getClassName() +
411                                " extends " + superFormat.getClassName());
412                       }
413                   }
414               }
415           }
416           /* Find entity format, if any. */
417           for (ComplexFormat format = this;
418                format != null;
419                format = format.getComplexSuper()) {
420               if (format.isEntity()) {
421                   entityFormat = format;
422                   break;
423               }
424           }
425  
426           /*
427            * Ensure that the current entity metadata is always referenced in
428            * order to return it to the user and to properly construct secondary
429            * key addresses. Secondary key metadata can change in an entity
430            * subclass or be created when a new subclass is used, but this will
431            * not cause evolution of the entity class; instead, the metadata is
432            * updated here. [#16467]
433            */
434           if (isEntity() && isCurrentVersion()) {
435               entityMeta = model.getEntityMetadata(getClassName());
436           }
437  
438           /* Disallow proxy class that extends an entity class. [#15950] */
439           if (clsMeta.getProxiedClassName() != null && entityFormat != null) {
440               throw new IllegalArgumentException
441                   ("A proxy may not be an entity: " + getClassName());
442           }
443           /* Disallow primary keys on entity subclasses. [#15757] */
444           if (entityFormat != null &&
445               entityFormat != this &&
446               priKeyField != null) {
447               throw new IllegalArgumentException
448                   ("A PrimaryKey may not appear on an Entity subclass: " +
449                    getClassName() + " field: " + priKeyField.getName());
450           }
451           /* Create the accessors. */
452           if (type != null) {
453               if (useEnhanced) {
454                   objAccessor = new EnhancedAccessor(catalog, type, this);
455               } else {
456                   Accessor superObjAccessor =
457                       (superFormat != null) ? superFormat.objAccessor : null;
458                   objAccessor = new ReflectionAccessor
459                       (catalog, type, superObjAccessor, priKeyField,
460                        secKeyFields, nonKeyFields);
461               }
462           }
463           Accessor superRawAccessor =
464               (superFormat != null) ? superFormat.rawAccessor : null;
465           rawAccessor = new RawAccessor
466               (this, superRawAccessor, priKeyField, secKeyFields, nonKeyFields);
467  
468           /* Initialize secondary key field addresses. */
469           EntityMetadata latestEntityMeta = null;
470           if (entityFormat != null) {
471               latestEntityMeta =
472                   entityFormat.getLatestVersion().getEntityMetadata();
473           }
474           if (latestEntityMeta != null) {
475               secKeyAddresses = new HashMap<String, FieldAddress>();
476               ComplexFormat thisLatest = getComplexLatest();
477               if (thisLatest != this) {
478                   thisLatest.initializeIfNeeded(catalog, model);
479               }
480               nextKeyLoop:
481               for (SecondaryKeyMetadata secKeyMeta :
482                    latestEntityMeta.getSecondaryKeys().values()) {
483                   String clsName = secKeyMeta.getDeclaringClassName();
484                   String fieldName = secKeyMeta.getName();
485                   int superLevel = 0;
486                   for (ComplexFormat format = this;
487                        format != null;
488                        format = format.getComplexSuper()) {
489                       if (clsName.equals
490                               (format.getLatestVersion().getClassName())) {
491                           String useFieldName = null;
492                           if (format.newToOldFieldMap != null &&
493                               format.newToOldFieldMap.containsKey(fieldName)) {
494                               useFieldName =
495                                   format.newToOldFieldMap.get(fieldName);
496                           } else {
497                               useFieldName = fieldName;
498                           }
499                           boolean isSecField;
500                           int fieldNum;
501                           FieldInfo info = FieldInfo.getField
502                               (format.secKeyFields, useFieldName);
503                           if (info != null) {
504                               isSecField = true;
505                               fieldNum = format.secKeyFields.indexOf(info);
506                           } else {
507                               isSecField = false;
508                               info = FieldInfo.getField
509                                   (format.nonKeyFields, useFieldName);
510                               if (info == null) {
511                                   /* Field not present in old format. */
512                                   assert thisLatest != this;
513                                   thisLatest.checkNewSecKeyInitializer
514                                       (secKeyMeta);
515                                   continue nextKeyLoop;
516                               }
517                               fieldNum = format.nonKeyFields.indexOf(info);
518                           }
519                           FieldAddress addr = new FieldAddress
520                               (isSecField, fieldNum, superLevel, format,
521                                info.getType());
522                           secKeyAddresses.put(secKeyMeta.getKeyName(), addr);
523                       }
524                       superLevel += 1;
525                   }
526               }
527           }
528       }

530       /**
531        * Checks that the type of a new secondary key is not a primitive and that
532        * the default contructor does not initialize it to a non-null value.
533        */
534       private void checkNewSecKeyInitializer(SecondaryKeyMetadata secKeyMeta) {
535           if (objAccessor != null) {
536          
537               /*
538                * If this format represents an abstract class, we will not do the
539                * following check. When initializing this abstract class's
540                * subclass, which is not abstract, the new added secondary key
541                * will be checked then. [#19358]
542                */
543               if (Modifier.isAbstract(this.getType().getModifiers())) {
544                   return;
545               }
546               FieldAddress addr = secKeyAddresses.get(secKeyMeta.getKeyName());
547               Object obj = objAccessor.newInstance();
548               Object val = objAccessor.getField
549                   (obj, addr.fieldNum, addr.superLevel, addr.isSecField);
550               if (val != null) {
551                   if (addr.keyFormat.isPrimitive()) {
552                       throw new IllegalArgumentException
553                           ("For a new secondary key field the field type must " +
554                            "not be a primitive -- class: " +
555                            secKeyMeta.getDeclaringClassName() + " field: " +
556                            secKeyMeta.getName());
557                   } else {
558                       throw new IllegalArgumentException
559                           ("For a new secondary key field the default " +
560                            "constructor must not initialize the field to a " +
561                            "non-null value -- class: " +
562                            secKeyMeta.getDeclaringClassName() + " field: " +
563                            secKeyMeta.getName());
564                   }
565               }
566           }

607       @Override

609                               boolean rawAccess,
610                               RawObject rawObject,
611                               IdentityHashMap converted)
612           throws RefreshException {
613  
614           /*
615            * Synchronization is not required since rawInputFields, rawInputLevels
616            * and rawInputDepth are immutable. If by chance we create duplicate
617            * values when two threads execute this block, no harm is done. But be
618            * sure to assign the fields only after the values are fully populated.
619            */
620           FieldInfo[] fields = rawInputFields;
621           int[] levels = rawInputLevels;
622           int depth = rawInputDepth;
623           if (fields == null || levels == null || depth == 0) {
624  
625               /*
626                * The volatile fields are not yet set. Prepare to process the
627                * class hierarchy, storing class formats in order from the highest
628                * superclass down to the current class.
629                */
630               depth = 0;
631               int nFields = 0;
632               for (ComplexFormat format = this;
633                    format != null;
634                    format = format.getComplexSuper()) {
635                   nFields += format.getNFields();
636                   depth += 1;
637               }
638               ComplexFormat[] hierarchy = new ComplexFormat[depth];
639               int level = depth;
640               for (ComplexFormat format = this;
641                    format != null;
642                    format = format.getComplexSuper()) {
643                   level -= 1;
644                   hierarchy[level] = format;
645               }
646               assert level == 0;
647  
648               /* Populate levels and fields in parallel. */
649               levels = new int[nFields];
650               fields = new FieldInfo[nFields];
651               int index = 0;
652  
653               /*
654                * The primary key is the first field read/written. We use the
655                * first primary key field encountered going from this class upward
656                * in the class hierarchy.
657                */
658               if (getEntityFormat() != null) {
659                   for (level = depth - 1; level >= 0; level -= 1) {
660                       ComplexFormat format = hierarchy[level];
661                       if (format.priKeyField != null) {
662                           levels[index] = level;
663                           fields[index] = format.priKeyField;
664                           index += 1;
665                           break;
666                       }
667                   }
668                   assert index == 1;
669               }
670  
671               /*
672                * Secondary key fields are read/written next, from the highest
673                * base class downward.
674                */
675               for (level = 0; level < depth; level += 1) {
676                   ComplexFormat format = hierarchy[level];
677                   for (FieldInfo field : format.secKeyFields) {
678                       levels[index] = level;
679                       fields[index] = field;
680                       index += 1;
681                   }
682               }
683  
684               /*
685                * Other fields are read/written last, from the highest base class
686                * downward.
687                */
688               for (level = 0; level < depth; level += 1) {
689                   ComplexFormat format = hierarchy[level];
690                   for (FieldInfo field : format.nonKeyFields) {
691                       levels[index] = level;
692                       fields[index] = field;
693                       index += 1;
694                   }
695               }
696  
697               /* We're finished -- update the volatile fields for next time. */
698               assert index == fields.length;
699               rawInputFields = fields;
700               rawInputLevels = levels;
701               rawInputDepth = depth;
702           }
703  
704           /*
705            * Create an objects array that is parallel to the fields and levels
706            * arrays, but contains the RawObject for each slot from which the
707            * field value can be retrieved. The predetermined level for each
708            * field determines which RawObject in the instance hierarchy to use.
709            */
710           RawObject[] objectsByLevel = new RawObject[depth];
711           int level = depth;
712           for (RawObject raw = rawObject; raw != null; raw = raw.getSuper()) {
713               if (level == 0) {
714                   throw new IllegalArgumentException
715                       ("RawObject has too many superclasses: " +
716                        rawObject.getType().getClassName());
717               }
718               level -= 1;
719               objectsByLevel[level] = raw;
720           }
721           if (level > 0) {
722               throw new IllegalArgumentException
723                   ("RawObject has too few superclasses: " +
724                    rawObject.getType().getClassName());
725           }
726           assert level == 0;
727           RawObject[] objects = new RawObject[fields.length];
728           for (int i = 0; i < objects.length; i += 1) {
729               objects[i] = objectsByLevel[levels[i]];
730           }
731  
732           /* Create the persistent object and convert all RawObject fields. */
733           EntityInput in = new RawComplexInput
734               (catalog, rawAccess, converted, fields, objects);
735           Object o = newInstance(in, rawAccess);
736           converted.put(rawObject, o);
737           if (getEntityFormat() != null) {
738               readPriKey(o, in, rawAccess);
739           }
740           return readObject(o, in, rawAccess);
741       }

922           /* Disallow evolution to a non-complex format. */

925                   (this, newFormatParam,
926                    "Converter is required when a complex type is changed " +
927                    "to a simple type or enum type");
928               return false;
929           }
930           ComplexFormat newFormat = (ComplexFormat) newFormatParam;
931           Mutations mutations = evolver.getMutations();
932           boolean thisChanged = false;
933           boolean hierarchyChanged = false;
934           Map<String, String> allKeyNameMap = new HashMap<String, String>();
935  
936           /* The Evolver has already ensured that entities evolve to entities. */
937           assert isEntity() == newFormat.isEntity();
938           assert isEntity() == (entityMeta != null);
939           assert newFormat.isEntity() == (newFormat.entityMeta != null);
940  
941           /*
942            * Keep track of the old and new entity class names for use in deleting
943            * and renaming secondary keys below. If the oldEntityClass is
944            * non-null this also signifies an entity class or subclass. Note that
945            * getEntityFormat cannot be called on a newly created format during
946            * evolution because its super format property is not yet initialized.
947            * [#16253]
948            */
949           String oldEntityClass;
950           String newEntityClass;
951           if (isEntity()) {
952               oldEntityClass = getClassName();
953               newEntityClass = newFormat.getClassName();
954           } else {
955               oldEntityClass = null;
956               newEntityClass = null;
957           }
958  
959           /*
960            * Evolve all superclass formats, even when a deleted class appears in
961            * the hierarchy. This ensures that the super format's
962            * getLatestVersion/getComplexLatest method can be used accurately
963            * below.
964            */
965           for (ComplexFormat oldSuper = getComplexSuper();
966                oldSuper != null;
967                oldSuper = oldSuper.getComplexSuper()) {
968               Converter converter = mutations.getConverter
969                   (oldSuper.getClassName(), oldSuper.getVersion(), null);
970               if (converter != null) {
971                   evolver.addMissingMutation
972                       (this, newFormatParam,
973                        "Converter is required for this subclass when a " +
974                        "Converter appears on its superclass: " + converter);
975                   return false;
976               }
977               if (!evolver.evolveFormat(oldSuper)) {
978                   return false;
979               }
980               if (!oldSuper.isCurrentVersion()) {
981                   if (oldSuper.isDeleted()) {
982                       if (!oldSuper.evolveDeletedClass(evolver)) {
983                           return false;
984                       }
985                   }
986                   if (oldSuper.oldToNewKeyMap != null) {
987                       allKeyNameMap.putAll(oldSuper.oldToNewKeyMap);
988                   }
989                   hierarchyChanged = true;
990               }
991           }
992  
993           /*
994            * Compare the old and new class hierarhies and decide whether each
995            * change is allowed or not:
996            * + Old deleted and removed superclass -- allowed
997            * + Old empty and removed superclass -- allowed
998            * + Old non-empty and removed superclass -- not allowed
999            * + Old superclass repositioned in the hierarchy -- not allowed
1000            * + New inserted superclass -- allowed
1001            */
1002           Class newFormatCls = newFormat.getExistingType();
1003           Class newSuper = newFormatCls;
1004           List<Integer> newLevels = new ArrayList<Integer>();
1005           int newLevel = 0;
1006           newLevels.add(newLevel);
1007  
1008           /*
1009            * When this format has a new superclass, we treat it as a change to
1010            * this format as well as to the superclass hierarchy.
1011            */
1012           if (getSuperFormat() == null) {
1013               if (newFormatCls.getSuperclass() != Object.class) {
1014                   thisChanged = true;
1015                   hierarchyChanged = true;
1016               }
1017           } else {
1018               if (!getSuperFormat().getLatestVersion().getClassName().equals
1019                       (newFormatCls.getSuperclass().getName())) {
1020                   thisChanged = true;
1021                   hierarchyChanged = true;
1022               }
1023           }
1024  
1025           for (ComplexFormat oldSuper = getComplexSuper();
1026                oldSuper != null;
1027                oldSuper = oldSuper.getComplexSuper()) {
1028  
1029               /* Find the matching superclass in the new hierarchy. */
1030               String oldSuperName = oldSuper.getLatestVersion().getClassName();
1031               Class foundNewSuper = null;
1032               int tryNewLevel = newLevel;
1033               for (Class newSuper2 = newSuper.getSuperclass();
1034                    newSuper2 != Object.class;

1082                           }

1104                                "hierarchy, the superclass or all of its " +
1105                                "persistent fields must be deleted with a " +
1106                                "Deleter: " +
1107                                oldSuperLatest.getClassName());

1109                       }

1114                           oldSuper.clsMeta.getSecondaryKeys();

1117                               getSecondaryKeyMetadataByFieldName
1118                                   (secKeys, field.getName());

1126                    * (no persistent fields) from being read via the Accessor.

1139           }

1145           if (thisChanged &&

1150           }
