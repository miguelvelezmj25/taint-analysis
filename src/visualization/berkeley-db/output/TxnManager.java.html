53  /**
54    * Class to manage transactions. Basically a Set of all transactions with add
55    * and remove methods and a latch around the set.
56    */
57  public class TxnManager { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
58  
59       /*
60        * All NullTxns share the same id so as not to eat from the id number
61        * space.
62        *
63        * Negative transaction ids are used by the master node of a replication
64        * group. That sequence begins at -10 to avoid conflict with the
65        * NULL_TXN_ID and leave room for other special purpose ids.
66        */
67       static final long NULL_TXN_ID = -1;
68       private static final long FIRST_NEGATIVE_ID = -10;
69       private LockManager lockManager;
70       private final EnvironmentImpl envImpl;
71       private final SharedLatch allTxnsLatch;
72       private final Map<Txn, Txn> allTxns;
73  
74       /* Maps Xids to Txns. */
75       private final Map<Xid, Txn> allXATxns;
76  
77       /* Maps Threads to Txns when there are thread implied transactions. */
78       private final Map<Thread, Transaction> thread2Txn;
79  
80       /*
81        * Positive and negative transaction ids are used in a replicated system,
82        * to let replicated transactions intermingle with local transactions.
83        */
84       private final AtomicLong lastUsedLocalTxnId;
85       private final AtomicLong lastUsedReplicatedTxnId;
86       private final AtomicInteger nActiveSerializable;
87  
88       /* Locker Stats */
89       private final StatGroup stats;
90       private final IntStat nActive;
91       private final LongStat numBegins;
92       private final LongStat numCommits;
93       private final LongStat numAborts;
94       private final LongStat numXAPrepares;
95       private final LongStat numXACommits;
96       private final LongStat numXAAborts;
97       private final ActiveTxnArrayStat activeTxns;
98       private volatile long nTotalCommits = 0;

100       public TxnManager(EnvironmentImpl envImpl) {
101           lockManager = new SyncedLockManager(envImpl);
102  
103           if (envImpl.isNoLocking()) {
104               lockManager = new DummyLockManager(envImpl, lockManager);
105           }
106  
107           this.envImpl = envImpl;
108           allTxnsLatch = LatchFactory.createSharedLatch(
109               envImpl, "TxnManager.allTxns", false /*exclusiveOnly*/);
110           allTxns = new ConcurrentHashMap<Txn, Txn>();
111           allXATxns = Collections.synchronizedMap(new HashMap<Xid, Txn>());
112           thread2Txn = new ConcurrentHashMap<Thread, Transaction>();
113  
114           lastUsedLocalTxnId = new AtomicLong(0);
115           lastUsedReplicatedTxnId = new AtomicLong(FIRST_NEGATIVE_ID);
116           nActiveSerializable = new AtomicInteger(0);
117  
118           /* Do the stats definition. */
119           stats = new StatGroup("Transaction", "Transaction statistics");
120           nActive = new IntStat(stats, TXN_ACTIVE);
121           numBegins = new LongStat(stats, TXN_BEGINS);
122           numCommits = new LongStat(stats, TXN_COMMITS);
123           numAborts = new LongStat(stats, TXN_ABORTS);
124           numXAPrepares = new LongStat(stats, TXN_XAPREPARES);
125           numXACommits = new LongStat(stats, TXN_XACOMMITS);
126           numXAAborts = new LongStat(stats, TXN_XAABORTS);
127           activeTxns = new ActiveTxnArrayStat(stats, TXN_ACTIVE_TXNS);
128       }

167       /*
168        * Tracks the lowest replicated transaction id used during a replay of the
169        * replication stream, so that it's available as the starting point if this
170        * replica transitions to being the master.
171        */
172       public void updateFromReplay(long replayTxnId) {
173           assert !envImpl.isMaster();
174           assert replayTxnId < 0 :
175               "replay txn id is unexpectedly positive " + replayTxnId;
176  
177           if (replayTxnId < lastUsedReplicatedTxnId.get()) {
178               lastUsedReplicatedTxnId.set(replayTxnId);
179           }
180       }

209       /**
210        * Called when txn is created.
211        */
212       public void registerTxn(Txn txn) {
213           allTxnsLatch.acquireShared();
214           try {
215               allTxns.put(txn, txn);
216               if (txn.isSerializableIsolation()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
217                   nActiveSerializable.incrementAndGet();
218               }
219               numBegins.increment();
220           } finally {
221               allTxnsLatch.release();
222           }
223       }
224  
225       /**
226        * Called when txn ends.
227        */
228       void unRegisterTxn(Txn txn, boolean isCommit) {
229           allTxnsLatch.acquireShared();
230           try {
231               allTxns.remove(txn);
232  
233               /* Remove any accumulated MemoryBudget delta for the Txn. */
234               envImpl.getMemoryBudget().
235                   updateTxnMemoryUsage(0 - txn.getBudgetedMemorySize());
236               if (isCommit) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
237                   numCommits.increment();
238                   nTotalCommits += 1;
239               } else {
240                   numAborts.increment();
241               }
242               if (txn.isSerializableIsolation()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
243                   nActiveSerializable.decrementAndGet();
244               }
245           } finally {
246               allTxnsLatch.release();
247           }
248       }

250       /**
251        * Called when txn is created.
252        */
253       public void registerXATxn(Xid xid, Txn txn, boolean isPrepare) {
254           if (!allXATxns.containsKey(xid)) {
255               allXATxns.put(xid, txn);
256               envImpl.getMemoryBudget().updateTxnMemoryUsage
257                   (MemoryBudget.HASHMAP_ENTRY_OVERHEAD);
258           }
259  
260           if (isPrepare) {
261               numXAPrepares.increment();
262           }
263       }

272       /**
273        * Called when txn ends.
274        *
275        * @throws IllegalStateException via XAResource
276        */
277       void unRegisterXATxn(Xid xid, boolean isCommit)
278           throws DatabaseException {
279  
280           if (allXATxns.remove(xid) == null) {
281               throw new IllegalStateException
282                   ("XA Transaction " + xid + " is not registered.");
283           }
284           envImpl.getMemoryBudget().updateTxnMemoryUsage
285               (0 - MemoryBudget.HASHMAP_ENTRY_OVERHEAD);
286           if (isCommit) {
287               numXACommits.increment();
288           } else {
289               numXAAborts.increment();
290           }
291       }

336       /**
337        * Returns whether there are any active serializable transactions,
338        * excluding the transaction given (if non-null). This is intentionally
339        * returned without latching, since latching would not make the act of
340        * reading an integer more atomic than it already is.
341        */
342       public boolean
343           areOtherSerializableTransactionsActive(Locker excludeLocker) {
344           int exclude = → [TRANSACTIONAL]
345               (excludeLocker != null &&
346                excludeLocker.isSerializableIsolation()) ? → [TRANSACTIONAL]
347               1 : 0;
348           return (nActiveSerializable.get() - exclude > 0); → [TRANSACTIONAL]
349       }

351       /**
352        * Get the earliest LSN of all the active transactions, for checkpoint.
353        * Returns NULL_LSN is no transaction is currently active.
354        */
355       public long getFirstActiveLsn() {
356  
357           /*
358            * Note that the latching hierarchy calls for synchronizing on
359            * allTxns first, then synchronizing on individual txns.
360            */
361           long firstActive = DbLsn.NULL_LSN;
362           allTxnsLatch.acquireExclusive();
363           try {
364               Iterator<Txn> iter = allTxns.keySet().iterator();
365               while (iter.hasNext()) {
366                   long txnFirstActive = iter.next().getFirstActiveLsn();
367                   if (firstActive == DbLsn.NULL_LSN) {
368                       firstActive = txnFirstActive;
369                   } else if (txnFirstActive != DbLsn.NULL_LSN) {
370                       if (DbLsn.compareTo(txnFirstActive, firstActive) < 0) {
371                           firstActive = txnFirstActive;
372                       }
373                   }
374               }
375           } finally {
376               allTxnsLatch.release();
377           }
378  
379           return firstActive;
380       }

383        * Statistics
384        */
385  
386       /**
387        * Collect transaction related stats.
388        */
389       public TransactionStats txnStat(StatsConfig config) {
390           TransactionStats txnStats = null;
391           allTxnsLatch.acquireShared();
392           try {
393               nActive.set(allTxns.size());
394               TransactionStats.Active[] activeSet =
395                   new TransactionStats.Active[nActive.get()];
396               Iterator<Txn> iter = allTxns.keySet().iterator();
397               int i = 0;
398               while (iter.hasNext() && i < activeSet.length) {
399                   Locker txn = iter.next();
400                   activeSet[i] = new TransactionStats.Active
401                       (txn.toString(), txn.getId(), 0);
402                   i++;
403               }
404               activeTxns.set(activeSet);
405               txnStats = new TransactionStats(stats.cloneGroup(false));
406               if (config.getClear()) {
407                   numCommits.clear();
408                   numAborts.clear();
409                   numXACommits.clear();
410                   numXAAborts.clear();
411               }
412           } finally {
413               allTxnsLatch.release();
414           }
415  
416           return txnStats;
417       }

432       /**
433        * Examine the transaction set and return Txn that match the class or are
434        * subclasses of that class. This method is used to obtain Master and Replay
435        * TXns for HA
436        */
437       public <T extends Txn> Set<T> getTxns(Class<T> txnClass) {
438           final Set<T> targetSet = new HashSet<>();
439  
440           allTxnsLatch.acquireShared();
441           try {
442               Set<Txn> all = allTxns.keySet();
443               for (Txn t: all) {
444                   if (txnClass.isAssignableFrom(t.getClass())) {
445                       @SuppressWarnings("unchecked")
446                       final T t2 = (T)t;
447                       targetSet.add(t2);
448                   }
449               }
450           } finally {
451               allTxnsLatch.release();
452           }
453  
454           return targetSet;
455       }
