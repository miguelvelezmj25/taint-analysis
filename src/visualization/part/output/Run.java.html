30  /**
31    * Helper class that executes Weka schemes from the command line. Performs
32    * Suffix matching on the scheme name entered by the user - e.g.<br>
33    * <br>
34    * <p>
35    * java weka.Run NaiveBayes <br>
36    * <br>
37    * <p>
38    * will prompt the user to choose among
39    * weka.classifiers.bayes.ComplementNaiveBayes,
40    * weka.classifiers.bayes.NaiveBayes,
41    * weka.classifiers.bayes.NaiveBayesMultinomial,
42    * weka.classifiers.bayes.NaiveBayesMultinomialUpdateable,
43    * weka.classifiers.bayes.NaiveBayesSimple,
44    * weka.classifiers.bayes.NaiveBayesUpdateable
45    *
46    * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
47    * @version $Revision: 13476 $
48    */

110       /**
111        * Main method for this class. -help or -h prints usage info.
112        *
113        * @param args
114        */
115       public static void main(String[] args) {
116           System.setProperty("apple.awt.UIElement", "true");
117           try {
118               if(args.length == 0 || args[0].equalsIgnoreCase("-h")
119                       || args[0].equalsIgnoreCase("-help")) {
120                   System.err
121                           .println("Usage:\n\tweka.Run [-no-scan] [-no-load] [-match-anywhere] <scheme name [scheme options]>");
122                   return;
123               }
124               boolean noScan = false;
125               boolean noLoad = false;
126               boolean matchAnywhere = false;
127               boolean dontPromptIfMultipleMatches = false;
128  
129               if(Utils.getFlag("list-packages", args)) {
130                   weka.core.WekaPackageManager.loadPackages(true, true, false);
131                   return;
132               }
133  
134               int schemeIndex = 0;
135               if(Utils.getFlag("no-load", args)) {
136                   noLoad = true;
137                   schemeIndex++;
138               }
139  
140               if(Utils.getFlag("no-scan", args)) {
141                   noScan = true;
142                   schemeIndex++;
143               }
144  
145               if(Utils.getFlag("match-anywhere", args)) {
146                   matchAnywhere = true;
147                   schemeIndex++;
148               }
149  
150               if(Utils.getFlag("do-not-prompt-if-multiple-matches", args)) {
151                   dontPromptIfMultipleMatches = true;
152                   schemeIndex++;
153               }
154  
155               if(!noLoad) {
156                   weka.core.WekaPackageManager.loadPackages(false, true, false);
157               }
158  
159               String schemeToRun = null;
160               String[] options = null;
161  
162               if(schemeIndex >= args.length) {
163                   System.err.println("No scheme name given.");
164                   return;
165               }
166               schemeToRun = args[schemeIndex];
167               options = new String[args.length - schemeIndex - 1];
168               if(options.length > 0) {
169                   System.arraycopy(args, schemeIndex + 1, options, 0, options.length);
170               }
171  
172               if(!noScan) {
173                   List<String> prunedMatches =
174                           findSchemeMatch(schemeToRun, matchAnywhere);
175  
176                   if(prunedMatches.size() == 0) {
177                       System.err.println("Can't find scheme " + schemeToRun
178                               + ", or it is not runnable.");
179                       // System.exit(1);
180                       return;
181                   }
182                   else if(prunedMatches.size() > 1) {
183                       if(dontPromptIfMultipleMatches) {
184                           System.out.println("There are multiple matches:");
185                           for(int i = 0; i < prunedMatches.size(); i++) {
186                               System.out.println("\t" + (i + 1) + ") " + prunedMatches.get(i));
187                           }
188                           System.out.println("\nPlease make your scheme name more specific "
189                                   + "(i.e. qualify it with more of the package name).");
190                           return;
191                       }
192                       java.io.BufferedReader br =
193                               new java.io.BufferedReader(new java.io.InputStreamReader(System.in));
194                       boolean done = false;
195                       while(!done) {
196                           System.out.println("Select a scheme to run, or <return> to exit:");
197                           for(int i = 0; i < prunedMatches.size(); i++) {
198                               System.out.println("\t" + (i + 1) + ") " + prunedMatches.get(i));
199                           }
200                           System.out.print("\nEnter a number > ");
201                           String choice = null;
202                           int schemeNumber = 0;
203                           try {
204                               choice = br.readLine();
205                               if(choice.equals("")) {
206                                   // System.exit(0);
207                                   return;
208                               }
209                               else {
210                                   schemeNumber = Integer.parseInt(choice);
211                                   schemeNumber--;
212                                   if(schemeNumber >= 0 && schemeNumber < prunedMatches.size()) {
213                                       schemeToRun = prunedMatches.get(schemeNumber);
214                                       done = true;
215                                   }
216                               }
217                           } catch(java.io.IOException ex) {
218                               // ignore
219                           }
220                       }
221                   }
222                   else {
223                       schemeToRun = prunedMatches.get(0);
224                   }
225               }
226  
227               Object scheme = null;
228               try {
229                   // scheme = java.beans.Beans.instantiate((new Run()).getClass()
230                   // .getClassLoader(), schemeToRun);
231                   scheme = WekaPackageClassLoaderManager.objectForName(schemeToRun);
232               } catch(Exception ex) {
233                   System.err.println(schemeToRun + " is not runnable!\n"
234                           + ex.getMessage());
235                   // System.exit(1);
236                   return;
237               }
238               // now see which interfaces/classes this scheme implements/extends
239               ArrayList<SchemeType> types = new ArrayList<SchemeType>();
240               if(scheme instanceof weka.core.CommandlineRunnable) {
241                   types.add(SchemeType.COMMANDLINE);
242               }
243               else {
244                   if(scheme instanceof weka.classifiers.Classifier) {
245                       types.add(SchemeType.CLASSIFIER);
246                   }
247                   if(scheme instanceof weka.clusterers.Clusterer) {
248                       types.add(SchemeType.CLUSTERER);
249                   }
250                   if(scheme instanceof weka.associations.Associator) {
251                       types.add(SchemeType.ASSOCIATOR);
252                   }
253                   if(scheme instanceof weka.attributeSelection.ASEvaluation) {
254                       types.add(SchemeType.ATTRIBUTE_SELECTION);
255                   }
256                   if(scheme instanceof weka.filters.Filter) {
257                       types.add(SchemeType.FILTER);
258                   }
259                   if(scheme instanceof weka.core.converters.AbstractFileLoader) {
260                       types.add(SchemeType.LOADER);
261                   }
262                   if(scheme instanceof weka.core.converters.AbstractFileSaver) {
263                       types.add(SchemeType.SAVER);
264                   }
265                   if(scheme instanceof weka.datagenerators.DataGenerator) {
266                       types.add(SchemeType.DATAGENERATOR);
267                   }
268               }
269  
270               SchemeType selectedType = null;
271               if(types.size() == 0) {
272                   System.err.println("" + schemeToRun + " is not runnable!");
273                   // System.exit(1);
274                   return;
275               }
276               if(types.size() == 1) {
277                   selectedType = types.get(0);
278               }
279               else {
280                   java.io.BufferedReader br =
281                           new java.io.BufferedReader(new java.io.InputStreamReader(System.in));
282                   boolean done = false;
283                   while(!done) {
284                       System.out.println("" + schemeToRun
285                               + " can be executed as any of the following:");
286                       for(int i = 0; i < types.size(); i++) {
287                           System.out.println("\t" + (i + 1) + ") " + types.get(i));
288                       }
289                       System.out.print("\nEnter a number > ");
290                       String choice = null;
291                       int typeNumber = 0;
292                       try {
293                           choice = br.readLine();
294                           if(choice.equals("")) {
295                               // System.exit(0);
296                               return;
297                           }
298                           else {
299                               typeNumber = Integer.parseInt(choice);
300                               typeNumber--;
301                               if(typeNumber >= 0 && typeNumber < types.size()) {
302                                   selectedType = types.get(typeNumber);
303                                   done = true;
304                               }
305                           }
306                       } catch(java.io.IOException ex) {
307                           // ignore
308                       }
309                   }
310               }
311  
312               if(selectedType == SchemeType.CLASSIFIER) {
313                   weka.classifiers.AbstractClassifier.runClassifier(
314                           (weka.classifiers.Classifier) scheme, options);
315               }
316               else if(selectedType == SchemeType.CLUSTERER) {
317                   weka.clusterers.AbstractClusterer.runClusterer(
318                           (weka.clusterers.Clusterer) scheme, options);
319               }
320               else if(selectedType == SchemeType.ATTRIBUTE_SELECTION) {
321                   weka.attributeSelection.ASEvaluation.runEvaluator(
322                           (weka.attributeSelection.ASEvaluation) scheme, options);
323               }
324               else if(selectedType == SchemeType.ASSOCIATOR) {
325                   weka.associations.AbstractAssociator.runAssociator(
326                           (weka.associations.Associator) scheme, options);
327               }
328               else if(selectedType == SchemeType.FILTER) {
329                   weka.filters.Filter.runFilter((weka.filters.Filter) scheme, options);
330               }
331               else if(selectedType == SchemeType.LOADER) {
332                   weka.core.converters.AbstractFileLoader.runFileLoader(
333                           (weka.core.converters.AbstractFileLoader) scheme, options);
334               }
335               else if(selectedType == SchemeType.SAVER) {
336                   weka.core.converters.AbstractFileSaver.runFileSaver(
337                           (weka.core.converters.AbstractFileSaver) scheme, options);
338               }
339               else if(selectedType == SchemeType.DATAGENERATOR) {
340                   weka.datagenerators.DataGenerator.runDataGenerator(
341                           (weka.datagenerators.DataGenerator) scheme, options);
342               }
343               else if(selectedType == SchemeType.COMMANDLINE) {
344                   ((weka.core.CommandlineRunnable) scheme).run(scheme, options);
345               }
346           } catch(Exception e) {
347               if(((e.getMessage() != null) && (e.getMessage().indexOf(
348                       "General options") == -1))
349                       || (e.getMessage() == null)) {
350                   e.printStackTrace();
351               }
352               else {
353                   System.err.println(e.getMessage());
354               }
355           }
356       }

358       public enum SchemeType {
359           CLASSIFIER("classifier"), CLUSTERER("clusterer"), ASSOCIATOR(
360                   "association rules"), ATTRIBUTE_SELECTION("attribute selection"), FILTER(
361                   "filter"), LOADER("loader"), SAVER("saver"), DATAGENERATOR(
362                   "data generator"), COMMANDLINE("general commandline runnable");
363  
364           private final String m_stringVal;
365  
366           SchemeType(String name) {
367               m_stringVal = name;
368           }
369  
370           @Override
371           public String toString() {
372               return m_stringVal;
373           }
374       }
