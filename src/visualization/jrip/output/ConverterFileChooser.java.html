389         addChoosableFileFilter(list.get(i));

393           setFileFilter(list.get(0));
394         } else {
395           setFileFilter(m_LastFilter);
396         }
397       }

399       // listener

401         removePropertyChangeListener(m_Listener);
402       }

404         @Override
405         public void propertyChange(PropertyChangeEvent evt) {
406           // filter changed
407           if (evt.getPropertyName().equals(FILE_FILTER_CHANGED_PROPERTY)) {
408             updateCurrentConverter();
409           }
410         }
411       };
412       addPropertyChangeListener(m_Listener);
413  
414       // initial setup
415       if (dialogType == LOADER_DIALOG) {
416         m_Editor.setClassType(AbstractFileLoader.class);
417         m_Editor.setValue(new weka.core.converters.ArffLoader());
418       } else {
419         m_Editor.setClassType(AbstractFileSaver.class);
420         m_Editor.setValue(new weka.core.converters.ArffSaver());
421       }
422  
423       updateCurrentConverter();
424     }

716      * sets the current converter according to the current filefilter.
717      */
718     protected void updateCurrentConverter() {
719       String[] extensions;
720       Object newConverter;
721  
722       if (getFileFilter() == null) {
723         return;
724       }
725  
726       if (!isAcceptAllFileFilterUsed()) {
727         // determine current converter
728         extensions = ((ExtensionFileFilter) getFileFilter()).getExtensions();
729         if (m_DialogType == LOADER_DIALOG) {
730           newConverter = ConverterUtils.getLoaderForExtension(extensions[0]);
731         } else {
732           newConverter = ConverterUtils.getSaverForExtension(extensions[0]);
733         }
734  
735         try {
736           if (m_CurrentConverter == null) {
737             m_CurrentConverter = newConverter;
738           } else {
739             if (!m_CurrentConverter.getClass().equals(newConverter.getClass())) {
740               m_CurrentConverter = newConverter;
741             }
742           }
743         } catch (Exception e) {
744           m_CurrentConverter = null;
745           e.printStackTrace();
746         }
747       } else {
748         m_CurrentConverter = null;
749       }
750     }

752     /**
753      * configures the current converter.
754      *
755      * @param dialogType the type of dialog to configure for
756      */
757     protected void configureCurrentConverter(int dialogType) {
758       String filename;
759       File currFile;
760  
761       if ((getSelectedFile() == null) || (getSelectedFile().isDirectory())) {
762         return;
763       }
764  
765       filename = getSelectedFile().getAbsolutePath();
766  
767       if (m_CurrentConverter == null) {
768         if (dialogType == LOADER_DIALOG) {
769           m_CurrentConverter = ConverterUtils.getLoaderForFile(filename);
770         } else if (dialogType == SAVER_DIALOG) {
771           m_CurrentConverter = ConverterUtils.getSaverForFile(filename);
772         } else {
773           throw new IllegalStateException("Cannot determine loader/saver!");
774         }
775  
776         // none found?
777         if (m_CurrentConverter == null) {
778           return;
779         }
780       }
781  
782       try {
783         currFile = ((FileSourcedConverter) m_CurrentConverter).retrieveFile();
784         if ((currFile == null) || (!currFile.getAbsolutePath().equals(filename))) {
785           ((FileSourcedConverter) m_CurrentConverter).setFile(new File(filename));
786         }
787       } catch (Exception e) {
788         e.printStackTrace();
789       }
790     }
