112       private void receiveTransactionFromServer() throws IOException, ClassNotFoundException {
113           Object transactionCandidate = _server.readObject();
114  
115           if(transactionCandidate.equals(ServerConnection.SUBSCRIBER_UP_TO_DATE)) {
116               synchronized (_upToDateMonitor) {
117                   _upToDateMonitor.notify();
118               }
119               return;
120           }
121  
122           if(transactionCandidate instanceof Date) {
123               Date clockTick = (Date) transactionCandidate;
124               _clock.advanceTo(clockTick);
125               return;
126           }
127  
128           if(transactionCandidate instanceof RuntimeException) {
129               _myTransactionRuntimeException = (RuntimeException) transactionCandidate;
130               notifyMyTransactionMonitor();
131               return;
132           }
133           if(transactionCandidate instanceof Error) {
134               _myTransactionError = (Error) transactionCandidate;
135               notifyMyTransactionMonitor();
136               return;
137           }
138  
139           TransactionTimestamp transactionTimestamp = (TransactionTimestamp) transactionCandidate;
140           Date timestamp = transactionTimestamp.executionTime();
141           long systemVersion = transactionTimestamp.systemVersion();
142  
143           _clock.advanceTo(timestamp);
144  
145           if(transactionTimestamp.capsule() == null) {
146               _subscriber.receive(new TransactionTimestamp(_myCapsule, systemVersion, timestamp));
147               notifyMyTransactionMonitor();
148               return;
149           }
150  
151           _subscriber.receive(new TransactionTimestamp(transactionTimestamp.capsule(), systemVersion, timestamp));
152       }
