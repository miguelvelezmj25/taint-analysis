177       private static Dimension getHqDimension(File image, Arguments args) throws IOException {
178           Dimension srcDimension = ImageUtil.getImageDimension(image);
179           Dimension hqDimension;
180           if(args.scaleMode == EScaleMode.FACTOR && args.scale < SVG_UPSCALE_FACTOR) {
181               hqDimension = new Dimension((int) args.round(SVG_UPSCALE_FACTOR / args.scale * (float) srcDimension.width), (int) args.round(SVG_UPSCALE_FACTOR / args.scale * (float) srcDimension.width));
182           }
183           else if(args.scaleMode == EScaleMode.DP_WIDTH && (args.scale * SVG_UPSCALE_FACTOR < srcDimension.width)) {
184               float scaleFactor = args.scale / (float) srcDimension.width * SVG_UPSCALE_FACTOR;
185               hqDimension = new Dimension((int) args.round(scaleFactor * (float) srcDimension.width), (int) args.round(scaleFactor * (float) srcDimension.height));
186           }
187           else if(args.scaleMode == EScaleMode.DP_HEIGHT && (args.scale * SVG_UPSCALE_FACTOR < srcDimension.height)) {
188               float scaleFactor = args.scale / (float) srcDimension.height * SVG_UPSCALE_FACTOR;
189               hqDimension = new Dimension((int) args.round(scaleFactor * (float) srcDimension.width), (int) args.round(scaleFactor * (float) srcDimension.height));
190           }
191           else {
192               hqDimension = srcDimension;
193           }
194           return hqDimension;
195       }
