261       private void initAndRetry(final String storePrefix,
262                                 final EntityModel modelParam,
263                                 final Mutations mutationsParam)
264           throws DatabaseException {

266           for (int i = 0;; i += 1) {
267               Transaction txn = null;
268               if (transactional && DbCompat.getThreadTransaction(env) == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
269                   txn =
270                       env.beginTransaction(null, store.getAutoCommitTxnConfig());
271               }
272               boolean success = false;
273               try {
274                   init(txn, storePrefix, modelParam, mutationsParam);
275                   success = true;
276                   return;
277               } catch (LockConflictException e) {
278  
279                   /*

281                    * EntityStore will cause a lock conflict. However, because we
282                    * read-modify-update the catalog record,
283                    * LockPreemptedException must be handled in a replicated JE
284                    * environment. Since LockPreemptedException is a
285                    * LockConfictException, it is simplest to retry when any
286                    * LockConfictException occurs.
287                    */
288                   if (i >= MAX_TXN_RETRIES) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
289                       throw e;

296                    * because a ReplicaWriteException may have occurred.
297                    * ReplicaWriteException invalidates the transaction, and there
298                    * are no writes to commit anyway. [#16655]
299                    */
300                   if (txn != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
301                       if (success && !isReadOnly()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]

306                   }
307               }
308           }
309       }
