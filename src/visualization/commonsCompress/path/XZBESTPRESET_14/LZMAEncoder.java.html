20       public static final int MODE_FAST = 1;

22  
23       /**
24        * LZMA2 chunk is considered full when its uncompressed size exceeds
25        * LZMA2_UNCOMPRESSED_LIMIT.
26        *

27        * A compressed LZMA2 chunk can hold 2 MiB of uncompressed data.
28        * A single LZMA symbol may indicate up to MATCH_LEN_MAX bytes
29        * of data, so the LZMA2 chunk is considered full when there is
30        * less space than MATCH_LEN_MAX bytes.
31        */
32       private static final int LZMA2_UNCOMPRESSED_LIMIT
33               = (2 << 20) - MATCH_LEN_MAX;
34  
35       /**
36        * LZMA2 chunk is considered full when its compressed size exceeds
37        * LZMA2_COMPRESSED_LIMIT.
38        *

39        * The maximum compressed size of a LZMA2 chunk is 64 KiB.
40        * A single LZMA symbol might use 20 bytes of space even though
41        * it usually takes just one byte or so. Two more bytes are needed
42        * for LZMA2 uncompressed chunks (see LZMA2OutputStream.writeChunk).
43        * Leave a little safety margin and use 26 bytes.
44        */
45       private static final int LZMA2_COMPRESSED_LIMIT = (64 << 10) - 26;

124       /**
125        * Gets an integer [0, 63] matching the highest two bits of an integer.
126        * This is like bit scan reverse (BSR) on x86 except that this also
127        * cares about the second highest bit.
128        */
129       public static int getDistSlot(int dist) {
130           if(dist <= DIST_MODEL_START && dist >= 0) {
131               return dist;
132           }
133  
134           int n = dist;
135           int i = 31;
136  
137           if((n & 0xFFFF0000) == 0) {
138               n <<= 16;
139               i = 15;
140           }
141  
142           if((n & 0xFF000000) == 0) {
143               n <<= 8;
144               i -= 8;
145           }
146  
147           if((n & 0xF0000000) == 0) {
148               n <<= 4;
149               i -= 4;
150           }
151  
152           if((n & 0xC0000000) == 0) {
153               n <<= 2;
154               i -= 2;
155           }
156  
157           if((n & 0x80000000) == 0) {  → 25
158               --i;
159           }
160  
161           return (i << 1) + ((dist >>> (i - 1)) & 1);
162       }

164       /**
165        * Gets the next LZMA symbol.
166        *

167        * There are three types of symbols: literal (a single byte),
168        * repeated match, and normal match. The symbol is indicated
169        * by the return value and by the variable back.
170        *

171        * Literal: back == -1 and return value is 1.
172        * The literal itself needs to be read from lz separately.
173        *

174        * Repeated match: back is in the range [0, 3] and
175        * the return value is the length of the repeated match.
176        *

177        * Normal match: back - REPS (back - 4)
178        * is the distance of the match and the return value is the length
179        * of the match.
180        */
181       abstract int getNextSymbol();

219       public void encodeLZMA1EndMarker() throws IOException {
220           // End of stream marker is encoded as a match with the maximum
221           // possible distance. The length is ignored by the decoder,
222           // but the minimum length has been used by the LZMA SDK.
223           //
224           // Distance is a 32-bit unsigned integer in LZMA.
225           // With Java's signed int, UINT32_MAX becomes -1.
226           int posState = (lz.getPos() - readAhead) & posMask;
227           rc.encodeBit(isMatch[state.get()], posState, 1);
228           rc.encodeBit(isRep, state.get(), 0);
229           encodeMatch(-1, MATCH_LEN_MIN, posState);
230       }

232       /**
233        * Compresses for LZMA2.
234        *
235        * @return true if the LZMA2 chunk became full, false otherwise
236        */
237       public boolean encodeForLZMA2() {
238           // LZMA2 uses RangeEncoderToBuffer so IOExceptions aren't possible.
239           try {
240               if(!lz.isStarted() && !encodeInit()) {
241                   return false;
242               }
243  
244               while (uncompressedSize <= LZMA2_UNCOMPRESSED_LIMIT
245                       && rc.getPendingSize() <= LZMA2_COMPRESSED_LIMIT)
246                   if(!encodeSymbol()) {
247                       return false;
248                   }
249           } catch (IOException e) {
250               throw new Error();
251           }
252  
253           return true;
254       }

256       private boolean encodeInit() throws IOException {
257           assert readAhead == -1;
258           if(!lz.hasEnoughData(0)) {
259               return false;
260           }
261  
262           // The first symbol must be a literal unless using
263           // a preset dictionary. This code isn't run if using
264           // a preset dictionary.
265           skip(1);
266           rc.encodeBit(isMatch[state.get()], 0, 0);
267           literalEncoder.encodeInit();
268  
269           --readAhead;
270           assert readAhead == -1;
271  
272           ++uncompressedSize;
273           assert uncompressedSize == 1;
274  
275           return true;
276       }
277  
278       private boolean encodeSymbol() throws IOException {
279           if(!lz.hasEnoughData(readAhead + 1)) {
280               return false;
281           }
282  
283           int len = getNextSymbol();
284  
285           assert readAhead >= 0;
286           int posState = (lz.getPos() - readAhead) & posMask;
287  
288           if(back == -1) {
289               // Literal i.e. eight-bit byte
290               assert len == 1;
291               rc.encodeBit(isMatch[state.get()], posState, 0);
292               literalEncoder.encode();
293           }
294           else {
295               // Some type of match
296               rc.encodeBit(isMatch[state.get()], posState, 1);
297               if(back < REPS) {
298                   // Repeated match i.e. the same distance
299                   // has been used earlier.
300                   assert lz.getMatchLen(-readAhead, reps[back], len) == len;
301                   rc.encodeBit(isRep, state.get(), 1);
302                   encodeRepMatch(back, len, posState);
303               }
304               else {
305                   // Normal match
306                   assert lz.getMatchLen(-readAhead, back - REPS, len) == len;
307                   rc.encodeBit(isRep, state.get(), 0);
308                   encodeMatch(back - REPS, len, posState);
309               }
310           }
311  
312           readAhead -= len;
313           uncompressedSize += len;
314  
315           return true;
316       }

318       private void encodeMatch(int dist, int len, int posState)
319               throws IOException {
320           state.updateMatch();
321           matchLenEncoder.encode(len, posState);
322  
323           int distSlot = getDistSlot(dist);
324           rc.encodeBitTree(distSlots[getDistState(len)], distSlot);
325  
326           if(distSlot >= DIST_MODEL_START) {
327               int footerBits = (distSlot >>> 1) - 1;
328               int base = (2 | (distSlot & 1)) << footerBits;
329               int distReduced = dist - base;
330  
331               if(distSlot < DIST_MODEL_END) {
332                   rc.encodeReverseBitTree(
333                           distSpecial[distSlot - DIST_MODEL_START],
334                           distReduced);
335               }
336               else {
337                   rc.encodeDirectBits(distReduced >>> ALIGN_BITS,
338                           footerBits - ALIGN_BITS);
339                   rc.encodeReverseBitTree(distAlign, distReduced & ALIGN_MASK);
340                   --alignPriceCount;
341               }
342           }
343  
344           reps[3] = reps[2];
345           reps[2] = reps[1];
346           reps[1] = reps[0];
347           reps[0] = dist;
348  
349           --distPriceCount;
350       }
351  
352       private void encodeRepMatch(int rep, int len, int posState)
353               throws IOException {
354           if(rep == 0) {
355               rc.encodeBit(isRep0, state.get(), 0);
356               rc.encodeBit(isRep0Long[state.get()], posState, len == 1 ? 0 : 1);
357           }
358           else {
359               int dist = reps[rep];
360               rc.encodeBit(isRep0, state.get(), 1);
361  
362               if(rep == 1) {
363                   rc.encodeBit(isRep1, state.get(), 0);
364               }
365               else {
366                   rc.encodeBit(isRep1, state.get(), 1);
367                   rc.encodeBit(isRep2, state.get(), rep - 2);
368  
369                   if(rep == 3) {
370                       reps[3] = reps[2];
371                   }
372  
373                   reps[2] = reps[1];
374               }
375  
376               reps[1] = reps[0];
377               reps[0] = dist;
378           }
379  
380           if(len == 1) {
381               state.updateShortRep();
382           }
383           else {
384               repLenEncoder.encode(len, posState);
385               state.updateLongRep();
386           }
387       }

446       int getLongRepAndLenPrice(int rep, int len, State state, int posState) {
447           int anyMatchPrice = getAnyMatchPrice(state, posState);
448           int anyRepPrice = getAnyRepPrice(anyMatchPrice, state);

453       int getMatchAndLenPrice(int normalMatchPrice,
454                               int dist, int len, int posState) {
455           int price = normalMatchPrice
456                   + matchLenEncoder.getPrice(len, posState);
457           int distState = getDistState(len);
458  
459           if(dist < FULL_DISTANCES) {
460               price += fullDistPrices[distState][dist];
461           }
462           else {
463               // Note that distSlotPrices includes also
464               // the price of direct bits.
465               int distSlot = getDistSlot(dist);
466               price += distSlotPrices[distState][distSlot]
467                       + alignPrices[dist & ALIGN_MASK];
468           }
469  
470           return price;
471       }

473       private void updateDistPrices() {
474           distPriceCount = DIST_PRICE_UPDATE_INTERVAL;
475  
476           for(int distState = 0; distState < DIST_STATES; ++distState) {
477               for(int distSlot = 0; distSlot < distSlotPricesSize; ++distSlot)
478                   distSlotPrices[distState][distSlot]
479                           = RangeEncoder.getBitTreePrice(
480                           distSlots[distState], distSlot);
481  
482               for(int distSlot = DIST_MODEL_END; distSlot < distSlotPricesSize;
483                   ++distSlot) {
484                   int count = (distSlot >>> 1) - 1 - ALIGN_BITS;
485                   distSlotPrices[distState][distSlot]
486                           += RangeEncoder.getDirectBitsPrice(count);
487               }
488  
489               for(int dist = 0; dist < DIST_MODEL_START; ++dist)
490                   fullDistPrices[distState][dist]
491                           = distSlotPrices[distState][dist];
492           }
493  
494           int dist = DIST_MODEL_START;
495           for(int distSlot = DIST_MODEL_START; distSlot < DIST_MODEL_END;
496               ++distSlot) {
497               int footerBits = (distSlot >>> 1) - 1;
498               int base = (2 | (distSlot & 1)) << footerBits;
499  
500               int limit = distSpecial[distSlot - DIST_MODEL_START].length;
501               for(int i = 0; i < limit; ++i) {
502                   int distReduced = dist - base;
503                   int price = RangeEncoder.getReverseBitTreePrice(
504                           distSpecial[distSlot - DIST_MODEL_START],
505                           distReduced);
506  
507                   for(int distState = 0; distState < DIST_STATES; ++distState)
508                       fullDistPrices[distState][dist]
509                               = distSlotPrices[distState][distSlot] + price;
510  
511                   ++dist;
512               }
513           }
514  
515           assert dist == FULL_DISTANCES;
516       }
517  
518       private void updateAlignPrices() {
519           alignPriceCount = ALIGN_PRICE_UPDATE_INTERVAL;
520  
521           for(int i = 0; i < ALIGN_SIZE; ++i)
522               alignPrices[i] = RangeEncoder.getReverseBitTreePrice(distAlign,
523                       i);
524       }

527        * Updates the lookup tables used for calculating match distance
528        * and length prices. The updating is skipped for performance reasons
529        * if the tables haven't changed much since the previous update.
530        */

542       }

547  
548           LiteralEncoder(int lc, int lp) {
549               super(lc, lp);
550  
551               subencoders = new LiteralSubencoder[1 << (lc + lp)];
552               for(int i = 0; i < subencoders.length; ++i)
553                   subencoders[i] = new LiteralSubencoder();
554           }

557               for(int i = 0; i < subencoders.length; ++i)
558                   subencoders[i].reset();
559           }

561           void encodeInit() throws IOException {
562               // When encoding the first byte of the stream, there is
563               // no previous byte in the dictionary so the encode function
564               // wouldn't work.
565               assert readAhead >= 0;
566               subencoders[0].encode();
567           }
568  
569           void encode() throws IOException {
570               assert readAhead >= 0;
571               int i = getSubcoderIndex(lz.getByte(1 + readAhead),
572                       lz.getPos() - readAhead);
573               subencoders[i].encode();
574           }
575  
576           int getPrice(int curByte, int matchByte,
577                        int prevByte, int pos, State state) {
578               int price = RangeEncoder.getBitPrice(
579                       isMatch[state.get()][pos & posMask], 0);
580  
581               int i = getSubcoderIndex(prevByte, pos);
582               price += state.isLiteral()
583                       ? subencoders[i].getNormalPrice(curByte)
584                       : subencoders[i].getMatchedPrice(curByte, matchByte);
585  
586               return price;
587           }
588  
589           private class LiteralSubencoder extends LiteralSubcoder {
590               void encode() throws IOException {
591                   int symbol = lz.getByte(readAhead) | 0x100;
592  
593                   if(state.isLiteral()) {
594                       int subencoderIndex;
595                       int bit;
596  
597                       do {
598                           subencoderIndex = symbol >>> 8;
599                           bit = (symbol >>> 7) & 1;
600                           rc.encodeBit(probs, subencoderIndex, bit);
601                           symbol <<= 1;
602                       } while (symbol < 0x10000);
603  
604                   }
605                   else {
606                       int matchByte = lz.getByte(reps[0] + 1 + readAhead);
607                       int offset = 0x100;
608                       int subencoderIndex;
609                       int matchBit;
610                       int bit;
611  
612                       do {
613                           matchByte <<= 1;
614                           matchBit = matchByte & offset;
615                           subencoderIndex = offset + matchBit + (symbol >>> 8);
616                           bit = (symbol >>> 7) & 1;
617                           rc.encodeBit(probs, subencoderIndex, bit);
618                           symbol <<= 1;
619                           offset &= ~(matchByte ^ symbol);
620                       } while (symbol < 0x10000);
621                   }
622  
623                   state.updateLiteral();
624               }
625  
626               int getNormalPrice(int symbol) {
627                   int price = 0;
628                   int subencoderIndex;
629                   int bit;
630  
631                   symbol |= 0x100;
632  
633                   do {
634                       subencoderIndex = symbol >>> 8;
635                       bit = (symbol >>> 7) & 1;
636                       price += RangeEncoder.getBitPrice(probs[subencoderIndex],
637                               bit);
638                       symbol <<= 1;
639                   } while (symbol < (0x100 << 8));
640  
641                   return price;
642               }
643  
644               int getMatchedPrice(int symbol, int matchByte) {
645                   int price = 0;
646                   int offset = 0x100;
647                   int subencoderIndex;
648                   int matchBit;
649                   int bit;
650  
651                   symbol |= 0x100;
652  
653                   do {
654                       matchByte <<= 1;
655                       matchBit = matchByte & offset;
656                       subencoderIndex = offset + matchBit + (symbol >>> 8);
657                       bit = (symbol >>> 7) & 1;
658                       price += RangeEncoder.getBitPrice(probs[subencoderIndex],
659                               bit);
660                       symbol <<= 1;
661                       offset &= ~(matchByte ^ symbol);
662                   } while (symbol < (0x100 << 8));
663  
664                   return price;
665               }
666           }
667       }

670       class LengthEncoder extends LengthCoder {
671           /**
672            * The prices are updated after at least
673            * PRICE_UPDATE_INTERVAL many lengths
674            * have been encoded with the same posState.
675            */
676           private static final int PRICE_UPDATE_INTERVAL = 32; // FIXME?
677  
678           private final int[] counters;
679           private final int[][] prices;
680  
681           LengthEncoder(int pb, int niceLen) {
682               int posStates = 1 << pb;
683               counters = new int[posStates];
684  
685               // Always allocate at least LOW_SYMBOLS + MID_SYMBOLS because
686               // it makes updatePrices slightly simpler. The prices aren't
687               // usually needed anyway if niceLen < 18.
688               int lenSymbols = Math.max(niceLen - MATCH_LEN_MIN + 1,
689                       LOW_SYMBOLS + MID_SYMBOLS);
690               prices = new int[posStates][lenSymbols];
691           }
692  
693           void reset() {
694               super.reset();
695  
696               // Reset counters to zero to force price update before
697               // the prices are needed.
698               for(int i = 0; i < counters.length; ++i)
699                   counters[i] = 0;
700           }
701  
702           void encode(int len, int posState) throws IOException {
703               len -= MATCH_LEN_MIN;
704  
705               if(len < LOW_SYMBOLS) {
706                   rc.encodeBit(choice, 0, 0);
707                   rc.encodeBitTree(low[posState], len);
708               }
709               else {
710                   rc.encodeBit(choice, 0, 1);
711                   len -= LOW_SYMBOLS;
712  
713                   if(len < MID_SYMBOLS) {
714                       rc.encodeBit(choice, 1, 0);
715                       rc.encodeBitTree(mid[posState], len);
716                   }
717                   else {
718                       rc.encodeBit(choice, 1, 1);
719                       rc.encodeBitTree(high, len - MID_SYMBOLS);
720                   }
721               }
722  
723               --counters[posState];
724           }
725  
726           int getPrice(int len, int posState) {
727               return prices[posState][len - MATCH_LEN_MIN];
728           }
729  
730           void updatePrices() {
731               for(int posState = 0; posState < counters.length; ++posState) {
732                   if(counters[posState] <= 0) {
733                       counters[posState] = PRICE_UPDATE_INTERVAL;
734                       updatePrices(posState);
735                   }
736               }
737           }
738  
739           private void updatePrices(int posState) {
740               int choice0Price = RangeEncoder.getBitPrice(choice[0], 0);
741  
742               int i = 0;
743               for(; i < LOW_SYMBOLS; ++i)
744                   prices[posState][i] = choice0Price
745                           + RangeEncoder.getBitTreePrice(low[posState], i);
746  
747               choice0Price = RangeEncoder.getBitPrice(choice[0], 1);
748               int choice1Price = RangeEncoder.getBitPrice(choice[1], 0);
749  
750               for(; i < LOW_SYMBOLS + MID_SYMBOLS; ++i)
751                   prices[posState][i] = choice0Price + choice1Price
752                           + RangeEncoder.getBitTreePrice(mid[posState],
753                           i - LOW_SYMBOLS);
754  
755               choice1Price = RangeEncoder.getBitPrice(choice[1], 1);
756  
757               for(; i < prices[posState].length; ++i)
758                   prices[posState][i] = choice0Price + choice1Price
759                           + RangeEncoder.getBitTreePrice(high, i - LOW_SYMBOLS
760                           - MID_SYMBOLS);
761           }
762       }
