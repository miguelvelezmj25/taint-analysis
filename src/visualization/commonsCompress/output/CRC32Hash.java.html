13  /**
14    * Provides a CRC32 table using the polynomial from IEEE 802.3.
15    */
16  class CRC32Hash {
17       static final int[] crcTable = new int[256];
18       private static final int CRC32_POLY = 0xEDB88320;
19  
20       static {
21           for(int i = 0; i < 256; ++i) { → [GZIPCOMPRESSOR]
22               int r = i;
23  
24               for(int j = 0; j < 8; ++j) { → [GZIPCOMPRESSOR]
25                   if((r & 1) != 0) { → [GZIPCOMPRESSOR]
26                       r = (r >>> 1) ^ CRC32_POLY;
27                   }
28                   else {
29                       r >>>= 1;
30                   }
31               }
32  
33               crcTable[i] = r;
34           }
35       }
36  }
