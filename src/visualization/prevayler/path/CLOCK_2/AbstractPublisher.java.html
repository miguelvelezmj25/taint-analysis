16  /**
17    * This class provides basic subscriber addition and notification.
18    */
19  public abstract class AbstractPublisher implements TransactionPublisher {
20  
21       protected final Clock _clock;
22       private final List _subscribers = new LinkedList();
23  
24  
25       public AbstractPublisher(Clock clock) {
26           _clock = clock;  → 9
27       }  → 10
28  
29       public Clock clock() {
30           return _clock;
31       }
32  
33       public synchronized void addSubscriber(TransactionSubscriber subscriber) {
34           _subscribers.add(subscriber);
35       }
36  
37       public synchronized void cancelSubscription(TransactionSubscriber subscriber) {
38           _subscribers.remove(subscriber);
39       }
40  
41       protected synchronized void notifySubscribers(TransactionTimestamp transactionTimestamp) {
42           Iterator i = _subscribers.iterator();
43           while (i.hasNext()) ((TransactionSubscriber) i.next()).receive(transactionTimestamp);
44       }
45  
46  }
