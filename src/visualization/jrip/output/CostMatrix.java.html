46  /**
47    * Class for storing and manipulating a misclassification cost matrix. The
48    * element at position i,j in the matrix is the penalty for classifying an
49    * instance of class j as class i. Cost values can be fixed or computed on a
50    * per-instance basis (cost sensitive evaluation only) from the value of an
51    * attribute or a mathematical expression involving attribute(s).<br>
52    * <br>
53    *
54    * Values in an instance are accessed in an expression by prefixing their index
55    * (starting at 1) with the character 'a'. E.g.<br>
56    * <br>
57    *
58    * a1 ˆ 2 * a5 / log(a7 * 4.0) <br>
59    *
60    * Supported opperators: +, -, *, /, ^, log, abs, cos, exp, sqrt, floor, ceil,
61    * rint, tan, sin, (, ).
62    *
63    *
64    *
65    *
66    * @author Mark Hall
67    * @author Richard Kirkby (rkirkby@cs.waikato.ac.nz)
68    * @version $Revision: 11868 $
69    * @see weka.core.UnsupportedAttributeTypeException
70    */

166     /**
167      * Applies the cost matrix to a set of instances. If a random number generator
168      * is supplied the instances will be resampled, otherwise they will be
169      * rewighted. Adapted from code once sitting in Instances.java
170      *
171      * @param data the instances to reweight.
172      * @param random a random number generator for resampling, if null then
173      * instances are rewighted.
174      * @return a new dataset reflecting the cost of misclassification.
175      * @exception Exception if the data has no class or the matrix in
176      * inappropriate.
177      */
178     public Instances applyCostMatrix(Instances data, Random random)
179       throws Exception {
180  
181       double sumOfWeightFactors = 0, sumOfMissClassWeights, sumOfWeights;
182       double[] weightOfInstancesInClass, weightFactor, weightOfInstances;
183  
184       if (data.classIndex() < 0) {
185         throw new Exception("Class index is not set!");
186       }
187  
188       if (size() != data.numClasses()) {
189         throw new Exception("Misclassification cost matrix has wrong format!");
190       }
191  
192       // are there any non-fixed, per-instance costs defined in the matrix?
193       if (replaceStrings(data)) {
194         // could reweight in the two class case
195         if (data.classAttribute().numValues() > 2) {
196           throw new Exception("Can't resample/reweight instances using "
197             + "non-fixed cost values when there are more " + "than two classes!");
198         } else {
199           // Store new weights
200           weightOfInstances = new double[data.numInstances()];
201           for (int i = 0; i < data.numInstances(); i++) {
202             Instance inst = data.instance(i);
203             int classValIndex = (int) inst.classValue();
204             double factor = 1.0;
205             Object element =
206               (classValIndex == 0) ? getCell(classValIndex, 1) : getCell(
207                 classValIndex, 0);
208             if (element instanceof Double) {
209               factor = ((Double) element).doubleValue();
210             } else {
211               factor = ((InstanceExpression) element).evaluate(inst);
212             }
213             weightOfInstances[i] = inst.weight() * factor;
214             /*
215              * System.err.println("Multiplying " +
216              * inst.classAttribute().value((int)inst.classValue()) +" by factor "
217              * + factor);
218              */
219           }
220  
221           // Change instances weight or do resampling
222           if (random != null) {
223             return data.resampleWithWeights(random, weightOfInstances);
224           } else {
225             Instances instances = new Instances(data);
226             for (int i = 0; i < data.numInstances(); i++) {
227               instances.instance(i).setWeight(weightOfInstances[i]);
228             }
229             return instances;
230           }
231         }
232       }
233  
234       weightFactor = new double[data.numClasses()];
235       weightOfInstancesInClass = new double[data.numClasses()];
236       for (int j = 0; j < data.numInstances(); j++) {
237         weightOfInstancesInClass[(int) data.instance(j).classValue()] +=
238           data.instance(j).weight();
239       }
240       sumOfWeights = Utils.sum(weightOfInstancesInClass);
241  
242       // normalize the matrix if not already
243       for (int i = 0; i < m_size; i++) {
244         if (!Utils.eq(((Double) getCell(i, i)).doubleValue(), 0)) {
245           CostMatrix normMatrix = new CostMatrix(this);
246           normMatrix.normalize();
247           return normMatrix.applyCostMatrix(data, random);
248         }
249       }
250  
251       for (int i = 0; i < data.numClasses(); i++) {
252         // Using Kai Ming Ting's formula for deriving weights for
253         // the classes and Breiman's heuristic for multiclass
254         // problems.
255  
256         sumOfMissClassWeights = 0;
257         for (int j = 0; j < data.numClasses(); j++) {
258           if (Utils.sm(((Double) getCell(i, j)).doubleValue(), 0)) {
259             throw new Exception("Neg. weights in misclassification "
260               + "cost matrix!");
261           }
262           sumOfMissClassWeights += ((Double) getCell(i, j)).doubleValue();
263         }
264         weightFactor[i] = sumOfMissClassWeights * sumOfWeights;
265         sumOfWeightFactors += sumOfMissClassWeights * weightOfInstancesInClass[i];
266       }
267       for (int i = 0; i < data.numClasses(); i++) {
268         weightFactor[i] /= sumOfWeightFactors;
269       }
270  
271       // Store new weights
272       weightOfInstances = new double[data.numInstances()];
273       for (int i = 0; i < data.numInstances(); i++) {
274         weightOfInstances[i] =
275           data.instance(i).weight()
276             * weightFactor[(int) data.instance(i).classValue()];
277       }
278  
279       // Change instances weight or do resampling
280       if (random != null) {
281         return data.resampleWithWeights(random, weightOfInstances);
282       } else {
283         Instances instances = new Instances(data);
284         for (int i = 0; i < data.numInstances(); i++) {
285           instances.instance(i).setWeight(weightOfInstances[i]);
286         }
287         return instances;
288       }
289     }
