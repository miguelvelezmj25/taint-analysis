33    * There are methods for adding and removing variables to this Environment

35    * method for replacing key names (enclosed by ${}) with their associated value
36    * in Strings.

74     /**

94     /**
95      * Tests for the presence of environment variables.
96      *
97      * @param source the string to test
98      * @return true if the argument contains one or more environment variables
99      */
100     public static boolean containsEnvVariables(String source) {
101       return (source.indexOf("${") >= 0);
102     }

138     /**
139      * Add a variable to the internal map of this properties object.
140      *
141      * @param key the name of the variable
142      * @param value its value
143      */
144     public void addVariable(String key, String value) {
145       m_envVars.put(key, value);
146     }

165     /**
166      * Remove a named variable from the map.
167      *
168      * @param key the name of the varaible to remove.
169      */
170     public void removeVariable(String key) {
171       m_envVars.remove(key);

174     /**
175      * Get the names of the variables (keys) stored in the internal map.
176      *
177      * @return a Set of variable names (keys)
178      */
179     public Set<String> getVariableNames() {

181     }
182  
183     /**
184      * Get the value for a particular variable.
185      *
186      * @param key the name of the variable to get
187      * @return the associated value or null if this variable is not in the
188      * internal map
189      */
190     public String getVariableValue(String key) {
191       return m_envVars.get(key);
192     }
