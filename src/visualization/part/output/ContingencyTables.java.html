68       /**
69        * Computes chi-squared statistic for a contingency table.
70        *
71        * @param matrix the contigency table
72        * @param useYates is Yates' correction to be used?
73        * @return the value of the chi-squared statistic
74        */
75       public static double chiVal(double[][] matrix, boolean useYates) {
76  
77           int df, nrows, ncols, row, col;
78           double[] rtotal, ctotal;
79           double expect = 0, chival = 0, n = 0;
80           boolean yates = true;
81  
82           nrows = matrix.length;
83           ncols = matrix[0].length;
84           rtotal = new double[nrows];
85           ctotal = new double[ncols];
86           for(row = 0; row < nrows; row++) {
87               for(col = 0; col < ncols; col++) {
88                   rtotal[row] += matrix[row][col];
89                   ctotal[col] += matrix[row][col];
90                   n += matrix[row][col];
91               }
92           }
93           df = (nrows - 1) * (ncols - 1);
94           if((df > 1) || (!useYates)) {
95               yates = false;
96           }
97           else if(df <= 0) {
98               return 0;
99           }
100           chival = 0.0;
101           for(row = 0; row < nrows; row++) {
102               if(Utils.gr(rtotal[row], 0)) {
103                   for(col = 0; col < ncols; col++) {
104                       if(Utils.gr(ctotal[col], 0)) {
105                           expect = (ctotal[col] * rtotal[row]) / n;
106                           chival += chiCell(matrix[row][col], expect, yates);
107                       }
108                   }
109               }
110           }
111           return chival;
112       }

114       /**
115        * Tests if Cochran's criterion is fullfilled for the given
116        * contingency table. Rows and columns with all zeros are not considered
117        * relevant.
118        *
119        * @param matrix the contigency table to be tested
120        * @return true if contingency table is ok, false if not
121        */
122       public static boolean cochransCriterion(double[][] matrix) {
123  
124           double[] rtotal, ctotal;
125           double n = 0, expect, smallfreq = 5;
126           int smallcount = 0, nonZeroRows = 0, nonZeroColumns = 0, nrows, ncols,
127                   row, col;
128  
129           nrows = matrix.length;
130           ncols = matrix[0].length;
131  
132           rtotal = new double[nrows];
133           ctotal = new double[ncols];
134           for(row = 0; row < nrows; row++) {
135               for(col = 0; col < ncols; col++) {
136                   rtotal[row] += matrix[row][col];
137                   ctotal[col] += matrix[row][col];
138                   n += matrix[row][col];
139               }
140           }
141           for(row = 0; row < nrows; row++) {
142               if(Utils.gr(rtotal[row], 0)) {
143                   nonZeroRows++;
144               }
145           }
146           for(col = 0; col < ncols; col++) {
147               if(Utils.gr(ctotal[col], 0)) {
148                   nonZeroColumns++;
149               }
150           }
151           for(row = 0; row < nrows; row++) {
152               if(Utils.gr(rtotal[row], 0)) {
153                   for(col = 0; col < ncols; col++) {
154                       if(Utils.gr(ctotal[col], 0)) {
155                           expect = (ctotal[col] * rtotal[row]) / n;
156                           if(Utils.sm(expect, smallfreq)) {
157                               if(Utils.sm(expect, 1)) {
158                                   return false;
159                               }
160                               else {
161                                   smallcount++;
162                                   if(smallcount > (nonZeroRows * nonZeroColumns) / smallfreq) {
163                                       return false;
164                                   }
165                               }
166                           }
167                       }
168                   }
169               }
170           }
171           return true;
172       }

174       /**
175        * Computes Cramer's V for a contingency table.
176        *
177        * @param matrix the contingency table
178        * @return Cramer's V
179        */
180       public static double CramersV(double[][] matrix) {
181  
182           int row, col, nrows, ncols, min;
183           double n = 0;
184  
185           nrows = matrix.length;
186           ncols = matrix[0].length;
187           for(row = 0; row < nrows; row++) {
188               for(col = 0; col < ncols; col++) {
189                   n += matrix[row][col];
190               }
191           }
192           min = nrows < ncols ? nrows - 1 : ncols - 1;
193           if((min == 0) || Utils.eq(n, 0)) {
194               return 0;
195           }
196           return Math.sqrt(chiVal(matrix, false) / (n * (double) min));
197       }

199       /**
200        * Computes the entropy of the given array.
201        *
202        * @param array the array
203        * @return the entropy
204        */
205       public static double entropy(double[] array) {
206  
207           double returnValue = 0, sum = 0;
208  
209           for(int i = 0; i < array.length; i++) {
210               returnValue -= lnFunc(array[i]);
211               sum += array[i];
212           }
213           if(Utils.eq(sum, 0)) {
214               return 0;
215           }
216           else {
217               return (returnValue + lnFunc(sum)) / (sum * log2);
218           }
219       }

221       /**
222        * Computes conditional entropy of the rows given
223        * the columns.
224        *
225        * @param matrix the contingency table
226        * @return the conditional entropy of the rows given the columns
227        */
228       public static double entropyConditionedOnColumns(double[][] matrix) {
229  
230           double returnValue = 0, sumForColumn, total = 0;
231  
232           for(int j = 0; j < matrix[0].length; j++) {
233               sumForColumn = 0;
234               for(int i = 0; i < matrix.length; i++) {
235                   returnValue = returnValue + lnFunc(matrix[i][j]);
236                   sumForColumn += matrix[i][j];
237               }
238               returnValue = returnValue - lnFunc(sumForColumn);
239               total += sumForColumn;
240           }
241           if(Utils.eq(total, 0)) {
242               return 0;
243           }
244           return -returnValue / (total * log2);
245       }

273       /**
274        * Computes conditional entropy of the columns given the rows
275        * of the test matrix with respect to the train matrix. Uses a
276        * Laplace prior. Does NOT normalize the entropy.
277        *
278        * @param train the train matrix
279        * @param test the test matrix
280        * @param numClasses the number of symbols for Laplace
281        * @return the entropy
282        */
283       public static double entropyConditionedOnRows(double[][] train,
284                                                     double[][] test,
285                                                     double numClasses) {
286  
287           double returnValue = 0, trainSumForRow, testSumForRow, testSum = 0;
288  
289           for(int i = 0; i < test.length; i++) {
290               trainSumForRow = 0;
291               testSumForRow = 0;
292               for(int j = 0; j < test[0].length; j++) {
293                   returnValue -= test[i][j] * Math.log(train[i][j] + 1);
294                   trainSumForRow += train[i][j];
295                   testSumForRow += test[i][j];
296               }
297               testSum = testSumForRow;
298               returnValue += testSumForRow * Math.log(trainSumForRow +
299                       numClasses);
300           }
301           return returnValue / (testSum * log2);
302       }

352       /**
353        * Computes gain ratio for contingency table (split on rows).
354        * Returns Double.MAX_VALUE if the split entropy is 0.
355        *
356        * @param matrix the contingency table
357        * @return the gain ratio
358        */
359       public static double gainRatio(double[][] matrix) {
360  
361           double preSplit = 0, postSplit = 0, splitEnt = 0,
362                   sumForRow, sumForColumn, total = 0, infoGain;
363  
364           // Compute entropy before split
365           for(int i = 0; i < matrix[0].length; i++) {
366               sumForColumn = 0;
367               for(int j = 0; j < matrix.length; j++)
368                   sumForColumn += matrix[j][i];
369               preSplit += lnFunc(sumForColumn);
370               total += sumForColumn;
371           }
372           preSplit -= lnFunc(total);
373  
374           // Compute entropy after split and split entropy
375           for(int i = 0; i < matrix.length; i++) {
376               sumForRow = 0;
377               for(int j = 0; j < matrix[0].length; j++) {
378                   postSplit += lnFunc(matrix[i][j]);
379                   sumForRow += matrix[i][j];
380               }
381               splitEnt += lnFunc(sumForRow);
382           }
383           postSplit -= splitEnt;
384           splitEnt -= lnFunc(total);
385  
386           infoGain = preSplit - postSplit;
387           if(Utils.eq(splitEnt, 0)) {
388               return 0;
389           }
390           return infoGain / splitEnt;
391       }

393       /**
394        * Returns negative base 2 logarithm of multiple hypergeometric
395        * probability for a contingency table.
396        *
397        * @param matrix the contingency table
398        * @return the log of the hypergeometric probability of the contingency table
399        */
400       public static double log2MultipleHypergeometric(double[][] matrix) {
401  
402           double sum = 0, sumForRow, sumForColumn, total = 0;
403  
404           for(int i = 0; i < matrix.length; i++) {
405               sumForRow = 0;
406               for(int j = 0; j < matrix[i].length; j++) {
407                   sumForRow += matrix[i][j];
408               }
409               sum += SpecialFunctions.lnFactorial(sumForRow);
410               total += sumForRow;
411           }
412           for(int j = 0; j < matrix[0].length; j++) {
413               sumForColumn = 0;
414               for(int i = 0; i < matrix.length; i++) {
415                   sumForColumn += matrix[i][j];
416               }
417               sum += SpecialFunctions.lnFactorial(sumForColumn);
418           }
419           for(int i = 0; i < matrix.length; i++) {
420               for(int j = 0; j < matrix[i].length; j++) {
421                   sum -= SpecialFunctions.lnFactorial(matrix[i][j]);
422               }
423           }
424           sum -= SpecialFunctions.lnFactorial(total);
425           return -sum / log2;
426       }

478       /**
479        * Calculates the symmetrical uncertainty for base 2.
480        *
481        * @param matrix the contingency table
482        * @return the calculated symmetrical uncertainty
483        */
484       public static double symmetricalUncertainty(double matrix[][]) {
485  
486           double sumForColumn, sumForRow, total = 0, columnEntropy = 0,
487                   rowEntropy = 0, entropyConditionedOnRows = 0, infoGain = 0;
488  
489           // Compute entropy for columns
490           for(int i = 0; i < matrix[0].length; i++) {
491               sumForColumn = 0;
492               for(int j = 0; j < matrix.length; j++) {
493                   sumForColumn += matrix[j][i];
494               }
495               columnEntropy += lnFunc(sumForColumn);
496               total += sumForColumn;
497           }
498           columnEntropy -= lnFunc(total);
499  
500           // Compute entropy for rows and conditional entropy
501           for(int i = 0; i < matrix.length; i++) {
502               sumForRow = 0;
503               for(int j = 0; j < matrix[0].length; j++) {
504                   sumForRow += matrix[i][j];
505                   entropyConditionedOnRows += lnFunc(matrix[i][j]);
506               }
507               rowEntropy += lnFunc(sumForRow);
508           }
509           entropyConditionedOnRows -= rowEntropy;
510           rowEntropy -= lnFunc(total);
511           infoGain = columnEntropy - entropyConditionedOnRows;
512           if(Utils.eq(columnEntropy, 0) || Utils.eq(rowEntropy, 0)) {
513               return 0;
514           }
515           return 2.0 * (infoGain / (columnEntropy + rowEntropy));
516       }

518       /**
519        * Computes Goodman and Kruskal's tau-value for a contingency table.
520        *
521        * @param matrix the contingency table
522        * @return Goodman and Kruskal's tau-value
523        */
524       public static double tauVal(double[][] matrix) {
525  
526           int nrows, ncols, row, col;
527           double[] ctotal;
528           double maxcol = 0, max, maxtotal = 0, n = 0;
529  
530           nrows = matrix.length;
531           ncols = matrix[0].length;
532           ctotal = new double[ncols];
533           for(row = 0; row < nrows; row++) {
534               max = 0;
535               for(col = 0; col < ncols; col++) {
536                   if(Utils.gr(matrix[row][col], max)) {
537                       max = matrix[row][col];
538                   }
539                   ctotal[col] += matrix[row][col];
540                   n += matrix[row][col];
541               }
542               maxtotal += max;
543           }
544           if(Utils.eq(n, 0)) {
545               return 0;
546           }
547           maxcol = ctotal[Utils.maxIndex(ctotal)];
548           return (maxtotal - maxcol) / (n - maxcol);
549       }

572       /**
573        * Computes chi-value for one cell in a contingency table.
574        *
575        * @param freq the observed frequency in the cell
576        * @param expected the expected frequency in the cell
577        * @return the chi-value for that cell; 0 if the expected value is
578        * too close to zero
579        */
580       private static double chiCell(double freq, double expected,
581                                     boolean yates) {
582  
583           // Cell in empty row and column?
584           if(Utils.smOrEq(expected, 0)) {
585               return 0;
586           }
587  
588           // Compute difference between observed and expected value
589           double diff = Math.abs(freq - expected);
590           if(yates) {
591  
592               // Apply Yates' correction if wanted
593               diff -= 0.5;
594  
595               // The difference should never be negative
596               if(diff < 0) {
597                   diff = 0;
598               }
599           }
600  
601           // Return chi-value for the cell
602           return (diff * diff / expected);
603       }
