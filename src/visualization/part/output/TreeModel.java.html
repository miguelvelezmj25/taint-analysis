76       /**
77        * Construct a ScoreDistribution entry
78        *
79        * @param scoreE the node containing the distribution
80        * @param miningSchema the mining schema
81        * @param baseCount the number of records at the node that owns this
82        * distribution entry
83        * @throws Exception if something goes wrong
84        */
85       protected ScoreDistribution(Element scoreE, MiningSchema miningSchema,
86         double baseCount) throws Exception {
87         // get the label
88         m_classLabel = scoreE.getAttribute("value");
89         Attribute classAtt = miningSchema.getFieldsAsInstances().classAttribute();
90         if (classAtt == null || classAtt.indexOfValue(m_classLabel) < 0) {
91           throw new Exception(
92             "[ScoreDistribution] class attribute not set or class value "
93               + m_classLabel + " not found!");
94         }
95  
96         m_classLabelIndex = classAtt.indexOfValue(m_classLabel);
97  
98         // get the frequency
99         String recordC = scoreE.getAttribute("recordCount");
100         m_recordCount = Double.parseDouble(recordC);
101  
102         // get the optional confidence
103         String confidence = scoreE.getAttribute("confidence");
104         if (confidence != null && confidence.length() > 0) {
105           m_confidence = Double.parseDouble(confidence);
106         } else if (!Utils.isMissingValue(baseCount) && baseCount > 0) {
107           m_confidence = m_recordCount / baseCount;
108         }
109       }
110  
111       /**
112        * Backfit confidence value (does nothing if the confidence value is already
113        * set).
114        *
115        * @param baseCount the total number of records (supplied either explicitly
116        * from the node that owns this distribution entry or most likely
117        * computed from summing the recordCounts of all the distribution
118        * entries in the distribution that owns this entry).
119        */
120       void deriveConfidenceValue(double baseCount) {
121         if (Utils.isMissingValue(m_confidence)
122           && !Utils.isMissingValue(baseCount) && baseCount > 0) {
123           m_confidence = m_recordCount / baseCount;
124         }
125       }

186       static Eval booleanToEval(boolean missing, boolean result) {
187         if (missing) {
188           return Eval.UNKNOWN;

196       /**
197        * Factory method to return the appropriate predicate for a given node in
198        * the tree.
199        *
200        * @param nodeE the XML node encapsulating the tree node.
201        * @param miningSchema the mining schema in use
202        * @return a Predicate
203        * @throws Exception of something goes wrong.
204        */
205       static Predicate getPredicate(Element nodeE, MiningSchema miningSchema)
206         throws Exception {
207  
208         Predicate result = null;
209         NodeList children = nodeE.getChildNodes();
210         for (int i = 0; i < children.getLength(); i++) {
211           Node child = children.item(i);
212           if (child.getNodeType() == Node.ELEMENT_NODE) {
213             String tagName = ((Element) child).getTagName();
214             if (tagName.equals("True")) {
215               result = new True();
216               break;
217             } else if (tagName.equals("False")) {
218               result = new False();
219               break;
220             } else if (tagName.equals("SimplePredicate")) {
221               result = new SimplePredicate((Element) child, miningSchema);
222               break;
223             } else if (tagName.equals("CompoundPredicate")) {
224               result = new CompoundPredicate((Element) child, miningSchema);
225               break;
226             } else if (tagName.equals("SimpleSetPredicate")) {
227               result = new SimpleSetPredicate((Element) child, miningSchema);
228               break;
229             }
230           }
231         }
232  
233         if (result == null) {
234           throw new Exception(
235             "[Predicate] unknown or missing predicate type in node");
236         }
237  
238         return result;
239       }
240     }

435       public SimplePredicate(Element simpleP, MiningSchema miningSchema) throws Exception {
436         Instances totalStructure = miningSchema.getFieldsAsInstances();
437  
438         // get the field name and set up the index
439         String fieldS = simpleP.getAttribute("field");
440         Attribute att = totalStructure.attribute(fieldS);
441         if (att == null) {
442           throw new Exception("[SimplePredicate] unable to find field " + fieldS
443             + " in the incoming instance structure!");
444         }
445  
446         // find the index
447         int index = -1;
448         for (int i = 0; i < totalStructure.numAttributes(); i++) {
449           if (totalStructure.attribute(i).name().equals(fieldS)) {
450             index = i;
451             m_fieldName = totalStructure.attribute(i).name();
452             break;
453           }
454         }
455         m_fieldIndex = index;
456         if (att.isNominal()) {
457           m_isNominal = true;
458         }
459  
460         // get the operator
461         String oppS = simpleP.getAttribute("operator");
462         for (Operator o : Operator.values()) {
463           if (o.toString().equals(oppS)) {
464             m_operator = o;
465             break;
466           }
467         }
468  
469         if (m_operator != Operator.ISMISSING
470           && m_operator != Operator.ISNOTMISSING) {
471           String valueS = simpleP.getAttribute("value");
472           if (att.isNumeric()) {
473             m_value = Double.parseDouble(valueS);
474           } else {
475             m_nominalValue = valueS;
476             m_value = att.indexOfValue(valueS);
477             if (m_value < 0) {
478               throw new Exception("[SimplePredicate] can't find value " + valueS
479                 + " in nominal " + "attribute " + att.name());
480             }
481           }
482         }
483       }
484  
485       @Override
486       public Predicate.Eval evaluate(double[] input) {
487         return m_operator.evaluate(input, m_value, m_fieldIndex);
488       }
489  
490       @Override
491       public String toString() {
492         StringBuffer temp = new StringBuffer();
493  
494         temp.append(m_fieldName + " " + m_operator.shortName());
495         if (m_operator != Operator.ISMISSING
496           && m_operator != Operator.ISNOTMISSING) {
497           temp.append(" " + ((m_isNominal) ? m_nominalValue : "" + m_value));
498         }
499  
500         return temp.toString();
501       }

520             for (Predicate p : constituents) {
521               Predicate.Eval temp = p.evaluate(input);
522               if (temp == Predicate.Eval.TRUE) {
523                 currentStatus = temp;
524                 break;
525               } else if (temp == Predicate.Eval.UNKNOWN) {
526                 currentStatus = temp;
527               }
528             }
529             return currentStatus;
530           }
531         },
532         AND("and") {
533           @Override
534           Predicate.Eval evaluate(ArrayList<Predicate> constituents,
535             double[] input) {
536             Predicate.Eval currentStatus = Predicate.Eval.TRUE;
537             for (Predicate p : constituents) {
538               Predicate.Eval temp = p.evaluate(input);
539               if (temp == Predicate.Eval.FALSE) {
540                 currentStatus = temp;
541                 break;
542               } else if (temp == Predicate.Eval.UNKNOWN) {
543                 currentStatus = temp;
544               }
545             }
546             return currentStatus;
547           }
548         },
549         XOR("xor") {
550           @Override
551           Predicate.Eval evaluate(ArrayList<Predicate> constituents,
552             double[] input) {
553             Predicate.Eval currentStatus = constituents.get(0).evaluate(input);
554             if (currentStatus != Predicate.Eval.UNKNOWN) {
555               for (int i = 1; i < constituents.size(); i++) {
556                 Predicate.Eval temp = constituents.get(i).evaluate(input);
557                 if (temp == Predicate.Eval.UNKNOWN) {
558                   currentStatus = temp;
559                   break;
560                 } else {
561                   if (currentStatus != temp) {
562                     currentStatus = Predicate.Eval.TRUE;
563                   } else {
564                     currentStatus = Predicate.Eval.FALSE;
565                   }
566                 }
567               }
568             }
569             return currentStatus;
570           }
571         },

573           @Override
574           Predicate.Eval evaluate(ArrayList<Predicate> constituents,
575             double[] input) {
576             Predicate.Eval currentStatus = constituents.get(0).evaluate(input);
577  
578             int i = 1;
579             while (currentStatus == Predicate.Eval.UNKNOWN) {
580               currentStatus = constituents.get(i).evaluate(input);

582  
583             // return false if all our surrogates evaluate to unknown.

589           }

613       public CompoundPredicate(Element compoundP, MiningSchema miningSchema) throws Exception {
614         // Instances totalStructure = miningSchema.getFieldsAsInstances();
615  
616         String booleanOpp = compoundP.getAttribute("booleanOperator");
617         for (BooleanOperator b : BooleanOperator.values()) {
618           if (b.toString().equals(booleanOpp)) {
619             m_booleanOperator = b;
620           }
621         }
622  
623         // now get all the encapsulated operators
624         NodeList children = compoundP.getChildNodes();
625         for (int i = 0; i < children.getLength(); i++) {
626           Node child = children.item(i);
627           if (child.getNodeType() == Node.ELEMENT_NODE) {
628             String tagName = ((Element) child).getTagName();
629             if (tagName.equals("True")) {
630               m_components.add(new True());
631             } else if (tagName.equals("False")) {
632               m_components.add(new False());
633             } else if (tagName.equals("SimplePredicate")) {
634               m_components
635                 .add(new SimplePredicate((Element) child, miningSchema));
636             } else if (tagName.equals("CompoundPredicate")) {
637               m_components.add(new CompoundPredicate((Element) child,
638                 miningSchema));
639             } else {
640               m_components.add(new SimpleSetPredicate((Element) child,
641                 miningSchema));
642             }
643           }
644         }
645       }

648       public Predicate.Eval evaluate(double[] input) {
649         return m_booleanOperator.evaluate(m_components, input);
650       }

653       public String toString() {
654         return toString(0, false);
655       }

660         for (int j = 0; j < level; j++) {
661           text.append("| ");
662         }
663  
664         text.append("Compound [" + m_booleanOperator.toString() + "]");
665         if (cr) {
666           text.append("\\n");
667         } else {
668           text.append("\n");
669         }
670         for (int i = 0; i < m_components.size(); i++) {
671           text.append(m_components.get(i).toString(level, cr).replace(":", ""));
672           if (i != m_components.size() - 1) {
673             if (cr) {
674               text.append("\\n");
675             } else {
676               text.append("\n");
677             }
678           }
679         }
680  
681         return text.toString();
682       }

697           @Override
698           Predicate.Eval evaluate(double[] input, int fieldIndex, Array set,
699             Attribute nominalLookup) {
700             if (set.getType() == Array.ArrayType.STRING) {
701               String value = "";
702               if (!Utils.isMissingValue(input[fieldIndex])) {
703                 value = nominalLookup.value((int) input[fieldIndex]);
704               }
705               return Predicate.booleanToEval(
706                 Utils.isMissingValue(input[fieldIndex]), set.contains(value));
707             } else if (set.getType() == Array.ArrayType.NUM
708               || set.getType() == Array.ArrayType.REAL) {
709               return Predicate.booleanToEval(
710                 Utils.isMissingValue(input[fieldIndex]),
711                 set.contains(input[fieldIndex]));
712             }
713             return Predicate.booleanToEval(
714               Utils.isMissingValue(input[fieldIndex]),
715               set.contains((int) input[fieldIndex]));
716           }
717         },
