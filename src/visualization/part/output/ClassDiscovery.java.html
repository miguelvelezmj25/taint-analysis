575        * <li>case insensitive</li>
576        * <li>german umlauts (ä , ö etc.) or other non-ASCII letters are
577        * treated as special chars</li>
578        * <li>special chars < numbers < letters</li>
579        * </ul>
580        */
581       public static class StringCompare implements Comparator<String>,
582               RevisionHandler {
583  
584           /**
585            * appends blanks to the string if its shorter than <code>len</code>.
586            *
587            * @param s the string to pad
588            * @param len the minimum length for the string to have
589            * @return the padded string
590            */
591           private String fillUp(String s, int len) {
592               while(s.length() < len) {
593                   s += " ";
594               }
595               return s;
596           }
597  
598           /**
599            * returns the group of the character: 0=special char, 1=number, 2=letter.
600            *
601            * @param c the character to check
602            * @return the group
603            */
604           private int charGroup(char c) {
605               int result;
606  
607               result = 0;
608  
609               if((c >= 'a') && (c <= 'z')) {
610                   result = 2;
611               }
612               else if((c >= '0') && (c <= '9')) {
613                   result = 1;
614               }
615  
616               return result;
617           }
618  
619           /**
620            * Compares its two arguments for order.
621            *
622            * @param o1 the first object
623            * @param o2 the second object
624            * @return -1 if o1<o2, 0 if o1=o2 and 1 if o1&;gt;o2
625            */
626           @Override
627           public int compare(String o1, String o2) {
628               String s1;
629               String s2;
630               int i;
631               int result;
632               int v1;
633               int v2;
634  
635               result = 0; // they're equal
636  
637               // get lower case string
638               s1 = o1.toString().toLowerCase();
639               s2 = o2.toString().toLowerCase();
640  
641               // same length
642               s1 = fillUp(s1, s2.length());
643               s2 = fillUp(s2, s1.length());
644  
645               for(i = 0; i < s1.length(); i++) {
646                   // same char?
647                   if(s1.charAt(i) == s2.charAt(i)) {
648                       result = 0;
649                   }
650                   else {
651                       v1 = charGroup(s1.charAt(i));
652                       v2 = charGroup(s2.charAt(i));
653  
654                       // different type (special, number, letter)?
655                       if(v1 != v2) {
656                           if(v1 < v2) {
657                               result = -1;
658                           }
659                           else {
660                               result = 1;
661                           }
662                       }
663                       else {
664                           if(s1.charAt(i) < s2.charAt(i)) {
665                               result = -1;
666                           }
667                           else {
668                               result = 1;
669                           }
670                       }
671  
672                       break;
673                   }
674               }
675  
676               return result;
677           }
678  
679           /**
680            * Indicates whether some other object is "equal to" this Comparator.
681            *
682            * @param obj the object to compare with this Comparator
683            * @return true if the object is a StringCompare object as well
684            */
685           @Override
686           public boolean equals(Object obj) {
687               return (obj instanceof StringCompare);
688           }

691            * Returns the revision string.
692            *
693            * @return the revision
694            */
695           @Override
696           public String getRevision() {
697               return RevisionUtils.extract("$Revision: 13476 $");
698           }
699       }
