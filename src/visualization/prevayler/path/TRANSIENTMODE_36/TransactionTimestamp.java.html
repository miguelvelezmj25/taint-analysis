5  package org.prevayler.implementation;
6  
7  import org.prevayler.foundation.Chunk;
8  
9  import java.io.Serializable;
10  import java.util.Date;
11  
12  public class TransactionTimestamp implements Serializable {
13  
14       static final long serialVersionUID = 1L;
15  
16       private final Capsule _capsule;
17       private final long _systemVersion;
18       private final long _executionTime;
19  
20       public TransactionTimestamp(Capsule capsule, long systemVersion, Date executionTime) {
21           this(capsule, systemVersion, executionTime.getTime());
22       }
23  
24       private TransactionTimestamp(Capsule capsule, long systemVersion, long executionTime) {
25           _capsule = capsule;
26           _systemVersion = systemVersion;
27           _executionTime = executionTime;
28       }
29  
30       public static TransactionTimestamp fromChunk(Chunk chunk) {
31           Capsule capsule = Capsule.fromChunk(chunk);
32           long systemVersion = Long.parseLong(chunk.getParameter("systemVersion"));
33           long executionTime = Long.parseLong(chunk.getParameter("executionTime"));
34           return new TransactionTimestamp(capsule, systemVersion, executionTime);
35       }
36  
37       public Capsule capsule() {
38           return _capsule;  → 31 → 32
39       }
40  
41       public long systemVersion() {
42           return _systemVersion;
43       }
44  
45       public Date executionTime() {
46           return new Date(_executionTime);
47       }
48  
49       public TransactionTimestamp cleanCopy() {
50           return new TransactionTimestamp(_capsule.cleanCopy(), _systemVersion, _executionTime);
51       }
52  
53       public Chunk toChunk() {
54           Chunk chunk = _capsule.toChunk();
55           chunk.setParameter("systemVersion", String.valueOf(_systemVersion));
56           chunk.setParameter("executionTime", String.valueOf(_executionTime));
57           return chunk;
58       }
59  
60  }
