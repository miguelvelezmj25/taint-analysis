265     /**

271      * <p>

275      * file is given, Saver tries to use standard out.

282     @Override
283     public void setOptions(String[] options) throws Exception {
284  
285       Option.setOptionsForHierarchy(options, this, AbstractFileSaver.class);
286       String outputString = Utils.getOption('o', options);
287       String inputString = Utils.getOption('i', options);
288  
289       ArffLoader loader = new ArffLoader();
290  
291       resetOptions();
292  
293       if (inputString.length() != 0) {
294         try {
295           File input = new File(inputString);
296           loader.setFile(input);
297           setInstances(loader.getDataSet());
298         } catch (Exception ex) {
299           ex.printStackTrace();
300           throw new IOException(
301             "No data set loaded. Data set has to be in ARFF format.");
302         }
303       }
304       if (outputString.length() != 0) {
305         boolean validExt = false;
306         for (String ext : getFileExtensions()) {
307           if (outputString.endsWith(ext)) {
308             validExt = true;
309             break;
310           }
311         }
312         // add appropriate file extension
313         if (!validExt) {
314           if (outputString.lastIndexOf('.') != -1) {
315             outputString =
316               (outputString.substring(0, outputString.lastIndexOf('.')))
317                 + FILE_EXTENSION;
318           } else {
319             outputString = outputString + FILE_EXTENSION;
320           }
321         }
322         try {
323           File output = new File(outputString);
324           setFile(output);
325         } catch (Exception ex) {
326           throw new IOException("Cannot create output file (Reason: "
327             + ex.toString() + "). Standard out is used.");
328         }
329       }
330     }
