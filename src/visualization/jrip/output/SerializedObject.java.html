104     /*

106      *
107      * @param compareTo the object to compare to
108      *
109      * @return whether or not the objects are equal
110      */
111     public final boolean equals(Object compareTo) {
112  
113       if (compareTo == null)
114         return false;
115       if (!compareTo.getClass().equals(this.getClass()))
116         return false;
117       byte[] compareArray = ((SerializedObject) compareTo).m_storedObjectArray;
118       if (compareArray.length != m_storedObjectArray.length)
119         return false;
120       for (int i = 0; i < compareArray.length; i++) {
121         if (compareArray[i] != m_storedObjectArray[i])
122           return false;
123       }
124       return true;
125     }
