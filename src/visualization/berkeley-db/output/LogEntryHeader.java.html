31        * Persistent fields. Layout on disk is
32        * (invariant) checksum - 4 bytes
33        * (invariant) entry type - 1 byte
34        * (invariant) entry flags - 1 byte
35        * (invariant) offset of previous log entry - 4 bytes
36        * (invariant) item size (not counting header size) - 4 bytes
37        * (optional) vlsn - 8 bytes
38        *
39        * Flags:
40        * The provisional bit can be set for any log type in the log. It's an
41        * indication to recovery that the entry shouldn't be processed when
42        * rebuilding the tree. See com.sleepycat.je.log.Provisional.java for
43        * the reasons why it's set.
44        *

46        * part of the replication stream and contains a VLSN in the header.
47        *
48        * The invisible bit is set when this log entry has been rolled back as

55        * that a VLSN is present. For all versions, if the replicated bit is set
56        * then a VLSN is always present. [#19476]

58        * first version of migrated LN
59        * a replicated LN
60        * --------------- -----------
61        * log version 7- replicated = true replicated = false
62        * (JE 4.1 and vlsn present = false vlsn present = false
63        * earlier) vlsn exists in header no vlsn in header
64        *
65        * log version 8+ replicated = true replicated = false

117       /*
118        * Prior to log version 6, a type-specific version was stored in each
119        * entry, and was packed together with the flags in a single byte.
120        *
121        * For version 6, we changed to use a global version (not type specific),
122        * but it was stored in each entry, packed with the flags as in earlier
123        * versions, as well as being stored redundantly in the FileHeader. The
124        * entry header and file header versions are always the same for all
125        * entries in a file. We flip the log file to guarantee this, when running
126        * for the first time with an upgraded JE with a new log version.
127        *
128        * For version 7 and above, the version is stored only in the FileHeader,
129        * freeing the space formerly taken by the version in each entry for use
130        * by flag bits. The version is not stored in each entry; however, the
131        * version is still maintained in this in-memory object for two reasons:
132        *
133        * 1. When reading log files prior to version 6, each entry potentially has
134        * a different version.
135        * 2. Convenience of access to the version when processing log entries.
136        *
137        * [#16939]
138        */

384           entryBuffer.position(ENTRYTYPE_OFFSET);
385           entryBuffer.put(entryType);

391           } else if (provisional == Provisional.BEFORE_CKPT_END) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]

395               flags |= REPLICATED_MASK;

420        * entryBuffer, and also assign the fields in this class.
421        * That's
422        * - the prev offset, which must be done within the log write latch to
423        * be sure what that lsn is

430                                                long lastOffset,
431                                                VLSN vlsn) {
432  
433           /* Add the prev pointer */
434           prevOffset = lastOffset;
435           entryBuffer.position(PREV_OFFSET);
436           LogUtils.writeUnsignedInt(entryBuffer, prevOffset);
437  
438           if (vlsn != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
439               this.vlsn = vlsn;
440               entryBuffer.position(VLSN_OFFSET);
441  
442               vlsn.writeToLog(entryBuffer);

444  
445           /*

447            * to set the field in this instance, for use later when printing or
448            * debugging the header.
449            */
450           Checksum checksum = Adler32.makeChecksum();

452                           entryBuffer.arrayOffset() + CHECKSUM_BYTES,
453                           entryBuffer.limit() - CHECKSUM_BYTES);
454           entryBuffer.position(0);
455           checksumVal = checksum.getValue();
456           LogUtils.writeUnsignedInt(entryBuffer, checksumVal);
457  
458           /* Leave this buffer ready for copying into another buffer. */
459           entryBuffer.position(0);

462       }

513       /**
514        * For use in special case where commits are transformed to aborts because
515        * of i/o errors during a logBuffer flush. See [11271].
516        * Assumes that the entryBuffer is positioned at the start of the item.
517        * Return with the entryBuffer positioned to the end of the log entry.
518        */
519       void convertCommitToAbort(ByteBuffer entryBuffer) {
520           assert (entryType == LogEntryType.LOG_TXN_COMMIT.getTypeNum());
521  
522           /* Remember the start of the entry item. */
523           int itemStart = entryBuffer.position();
524  
525           /* Back up to where the type is stored and change the type. */
526           int entryTypePosition =
527               itemStart - (getSize() - ENTRYTYPE_OFFSET);
528           entryBuffer.position(entryTypePosition);
529           entryBuffer.put(LogEntryType.LOG_TXN_ABORT.getTypeNum());
530  
531           /*
532            * Recalculate the checksum. This byte buffer could be large,
533            * so don't just turn the whole buffer into an array to pass
534            * into the checksum object.
535            */
536           Checksum checksum = Adler32.makeChecksum();
537           int checksumSize = itemSize + (getSize() - CHECKSUM_BYTES);
538           checksum.update(entryBuffer.array(),
539                           entryTypePosition + entryBuffer.arrayOffset(),
540                           checksumSize);
541           entryBuffer.position(itemStart - getSize());
542           checksumVal = checksum.getValue();
543           LogUtils.writeUnsignedInt(entryBuffer, checksumVal);
544       }

588            * on-disk compression can make itemSize vary if the entry has VLSNs
589            * that were packed differently.
590            */
591           return ((getVLSN().equals(other.getVLSN())) &&
592                   (getReplicated() == other.getReplicated()) &&
593                   (isInvisible() == other.isInvisible()) &&
594                   (LogEntryType.compareTypeAndVersion(getVersion(), getType(),
595                                                       other.getVersion(),
596                                                       other.getType())));
597       }

600        * May be called after reading MIN_HEADER_SIZE bytes to determine
601        * whether more bytes (getVariablePortionSize) should be read.
602        */
603       public boolean isVariableLength() {
604           /* Currently only entries with VLSNs are variable length. */
605           return vlsnPresent;
606       }

615       /**
616        * Turn off the invisible bit in the byte buffer which backs this log entry
617        * header.
618        * @param logHeaderStartPosition the byte position of the start of the log
619        * entry header.
620        */
621       public static void turnOffInvisible(ByteBuffer buffer,
622                                           int logHeaderStartPosition) {
623
