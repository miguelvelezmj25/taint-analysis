243             continue;
244           }
245           OptionMetadata parameterDescription = null;
246           parameterDescription = getter.getAnnotation(OptionMetadata.class);
247           if (parameterDescription == null) {
248             parameterDescription = setter.getAnnotation(OptionMetadata.class);

251           if (parameterDescription != null
252             && parameterDescription.commandLineParamName().length() > 0) {
253             Object value = getter.invoke(target, args);
254             if (value != null) {

256                 options.add("-" + parameterDescription.commandLineParamName());
257               }

259               if (value.getClass().isArray()) {
260                 if (parameterDescription.commandLineParamIsFlag()) {

266                   // remove the initial option identifier (added above)
267                   // as the array is empty
268                   options.remove(options.size() - 1);
269                 }
270                 int index = 0;
271                 for (Object element : (Object[]) value) {
272                   if (index > 0) {
273                     options
274                       .add("-" + parameterDescription.commandLineParamName());
275                   }

277                     options

279                   } else {
280                     options.add(element.toString());
281                   }

283                 }

285                 if (parameterDescription.commandLineParamIsFlag()) {

288                       + "line flag should return a boolean value");

301                         + "line flag should return a boolean value");
