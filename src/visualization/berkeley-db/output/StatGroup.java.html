25  /**
26    * The Stats infrastructure provides context for JE statistics. Each statistic
27    * has these attributes:
28    * - metadata - specifically, a name and description
29    * - each statistic is associated with a parent stat group, which itself has
30    * a name and description.
31    * - support for the StatsConfig.clear semantics
32    * - a way to print statistics in a user friendly way.
33    *
34    * To create a statistic variable, instantiate one of the concrete subclasses
35    * of Stat. Each concrete subclass should hold the methods that are needed to
36    * best set and display the value. For example, instead of using LongStat to
37    * hold a timestamp or LSN value, use TimestampStat or LSNStat. A Stat instance
38    * needs to specify a StatDefinition. There may be multiple Stat variables in
39    * different components that share a StatDefinition. They are differentiated
40    * when displayed by their parent StatGroup.
41    *
42    * Each Stat instance is associated with a StatGroup, which holds the
43    * collection of stats that belong to a given component. Each member of the
44    * StatGroup has a unique StatDefinition. StatGroups can be combined, in order
45    * to accumulate values. For example, the LockManager may have multiple lock
46    * tables. Each lock table keeps its own latch statistics. When LockStats are
47    * generated, the StatsGroup for each latch is collected and rolled up into a
48    * single StatGroup, using the addAll(StatGroup) method.
49    *
50    * The Stats infrastructure is for internal use only. Public API classes like
51    * EnvironmentStats, LockStats, etc, contain StatGroups. A call to retrieve
52    * stats is implemented by getting a clone of the StatGroups held by the
53    * components like the cleaner, the incompressor, the LockManager, etc. The
54    * public API classes provide getter methods that reach into the StatGroups to
55    * return the specific stat value.
56    *
57    * To add a statistic, create the Stat variable in the component where it is
58    * being used and associate it with a StatGroup. The Stat infrastructure does
59    * the rest of the work for plumbing that statistic up to the public API
60    * class. Each API class must provide a getter method to access the specific
61    * statistic. Currently, this is done manually.
62    */
63  public class StatGroup implements Serializable {
64       private static final long serialVersionUID = 1L;
65  
66       /*
67        * User understandable description of the grouping. The description may
68        * indicate that these stats are meant for internal use.
69        */

71       private final String groupDescription;
72       private final Map<StatDefinition, Stat<?>> stats;

99       void register(Stat<?> oneStat) {
100           Stat<?> prev = stats.put(oneStat.getDefinition(), oneStat);
101           assert (prev == null) : "prev = " + prev + " oneStat=" + → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
102               oneStat.getDefinition();

105       /**
106        * Add all the stats from the other group into this group. If both groups
107        * have the same stat, add the values. The caller must make sure that no
108        * stats are added to or removed from the argument during this call.
109        *
110        * @throws ConcurrentModificationException if the addition or removal of
111        * stats in the argument is detected
112        */
113       @SuppressWarnings("unchecked")
114       public void addAll(StatGroup other) {
115  
116           for (Entry<StatDefinition, Stat<?>> entry :
117                other.stats.entrySet()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
118  
119               StatDefinition definition = entry.getKey();
120               Stat<?> localStat;
121               synchronized (stats) {
122                   localStat = stats.get(definition);
123                   if (localStat == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
124                       stats.put(definition, entry.getValue());
125                       continue;
126                   }
127               }
128  
129               /*
130                * Cast to get around type problem. We know it's the same stat type
131                * because the definition is the same, but the compiler doesn't
132                * know that.
133                */
134               @SuppressWarnings("rawtypes")
135               Stat additionalValue = entry.getValue();
136               localStat.add(additionalValue);
137           }
138       }
139  
140       /**
141        * The caller must make sure that no stats are added to or removed from
142        * this stat group while this method is being called.
143        *
144        * @throws ConcurrentModificationException if the addition or removal of
145        * stats in this group is detected
146        */
147       @SuppressWarnings("unchecked")
148       public StatGroup computeInterval(StatGroup baseGroup) {
149           Map<StatDefinition, Stat<?>> intervalValues =
150                   new HashMap<StatDefinition, Stat<?>>();
151  
152           for (Entry<StatDefinition, Stat<?>> entry :
153                stats.entrySet()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
154               StatDefinition definition = entry.getKey();
155               Stat<?> statValue = entry.getValue();
156               @SuppressWarnings("rawtypes")
157               Stat baseStat = baseGroup.stats.get(definition);
158               if (baseStat == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
159                   intervalValues.put(definition, statValue.copy());
160               } else {
161                   intervalValues.put(definition,
162                                      statValue.computeInterval(baseStat));
163               }
164           }
165           return new StatGroup(groupName, groupDescription, intervalValues);
166       }

198       /**
199        * @return a Stats class that copies the value of all stats in the group
200        */
201       public StatGroup cloneGroup(boolean clear) {
202  
203           Map<StatDefinition, Stat<?>> copyValues =
204               new HashMap<StatDefinition, Stat<?>>();
205  
206           synchronized (stats) {
207               for (Stat<?> s : stats.values()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
208                   if (clear) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
209                       copyValues.put(s.getDefinition(), s.copyAndClear());
210                   } else {
211                       copyValues.put(s.getDefinition(), s.copy());
212                   }
213               }
214           }
215           return new StatGroup(groupName, groupDescription, copyValues);
216       }

228       public int getInt(StatDefinition definition) {
229           int retval;
230           Stat<?> s = stats.get(definition);
231           if (s == null) {
232               retval = 0;
233           } else if (s instanceof IntStat) {
234               retval = ((IntStat) s).get();
235           } else if (s instanceof AtomicIntStat) {
236               retval = ((AtomicIntStat) s).get();
237           } else {
238               assert false : "Internal error calling getInt with" +
239                   " unexpected stat type: " + s.getClass().getName();
240               retval = 0;
241           }
242           return retval;
243       }

348       @Override
349       public String toString() {
350           StringBuilder sb = new StringBuilder();
351           sb.append(groupName).append(": ");
352           sb.append(groupDescription).append("\n");
353  
354           /* Order the stats for consistent display. */
355           Map<StatDefinition, Stat<?>> sortedStats;
356           synchronized (stats) {
357               sortedStats = new TreeMap<StatDefinition, Stat<?>>(stats);
358           }
359           for (Stat<?> s : sortedStats.values()) {

361           }

385       /**
386        * Only print values that are not null or zero.
387        */
388       public String toStringConcise() {
389  
390           boolean headerPrinted = false;
391           StringBuilder sb = new StringBuilder();
392  
393           /* Order the stats for consistent display.*/
394           Map<StatDefinition, Stat<?>> sortedStats;
395           synchronized (stats) {
396               sortedStats = new TreeMap<StatDefinition, Stat<?>>(stats);
397           }
398  
399           for (Stat<?> s : sortedStats.values()) {
400  
401               if (s.isNotSet()) {
402                   continue;
403               }
404  
405               /*
406                * Print the group name lazily, in case no fields in this group are
407                * set at all. In that case, this method will not print anything.
408                */
409               if (!headerPrinted) {
410                   sb.append(groupName + "\n");
411                   headerPrinted = true;
412               }
413               sb.append("\t").append(s).append("\n");
414           }
415           return sb.toString();
416       }
