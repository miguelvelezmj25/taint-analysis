42  /**
43    * A cache of the histograms for all files, except for the last file. Also
44    * caches the number of bytes expired in the current interval.
45    *
46    * No memory budgeting is performed because the size of these data structures
47    * is small compared to the Btree they represent. The serialized form of the
48    * histogram is cached, which is a small number of bytes per file. If no data
49    * in a file expires, it will not have a cache entry.
50    *
51    * Possible future optimization: If there is contention on this data structure,
52    * the refresh method could create a read-only map containing the current number
53    * of expired bytes, for access by getExpiredBytes without synchronization.
54    */
55  public class ExpirationProfile {
56  
57       private static Pair<Integer, Integer> PAIR_OF_ZEROS = new Pair<>(0, 0);
58  
59       private final EnvironmentImpl env;
60  
61       /*
62        * The 'map' of file number to Histogram, protected by its own mutex.
63        *
64        * Note that if the map and completedTrackers mutexes are both held, they
65        * must be acquired in that order.
66        */
67       private final Map<Long, ExpInfo> map;
68  
69       /*
70        * The the expiration times in days and hours of last refresh, i.e, the
71        * cached bytes in each Histogram are those that were expired on this
72        * day/hour. Both fields are protected by the 'map' mutex.
73        */
74       private int lastRefreshHour = -1;
75       private int lastRefreshDay = -1;
76  
77       /*
78        * Whether any expiration times are in hours. If false, all intervals are
79        * in days. Protected by the 'map' mutex.
80        */
81       private boolean anyExpirationInHours;
82  
83       /*
84        * Map of file number to completed tracker. Protected by its own mutex.
85        *
86        * Note that if the map and completedTrackers mutexes are both held, they
87        * must be acquired in that order.
88        */
89       private final Map<Long, ExpirationTracker> completedTrackers;
90  
91       /*
92        * The expiration summary DB. Its key is the file number, a long. Its data
93        * is the serialized form of the histogram, or is empty (zero length) if
94        * the file has no expired data. The latter case includes files created
95        * before the TTL feature was added.
96        */
97       private DatabaseImpl db;
98  
99       public ExpirationProfile(final EnvironmentImpl env) {
100           this.env = env;
101           map = new HashMap<>();
102           completedTrackers = new HashMap<>();
103       }
104  
105       /** Make a copy for used in utilities, etc. */
106       public ExpirationProfile(final ExpirationProfile other) {
107           env = other.env;
108           db = other.db;
109           synchronized (other.map) {
110               map = new HashMap<>(other.map);
111           }
112           completedTrackers = Collections.emptyMap();
113       }
114  
115       /**
116        * Called at the end of recovery to open the expiration DB, and cache its
117        * records in the profile's map.
118        *
119        * Also collects expiration info for any complete file having expiration
120        * info that was not written to the DB earlier, due to a crash for example.
121        *
122        * Also initializes the tracker for the current file in the log manager,
123        * reading/tracking the existing entries in that file.
124        */
125       public void populateCache(
126           final StartupTracker.Counter counter,
127           final ProgressListener<RecoveryProgress> listener) {
128  
129           synchronized (map) {
130  
131               assert db == null;
132               assert completedTrackers.isEmpty();
133  
134               db = env.getDbTree().openNonRepInternalDB(DbType.EXPIRATION);
135  
136               if (db == null) { → [ALLOWCREATE, TRANSACTIONAL]
137                   /* Read-only env with no expiration summary DB. */
138                   return;
139               }
140  
141               final DatabaseEntry key = new DatabaseEntry();
142               final DatabaseEntry data = new DatabaseEntry();
143  
144               /* Ordered array of file numbers. */
145               final Long[] existingFiles =
146                   env.getFileManager().getAllFileNumbers();
147  
148               /* Parallel array to existingFiles. */
149               final boolean[] filesHaveRecords =
150                   new boolean[existingFiles.length];
151  
152               /*
153                * For the last file we must always get its expiration info and
154                * then initialize the log manager's tracker. Note that its DB
155                * record, if any, is deleted below.
156                */
157               final FileProcessor processor = env.getCleaner().createProcessor();
158               final long lastFileNum = env.getFileManager().getCurrentFileNum();
159  
160               if (!env.isReadOnly()) { → [ALLOWCREATE, TRANSACTIONAL]
161                   if (existingFiles.length > 0) { → [ALLOWCREATE, TRANSACTIONAL]
162  
163                       /* Flush to ensure the cleaner can read all entries. */
164                       env.flushLog(false /*fsync*/);
165  
166                       final ExpirationTracker tracker =
167                           processor.countExpiration(lastFileNum);
168  
169                       env.getLogManager().initExpirationTracker(tracker);
170                   } else {
171                       env.getLogManager().initExpirationTracker(
172                           new ExpirationTracker(0));
173                   }
174               }
175  
176               /*
177                * Populate map with existing records in the DB that correspond to
178                * existing files. Delete records in the DB that do not correspond
179                * to existing files, to clean-up past errors. Also delete the
180                * record for the last file, if it exists.
181                */
182               final Locker locker = BasicLocker.createBasicLocker(
183                   env, false /*noWait*/);
184  
185               try (final Cursor cursor =
186                    DbInternal.makeCursor(db, locker, null)) {
187  
188                   while (cursor.get(key, data, Get.NEXT, null) != null) { → [ALLOWCREATE, TRANSACTIONAL]
189  
190                       counter.incNumRead();
191  
192                       final long fileNum =
193                           SortedPackedLongBinding.entryToLong(key);
194  
195                       final int i = Arrays.binarySearch(existingFiles, fileNum);
196  
197                       if (i >= 0 && existingFiles[i] < lastFileNum) { → [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL]
198  
199                           filesHaveRecords[i] = true;
200  
201                           final byte[] serializedForm = data.getData();
202  
203                           if (serializedForm.length > 0) { → [ALLOWCREATE, TRANSACTIONAL]
204                               counter.incNumProcessed();
205                               map.put(
206                                   fileNum,
207                                   new ExpInfo(serializedForm, 0));
208                           }
209                       } else if (!env.isReadOnly()) { → [ALLOWCREATE, TRANSACTIONAL]
210                           counter.incNumDeleted();
211                           cursor.delete();
212                       }
213                   }
214               } finally { → [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL]
215                   locker.operationEnd();
216               }
217  
218               /*
219                * If a record is missing for an existing file, use the cleaner to
220                * get the expiration info for the file, and then add a record to
221                * the DB and to the map. Note that the last file is not processed.
222                */
223               for (int i = 0;
224                    i < existingFiles.length && existingFiles[i] < lastFileNum; → [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL]
225                    i += 1) {
226  
227                   if (filesHaveRecords[i]) { → [ALLOWCREATE, TRANSACTIONAL]
228                       continue;
229                   }
230  
231                   final long fileNum = existingFiles[i];
232  
233                   counter.incNumAux();
234  
235                   if (listener != null) { → [ALLOWCREATE, TRANSACTIONAL]
236                       listener.progress(
237                           RecoveryProgress.POPULATE_EXPIRATION_PROFILE,
238                           1, -1);
239                   }
240  
241                   final ExpirationTracker tracker =
242                       processor.countExpiration(fileNum);
243  
244                   putFile(tracker, 0);
245  
246                   LoggerUtils.info(
247                       env.getLogger(), env,
248                       "Loaded missing expiration data from file 0x" +
249                           Long.toHexString(fileNum));
250               }
251           }
252       }
253  
254       /**
255        * Writes a record in the expiration summary DB for the given tracker,
256        * and (if there is any data with an expiration time) adds it to the map.
257        *
258        * Because this method and {@link #removeFile} perform Btree operations
259        * while synchronized on the 'map', it is important that an IN is not
260        * latched when calling these methods, which could cause a deadlock. Also,
261        * an IN latch should not be held while calling a method that synchronizes
262        * on {@link FileSelector}, since methods in this class are called while
263        * synchronized on FileSelector, and this could cause a 3-way deadlock
264        * [#25613].
265        */
266       void putFile(final ExpirationTracker tracker, final int expiredSize) {
267  
268           final long fileNum = tracker.getFileNum();
269           final byte[] serializedForm = tracker.serialize();
270  
271           synchronized (map) {
272  
273               if (db != null && !env.isReadOnly()) { → [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL]
274                   final DatabaseEntry key = new DatabaseEntry();
275                   final DatabaseEntry data = new DatabaseEntry();
276  
277                   SortedPackedLongBinding.longToEntry(fileNum, key);
278  
279                   data.setData(serializedForm);
280  
281                   final Locker locker = BasicLocker.createBasicLocker(
282                       env, false /*noWait*/);
283  
284                   try (final Cursor cursor =
285                            DbInternal.makeCursor(db, locker, null)) {
286  
287                       cursor.put(key, data, Put.OVERWRITE, null);
288  
289                   } finally { → [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL] & [ALLOWCREATE, TRANSACTIONAL]
290                       locker.operationEnd();
291                   }
292               }
293  
294               if (serializedForm.length > 0) { → [ALLOWCREATE, TRANSACTIONAL]
295                   map.put(
296                       fileNum,
297                       new ExpInfo(serializedForm, expiredSize));
298               }
299           }
300       }
301  
302       /**
303        * Remove entry for a file from the map and DB, when the file is deleted.
304        */
305       void removeFile(final long fileNum) {
306  
307           if (db == null || env.isReadOnly()) {
308               return;
309           }
310  
311           synchronized (map) {
312  
313               map.remove(fileNum);
314  
315               final DatabaseEntry key = new DatabaseEntry();
316               SortedPackedLongBinding.longToEntry(fileNum, key);
317  
318               final Locker locker = BasicLocker.createBasicLocker(
319                   env, false /*noWait*/);
320  
321               try (final Cursor cursor =
322                    DbInternal.makeCursor(db, locker, null)) {
323  
324                   if (cursor.get( → [TRANSACTIONAL]
325                       key, null, Get.SEARCH,
326                       LockMode.RMW.toReadOptions()) != null) {
327  
328                       cursor.delete(null);
329                   }
330               } finally { → [TRANSACTIONAL] & [TRANSACTIONAL]
331                   locker.operationEnd();
332               }
333           }
334       }

336       /**
337        * Called after a file flip. The tracker is completed in the sense that the
338        * file is completely written, but there may be pending calls to
339        * {@link ExpirationTracker#track(int, boolean, int)} for some writing
340        * threads. This is because track is not called under the LWL.
341        */
342       public void addCompletedTracker(final ExpirationTracker tracker) {
343  
344           if (db == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
345               return;
346           }
347  
348           final long fileNum = tracker.getFileNum();
349  
350           synchronized (completedTrackers) {
351               assert !completedTrackers.containsKey(fileNum); → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
352  
353               completedTrackers.put(fileNum, tracker);
354           }
355       }
356  
357       /**
358        * Periodically, and when refreshing the profile, we process completed
359        * trackers that were added to the completedTrackers queue at the time of a
360        * file flip. If truly complete, they are added to the profile DB and map.
361        */
362       void processCompletedTrackers() {
363  
364           /* Only one thread at a time should process them. */
365           synchronized (map) {
366  
367               /*
368                * Make a copy in order to process them without holding their
369                * mutex, to avoid blocking addCompletedTracker, which is in the
370                * main write path.
371                */
372               final List<ExpirationTracker> trackers;
373  
374               synchronized (completedTrackers) {
375                   trackers = new ArrayList<>(completedTrackers.values());
376               }
377  
378               for (final ExpirationTracker tracker : trackers) {
379  
380                   if (tracker.hasPendingTrackCalls()) {
381                       /* Not quite completed. */
382                       continue;
383                   }
384  
385                   putFile(tracker, 0);
386  
387                   synchronized (completedTrackers) {
388                       completedTrackers.remove(tracker.getFileNum());
389                   }
390               }
391           }
392       }

394       /**
395        * Updates the expired bytes in the expiration profile according to the
396        * data that has expired at the given time. Should be called periodically,
397        * and before calling {@link #getExpiredBytes}.
398        *
399        * Also processes any completed trackers by adding them to the DB and to
400        * the histogram map.
401        *
402        * This method only does any real work once per hour, on hour boundaries,
403        * since data expires on (at most) hour boundaries.
404        */
405       public void refresh(final long time) {
406  
407           if (db == null) {
408               return;
409           }
410  
411           /* Synchronize to protect map and the lastRefreshXxx fields. */
412           synchronized (map) {
413  
414               /*
415                * Get last hour boundary, rounding down to the closest hour. If
416                * an hour has not passed (and this is not the first time called),
417                * then return and expect that we'll try again later.
418                */
419               final int hourLimit =
420                   (int) (time / TTL.MILLIS_PER_HOUR);
421  
422               if (hourLimit == lastRefreshHour) {
423                   return;
424               }
425  
426               final int dayLimit = hourLimit / 24;
427               final boolean newDayLimit = (dayLimit != lastRefreshDay);
428  
429               processCompletedTrackers();
430  
431               lastRefreshHour = hourLimit;
432               lastRefreshDay = dayLimit;
433               anyExpirationInHours = false;
434  
435               for (final ExpInfo info : map.values()) {
436                   if (ExpirationTracker.isExpirationInHours(
437                       info.serializedForm)) {
438                       anyExpirationInHours = true;
439                       break;
440                   }
441               }
442  
443               /*
444                * If all expiration times are on day boundaries, and we have not
445                * started a new day, there is nothing more to do.
446                */
447               if (!newDayLimit && !anyExpirationInHours) {
448                   return;
449               }
450  
451               /*
452                * Recalculate expired bytes for the current day/hour, saving the
453                * previous value.
454                */
455               for (final ExpInfo info : map.values()) {
456  
457                   info.previousExpiredBytes = info.currentExpiredBytes;
458  
459                   info.currentExpiredBytes = ExpirationTracker.getExpiredBytes(
460                       info.serializedForm, dayLimit, hourLimit);
461               }
462           }
463       }
464  
465       /**
466        * Returns the number of expired bytes for the given file. Uses the value
467        * calculated by the last call to {@link #refresh}.
468        */
469       public int getExpiredBytes(final long fileNum) {
470           synchronized (map) {
471               final ExpInfo info = map.get(fileNum);
472               return (info != null) ? info.currentExpiredBytes : 0;
473           }
474       }
475  
476       /**
477        * Returns the number of expired bytes for the given file. Two values are
478        * returned: the total expired at the given time, and a potentially smaller
479        * amount that gradually expires over the current hour or day interval.
480        * Uses the values calculated by the last call to {@link #refresh}.
481        *
482        * The amount that gradually expires is the amount that expires in the
483        * current time interval, which is one day if all data in the profile
484        * expires on day boundaries, and otherwise is one hour. If this is the
485        * first interval for the file (after a restart or a revisal run), all
486        * expired bytes are considered expired in the current interval and expire
487        * gradually.
488        *
489        * @return pair of {allExpiredBytes, gradualExpiredBytes}.
490        */
491       public Pair<Integer, Integer> getExpiredBytes(
492           final long fileNum,
493           final long time) {
494  
495           synchronized (map) {
496  
497               final ExpInfo info = map.get(fileNum);
498  
499               if (info == null) {
500                   return PAIR_OF_ZEROS;
501               }
502  
503               final int newlyExpiredBytes =
504                   info.currentExpiredBytes - info.previousExpiredBytes;
505  
506               if (newlyExpiredBytes == 0) {
507                   return new Pair<>(
508                       info.currentExpiredBytes, info.currentExpiredBytes);
509               }
510  
511               final long intervalMs = anyExpirationInHours ?
512                   TTL.MILLIS_PER_HOUR : TTL.MILLIS_PER_DAY;
513  
514               final long currentMs = time % intervalMs;
515  
516               final int gradualBytes = info.previousExpiredBytes +
517                   ((int) ((newlyExpiredBytes * currentMs) / intervalMs));
518  
519               return new Pair<>(info.currentExpiredBytes, gradualBytes);
520           }
521       }
522  
523       public String toString(final long fileNum) {
524           synchronized (map) {
525               final ExpInfo info = map.get(fileNum);
526               return (info != null) ? info.toString() : "NoExpInfo";
527           }
528       }

530       /**
531        * Contains cached information about expiration for a data file.
532        */
533       private static class ExpInfo {
534  
535           /**
536            * Cached serialized form, use to recompute the current expired bytes
537            * for a given expiration time.
538            */
539           final byte[] serializedForm;
540  
541           /**
542            * The number of expired bytes for the given file. This is the value
543            * calculated by the last call to {@link ExpirationProfile#refresh}.
544            */
545           int currentExpiredBytes = 0;
546  
547           /**
548            * The number of bytes that expired prior to the current interval.
549            * Calculated by the previous refresh for which the interval changed.
550            */
551           int previousExpiredBytes = 0;
552  
553           ExpInfo(final byte[] serializedForm,
554                   final int currentExpiredBytes) {
555  
556               this.serializedForm = serializedForm;
557               this.currentExpiredBytes = currentExpiredBytes;
558           }
559  
560           @Override
561           public String toString() {
562               return "{ExpInfo currentBytes = " + currentExpiredBytes +
563                   " " + ExpirationTracker.toString(serializedForm) + '}';
564           }
565       }
