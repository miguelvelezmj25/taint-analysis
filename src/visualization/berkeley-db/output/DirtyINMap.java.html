42  /**
43    * Manages the by-level map of checkpoint references that are to be flushed by
44    * a checkpoint or Database.sync, the MapLNs to be flushed, the highest level
45    * by database to be flushed, and the state of the checkpoint.
46    *
47    * An single instance of this class is used for checkpoints and has the same
48    * lifetime as the checkpointer and environment. An instance per Database.sync
49    * is created as needed. Only one checkpoint can occur at a time, but multiple
50    * syncs may occur concurrently with each other and with the checkpoint.
51    *
52    * The methods in this class are synchronized to protect internal state from
53    * concurrent access by the checkpointer and eviction, and to coordinate state
54    * changes between the two. Eviction must participate in the checkpoint so
55    * that INs cascade up properly; see coordinateEvictionWithCheckpoint.
56    *
57    * When INs are latched along with synchronization on a DirtyINMap, the order
58    * must be: 1) IN latches and 2) synchronize on DirtyINMap. For example,
59    * the evictor latches the parent and child IN before calling the synchronized
60    * method coordinateEvictionWithCheckpoint, and selectDirtyINsForCheckpoint
61    * latches the IN before calling the synchronized method selectForCheckpoint.
62    */
63  class DirtyINMap {

97       /**
98        * Coordinates an eviction with an in-progress checkpoint and returns
99        * whether or not provisional logging is needed.
100        *
101        * @return the provisional status to use for logging the target.
102        */
103       synchronized Provisional coordinateEvictionWithCheckpoint(
104           final DatabaseImpl db,
105           final int targetLevel,
106           final IN parent) {
107  
108           /*
109            * If the checkpoint is in-progress and has not finished dirty map
110            * construction, we must add the parent to the dirty map. That way the
111            * dirtiness and logging will cascade up in the same way as if the
112            * target were not evicted, and instead were encountered during dirty
113            * map construction. We don't want the evictor's actions to introduce
114            * an IN in the log that has not cascaded up properly.
115            *
116            * Note that we add the parent even if it is not dirty here. It will
117            * become dirty after the target child is logged, but that hasn't
118            * happened yet.
119            *
120            * We do not add the parent if it is null, which is the case when the
121            * root is being evicted.
122            */
123           if (ckptState == CkptState.DIRTY_MAP_INCOMPLETE && → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
124               parent != null) {
125  
126               /* Add latched parent IN to dirty map. */
127               selectForCheckpoint(parent, -1 /*index*/);
128  
129               /* Save dirty/temp DBs for later. */
130               saveMapLNsToFlush(parent);
131           }
132  
133           /*
134            * The evictor has to log provisionally in three cases:
135            *
136            * 1 - The eviction target is part of a deferred write database.
137            */
138           if (db.isDeferredWriteMode()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
139               return Provisional.YES;
140           }
141  
142           /*
143            * 2 - The checkpoint is in-progress and has not finished dirty map
144            * construction, and the target is not the root. The parent IN has
145            * been added to the dirty map, so we know the child IN is at a
146            * level below the max flush level.
147            */
148           if (ckptState == CkptState.DIRTY_MAP_INCOMPLETE && → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, READONLY]
149               parent != null) {
150               return Provisional.YES;
151           }
152  
153           /*
154            * 3 - The checkpoint is in-progress and has finished dirty map
155            * construction, and is at a level above the eviction target.
156            */
157           if (ckptState == CkptState.DIRTY_MAP_COMPLETE && → [ALLOWCREATE, READONLY]
158               targetLevel < getHighestFlushLevel(db)) { → [ALLOWCREATE, READONLY]
159               return Provisional.YES;
160           }
161  
162           /* Otherwise, log non-provisionally. */
163           return Provisional.NO;
164       }

166       /**
167        * Coordinates a split with an in-progress checkpoint.
168        *
169        * TODO:
170        * Is it necessary to perform MapLN flushing for nodes logged by a split
171        * (and not just the new sibling)?
172        *
173        * @param newSibling the sibling IN created by the split.
174        */
175       void coordinateSplitWithCheckpoint(final IN newSibling) {
176  
177           assert newSibling.isLatchExclusiveOwner();
178  
179           /*
180            * If the checkpoint is in-progress and has not finished dirty map
181            * construction, we must add the BIN children of the new sibling to the
182            * dirty map. The new sibling will be added to the INList but it may or
183            * may not be seen by the in-progress INList iteration, and we must
184            * ensure that its dirty BIN children are logged by the checkpoint.
185            *
186            * Note that we cannot synchronize on 'this' before calling
187            * selectDirtyBINChildrenForCheckpoint, since it latches BIN children.
188            * IN latching must come before synchronization on 'this'. Eventually
189            * after latching the IN, selectForCheckpoint is called , which is
190            * synchronized and checks for ckptState == DIRTY_MAP_INCOMPLETE.
191            */
192           selectDirtyBINChildrenForCheckpoint(newSibling);
193       }

224       /**
225        * Scan the INList for all dirty INs, excluding temp DB INs. Save them in
226        * a tree-level ordered map for level ordered flushing.
227        *
228        * Take this opportunity to recalculate the memory budget tree usage.
229        *
230        * This method itself is not synchronized to allow concurrent eviction.
231        * Synchronization is performed on a per-IN basis to protect the data
232        * structures here, and eviction can occur in between INs.
233        */
234       void selectDirtyINsForCheckpoint() {
235           assert ckptState == CkptState.DIRTY_MAP_INCOMPLETE;
236  
237           /*
238            * Opportunistically recalculate the INList memory budget while
239            * traversing the entire INList.
240            */
241           final INList inMemINs = envImpl.getInMemoryINs();
242           inMemINs.memRecalcBegin();
243  
244           boolean completed = false;
245           try {
246               for (IN in : inMemINs) {
247                   in.latchShared(CacheMode.UNCHANGED);
248                   try {
249                       if (!in.getInListResident()) {
250                           continue;
251                       }
252  
253                       inMemINs.memRecalcIterate(in);
254  
255                       /* Add dirty UIN to dirty map. */
256                       if (in.getDirty() && !in.isBIN()) {
257                           selectForCheckpoint(in, -1 /*index*/);
258                       }
259  
260                       /* Add dirty level 2 children to dirty map. */
261                       selectDirtyBINChildrenForCheckpoint(in);
262  
263                       /* Save dirty/temp DBs for later. */
264                       saveMapLNsToFlush(in);
265                   } finally {
266                       in.releaseLatch();
267                   }
268  
269                   /* Call test hook after releasing latch. */
270                   TestHookExecute.doHookIfSet(
271                       Checkpointer.examineINForCheckpointHook, in);
272               }
273               completed = true;
274           } finally {
275               inMemINs.memRecalcEnd(completed);
276           }
277  
278           /*
279            * Finish filling out the highestFlushLevels map. For each entry in
280            * highestFlushLevels that has a null level Integer value (set by
281            * selectForCheckpoint), we call DbTree.getHighestLevel and replace the
282            * null level. We must call DbTree.getHighestLevel, which latches the
283            * root, only when not synchronized, to avoid breaking the
284            * synchronization rules described in the class comment. This must be
285            * done in several steps to follow the sychronization rules, yet
286            * protect the highestFlushLevels using synchronization.
287            */
288           final Map<DatabaseImpl, Integer> maxFlushDbs = new HashMap<>();
289  
290           /* Copy entries with a null level. */
291           synchronized (this) {
292               for (DatabaseImpl db : highestFlushLevels.keySet()) {
293  
294                   if (highestFlushLevels.get(db) == null) {
295                       maxFlushDbs.put(db, null);
296                   }
297               }
298           }
299  
300           /* Call getHighestLevel without synchronization. */
301           final DbTree dbTree = envImpl.getDbTree();
302  
303           for (Map.Entry<DatabaseImpl, Integer> entry : maxFlushDbs.entrySet()) {
304  
305               entry.setValue(dbTree.getHighestLevel(entry.getKey()));
306           }
307  
308           /* Fill in levels in highestFlushLevels. */
309           synchronized (this) {
310  
311               for (Map.Entry<DatabaseImpl, Integer> entry :
312                    maxFlushDbs.entrySet()) {
313  
314                   highestFlushLevels.put(entry.getKey(), entry.getValue());
315               }
316           }
317  
318           /* Complete this phase of the checkpoint. */
319           synchronized (this) {
320               addCostToMemoryBudget();
321               ckptState = CkptState.DIRTY_MAP_COMPLETE;
322           }
323  
324           if (DIRTY_SET_DEBUG_TRACE) {
325               traceDirtySet();
326           }
327       }
328  
329       /**
330        * Add the IN to the dirty map if dirty map construction is in progress and
331        * the IN is not in a temp DB. If added, the highest flush level map is
332        * also updated.
333        */
334       synchronized void selectForCheckpoint(final IN in, final int index) {
335  
336           /*
337            * Must check state while synchronized. The state may not be
338            * DIRTY_MAP_INCOMPLETE when called from eviction or a split.
339            */
340           if (ckptState != CkptState.DIRTY_MAP_INCOMPLETE) { → [ALLOWCREATE, READONLY]
341               return;
342           }
343  
344           final DatabaseImpl db = in.getDatabase();
345  
346           if (db.isTemporary()) { → [ALLOWCREATE, READONLY]
347               return;
348           }
349  
350           addIN(in, index,
351               true /*updateFlushLevels*/,
352               false /*updateMemoryBudget*/);
353       }
354  
355       /**
356        * Adds the the dirty child BINs of the 'in' if dirty map construction is
357        * in progress and the IN is not in a temp DB.
358        *
359        * Main cache resident BINs are added when their parent is encountered in
360        * the INList iteration, rather than when the BIN is encountered in the
361        * iteration. This is because a BIN can transition between main and
362        * off-heap caches during the construction of the dirty map. When a BIN is
363        * loaded from off-heap and added to the main cache, it is added to the
364        * INList at that time, and such a BIN may not be encountered in the
365        * iteration. (ConcurrentHashMap iteration only guarantees that nodes will
366        * be encountered if they are present when the iterator is created). So if
367        * we relied on encountering BINs in the iteration, some might be missed.
368        *
369        * Note that this method is not synchronized because it latches the BIN
370        * children. IN latching must come before synchronizing on 'this'. The
371        * selectForCheckpoint method, which is called after latching the BIN, is
372        * synchronized.
373        */
374       private void selectDirtyBINChildrenForCheckpoint(final IN in) {
375  
376           if (in.getNormalizedLevel() != 2) {
377               return;
378           }
379  
380           for (int i = 0; i < in.getNEntries(); i += 1) {
381  
382               final IN bin = (IN) in.getTarget(i);
383  
384               if (bin != null) {
385  
386                   /* When called via split a child may already be latched. */
387                   final boolean latchBinHere = !bin.isLatchOwner();
388  
389                   if (latchBinHere) {
390                       bin.latchShared(CacheMode.UNCHANGED);
391                   }
392  
393                   try {
394                       if (bin.getDirty()) {
395                           selectForCheckpoint(bin, -1);
396                       }
397                   } finally {
398                       if (latchBinHere) {
399                           bin.releaseLatch();
400                       }
401                   }
402               } else {
403                   if (in.isOffHeapBINDirty(i)) {
404                       selectForCheckpoint(in, i);
405                   }
406               }
407           }
408       }

410       private void updateFlushLevels(Integer level,
411                                      final DatabaseImpl db,
412                                      final boolean isBIN,
413                                      final boolean isRoot) {
414  
415           /*
416            * IN was added to the dirty map. Update the highest level seen
417            * for the database. Use one level higher when ckptFlushExtraLevel
418            * is set. When ckptFlushAll is set, use the maximum level for the
419            * database. Durable deferred-write databases must be synced, so
420            * also use the maximum level.
421            *
422            * Always flush at least one level above the bottom-most BIN level so
423            * that the BIN level is logged provisionally and the expense of
424            * processing BINs during recovery is avoided.
425            */
426           if (ckptFlushAll || db.isDurableDeferredWrite()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
427               if (!highestFlushLevels.containsKey(db)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
428  
429                   /*
430                    * Null is used as an indicator that getHighestLevel should be
431                    * called in selectDirtyINsForCheckpoint, when not
432                    * synchronized.
433                    */
434                   highestFlushLevels.put(db, null);
435               }
436           } else {
437               if ((ckptFlushExtraLevel || isBIN) && !isRoot) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
438                   /* Next level up in the same tree. */
439                   level += 1;
440               }
441  
442               final Integer highestLevelSeen = highestFlushLevels.get(db);
443  
444               if (highestLevelSeen == null || level > highestLevelSeen) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
445                   highestFlushLevels.put(db, level);
446               }
447           }
448       }

450       /**
451        * Scan the INList for all dirty INs for a given database. Arrange them in
452        * level sorted map for level ordered flushing.
453        *
454        * This method is not synchronized to allow concurrent eviction.
455        * Coordination between eviction and Database.sync is not required.
456        */
457       void selectDirtyINsForDbSync(DatabaseImpl dbImpl) {
458  
459           assert ckptState == CkptState.NONE; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
460  
461           final DatabaseId dbId = dbImpl.getId();
462  
463           for (IN in : envImpl.getInMemoryINs()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
464               if (in.getDatabaseId().equals(dbId)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
465                   in.latch(CacheMode.UNCHANGED);
466                   try {
467                       if (in.getInListResident() && in.getDirty()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
468                           addIN(
469                               in, -1 /*index*/,
470                               false /*updateFlushLevels*/,
471                               false /*updateMemoryBudget*/);
472                       }
473                   } finally {
474                       in.releaseLatch();
475                   }
476               }
477           }
478  
479           /*
480            * Create a single entry map that forces all levels of this DB to
481            * be flushed.
482            */
483           highestFlushLevels.put(
484               dbImpl, envImpl.getDbTree().getHighestLevel(dbImpl));
485  
486           /* Add the dirty map to the memory budget. */
487           addCostToMemoryBudget();
488       }

490       synchronized int getHighestFlushLevel(DatabaseImpl db) {
491  
492           assert ckptState != CkptState.DIRTY_MAP_INCOMPLETE; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
493  
494           /*
495            * This method is only called while flushing dirty nodes for a
496            * checkpoint or Database.sync, not for an eviction, so an entry for
497            * this database should normally exist. However, if the DB root (and
498            * DatabaseImpl) have been evicted since the highestFlushLevels was
499            * constructed, the new DatabaseImpl instance will not be present in
500            * the map. In this case, we do not need to checkpoint the IN and
501            * eviction should be non-provisional.
502            */
503           Integer val = highestFlushLevels.get(db);
504           return (val != null) ? val : IN.MIN_LEVEL; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
505       }

512           final MemoryBudget mb = envImpl.getMemoryBudget();

514               ((long) numEntries) * MemoryBudget.CHECKPOINT_REFERENCE_SIZE;
515           mb.updateAdminMemoryUsage(cost);

518       private synchronized void removeCostFromMemoryBudget() {
519           final MemoryBudget mb = envImpl.getMemoryBudget();

521               ((long) numEntries) * MemoryBudget.CHECKPOINT_REFERENCE_SIZE;
522           mb.updateAdminMemoryUsage(0 - cost);

525       /**
526        * Add a node unconditionally to the dirty map.
527        *
528        * @param in is the IN to add, or the parent of an off-heap IN to add when
529        * index >= 0.
530        *
531        * @param index is the index of the off-heap child to add, or -1 to add the
532        * 'in' itself.
533        *
534        * @param updateMemoryBudget if true then update the memory budget as the
535        * map is changed; if false then addCostToMemoryBudget must be called
536        * later.
537        */
538       synchronized void addIN(final IN in,
539                               final int index,
540                               final boolean updateFlushLevels,
541                               final boolean updateMemoryBudget) {
542           final Integer level;
543           final long lsn;
544           final long nodeId;
545           final boolean isRoot;
546           final byte[] idKey;
547           final boolean isBin;
548  
549           if (index >= 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
550               level = in.getLevel() - 1;
551               lsn = in.getLsn(index);
552               nodeId = -1;
553               isRoot = false;
554               idKey = in.getKey(index);
555               isBin = true;
556           } else {
557               level = in.getLevel();
558               lsn = in.getLastLoggedLsn();
559               nodeId = in.getNodeId();
560               isRoot = in.isRoot();
561               idKey = in.getIdentifierKey();
562               isBin = in.isBIN();
563           }
564  
565           final Map<Long, CheckpointReference> lsnMap;
566           final Map<Long, CheckpointReference> nodeMap;
567  
568           Pair<Map<Long, CheckpointReference>,
569                Map<Long, CheckpointReference>> pairOfMaps = levelMap.get(level);
570  
571           if (pairOfMaps != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
572               lsnMap = pairOfMaps.first();
573               nodeMap = pairOfMaps.second();
574           } else {
575               /*
576                * We use TreeMap rather than HashMap because HashMap.iterator() is
577                * a slow way of getting the first element (see removeNextNode).
578                */
579               lsnMap = new TreeMap<>();
580               nodeMap = new TreeMap<>();
581               pairOfMaps = new Pair<>(lsnMap, nodeMap);
582               levelMap.put(level, pairOfMaps);
