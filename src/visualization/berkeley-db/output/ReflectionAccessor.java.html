42       private ReflectionAccessor(Class type, Accessor superAccessor) {
43           this.type = type;
44           this.superAccessor = superAccessor;
45           try {
46               constructor = type.getDeclaredConstructor();
47           } catch (NoSuchMethodException e) {
48               throw DbCompat.unexpectedState(type.getName());
49           }
50           if (!Modifier.isPublic(type.getModifiers()) ||  → [ALLOWCREATE, TRANSACTIONAL, READONLY]
51               !Modifier.isPublic(constructor.getModifiers())) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
52               setAccessible(constructor, type.getName() + "()");
53           }
54       }

56       /**
57        * Creates an accessor for a complex type.
58        */
59       ReflectionAccessor(Catalog catalog,
60                          Class type,
61                          Accessor superAccessor,
62                          FieldInfo priKeyField,
63                          List<FieldInfo> secKeyFields,
64                          List<FieldInfo> nonKeyFields) {
65           this(type, superAccessor);
66           if (priKeyField != null) {
67               priKey = getField(catalog, priKeyField,
68                                 true /*isRequiredKeyField*/);
69           } else {
70               priKey = null;
71           }
72           if (secKeyFields.size() > 0) {
73               secKeys = getFields(catalog, secKeyFields,
74                                   false /*isRequiredKeyField*/);
75           } else {
76               secKeys = EMPTY_KEYS;
77           }
78           if (nonKeyFields.size() > 0) {
79               nonKeys = getFields(catalog, nonKeyFields,
80                                   false /*isRequiredKeyField*/);
81           } else {
82               nonKeys = EMPTY_KEYS;
83           }
84       }

86       /**
87        * Creates an accessor for a composite key type.
88        */
89       ReflectionAccessor(Catalog catalog,
90                          Class type,
91                          List<FieldInfo> fieldInfos) {
92           this(type, null);
93           priKey = null;
94           secKeys = EMPTY_KEYS;
95           nonKeys = getFields(catalog, fieldInfos, true /*isRequiredKeyField*/);
96       }

98       private FieldAccess[] getFields(Catalog catalog,
99                                       List<FieldInfo> fieldInfos,
100                                       boolean isRequiredKeyField) {
101           int index = 0;
102           FieldAccess[] fields = new FieldAccess[fieldInfos.size()];
103           for (FieldInfo info : fieldInfos) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
104               fields[index] = getField(catalog, info, isRequiredKeyField);
105               index += 1;
106           }
107           return fields;
108       }
109  
110       private FieldAccess getField(Catalog catalog,
111                                    FieldInfo fieldInfo,
112                                    boolean isRequiredKeyField) {
113           Field field;
114           try {
115               field = type.getDeclaredField(fieldInfo.getName());
116           } catch (NoSuchFieldException e) {
117               throw DbCompat.unexpectedException(e);
118           }
119           if (!Modifier.isPublic(type.getModifiers()) ||  → [ALLOWCREATE, TRANSACTIONAL, READONLY]
120               !Modifier.isPublic(field.getModifiers())) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
121               setAccessible(field, field.getName());
122           }
123           Class fieldCls = field.getType();
124           if (fieldCls.isPrimitive()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
125               assert SimpleCatalog.isSimpleType(fieldCls); → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
126               return new PrimitiveAccess
127                   (field, (SimpleFormat) catalog.getFormat(fieldCls.getName()));
128           } else if (isRequiredKeyField) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
129               Format format = catalog.getFormat(fieldInfo.getClassName());
130               assert format != null; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
131               return new KeyObjectAccess(field, format);
132           } else if (fieldCls == String.class) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
133               return new StringAccess(field);
134           } else {
135               return new ObjectAccess(field);
136           }
137       }

139       private void setAccessible(AccessibleObject object, String memberName) {
140           try {
141               object.setAccessible(true);
142           } catch (SecurityException e) {
143               throw new IllegalStateException
144                   ("Unable to access non-public member: " +
145                    type.getName() + '.' + memberName +
146                    ". Please configure the Java Security Manager setting: " +
147                    " ReflectPermission suppressAccessChecks", e);
148           }
149       }

213       public void writeSecKeyFields(Object o, EntityOutput output)
214           throws RefreshException {
215  
216           try {
217          
218               /*
219                * In JE 5.0, String is treated as primitive type, so String does
220                * not need to be registered. [#19247]
221                */
222               if (priKey != null && !priKey.isPrimitive && !priKey.isString) {
223                   output.registerPriKeyObject(priKey.field.get(o));
224               }
225               if (superAccessor != null) {
226                   superAccessor.writeSecKeyFields(o, output);
227               }
228               for (int i = 0; i < secKeys.length; i += 1) {
229                   secKeys[i].write(o, output);
230               }
231           } catch (IllegalAccessException e) {
232               throw DbCompat.unexpectedException(e);
233           }
234       }

236       public void readSecKeyFields(Object o,
237                                    EntityInput input,
238                                    int startField,
239                                    int endField,
240                                    int superLevel)
241           throws RefreshException {
242  
243           try {
244               if (priKey != null && !priKey.isPrimitive && !priKey.isString) {
245                   input.registerPriKeyObject(priKey.field.get(o));
246               } else if (priKey != null && priKey.isString) {
247                   input.registerPriStringKeyObject(priKey.field.get(o));
248               }
249               if (superLevel != 0 && superAccessor != null) {
250                   superAccessor.readSecKeyFields
251                       (o, input, startField, endField, superLevel - 1);
252               } else {
253                   if (superLevel > 0) {
254                       throw DbCompat.unexpectedState
255                           ("Superclass does not exist");
256                   }
257               }
258               if (superLevel <= 0) {
259                   for (int i = startField;
260                        i <= endField && i < secKeys.length;
261                        i += 1) {
262                       secKeys[i].read(o, input);
263                   }
264               }
265           } catch (IllegalAccessException e) {
266               throw DbCompat.unexpectedException(e);
267           }
268       }

338       public Object getField(Object o,
339                              int field,
340                              int superLevel,
341                              boolean isSecField) {
342           if (superLevel > 0) {
343               return superAccessor.getField
344                   (o, field, superLevel - 1, isSecField);
345           }
346           try {
347               Field fld =
348                   isSecField ? secKeys[field].field : nonKeys[field].field;
349               return fld.get(o);
350           } catch (IllegalAccessException e) {
351               throw DbCompat.unexpectedException(e);
352           }
353       }

355       public void setField(Object o,
356                            int field,
357                            int superLevel,
358                            boolean isSecField,
359                            Object value) {
360           if (superLevel > 0) {
361               superAccessor.setField
362                   (o, field, superLevel - 1, isSecField, value);
363               return;
364           }
365           try {
366               Field fld =
367                   isSecField ? secKeys[field].field : nonKeys[field].field;
368               fld.set(o, value);
369           } catch (IllegalAccessException e) {
370               throw DbCompat.unexpectedException(e);
371           }
372       }

388       /**
389        * Abstract base class for field access classes.
390        */
391       private static abstract class FieldAccess {
392  
393           Field field;
394           boolean isPrimitive;
395           boolean isString = false;
396  
397           FieldAccess(Field field) {
398               this.field = field;
399               isPrimitive = field.getType().isPrimitive();
400               isString =
401                   field.getType().getName().equals(String.class.getName());
402           }
403  
404           /**
405            * Writes a field.
406            */
407           abstract void write(Object o, EntityOutput out)
408               throws IllegalAccessException, RefreshException;
409  
410           /**
411            * Reads a field.
412            */
413           abstract void read(Object o, EntityInput in)
414               throws IllegalAccessException, RefreshException;
415  
416           /**
417            * Returns whether a field is null (for reference types) or zero (for
418            * primitive integer types). This implementation handles the reference
419            * types.
420            */
421           boolean isNullOrZero(Object o)
422               throws IllegalAccessException {
423  
424               return field.get(o) == null;
425           }
426       }

428       /**
429        * Access for fields with object types.
430        */
431       private static class ObjectAccess extends FieldAccess {
432  
433           ObjectAccess(Field field) {
434               super(field);
435           }
436  
437           @Override
438           void write(Object o, EntityOutput out)
439               throws IllegalAccessException, RefreshException {
440  
441               out.writeObject(field.get(o), null);
442           }
443  
444           @Override
445           void read(Object o, EntityInput in)
446               throws IllegalAccessException, RefreshException {
447  
448               field.set(o, in.readObject());
449           }
450       }
