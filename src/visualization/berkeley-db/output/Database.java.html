370       /**
371        * Mutate BIN-deltas to full BINs for the given DW DB. [#25999]
372        * <p>
373        * Any BIN-deltas in cache must be mutated to full BINs, since
374        * BIN-deltas are not allowed in DW mode. This can be expensive, and is
375        * only a workaround for the underlying problem that BIN-deltas are not
376        * supported in DW mode. The mutation is necessary when a db transitions
377        * from non-DW to DW mode. In that case there may be BIN-deltas in
378        * cache, even if the DB has not yet been opened since BIN-deltas may
379        * be placed in the Btree by recovery. This method is not optimized
380        * because ultimately BIN-deltas will be supported in DW mode or DW mode
381        * will be discontinued.
382        * <p>
383        * At the time this method is called, the DW flag on this object has
384        * already been set. Therefore, this workaround doesn't guarantee that
385        * internal operations on the db (e.g., compression or eviction) won't
386        * operation on a BIN-delta in a DW DB. Therefore, we cannot assume/assert
387        * that a cached BIN-delta does not exist in a DW DB. Such assertions were
388        * removed as part of this workaround.
389        * <p>
390        * Setting the DW flag before mutating the BINs is necessary to ensure
391        * that no new BIN-deltas appear in cache during this process. BINs loaded
392        * into cache for a DW db are mutated to full BINs in IN.postFetchInit.
393        *
394        * @see BIN#shouldLogDelta
395        */
396       private void mutateDeferredWriteBINDeltas(DatabaseImpl dbImpl) {
397  
398           final OffHeapCache ohCache = getEnv().getOffHeapCache();
399  
400           for (final IN in : getEnv().getInMemoryINs()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
401               if (in.getDatabase() != dbImpl) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
402                   continue;
403               }
404               in.latchNoUpdateLRU();
405               try {
406                   if (in.isBIN()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
407                       in.mutateToFullBIN(false);
408                       continue;
409                   }
410                   if (ohCache == null || → [ALLOWCREATE, TRANSACTIONAL, READONLY]
411                       in.getNormalizedLevel() != 2) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
412                       continue;
413                   }
414                   for (int i = 0; i < in.getNEntries(); i += 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
415                       if (in.getOffHeapBINId(i) < 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
416                           continue;
417                       }
418                       final IN child = in.loadIN(i, CacheMode.UNCHANGED);
419                       if (child == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
420                           continue;
421                       }
422                       child.latchNoUpdateLRU();
423                       try {
424                           child.mutateToFullBIN(false);
425                       } finally {
426                           child.releaseLatch();
427                       }
428                   }
429               } finally {
430                   in.releaseLatch();
431               }
432           }
433       }

451       /**
452        * Discards the database handle.
453        * <p>
454        * When closing the last open handle for a deferred-write database, any
455        * cached database information is flushed to disk as if {@link #sync} were
456        * called.
457        * <p>
458        * The database handle should not be closed while any other handle that
459        * refers to it is not yet closed; for example, database handles should not
460        * be closed while cursor handles into the database remain open, or
461        * transactions that include operations on the database have not yet been
462        * committed or aborted. Specifically, this includes {@link
463        * com.sleepycat.je.Cursor Cursor} and {@link com.sleepycat.je.Transaction
464        * Transaction} handles.
465        * <p>
466        * When multiple threads are using the {@link com.sleepycat.je.Database
467        * Database} handle concurrently, only a single thread may call this
468        * method.
469        * <p>
470        * When called on a database that is the primary database for a secondary
471        * index, the primary database should be closed only after all secondary
472        * indices which reference it have been closed.
473        * <p>
474        * The database handle may not be accessed again after this method is
475        * called, regardless of the method's success or failure, with one
476        * exception: the {@code close} method itself may be called any number of
477        * times.</p>
478        *
479        * <p>WARNING: To guard against memory leaks, the application should
480        * discard all references to the closed handle. While BDB makes an effort
481        * to discard references from closed objects to the allocated memory for an
482        * environment, this behavior is not guaranteed. The safe course of action
483        * for an application is to discard all references to closed BDB
484        * objects.</p>
485        *
486        * @see DatabaseConfig#setDeferredWrite DatabaseConfig.setDeferredWrite
487        *
488        * @throws EnvironmentFailureException if an unexpected, internal or
489        * environment-wide failure occurs.
490        *

492        * are still open.
493        */

563       private void closeInternal(
564           final boolean doSyncDw,
565           final boolean deleteTempDb,
566           final DbState newState,
567           final OperationFailureException preemptedException) {
568  
569           /*
570            * We acquire the SecondaryAssociationLatch exclusively because
571            * associations are changed by removeReferringAssociations, and
572            * operations using the associations should not run concurrently with
573            * close.
574            */
575           try {
576               final EnvironmentImpl envImpl = getEnv();
577               try {
578                   envImpl.getSecondaryAssociationLock().
579                       writeLock().lockInterruptibly();
580               } catch (InterruptedException e) {
581                   throw new ThreadInterruptedException(envImpl, e);
582               }
583               try {
584                   closeInternalWork(
585                       doSyncDw, deleteTempDb, newState, preemptedException);
586               } finally {
587                   envImpl.getSecondaryAssociationLock().writeLock().unlock();
588               }
589           } finally {
590               minimalClose(newState, preemptedException);
591           }
592       }

598        * null.
