34  /**
35    * Store and calculate elapsed time, counts, and other statistics about
36    * environment open. No synchronization is used, which generally works because
37    * with the exception of replication, environment startup is currently a
38    * serial, single threaded event. Replicated environments must be sure to
39    * record startup times only at thread safe points.
40    */
41  public class StartupTracker {
42  
43       /*
44        * Statistics are kept about startup phases, defined below. Phases can
45        * be nested, so the child and root fields are used to express this
46        * relationship. For example:
47        * TotalEnvOpen
48        * TotalRecovery
49        * FindEndOfLog
50        * ..
51        * BuildTree
52        * ReadMapIN
53        * ..
54        * Ckpt
55        * TotalJoinGroup encompasses the following two phases.
56        * FindMaster
57        * BecomeConsistent
58        * Keep these enums in order of execution, so that the display is easier to
59        * comprehend. Of course, some phases subsume other phases, but in general,
60        * this enum order follows the order of execution.
61        */

149       public StartupTracker(EnvironmentImpl envImpl) {
150  
151           elapsed = new EnumMap<Phase, Elapsed>(Phase.class);
152           counters = new EnumMap<Phase, Counter>(Phase.class);
153           stats = new EnumMap<Phase, StatGroup>(Phase.class);
154           for (Phase p : Phase.values()){
155               elapsed.put(p, new Elapsed());
156           }
157  
158           this.envImpl = envImpl;
159  
160           logger = LoggerUtils.getLogger(getClass());
161           lastDumpMillis = System.currentTimeMillis();
162       }

168       /**
169        * Note that a particular phase is starting.
170        */
171       public void start(Phase phase) {
172           String msg = "Starting " + phase;
173           if (info != null) {
174               msg += " " + info;
175           }
176           LoggerUtils.logMsg(logger, envImpl, Level.CONFIG, msg);
177  
178           elapsed.get(phase).start();
179           Counter c = new Counter();
180           counters.put(phase, c);
181           if (!phase.equals(Phase.TOTAL_ENV_OPEN)) {
182  
183               /*
184                * LogManager does not exist yet so we can't reference it. Anyway,
185                * cache misses are 0 to start with, so TOTAL_ENV_OPEN does not
186                * have to set the starting cache miss count.
187                */
188               c.setCacheMissStart(envImpl.getLogManager().getNCacheMiss());
189           }
190       }
191      
192       /**
193        * Note that a particular phase is ending.
194        */
195       public void stop(Phase phase) {
196           Elapsed e = elapsed.get(phase);
197           e.end();
198           Counter c = getCounter(phase);
199           c.setCacheMissEnd(envImpl.getLogManager().getNCacheMiss());
200  
201           /* Log this phase to the je.info file. */
202           String msg = "Stopping " + phase;
203           if (info != null) { → [ALLOWCREATE, TRANSACTIONAL]
204               msg += " " + info;
205           }
206           LoggerUtils.logMsg(logger, envImpl, Level.CONFIG, msg);
207  
208           /*
209            * Conditionally log the whole report to the je.info file, either
210            * because this family of phases has ended, or because this startup
211            * is taking a very long time.
212            *
213            * Take care to only check the value of dumpThreshold here, rather than
214            * setting it in the StartupTracker constructor, because StartupTracker
215            * is instantiated before the DbConfigManager and the
216            * STARTUP_DUMP_THRESHOLD param cannot be read.
217            */
218           int dumpThreshold = envImpl.getConfigManager().getDuration
219               (EnvironmentParams.STARTUP_DUMP_THRESHOLD);
220  
221           /* We're at the end of a family of phases. */
222           if (phase.root == null) { → [ALLOWCREATE, TRANSACTIONAL]
223               if ((e.getEnd() - e.getStart()) > dumpThreshold) { → [ALLOWCREATE, TRANSACTIONAL]
224                   ByteArrayOutputStream baos = new ByteArrayOutputStream();
225                   PrintStream p = new PrintStream(baos);
226                   displayStats(p, phase);
227                   LoggerUtils.logMsg(logger, envImpl, Level.INFO,
228                                      baos.toString());
229                   return;
230               }
231           }
232           
233           /*
234            * It's not the ending phase, but this has been taking a very long
235            * time, so dump some information.
236            */
237            if ((System.currentTimeMillis() - lastDumpMillis) > dumpThreshold) { → [ALLOWCREATE, TRANSACTIONAL]
238                ByteArrayOutputStream baos = new ByteArrayOutputStream();
239                PrintStream p = new PrintStream(baos);
240                displayInterim(p, phase);
241                LoggerUtils.logMsg(logger, envImpl, Level.INFO, baos.toString());
242            }
243       }

278       /**
279        * Generate a description of the four recovery locations (firstActive,
280        * ckptStart, ckptend, end of Log) and the distance inbetween.
281        */
282       private String displayRecoveryInterval() {
283           StringBuilder returnInfo = new StringBuilder();
284  
285           CheckpointEnd cEnd = info.checkpointEnd;
286           if (cEnd != null) { → [ALLOWCREATE]
287               returnInfo.append("checkpointId = ");
288               returnInfo.append(cEnd.getId());
289               if (cEnd.getInvoker() == null) { → [ALLOWCREATE]
290                   returnInfo.append(" ");
291               } else {
292                   returnInfo.append("[").append(cEnd.getInvoker());
293                   returnInfo.append("] ");
294               }
295           }
296  
297           long fileMax =
298               envImpl.getConfigManager().getLong(EnvironmentParams.LOG_FILE_MAX);
299  
300           long useStart = info.checkpointStartLsn == DbLsn.NULL_LSN ? → [ALLOWCREATE]
301               0 : info.checkpointStartLsn;
302           long head = DbLsn.getNoCleaningDistance(useStart,
303                                                   info.firstActiveLsn,
304                                                   fileMax);
305  
306           long useEnd = info.checkpointEndLsn == DbLsn.NULL_LSN ?
307               0 : info.checkpointEndLsn;
308           long ckpt = DbLsn.getNoCleaningDistance(useEnd,
309                                                   info.checkpointStartLsn,
310                                                   fileMax);
311  
312           long useLast = info.lastUsedLsn == DbLsn.NULL_LSN ?
313               0 : info.lastUsedLsn;
314           long tail = DbLsn.getNoCleaningDistance(useLast,
315                                                   info.checkpointEndLsn,
316                                                   fileMax);
317           returnInfo.append(
318               "firstActive[" +
319               DbLsn.getNoFormatString(info.firstActiveLsn) +
320               "], ckptStart[" +
321               DbLsn.getNoFormatString(info.checkpointStartLsn) +
322               "], ckptEnd[" +
323               DbLsn.getNoFormatString(info.checkpointEndLsn) +
324               "], lastUsed[" +
325               DbLsn.getNoFormatString(info.lastUsedLsn) +
326               "]\n");
327           StringBuilder sb = new StringBuilder();
328           Formatter f = new Formatter(sb);
329           f.format("%24s bytes = %,d\n%24s bytes = %,d\n%24s bytes = %,d",
330                    "firstActive->ckptStart", head,
331                    "ckptStart->ckptEnd", ckpt,
332                    "ckptEnd->end bytes", tail);
333          
334           return returnInfo.toString() + "\nApproximate distances:\n" +
335               sb.toString();
336       }

346       /**
347        * Display a phase and its children, showing elapsed time as a
348        * percentage of the phases' root.
349        */
350       private void displayPhaseSubtree(PrintStream stream,
351                                        Phase parent,
352                                        Elapsed parentTime,
353                                        Elapsed rootElapsed) {
354  
355           String headerFormat = "%24s %% of total %s\n";
356           String parentFormat = "%20s %3d %s\n";
357           String dataFormat = "%24s %3d %s\n";
358           String divider = " "+
359               "-------------------------";
360  
361           if (parent.children == null) {
362               return;
363           }
364  
365           if ((parentTime.getEnd() - parentTime.getStart()) ==0) {
366               return;
367           }
368  
369           stream.println("\n");
370           stream.printf(headerFormat, " ", Elapsed.DISPLAY_COLUMNS);
371           stream.printf(parentFormat, parent,
372                         parentTime.getPercentage(rootElapsed), parentTime);
373           stream.println(divider);
374  
375           for (Phase child : parent.children) {
376               Elapsed time = elapsed.get(child);
377               if (time.getStart() == 0) {
378                   continue;
379               }
380               stream.printf(dataFormat,
381                             child,
382                             time.getPercentage(rootElapsed),
383                             time);
384           }
385       }

387       private void displayCounters(PrintStream stream, Phase root) {
388           String basicFormat = "%20s %s\n";
389           boolean headerNotPrinted = true;
390           for (Map.Entry<Phase, Counter> c : counters.entrySet()) { → [ALLOWCREATE]
391               Phase p = c.getKey();
392               if (p.root != root) {
393                   continue;
394               }
395               Counter counter = c.getValue();
396               if (counter.isEmpty()) {
397                   continue;
398               }
399  
400               if (headerNotPrinted) {
401                   stream.println();
402                   stream.printf(basicFormat, " " , Counter.DISPLAY_COLUMNS);
403                   headerNotPrinted = false;
404               }
405               stream.printf(basicFormat, c.getKey(), counter);
406           }
407       }

409       /**
410        * Display all information that has been tracked for this family of
411        * phases.
412        */
413       public void displayStats(PrintStream stream, Phase root ) {
414           lastDumpMillis = System.currentTimeMillis();
415           Elapsed rootTime = elapsed.get(root);
416  
417           stream.println("\n=== " + root.reportLabel + " Report ===");
418           stream.println("start = " + displayTimestamp(rootTime.getStart()));
419           stream.println("end = " + displayTimestamp(rootTime.getEnd()));
420           if (root == Phase.TOTAL_ENV_OPEN) {
421               stream.print(displayRecoveryInterval());
422           }
423  
424           /* Elapsed time. */
425           for (Map.Entry<Phase, Elapsed> x : elapsed.entrySet()) {
426               Phase p = x.getKey();
427               if (p.root == null) {
428                  if (p != root) {
429                      continue;
430                  }
431               } else if (p.root != root) {
432                  continue;
433               }
434              
435               displayPhaseSubtree(stream, x.getKey(),x.getValue(), rootTime);
436           }
437  
438           /* Counters */
439           displayCounters(stream, root);
440  
441           /* Stats */
442           for (Map.Entry<Phase, StatGroup> s : stats.entrySet()) {
443               Phase p = s.getKey();
444               if (p.root != root) {
445                   continue;
446               }
447               stream.println(s.getKey() + " stats:");
448               stream.println(s.getValue());
449           }
450       }

452       /**
453        * Display all information available so far.
454        */
455       private void displayInterim(PrintStream stream, Phase phase ) {
456           lastDumpMillis = System.currentTimeMillis();
457  
458           stream.println("\n=== Interim " + phase + " Report ===");
459  
460           stream.println(displayRecoveryInterval());
461  
462           /* Elapsed time. */
463           boolean headerNotPrinted = true;
464           for (Map.Entry<Phase, Elapsed> x : elapsed.entrySet()) { → [ALLOWCREATE]
465               Phase p = x.getKey();
466               Elapsed e = x.getValue();
467               if (e.start == 0) { → [ALLOWCREATE]
468                   continue;
469               }
470               if (headerNotPrinted) { → [ALLOWCREATE]
471                   stream.println(" Elapsed(ms)");
472                   headerNotPrinted = false;
473               }
474               stream.printf("%20s : %s\n", p, e);
475           }
476  
477           /* Counters */
478           displayCounters(stream, phase.root);
479  
480           /* Stats */
481           for (Map.Entry<Phase, StatGroup> s : stats.entrySet()) { → [ALLOWCREATE]
482               stream.println(s.getKey() + " stats:");
483               stream.println(s.getValue());
484           }
485       }

487       /** Measures elapsed time in millisecond granularity. */
488       static private class Elapsed {
489  
490           /* For dumping elapsed values in a column */
491           static String DISPLAY_COLUMNS = " Elapsed(ms)";
492  
493           private long start;
494           private long end;
495  
496           public long getStart() {
497               return start;
498           }
499  
500           public long getEnd() {
501               return end;
502           }
503  
504           /* Mark the start of a phase. */
505           private void start() {
506               start = System.currentTimeMillis();
507           }
508  
509           /* Mark the end of a phase. */
510           private void end() {
511               end = System.currentTimeMillis();
512           }
513  
514           private int getPercentage(Elapsed rootTime) {
515               if (rootTime == null) {
516                   return 0;
517               }
518  
519               long rootTotal = rootTime.end-rootTime.start;
520               if (rootTotal <= 0) {
521                   return 0;
522               }
523  
524               if (end == 0) {
525                   return 0;
526               }
527               return (int)(((float) (end-start)/ rootTotal) * 100);
528           }
529  
530           @Override
531           public String toString() {
532               StringBuilder sb = new StringBuilder();
533               Formatter f = new Formatter(sb);
534               if (end != 0) {
535                   f.format("%,13d", (end-start));
536               } else {
537                   if (start != 0) {
538                       f.format("%13s %tD,%tH:%tM:%tS:%tL",
539                                "started at", start, start, start, start, start);
540                   } else {
541                       f.format("%13s", "none");
542                   }
543               }
544               return sb.toString();
545           }
546       }

548       /**
549        * Record number of log entries processed during a given recovery phase.
550        */
551       static public class Counter {
552           private int numRead;
553           private int numProcessed;
554           private int numDeleted;
555           private int numAux;
556           private long numRepeatIteratorReads;
557           private long startCacheMiss;
558           private long endCacheMiss;
559  
560           /* If nothing is set, don't print this one. */
561           private boolean isEmpty() {
562               return((numRead==0) &&
563                      (numProcessed==0) &&
564                      (numDeleted==0) &&
565                      (numAux==0) &&
566                      (numRepeatIteratorReads==0) &&
567                      ((endCacheMiss-startCacheMiss)==0));
568           }
569  
570           public void incNumRead() {
571               numRead++;
572           }
573  
574           public void incNumProcessed() {
575               numProcessed++;
576           }
577  
578           public void incNumDeleted() {
579               numDeleted++;
580           }
581  
582           /**
583            * Keep track of auxiliary log entries processed during this pass.
584            * For example, LNs are the main target of the undoLN pass, but we
585            * also read aborts and commits.
586            */
587           public void incNumAux() {
588               numAux++;
589           }
590  
591           public void setRepeatIteratorReads(long repeats) {
592               numRepeatIteratorReads = repeats;

594
