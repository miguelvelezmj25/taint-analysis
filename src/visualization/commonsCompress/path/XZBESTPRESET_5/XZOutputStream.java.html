25    *

26    * Getting an output stream to compress with LZMA2 using the default

28    * 



29    * FileOutputStream outfile = new FileOutputStream("foo.xz");


30    * XZOutputStream outxz = new XZOutputStream(outfile, new LZMA2Options());

171            *


172            * @param filterOptions new filter chain to use


173            * @throws UnsupportedOptionsException unsupported filter chain, or trying to change


174            *                                     the filter chain in the middle of a Block


175            */


176          public void updateFilters(FilterOptions[] filterOptions)


177                          throws XZIOException {


178                  if(blockEncoder != null) {


179                          throw new UnsupportedOptionsException("Changing filter options "


180                                          + "in the middle of a XZ Block not implemented");


181                  }


182  


183                  if(filterOptions.length < 1 || filterOptions.length > 4) {


184                          throw new UnsupportedOptionsException(


185                                          "XZ filter chain must be 1-4 filters");


186                  }


187  


188                  filtersSupportFlushing = true;


189                  FilterEncoder[] newFilters = new FilterEncoder[filterOptions.length];


190                  for(int i = 0; i < filterOptions.length; ++i) {


191                          newFilters[i] = filterOptions[i].getFilterEncoder();  → 17 → 18


192                          filtersSupportFlushing &= newFilters[i].supportsFlushing();


193                  }

196                  filters = newFilters;

212          /**


213            * Writes an array of bytes to be compressed.


214            * The compressors tend to do internal buffering and thus the written


215            * data won't be readable from the compressed output immediately.


216            * Use flush() to force everything written so far to


217            * be written to the underlaying output stream, but be aware that


218            * flushing reduces compression ratio.


219            *


220            * @param buf buffer of bytes to be written


221            * @param off start offset in buf


222            * @param len number of bytes to write


223            * @throws XZIOException XZ Stream has grown too big: total file size


224            *                       about 8 EiB or the Index field exceeds


225            *                       16 GiB; you shouldn't reach these sizes


226            *                       in practice


227            * @throws XZIOException finish() or close()


228            *                       was already called and len > 0


229            * @throws IOException   may be thrown by the underlying output stream


230            */


231          public void write(byte[] buf, int off, int len) throws IOException {


232                  if(off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {


233                          throw new IndexOutOfBoundsException();


234                  }


235  


236                  if(exception != null) {


237                          throw exception;


238                  }


239  


240                  if(finished) {


241                          throw new XZIOException("Stream finished or closed");


242                  }


243  


244                  try {


245                          if(blockEncoder == null) {


246                                  blockEncoder = new BlockOutputStream(out, filters, check);


247                          }


248  


249                          blockEncoder.write(buf, off, len);


250                  } catch (IOException e) {


251                          exception = e;


252                          throw e;


253                  }


254          }

256          /**


257            * Finishes the current XZ Block (but not the whole XZ Stream).


258            * This doesn't flush the stream so it's possible that not all data will


259            * be decompressible from the output stream when this function returns.


260            * Call also flush() if flushing is wanted in addition to


261            * finishing the current XZ Block.


262            * 


263            * If there is no unfinished Block open, this function will do nothing.


264            * (No empty XZ Block will be created.)


265            * 


266            * This function can be useful, for example, to create


267            * random-accessible .xz files.


268            * 


269            * Starting a new XZ Block means that the encoder state is reset.


270            * Doing this very often will increase the size of the compressed


271            * file a lot (more than plain flush() would do).


272            *


273            * @throws XZIOException XZ Stream has grown too big


274            * @throws XZIOException stream finished or closed


275            * @throws IOException   may be thrown by the underlying output stream


276            */


277          public void endBlock() throws IOException {


278                  if(exception != null) {


279                          throw exception;


280                  }


281  


282                  if(finished) {


283                          throw new XZIOException("Stream finished or closed");


284                  }


285  


286                  // NOTE: Once there is threading with multiple Blocks, it's possible


287                  // that this function will be more like a barrier that returns


288                  // before the last Block has been finished.


289                  if(blockEncoder != null) {


290                          try {


291                                  blockEncoder.finish();


292                                  index.add(blockEncoder.getUnpaddedSize(),


293                                                  blockEncoder.getUncompressedSize());


294                                  blockEncoder = null;


295                          } catch (IOException e) {


296                                  exception = e;


297                                  throw e;


298                          }


299                  }


300          }

302          /**


303            * Flushes the encoder and calls out.flush().


304            * All buffered pending data will then be decompressible from


305            * the output stream.


306            * 


307            * Calling this function very often may increase the compressed


308            * file size a lot. The filter chain options may affect the size


309            * increase too. For example, with LZMA2 the HC4 match finder has


310            * smaller penalty with flushing than BT4.


311            * 


312            * Some filters don't support flushing. If the filter chain has


313            * such a filter, flush() will call endBlock()


314            * before flushing.


315            *


316            * @throws XZIOException XZ Stream has grown too big


317            * @throws XZIOException stream finished or closed


318            * @throws IOException   may be thrown by the underlying output stream


319            */


320          public void flush() throws IOException {


321                  if(exception != null) {


322                          throw exception;


323                  }


324  


325                  if(finished) {


326                          throw new XZIOException("Stream finished or closed");


327                  }


328  


329                  try {


330                          if(blockEncoder != null) {


331                                  if(filtersSupportFlushing) {


332                                          // This will eventually call out.flush() so


333                                          // no need to do it here again.


334                                          blockEncoder.flush();


335                                  }


336                                  else {


337                                          endBlock();


338                                          out.flush();


339                                  }


340                          }


341                          else {


342                                  out.flush();


343                          }


344                  } catch (IOException e) {


345                          exception = e;


346                          throw e;


347                  }


348          }

350          /**


351            * Finishes compression without closing the underlying stream.


352            * No more data can be written to this stream after finishing


353            * (calling write with an empty buffer is OK).


354            * 


355            * Repeated calls to finish() do nothing unless


356            * an exception was thrown by this stream earlier. In that case


357            * the same exception is thrown again.


358            * 


359            * After finishing, the stream may be closed normally with


360            * close(). If the stream will be closed anyway, there


361            * usually is no need to call finish() separately.


362            *


363            * @throws XZIOException XZ Stream has grown too big


364            * @throws IOException   may be thrown by the underlying output stream


365            */


366          public void finish() throws IOException {


367                  if(!finished) {


368                          // This checks for pending exceptions so we don't need to


369                          // worry about it here.


370                          endBlock();


371  


372                          try {


373                                  index.encode(out);


374                                  encodeStreamFooter();


375                          } catch (IOException e) {


376                                  exception = e;


377                                  throw e;


378                          }


379  


380                          // Set it to true only if everything goes fine. Setting it earlier


381                          // would cause repeated calls to finish() do nothing instead of


382                          // throwing an exception to indicate an earlier error.


383                          finished = true;


384                  }


385          }

387          /**


388            * Finishes compression and closes the underlying stream.


389            * The underlying stream out is closed even if finishing


390            * fails. If both finishing and closing fail, the exception thrown


391            * by finish() is thrown and the exception from the failed


392            * out.close() is lost.


393            *


394            * @throws XZIOException XZ Stream has grown too big


395            * @throws IOException   may be thrown by the underlying output stream


396            */


397          public void close() throws IOException {


398                  if(out != null) {


399                          // If finish() throws an exception, it stores the exception to


400                          // the variable "exception". So we can ignore the possible


401                          // exception here.


402                          try {


403                                  finish();


404                          } catch (IOException e) {


405                          }


406  


407                          try {


408                                  out.close();


409                          } catch (IOException e) {


410                                  // Remember the exception but only if there is no previous


411                                  // pending exception.


412                                  if(exception == null) {


413                                          exception = e;


414                                  }


415                          }


416  


417                          out = null;


418                  }


419  


420                  if(exception != null) {


421                          throw exception;


422                  }


423          }

440          private void encodeStreamFooter() throws IOException {


441                  byte[] buf = new byte[6];


442                  long backwardSize = index.getIndexSize() / 4 - 1;


443                  for(int i = 0; i < 4; ++i)


444                          buf[i] = (byte) (backwardSize >>> (i * 8));

446                  encodeStreamFlags(buf, 4);


447  


448                  EncoderUtil.writeCRC32(out, buf);


449                  out.write(buf);


450                  out.write(XZ.FOOTER_MAGIC);


451          }
