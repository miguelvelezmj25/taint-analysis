399     /**
400      * Subset description length: <br>
401      * S(t,k,p) = -k*log2(p)-(n-k)log2(1-p)
402      *
403      * Details see Quilan: "MDL and categorical theories (Continued)",ML95
404      *
405      * @param t the number of elements in a known set
406      * @param k the number of elements in a subset
407      * @param p the expected proportion of subset known by recipient
408      * @return the subset description length
409      */
410     public static double subsetDL(double t, double k, double p) {
411       double rt = Utils.gr(p, 0.0) ? (-k * Utils.log2(p)) : 0.0; → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
412       rt -= (t - k) * Utils.log2(1 - p);

416     /**
417      * The description length of the theory for a given rule. Computed as:<br>
418      * 0.5* [||k||+ S(t, k, k/t)]<br>
419      * where k is the number of antecedents of the rule; t is the total possible
420      * antecedents that could appear in a rule; ||K|| is the universal prior for k
421      * , log2*(k) and S(t,k,p) = -k*log2(p)-(n-k)log2(1-p) is the subset encoding
422      * length.
423      * <p>
424      *
425      * Details see Quilan: "MDL and categorical theories (Continued)",ML95
426      *
427      * @param index the index of the given rule (assuming correct)
428      * @return the theory DL, weighted if weight != 1.0
429      */
430     public double theoryDL(int index) {
431  
432       double k = m_Ruleset.get(index).size();
433  
434       if (k == 0) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
435         return 0.0;
436       }
437  
438       double tdl = Utils.log2(k);
439       if (k > 1) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
440         tdl += 2.0 * Utils.log2(tdl); // of log2 star
441       }
442       tdl += subsetDL(m_Total, k, k / m_Total);
443       // System.out.println("!!!theory: "+MDL_THEORY_WEIGHT * REDUNDANCY_FACTOR *
444       // tdl);
445       return MDL_THEORY_WEIGHT * REDUNDANCY_FACTOR * tdl;
446     }

448     /**
449      * The description length of data given the parameters of the data based on
450      * the ruleset.
451      * <p>
452      * Details see Quinlan: "MDL and categorical theories (Continued)",ML95
453      * <p>
454      *
455      * @param expFPOverErr expected FP/(FP+FN)
456      * @param cover coverage
457      * @param uncover uncoverage
458      * @param fp False Positive
459      * @param fn False Negative
460      * @return the description length
461      */
462     public static double dataDL(double expFPOverErr, double cover,
463       double uncover, double fp, double fn) {
464       double totalBits = Utils.log2(cover + uncover + 1.0); // how many data?
465       double coverBits, uncoverBits; // What's the error?
466       double expErr; // Expected FP or FN
467  
468       if (Utils.gr(cover, uncover)) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
469         expErr = expFPOverErr * (fp + fn);
470         coverBits = subsetDL(cover, fp, expErr / cover);
471         uncoverBits = Utils.gr(uncover, 0.0) ? subsetDL(uncover, fn, fn / uncover) → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
472           : 0.0;
473       } else {
474         expErr = (1.0 - expFPOverErr) * (fp + fn);
475         coverBits = Utils.gr(cover, 0.0) ? subsetDL(cover, fp, fp / cover) : 0.0; → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
476         uncoverBits = subsetDL(uncover, fn, expErr / uncover);
477       }
478  
479       /*
480        * System.err.println("!!!cover: " + cover + "|uncover" + uncover +
481        * "|coverBits: "+coverBits+"|uncBits: "+ uncoverBits+
482        * "|FPRate: "+expFPOverErr + "|expErr: "+expErr+
483        * "|fp: "+fp+"|fn: "+fn+"|total: "+totalBits);
484        */
485       return (totalBits + coverBits + uncoverBits);
486     }

488     /**
489      * Calculate the potential to decrease DL of the ruleset, i.e. the possible DL
490      * that could be decreased by deleting the rule whose index and simple
491      * statstics are given. If there's no potentials (i.e. smOrEq 0 && error rate
492      * < 0.5), it returns NaN.
493      * <p>
494      *
495      * The way this procedure does is copied from original RIPPER implementation
496      * and is quite bizzare because it does not update the following rules' stats
497      * recursively any more when testing each rule, which means it assumes after
498      * deletion no data covered by the following rules (or regards the deleted
499      * rule as the last rule). Reasonable assumption?
500      * <p>
501      *

503      * @param expFPOverErr expected FP/(FP+FN)

505      * rule should be deleted
506      * @param ruleStat the simple statistics of the rule to be deleted
507      * @param checkErr whether check if error rate >= 0.5
508      * @return the potential DL that could be decreased
509      */
510     public double potential(int index, double expFPOverErr, double[] rulesetStat,
511       double[] ruleStat, boolean checkErr) {
512       // System.out.println("!!!inside potential: ");
513       // Restore the stats if deleted
514       double pcov = rulesetStat[0] - ruleStat[0];
515       double puncov = rulesetStat[1] + ruleStat[0];
516       double pfp = rulesetStat[4] - ruleStat[4];
517       double pfn = rulesetStat[5] + ruleStat[2];
518  
519       double dataDLWith = dataDL(expFPOverErr, rulesetStat[0], rulesetStat[1],
520         rulesetStat[4], rulesetStat[5]);
