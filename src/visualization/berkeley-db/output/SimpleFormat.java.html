27  /**
28    * Format for simple types, including primitives. Additional methods are
29    * included to optimize the handling of primitives. Other classes such as
30    * PrimitiveArrayFormat and ReflectAccessor take advantage of these methods.
31    *
32    * @author Mark Hayes
33    */
34  public abstract class SimpleFormat extends Format {
35  
36       private static final long serialVersionUID = 4595245575868697702L;
37  
38       private final boolean primitive;
39       private SimpleFormat wrapperFormat;
40  
41       SimpleFormat(Catalog catalog, Class type, boolean primitive) {
42           super(catalog, type);
43           this.primitive = primitive;
44       }
45  
46       void setWrapperFormat(SimpleFormat wrapperFormat) {
47           this.wrapperFormat = wrapperFormat;
48       }
49  
50       @Override
51       Format getWrapperFormat() {
52           return wrapperFormat;
53       }
54  
55       @Override
56       public boolean isSimple() {
57           return true;
58       }
59  
60       @Override
61       public boolean isPrimitive() {
62           return primitive;
63       }
64  
65       @Override
66       void collectRelatedFormats(Catalog catalog,
67                                  Map<String, Format> newFormats) {
68       }
69  
70       @Override
71       void initialize(Catalog catalog, EntityModel model, int initVersion) {
72       }
73  
74       @Override
75       public Object readObject(Object o, EntityInput input, boolean rawAccess) {
76           /* newInstance reads the value -- do nothing here. */
77           return o;
78       }
79  
80       @Override
81       boolean evolve(Format newFormat, Evolver evolver) {
82           evolver.useOldFormat(this, newFormat);
83           return true;
84       }
85  
86       /* -- Begin methods to be overridden by primitive formats only. -- */
87  
88       Object newPrimitiveArray(int len, EntityInput input)
89           throws RefreshException {
90  
91           throw DbCompat.unexpectedState();
92       }
93  
94       void writePrimitiveArray(Object o, EntityOutput output) {
95           throw DbCompat.unexpectedState();
96       }
97  
98       int getPrimitiveLength() {
99           throw DbCompat.unexpectedState();
100       }
101  
102       /**
103        * @throws IllegalAccessException from subclasses.
104        */
105       void readPrimitiveField(Object o, EntityInput input, Field field)
106           throws IllegalAccessException, RefreshException {
107  
108           throw DbCompat.unexpectedState();
109       }
110  
111       /**
112        * @throws IllegalAccessException from subclasses.
113        */
114       void writePrimitiveField(Object o, EntityOutput output, Field field)
115           throws IllegalAccessException {
116  
117           throw DbCompat.unexpectedState();
118       }
119  
120       /* -- End methods to be overridden by primitive formats only. -- */
121  
122       void skipPrimitiveArray(int len, RecordInput input) {
123           input.skipFast(len * getPrimitiveLength());
124       }
125  
126       void copySecMultiKeyPrimitiveArray(int len,
127                                          RecordInput input,
128                                          Set results) {
129           int primLen = getPrimitiveLength();
130           for (int i = 0; i < len; i += 1) {
131               DatabaseEntry entry = new DatabaseEntry
132                   (input.getBufferBytes(), input.getBufferOffset(), primLen);
133               results.add(entry);
134               input.skipFast(primLen);
135           }
136       }
