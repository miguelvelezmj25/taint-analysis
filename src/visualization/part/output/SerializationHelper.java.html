28  /**
29    * A helper class for determining serialVersionUIDs and checking whether classes
30    * contain one and/or need one. One can also serialize and deserialize objects
31    * to and from files or streams.
32    *
33    * @author fracpete (fracpete at waikato dot ac dot nz)
34    * @version $Revision: 13874 $
35    */
36  public class SerializationHelper implements RevisionHandler {
37  
38       /**
39        * the field name of serialVersionUID.
40        */
41       public final static String SERIAL_VERSION_UID = "serialVersionUID";
42  
43       /**
44        * checks whether a class is serializable.
45        *
46        * @param classname the class to check
47        * @return true if the class or one of its ancestors implements the
48        * Serializable interface, otherwise false (also if the class cannot
49        * be loaded)
50        */
51       public static boolean isSerializable(String classname) {
52           boolean result;
53  
54           try {
55               // result = isSerializable(Class.forName(classname));
56               result = isSerializable(WekaPackageClassLoaderManager.forName(classname));
57           } catch(Exception e) {
58               result = false;
59           }
60  
61           return result;
62       }
63  
64       /**
65        * checks whether a class is serializable.
66        *
67        * @param c the class to check
68        * @return true if the class or one of its ancestors implements the
69        * Serializable interface, otherwise false
70        */
71       public static boolean isSerializable(Class<?> c) {
72           return ClassDiscovery.hasInterface(Serializable.class, c);
73       }
74  
75       /**
76        * checks whether the given class contains a serialVersionUID.
77        *
78        * @param classname the class to check
79        * @return true if the class contains a serialVersionUID, otherwise false
80        * (also if the class is not implementing serializable or cannot be
81        * loaded)
82        */
83       public static boolean hasUID(String classname) {
84           boolean result;
85  
86           try {
87               // result = hasUID(Class.forName(classname));
88               result = hasUID(WekaPackageClassLoaderManager.forName(classname));
89           } catch(Exception e) {
90               result = false;
91           }
92  
93           return result;
94       }
95  
96       /**
97        * checks whether the given class contains a serialVersionUID.
98        *
99        * @param c the class to check
100        * @return true if the class contains a serialVersionUID, otherwise false
101        * (also if the class is not implementing serializable)
102        */
103       public static boolean hasUID(Class<?> c) {
104           boolean result;
105  
106           result = false;
107  
108           if(isSerializable(c)) {
109               try {
110                   c.getDeclaredField(SERIAL_VERSION_UID);
111                   result = true;
112               } catch(Exception e) {
113                   result = false;
114               }
115           }
116  
117           return result;
118       }
119  
120       /**
121        * checks whether a class needs to declare a serialVersionUID, i.e., it
122        * implements the java.io.Serializable interface but doesn't declare a
123        * serialVersionUID.
124        *
125        * @param classname the class to check
126        * @return true if the class needs to declare one, false otherwise (also if
127        * the class cannot be loaded!)
128        */
129       public static boolean needsUID(String classname) {
130           boolean result;
131  
132           try {
133               // result = needsUID(Class.forName(classname));
134               result = needsUID(WekaPackageClassLoaderManager.forName(classname));
135           } catch(Exception e) {
136               result = false;
137           }
138  
139           return result;
140       }
141  
142       /**
143        * checks whether a class needs to declare a serialVersionUID, i.e., it
144        * implements the java.io.Serializable interface but doesn't declare a
145        * serialVersionUID.
146        *
147        * @param c the class to check
148        * @return true if the class needs to declare one, false otherwise
149        */
150       public static boolean needsUID(Class<?> c) {
151           boolean result;
152  
153           if(isSerializable(c)) {
154               result = !hasUID(c);
155           }
156           else {
157               result = false;
158           }
159  
160           return result;
161       }

168        * loaded)
169        */
170       public static long getUID(String classname) {
171           long result;
172  
173           try {
174               // result = getUID(Class.forName(classname));
175               result = getUID(WekaPackageClassLoaderManager.forName(classname));
176           } catch(Exception e) {
177               result = 0L;
178           }
179  
180           return result;
181       }
182  
183       /**
184        * reads or creates the serialVersionUID for the given class.
185        *
186        * @param c the class to get the serialVersionUID for
187        * @return the UID, 0L for non-serializable classes

191       }

204       /**
205        * serializes the given object to the specified stream.
206        *
207        * @param stream the stream to write the object to
208        * @param o the object to serialize
209        * @throws Exception if serialization fails
210        */
211       public static void write(OutputStream stream, Object o) throws Exception {
212           ObjectOutputStream oos;
213  
214           if(!(stream instanceof BufferedOutputStream)) {
215               stream = new BufferedOutputStream(stream);
216           }
217  
218           oos = new ObjectOutputStream(stream);
219           oos.writeObject(o);
220           oos.flush();
221           oos.close();
222       }

235       /**
236        * serializes the given objects to the specified stream.
237        *
238        * @param stream the stream to write the object to
239        * @param o the objects to serialize
240        * @throws Exception if serialization fails
241        */
242       public static void writeAll(OutputStream stream, Object[] o) throws Exception {
243           ObjectOutputStream oos;
244           int i;
245  
246           if(!(stream instanceof BufferedOutputStream)) {
247               stream = new BufferedOutputStream(stream);
248           }
249  
250           oos = new ObjectOutputStream(stream);
251           for(i = 0; i < o.length; i++) {
252               oos.writeObject(o[i]);
253           }
254           oos.flush();
255           oos.close();
256       }

289        * for reading objects from the supplied input stream
290        *

292        * @return an {@code ObjectInputStream} instance that is aware of of Weka
293        * package classloaders
294        * @throws IOException if a problem occurs

296       public static ObjectInputStream getObjectInputStream(InputStream stream)
297               throws IOException {

299               stream = new BufferedInputStream(stream);

311                   // what we're going to look up!
312                   String arrayStripped =
313                           desc.getName().replace("[L", "").replace("[", "").replace(";", "");

322                   }
323  
324                   Class<?> result = null;
325                   try {
326                       result = Class.forName(desc.getName(), true, cl);
327                   } catch(ClassNotFoundException ex) {
328                       for(WekaPackageLibIsolatingClassLoader l : m_thirdPartyLoaders) {
329                           if(l.hasThirdPartyClass(arrayStripped)) {

333                   }

356       /**
357        * deserializes from the given stream and returns the object from it.
358        *
359        * @param stream the stream to deserialize from
360        * @return the deserialized object
361        * @throws Exception if deserialization fails
362        */
363       public static Object[] readAll(InputStream stream) throws Exception {
364           ObjectInputStream ois;
365           Vector<Object> result;
366  
367           ois = getObjectInputStream(stream);
368  
369           result = new Vector<Object>();
370           try {
371               while(true) {
372                   result.add(ois.readObject());
373               }
374           } catch(IOException e) {
375               // ignored
376           }
377           ois.close();
378  
379           return result.toArray(new Object[result.size()]);
380       }

382       /**
383        * Outputs information about a class on the commandline, takes class name as
384        * arguments.
385        *
386        * @param args the classnames to check
387        * @throws Exception if something goes wrong
388        */
389       public static void main(String[] args) throws Exception {
390           if(args.length == 0) {
391               System.out.println("\nUsage: " + SerializationHelper.class.getName()
392                       + " classname [classname [classname [...]]]\n");
393               System.exit(1);
394           }
395  
396           // check all the classes
397           System.out.println();
398           for(String arg : args) {
399               System.out.println(arg);
400               System.out.println("- is serializable: " + isSerializable(arg));
401               System.out.println("- has " + SERIAL_VERSION_UID + ": " + hasUID(arg));
402               System.out
403                       .println("- needs " + SERIAL_VERSION_UID + ": " + needsUID(arg));
404               System.out.println("- " + SERIAL_VERSION_UID
405                       + ": private static final long serialVersionUID = " + getUID(arg)
406                       + "L;");
407               System.out.println();
408           }
409       }
