36  /**
37    * Keeps track of the status of files for which cleaning is in progress.
38    */
39  public class FileSelector {
40  
41       /**
42        * Each file for which cleaning is in progress has one of the following
43        * status values. Files numbers migrate from one status to another, in
44        * the order declared below.
45        */
46       private enum FileStatus {
47  
48           /**
49            * A file's status is initially TO_BE_CLEANED when it is selected as
50            * part of a batch of files that, when deleted, will bring total
51            * utilization down to the minimum configured value. All files with
52            * this status will be cleaned in lowest-cost-to-clean order. For two
53            * files of equal cost to clean, the lower numbered (oldest) files is
54            * selected; this is why the fileInfoMap is sorted by key (file
55            * number).
56            */
57           TO_BE_CLEANED,
58  
59           /**
60            * When a TO_BE_CLEANED file is selected for processing by
61            * FileProcessor, it is moved to the BEING_CLEANED status. This
62            * distinction is used to prevent a file from being processed by more
63            * than one thread.
64            */
65           BEING_CLEANED,
66  
67           /**
68            * A file is moved to the CLEANED status when all its log entries have
69            * been read and processed. However, entries needing migration will be
70            * marked with the BIN entry MIGRATE flag, entries that could not be
71            * locked will be in the pending LN set, and the DBs that were pending
72            * deletion will be in the pending DB set.
73            */
74           CLEANED,
75  
76           /**
77            * A file is moved to the CHECKPOINTED status at the end of a
78            * checkpoint if it was CLEANED at the beginning of the checkpoint.
79            * Because all dirty BINs are flushed during the checkpoints, no files
80            * in this set will have entries with the MIGRATE flag set. However,
81            * some entries may be in the pending LN set and some DBs may be in the
82            * pending DB set.
83            */
84           CHECKPOINTED,
85  
86           /**
87            * A file is moved from the CHECKPOINTED status to the FULLY_PROCESSED
88            * status when the pending LN/DB sets become empty. Since a pending LN
89            * was not locked successfully, we don't know its original file. But
90            * we do know that when no pending LNs are present for any file, all
91            * log entries in CHECKPOINTED files are either obsolete or have been
92            * migrated. Note, however, that the parent BINs of the migrated
93            * entries may not have been logged yet.
94            *
95            * No special handling is required to coordinate syncing of deferred
96            * write databases for pending, deferred write LNs, because
97            * non-temporary deferred write DBs are always synced during
98            * checkpoints, and temporary deferred write DBs are not recovered.
99            * Note that although DW databases are non-txnal, their LNs may be
100            * pended because of lock collisions.
101            */
102           FULLY_PROCESSED,
103  
104           /**
105            * A file is moved to the SAFE_TO_DELETE status at the end of a
106            * checkpoint if it was FULLY_PROCESSED at the beginning of the
107            * checkpoint. All parent BINs of migrated entries have now been
108            * logged.
109            */
110           SAFE_TO_DELETE
111       }

127                      " firstVlsn = " + firstVlsn +

129           }
130       }

132       /**
133        * Information about files being cleaned, keyed by file number. The map is
134        * sorted by file number to clean older files before newer files.
135        */
136       private SortedMap<Long, FileInfo> fileInfoMap;
137  
138       /**
139        * Pending LN info, keyed by original LSN. These are LNs that could not be
140        * locked, either during processing or during migration.
141        */
142       private Map<Long, LNInfo> pendingLNs;
143  
144       /**
145        * For processed entries with DBs that are pending deletion, we consider
146        * them to be obsolete but we store their DatabaseIds in a set. Until the
147        * DB deletion is complete, we can't delete the log files containing those
148        * entries.
149        */
150       private Set<DatabaseId> pendingDBs;
151  
152       /**
153        * If during a checkpoint there are no pending LNs or DBs added, we can
154        * move CLEANED files to SAFE_TO_DELETE files at the end of the checkpoint.
155        * This is an optimization that allows deleting files more quickly when
156        * possible. In particular this impacts the checkpoint during environment
157        * close, since no user operations are active during that checkpoint; this
158        * optimization allows us to delete all cleaned files after the final
159        * checkpoint.
160        */
161       private boolean anyPendingDuringCheckpoint;

163       FileSelector() {

169       /**
170        * Returns the best file that qualifies for cleaning, or null if no file
171        * qualifies.
172        *
173        * @param forceCleaning is true to always select a file, even if its
174        * utilization is above the minimum utilization threshold.
175        *
176        * @return {file number, required utilization for 2-pass cleaning},
177        * or null if no file qualifies for cleaning.
178        */
179       synchronized Pair<Long, Integer> selectFileForCleaning(
180           UtilizationCalculator calculator,
181           SortedMap<Long, FileSummary> fileSummaryMap,
182           boolean forceCleaning) {
183  
184           final Set<Long> toBeCleaned = getToBeCleanedFiles();
185  
186           if (!toBeCleaned.isEmpty()) {
187               final Long fileNum = toBeCleaned.iterator().next();
188               final FileInfo info = setStatus(fileNum, FileStatus.BEING_CLEANED);
189               return new Pair<>(fileNum, info.requiredUtil);
190           }
191  
192           final Pair<Long, Integer> result = calculator.getBestFile(
193               fileSummaryMap, forceCleaning);
194  
195           if (result == null) {
196               return null;
197           }
198  
199           final Long fileNum = result.first();
200           final int requiredUtil = result.second();
201  
202           final FileInfo info = setStatus(fileNum, FileStatus.BEING_CLEANED);
203           info.requiredUtil = requiredUtil;
204  
205           return result;
206       }

275       /**

277        * assertion to avoid the overhead of the method call and synchronization.

395       /**
396        * When a checkpoint is complete, move the previously CLEANED and
397        * FULLY_PROCESSED files to the CHECKPOINTED and SAFE_TO_DELETE status.
398        */
399       synchronized void
400           updateFilesAtCheckpointEnd(CheckpointStartCleanerState info) {
401  
402           if (!info.isEmpty()) {
403  
404               Set<Long> previouslyCleanedFiles = info.getCleanedFiles();
405  
406               if (previouslyCleanedFiles != null) {
407  
408                   assert checkStatus(previouslyCleanedFiles, FileStatus.CLEANED);
409  
410                   setStatus(previouslyCleanedFiles, anyPendingDuringCheckpoint ?
411                                                     FileStatus.CHECKPOINTED :
412                                                     FileStatus.SAFE_TO_DELETE);
413               }
414  
415               Set<Long> previouslyProcessedFiles = info.getFullyProcessedFiles();
416  
417               if (previouslyProcessedFiles != null) {
418  
419                   assert checkStatus(previouslyProcessedFiles,
420                                      FileStatus.FULLY_PROCESSED);
421  
422                   setStatus(previouslyProcessedFiles, FileStatus.SAFE_TO_DELETE);
423               }
424  
425               updateProcessedFiles();
426           }
427       }

592       /**
593        * Adjust the memory budget when an entry is added to or removed from the
594        * cleanedFilesDatabases map.
595        */
596       private void adjustMemoryBudget(MemoryBudget budget,
597                                       Set<DatabaseId> oldDatabases,
598                                       Set<DatabaseId> newDatabases) {
599           long adjustMem = 0;
600           if (oldDatabases != null) {
601               adjustMem -= getCleanedFilesDatabaseEntrySize(oldDatabases);
602           }
603           if (newDatabases != null) {
604               adjustMem += getCleanedFilesDatabaseEntrySize(newDatabases);
605           }
606           budget.updateAdminMemoryUsage(adjustMem);
607       }

634       /**
635        * Holds copy of all checkpoint-dependent cleaner state.
636        */
637       public static class CheckpointStartCleanerState {
638  
639           /* A snapshot of the cleaner collections at the checkpoint start. */
640           private Set<Long> cleanedFiles;
