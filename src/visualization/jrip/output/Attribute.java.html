33  /**
34    * Class for handling an attribute. Once an attribute has been created, it can't
35    * be changed.
36    * <p>
37    *
38    * The following attribute types are supported:
39    * <ul>
40    * <li>numeric: <br/>
41    * This type of attribute represents a floating-point number.</li>
42    * <li>nominal: <br/>
43    * This type of attribute represents a fixed set of nominal values.</li>
44    * <li>string: <br/>
45    * This type of attribute represents a dynamically expanding set of nominal
46    * values. Usually used in text classification.</li>
47    * <li>date: <br/>
48    * This type of attribute represents a date, internally represented as
49    * floating-point number storing the milliseconds since January 1, 1970,
50    * 00:00:00 GMT. The string representation of the date must be <a
51    * href="http://www.iso.org/iso/en/prods-services/popstds/datesandtime.html"
52    * target="_blank"> ISO-8601</a> compliant, the default is
53    * <code>yyyy-MM-dd'T'HH:mm:ss</code>.</li>
54    * <li>relational: <br/>
55    * This type of attribute can contain other attributes and is, e.g., used for
56    * representing Multi-Instance data. (Multi-Instance data consists of a nominal
57    * attribute containing the bag-id, then a relational attribute with all the
58    * attributes of the bag, and finally the class attribute.)</li>
59    * </ul>
60    *
61    * Typical usage (code from the main() method of this class):
62    * <p>
63    *
64    * <code>
65    * ... <br>
66    *
67    * // Create numeric attributes "length" and "weight" <br>
68    * Attribute length = new Attribute("length"); <br>
69    * Attribute weight = new Attribute("weight"); <br><br>
70    *
71    * // Create list to hold nominal values "first", "second", "third" <br>
72    * List<String> my_nominal_values = new ArrayList<String>(3); <br>
73    * my_nominal_values.add("first"); <br>
74    * my_nominal_values.add("second"); <br>
75    * my_nominal_values.add("third"); <br><br>
76    *
77    * // Create nominal attribute "position" <br>
78    * Attribute position = new Attribute("position", my_nominal_values);<br>
79    *
80    * ... <br>
81    * </code>
82    * <p>
83    *
84    * @author Eibe Frank (eibe@cs.waikato.ac.nz)
85    * @version $Revision: 12891 $
86    */
