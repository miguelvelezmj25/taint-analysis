414        *

416        * key.

418        * <p>Specifying that only the primary key is needed can be used to
419        * optimize primary database updates and deletions. If a primary record is

426        *
427        * <p>Note that if this property is true, either null or a non-null value
428        * may be passed to the key extractor/creator for the primary data
429        * parameter. The key extractor/creator is expected to ignore this
430        * parameter.<p>
431        *

482        * <p>Be sure to set this property to true only if the secondary key in the
483        * primary record is never changed. If this rule is violated, the
484        * secondary index will become corrupted, that is, it will become out of
485        * sync with the primary.</p>

528       /**
529        * For JCA Database handle caching.
530        *
531        * @throws IllegalArgumentException via JEConnection.openSecondaryDatabase.
532        */
533       @Override
534       void validate(DatabaseConfig configArg)
535           throws DatabaseException {
536  
537           super.validate(configArg);
538  
539           if (configArg == null ||
540               !(configArg instanceof SecondaryConfig)) {
541               throw new IllegalArgumentException
542                   ("The SecondaryConfig argument is null.");
543           }
544  
545           SecondaryConfig config = (SecondaryConfig) configArg;
546  
547           boolean kcMatch = equalOrBothNull
548               (config.getKeyCreator(), keyCreator);
549           boolean mkcMatch = equalOrBothNull
550               (config.getMultiKeyCreator(), multiKeyCreator);
551           boolean fkdMatch =
552               (config.getForeignKeyDatabase() == foreignKeyDatabase);
553           boolean fkdaMatch =
554               (config.getForeignKeyDeleteAction() == foreignKeyDeleteAction);
555           boolean fknMatch = equalOrBothNull
556               (config.getForeignKeyNullifier(), foreignKeyNullifier);
557           boolean fmknMatch = equalOrBothNull
558               (config.getForeignMultiKeyNullifier(), foreignMultiKeyNullifier);
559           boolean imskMatch =
560               (config.getImmutableSecondaryKey() == immutableSecondaryKey);
561           if (kcMatch &&
562               mkcMatch &&
563               fkdMatch &&
564               fkdaMatch &&
565               fknMatch &&
566               fmknMatch &&
567               imskMatch) {
568               return;
569           }
570  
571           String message =
572               genSecondaryConfigMismatchMessage(
573                   config, kcMatch, mkcMatch, fkdMatch, fkdaMatch,
574                   fknMatch, fmknMatch, imskMatch);
575           throw new IllegalArgumentException(message);
576       }

604       private boolean equalOrBothNull(Object o1, Object o2) {
605           return (o1 != null) ? o1.equals(o2) : (o2 == null);
606       }
607  
608       String genSecondaryConfigMismatchMessage(DatabaseConfig config,
609                                                boolean kcMatch,
610                                                boolean mkcMatch,
611                                                boolean fkdMatch,
612                                                boolean fkdaMatch,
613                                                boolean fknMatch,
614                                                boolean fmknMatch,
615                                                boolean imskMatch) {
616           StringBuilder ret = new StringBuilder
617               ("The following SecondaryConfig parameters for the\n" +
618                "cached Database do not match the parameters for the\n" +
619                "requested Database:\n");
620           if (!kcMatch) {
621               ret.append(" SecondaryKeyCreator\n");
622           }
623  
624           if (!mkcMatch) {
625               ret.append(" SecondaryMultiKeyCreator\n");
626           }
627  
628           if (!fkdMatch) {
629               ret.append(" ForeignKeyDelete\n");
630           }
631  
632           if (!fkdaMatch) {
633               ret.append(" ForeignKeyDeleteAction\n");
634           }
635  
636           if (!fknMatch) {
637               ret.append(" ForeignKeyNullifier\n");
638           }
639  
640           if (!fknMatch) {
641               ret.append(" ForeignMultiKeyNullifier\n");
642           }
643  
644           if (!imskMatch) {
645               ret.append(" ImmutableSecondaryKey\n");
646           }
647  
648           return ret.toString();
649       }
