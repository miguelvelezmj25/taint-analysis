19  /**
20    * Defines how float numbers will be rounded
21    */
22  public class RoundingHandler {
23       public enum Strategy {
24           ROUND_HALF_UP,
25           CEIL,
26           FLOOR
27       }
28  
29       private final Strategy strategy;
30  
31       public RoundingHandler(Strategy strategy) {
32           this.strategy = strategy;
33       }
34  
35       public long round(double value) {
36           switch (strategy) { → [PLATFORM, POSTCONVERTWEBP, ENABLEPNGCRUSH, ENABLEMOZJPEG, SKIPUPSCALING, SCALE]
37               case CEIL:
38                   return Math.max(1, (long) Math.ceil(value));
39               case FLOOR:
40                   return Math.max(1, (long) Math.floor(value));
41               default:
42               case ROUND_HALF_UP:
43                   return Math.max(1, Math.round(value));
44           }
45       }
46  }
