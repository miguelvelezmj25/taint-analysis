48       /**
49        * @hidden
50        * For internal use only.
51        *
52        * Maps the existing sync settings to the equivalent durability settings.
53        * Figure out what we should do on commit. TransactionConfig could be
54        * set with conflicting values; take the most stringent ones first.
55        * All environment level defaults were applied by the caller.
56        *
57        * ConfigSync ConfigWriteNoSync ConfigNoSync default
58        * 0 0 0 sync
59        * 0 0 1 nosync
60        * 0 1 0 write nosync
61        * 0 1 1 write nosync
62        * 1 0 0 sync
63        * 1 0 1 sync
64        * 1 1 0 sync
65        * 1 1 1 sync
66        *
67        * @return the equivalent durability
68        */
69       public Durability getDurabilityFromSync(final EnvironmentImpl envImpl) {
70           if (sync) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
71               return Durability.COMMIT_SYNC;
72           } else if (writeNoSync) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
73               return Durability.COMMIT_WRITE_NO_SYNC;
74           } else if (noSync) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
75               return Durability.COMMIT_NO_SYNC;
76           }
77  
78           /*
79            * Replicated environments default to commitNoSync, while standalone
80            * default to commitSync.
81            */
82           if (envImpl.isReplicated()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
83               return Durability.COMMIT_NO_SYNC;
84           } else {
85               return Durability.COMMIT_SYNC;
86           }
87       }

534        * write to either replicated databases or non-replicated databases, but

544        * always false for a replicated database.</p>

547        * uses {@link com.sleepycat.je.Durability.ReplicaAckPolicy#NONE}.

549        * throw {@link com.sleepycat.je.rep.InsufficientReplicasException}, if the

551        * time the transaction is initiated.</p>
552        *
553        * <p>By default the local-write setting is false, meaning that the
554        * transaction may only write to replicated Databases in a replicated

558        * since no databases are replicated.</p>
559        *
560        * @return this

601       /**
602        *
603        * Checks to catch mixing of deprecated and non-deprecated forms of the
604        * API. It's invoked before setting any of the config parameters. The
605        * arguments represent the new state of the durability configuration,
606        * before it has been changed.
607        *
608        * @throws IllegalArgumentException via TransactionConfig and
609        * EnvironmentMutableConfig setters
610        */
611       static void checkMixedMode(final boolean sync,
612                                  final boolean noSync,
613                                  final boolean writeNoSync,
614                                  final Durability durability)
615           throws IllegalArgumentException {
616  
617           if ((sync || noSync || writeNoSync) && (durability != null)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
618               throw new IllegalArgumentException
619                   ("Mixed use of deprecated and current durability APIs is " +
620                    "not supported");
621           }
622  
623           if ((sync && noSync) || → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
624               (sync && writeNoSync) ||
625               (noSync && writeNoSync)) {
626               throw new IllegalArgumentException
627                   ("Only one of TxnSync, TxnNoSync, and TxnWriteNoSync " +
628                    "can be set.");
629           }
630       }
