25  public class ByteFunctionFactory {
26       // Up to 15 transforms can be declared (4 bit index)
27       public static final short NULL_TRANSFORM_TYPE = 0; // copy
28       public static final short BWT_TYPE = 1; // Burrows Wheeler
29       public static final short BWTS_TYPE = 2; // Burrows Wheeler Scott
30       public static final short LZ4_TYPE = 3; // LZ4
31       public static final short SNAPPY_TYPE = 4; // Snappy
32       public static final short RLT_TYPE = 5; // Run Length
33       public static final short ZRLT_TYPE = 6; // Zero Run Length
34       public static final short MTFT_TYPE = 7; // Move To Front
35       public static final short RANK_TYPE = 8; // Rank
36       public static final short TIMESTAMP_TYPE = 9; // TimeStamp
37       public static final short TEXTCODEC_TYPE = 10; // Text codec

39       private static ByteTransform newFunctionToken(int size, short functionType) {
40           switch (functionType & 0x0F) {
41               case SNAPPY_TYPE:
42                   return new SnappyCodec();
43  
44               case LZ4_TYPE:
45                   return new LZ4Codec();
46  
47               case BWT_TYPE:
48                   return new BWTBlockCodec();
49  
50               case BWTS_TYPE:
51                   return new BWTS();
52  
53               case MTFT_TYPE:
54                   return new MTFT();
55  
56               case ZRLT_TYPE:
57                   return new ZRLT();
58  
59               case RLT_TYPE:
60                   return new RLT();
61  
62               case RANK_TYPE:
63                   return new SBRT(SBRT.MODE_RANK);
64  
65               case TEXTCODEC_TYPE:
66                   // Select an appropriate initial dictionary size
67                   int dictSize = 1 << 12;
68  
69                   for(int i = 14; i <= 24; i += 2) {
70                       if(size >= 1 << i) {
71                           dictSize <<= 1;
72                       }
73                   }
74  
75                   return new TextCodec(dictSize);
76  
77               case TIMESTAMP_TYPE:
78                   return new SBRT(SBRT.MODE_TIMESTAMP);
79  
80               case NULL_TRANSFORM_TYPE:
81                   return new NullFunction();
82  
83               default:
84                   throw new IllegalArgumentException("Unknown transform type: " + functionType);
85           }
86       }

128       // The returned type contains 4 (nibble based) transform values
129       public short getType(String name) {
130           if(name.indexOf('+') < 0) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
131               return (short) (this.getTypeToken(name) << 12);
132           }
133  
134           String[] tokens = name.split("\\+");
135  
136           if(tokens.length == 0) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
137               throw new IllegalArgumentException("Unknown transform type: " + name);
138           }
139  
140           if(tokens.length > 4) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
141               throw new IllegalArgumentException("Only 4 transforms allowed: " + name);
142           }
143  
144           int res = 0;
145           int shift = 12;
146  
147           for(String token : tokens) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
148               final int typeTk = this.getTypeToken(token);
149  
150               // Skip null transform
151               if(typeTk != NULL_TRANSFORM_TYPE) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
152                   res |= (typeTk << shift);
153                   shift -= 4;
154               }
155           }
156  
157           return (short) res;
158       }

161           // Strings in switch not supported in JDK 6
162           name = name.toUpperCase();
163  
164           switch (name) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY] & [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
165               case "BWT":
166                   return BWT_TYPE;
167  
168               case "BWTS":

170  
171               case "SNAPPY":
172                   return SNAPPY_TYPE;
173  
174               case "LZ4":

176  
177               case "MTFT":

180               case "ZRLT":

203       public ByteTransformSequence newFunction(int size, short functionType) {
204           int nbtr = 0;
205  
206           // Several transforms
207           for(int i = 0; i < 4; i++) {
208               if(((functionType >>> (12 - 4 * i)) & 0x0F) != NULL_TRANSFORM_TYPE) {
209                   nbtr++;
210               }
211           }
212  
213           // Only null transforms ? Keep first.
214           if(nbtr == 0) {
215               nbtr = 1;
216           }
217  
218           ByteTransform[] transforms = new ByteTransform[nbtr];
219           nbtr = 0;
220  
221           for(int i = 0; i < transforms.length; i++) {
222               int t = (functionType >>> (12 - 4 * i)) & 0x0F;
223  
224               if((t != NULL_TRANSFORM_TYPE) || (i == 0)) {
225                   transforms[nbtr++] = newFunctionToken(size, (short) t);
226               }
227           }
228  
229           return new ByteTransformSequence(transforms);
230       }
231  
232       public String getName(short functionType) {
233           StringBuilder sb = new StringBuilder();
234  
235           for(int i = 0; i < 4; i++) { → [TRANSFORM, LEVEL]
236               int t = functionType >>> (12 - 4 * i);
237  
238               if((t & 0x0F) == NULL_TRANSFORM_TYPE) { → [TRANSFORM, LEVEL]
239                   continue;
240               }
241  
242               String name = getNameToken(t);
243  
244               if(sb.length() != 0) { → [TRANSFORM, LEVEL]
245                   sb.append('+');
246               }
247  
248               sb.append(name);
249           }
250  
251           if(sb.length() == 0) { → [TRANSFORM, LEVEL]
252               sb.append(getNameToken(NULL_TRANSFORM_TYPE));
253           }
254  
255           return sb.toString();
256       }
