14  package berkeley.com.sleepycat.persist.impl;
15  
16  import java.util.Map;
17  
18  /**
19    * Read-only catalog used by a PersistComparator to return simple formats plus
20    * reconstituted enum formats.
21    *
22    * @author Mark Hayes
23    */
24  class ComparatorCatalog extends SimpleCatalog {
25  
26       private final Map<String, Format> formatMap;
27  
28       ComparatorCatalog(final ClassLoader classLoader,
29                         final Map<String, Format> formatMap) {
30           super(classLoader);
31           this.formatMap = formatMap;
32       }
33  
34       public Format getFormat(final String className) {
35           if (formatMap != null) { → [ALLOWCREATE, READONLY]
36               final Format f = formatMap.get(className);
37               if (f != null) { → [ALLOWCREATE, READONLY]
38                   return f;
39               }
40           }
41           return super.getFormat(className);
42       }
43  }
