16  package kanzi.entropy;
17  
18  import kanzi.EntropyDecoder;
19  import kanzi.EntropyEncoder;
20  import kanzi.InputBitStream;
21  import kanzi.OutputBitStream;
22  
23  
24  public class EntropyCodecFactory {
25       public static final byte NONE_TYPE = 0; // No compression
26       public static final byte HUFFMAN_TYPE = 1; // Huffman
27       public static final byte FPAQ_TYPE = 2; // Fast PAQ (order 0)
28       public static final byte PAQ_TYPE = 3; // PAQ (stripped from many models for speed)
29       public static final byte RANGE_TYPE = 4; // Range
30       public static final byte ANS0_TYPE = 5; // Asymmetric Numerical System order 0
31       public static final byte CM_TYPE = 6; // Context Model
32       public static final byte TPAQ_TYPE = 7; // Tangelo PAQ
33       public static final byte ANS1_TYPE = 8; // Asymmetric Numerical System order 1
34  
35  
36       public EntropyDecoder newDecoder(InputBitStream ibs, short entropyType) {
37           if(ibs == null) {
38               throw new NullPointerException("Invalid null input bitstream parameter");
39           }
40  
41           switch (entropyType) {
42               // Each block is decoded separately
43               // Rebuild the entropy decoder to reset block statistics
44               case HUFFMAN_TYPE:
45                   return new HuffmanDecoder(ibs);
46               case ANS0_TYPE:
47                   return new ANSRangeDecoder(ibs, 0);
48               case ANS1_TYPE:
49                   return new ANSRangeDecoder(ibs, 1);
50               case RANGE_TYPE:
51                   return new RangeDecoder(ibs);
52               case PAQ_TYPE:
53                   return new BinaryEntropyDecoder(ibs, new PAQPredictor());
54               case FPAQ_TYPE:
55                   return new BinaryEntropyDecoder(ibs, new FPAQPredictor());
56               case CM_TYPE:
57                   return new BinaryEntropyDecoder(ibs, new CMPredictor());
58               case TPAQ_TYPE:
59                   return new BinaryEntropyDecoder(ibs, new TPAQPredictor());
60               case NONE_TYPE:
61                   return new NullEntropyDecoder(ibs);
62               default:
63                   throw new IllegalArgumentException("Unsupported entropy codec type: " + (char) entropyType);
64           }
65       }
66  
67  
68       public EntropyEncoder newEncoder(OutputBitStream obs, short entropyType) {
69           if(obs == null) {
70               throw new NullPointerException("Invalid null output bitstream parameter");
71           }
72  
73           switch (entropyType) {
74               case HUFFMAN_TYPE:
75                   return new HuffmanEncoder(obs);
76               case ANS0_TYPE:
77                   return new ANSRangeEncoder(obs, 0);
78               case ANS1_TYPE:
79                   return new ANSRangeEncoder(obs, 1);
80               case RANGE_TYPE:
81                   return new RangeEncoder(obs);
82               case PAQ_TYPE:
83                   return new BinaryEntropyEncoder(obs, new PAQPredictor());
84               case FPAQ_TYPE:
85                   return new BinaryEntropyEncoder(obs, new FPAQPredictor());
86               case CM_TYPE:
87                   return new BinaryEntropyEncoder(obs, new CMPredictor());
88               case TPAQ_TYPE:
89                   return new BinaryEntropyEncoder(obs, new TPAQPredictor());
90               case NONE_TYPE:
91                   return new NullEntropyEncoder(obs);
92               default:
93                   throw new IllegalArgumentException("Unknown entropy codec type: " + (char) entropyType);
94           }
95       }
96  
97  
98       public String getName(short entropyType) {
99           switch (entropyType) {
100               case HUFFMAN_TYPE:
101                   return "HUFFMAN";
102               case ANS0_TYPE:
103                   return "ANS0";
104               case ANS1_TYPE:
105                   return "ANS1";
106               case RANGE_TYPE:
107                   return "RANGE";
108               case PAQ_TYPE:
109                   return "PAQ";
110               case FPAQ_TYPE:
111                   return "FPAQ";
112               case CM_TYPE:
113                   return "CM";
114               case TPAQ_TYPE:
115                   return "TPAQ";
116               case NONE_TYPE:
117                   return "NONE";
118               default:
119                   throw new IllegalArgumentException("Unknown entropy codec type: " + (char) entropyType);
120           }
121       }
122  
123       // Do not use switch on String because some versions of Java do not support it
124       public short getType(String name) {
125           name = String.valueOf(name).toUpperCase();
126  
127           if(name.equals("HUFFMAN")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
128               return HUFFMAN_TYPE;
129           }
130  
131           if(name.equals("ANS0")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
132               return ANS0_TYPE;
133           }
134  
135           if(name.equals("ANS1")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
136               return ANS1_TYPE;
137           }
138  
139           if(name.equals("FPAQ")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
140               return FPAQ_TYPE;
141           }
142  
143           if(name.equals("PAQ")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
144               return PAQ_TYPE;
145           }
146  
147           if(name.equals("RANGE")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
148               return RANGE_TYPE;
149           }
150  
151           if(name.equals("CM")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
152               return CM_TYPE;
153           }
154  
155           if(name.equals("NONE")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
156               return NONE_TYPE;
157           }
158  
159           if(name.equals("TPAQ")) { → [FORCE, BLOCK, TRANSFORM, LEVEL, ENTROPY]
160               return TPAQ_TYPE;
161           }
162  
163           throw new IllegalArgumentException("Unsupported entropy codec type: " + name);
164       }
165  
166  }
