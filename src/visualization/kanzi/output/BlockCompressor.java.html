59       public BlockCompressor(Map<String, Object> map, ExecutorService threadPool) {
60           this.level = (Integer) map.remove("level");
61           this.verbosity = (Integer) map.remove("verbose");
62           Boolean bForce = (Boolean) map.remove("overwrite");
63           this.overwrite = (bForce == null) ? false : bForce;
64           this.inputName = (String) map.remove("inputName");
65           this.outputName = (String) map.remove("outputName");
66           String strTransf;
67           String strCodec;
68  
69           if(this.level >= 0) {
70               String tranformAndCodec = getTransformAndCodec(this.level);
71               String[] tokens = tranformAndCodec.split("&");
72               strTransf = tokens[0];
73               strCodec = tokens[1];
74           }
75           else {
76               strTransf = (String) map.remove("transform");
77               strCodec = (String) map.remove("entropy");
78           }
79  
80           this.codec = (strCodec == null) ? "ANS0" : strCodec;
81           Integer iBlockSize = (Integer) map.remove("block");
82           this.blockSize = (iBlockSize == null) ? DEFAULT_BLOCK_SIZE : iBlockSize;
83  
84           // Extract transform names. Curate input (EG. NONE+NONE+xxxx => xxxx)
85           ByteFunctionFactory bff = new ByteFunctionFactory();
86           this.transform = (strTransf == null) ? "BWT+RANK+ZRLT" : bff.getName(bff.getType(strTransf));
87           Boolean bChecksum = (Boolean) map.remove("checksum");
88           this.checksum = (bChecksum == null) ? false : bChecksum;
89           this.jobs = (Integer) map.remove("jobs");
90           this.pool = (this.jobs < 2) ? null :
91                   ((threadPool == null) ? Executors.newCachedThreadPool() : threadPool);
92           this.ownPool = (threadPool == null) && (this.pool != null);
93           this.listeners = new ArrayList<>(10);
94  
95           if(this.verbosity > 2) {
96               this.addListener(new InfoPrinter(this.verbosity, InfoPrinter.Type.ENCODING, System.out));
97           }
98  
99           if((this.verbosity > 0) && (map.size() > 0)) {
100               for(String k : map.keySet())
101                   printOut("Ignoring invalid option [" + k + "]", this.verbosity > 0);
102           }
103       }

107           this.level = level;
108           this.verbosity = verbose;
109           Boolean bForce = force;
110           this.overwrite = (bForce == null) ? false : bForce; → [FORCE]
111           this.inputName = (String) map.remove("inputName");
112           this.outputName = (String) map.remove("outputName");
113           String strTransf;
114           String strCodec;
115  
116           if(this.level >= 0) { → [LEVEL]
117               String tranformAndCodec = getTransformAndCodec(this.level);
118               String[] tokens = tranformAndCodec.split("&");
119               strTransf = tokens[0];
120               strCodec = tokens[1];
121           }

123               strTransf = transform;

125           }

130

132           ByteFunctionFactory bff = new ByteFunctionFactory();
133           this.transform = (strTransf == null) ? "BWT+RANK+ZRLT" : bff.getName(bff.getType(strTransf)); → [TRANSFORM, LEVEL]

168       private static String getTransformAndCodec(int level) {
169           switch (level) { → [LEVEL]
170               case 0:
171                   return "NONE&NONE";
172  
173               case 1:
174                   return "TEXT+LZ4&HUFFMAN";
175  
176               case 2:
177                   return "BWT+RANK+ZRLT&ANS0";
178  
179               case 3:
180                   return "BWT+RANK+ZRLT&FPAQ";
181  
182               case 4:
183                   return "BWT&CM";
184  
185               case 5:
186                   return "RLT+TEXT&TPAQ";
187  
188               default:
189                   return "Unknown&Unknown";
190           }

193       public void dispose() {
194           try {
195               if(this.is != null) { → [FORCE]
196                   this.is.close();
197               }
198           } catch (IOException ioe) {

208               System.exit(Error.ERR_WRITE_FILE);

212               this.pool.shutdown();
213           }

221       // Return status (success = 0, error < 0)
222       @Override
223       public Integer call() {
224           boolean printFlag = this.verbosity > 2; → [VERBOSE]
225           printOut("Kanzi 1.2 (C) 2017, Frederic Langlet", this.verbosity >= 1); → [VERBOSE]
226           printOut("Input file name set to '" + this.inputName + "'", printFlag);
227           printOut("Output file name set to '" + this.outputName + "'", printFlag);
228           printOut("Block size set to " + this.blockSize + " bytes", printFlag);
229           printOut("Verbosity set to " + this.verbosity, printFlag);
230           printOut("Overwrite set to " + this.overwrite, printFlag);
231           printOut("Checksum set to " + this.checksum, printFlag);
232  
233           if(this.level < 0) { → [LEVEL]
234               String etransform = ("NONE".equals(this.transform)) ? "no" : this.transform; → [TRANSFORM, LEVEL]
235               printOut("Using " + etransform + " transform (stage 1)", printFlag);
236               String ecodec = ("NONE".equals(this.codec)) ? "no" : this.codec; → [LEVEL, ENTROPY]
237               printOut("Using " + ecodec + " entropy codec (stage 2)", printFlag);
238           }
239           else {
240               printOut("Compression level set to " + this.level, printFlag);
241           }
242  
243           if(this.jobs > 0) {
244               printOut("Using " + this.jobs + " job" + ((this.jobs > 1) ? "s" : ""), printFlag);
245           }
246  
247           OutputStream os;
248  
249           try {
250               if(this.outputName.equalsIgnoreCase("NONE")) {
251                   os = new NullOutputStream();
252               }
253               else if(this.outputName.equalsIgnoreCase("STDOUT")) {
254                   os = System.out;
255               }
256               else {
257                   File output = new File(this.outputName);
258  
259                   if(output.exists()) {
260                       if(output.isDirectory()) {
261                           System.err.println("The output file is a directory");
262                           return Error.ERR_OUTPUT_IS_DIR;
263                       }
264  
265                       if(this.overwrite == false) { → [FORCE]
266                           System.err.println("The output file exists and the 'overwrite' command "
267                                   + "line option has not been provided");
268                           return Error.ERR_OVERWRITE_FILE;
269                       }
270  
271                       Path path1 = FileSystems.getDefault().getPath(this.inputName).toAbsolutePath();
272                       Path path2 = FileSystems.getDefault().getPath(this.outputName).toAbsolutePath();
273  
274                       if(path1.equals(path2)) { → [FORCE]
275                           System.err.println("The input and output files must be different");
276                           return Error.ERR_CREATE_FILE;
277                       }
278                   }
279  
280                   os = new FileOutputStream(output);
281               }
282  
283               try {
284                   Map<String, Object> ctx = new HashMap<>();
285                   ctx.put("blockSize", this.blockSize);
286                   ctx.put("checksum", this.checksum);
287                   ctx.put("pool", this.pool);
288                   ctx.put("jobs", this.jobs);
289                   ctx.put("codec", this.codec);
290                   ctx.put("transform", this.transform);
291                   this.cos = new CompressedOutputStream(os, ctx);
292  
293                   for(Listener bl : this.listeners) → [FORCE]
294                       this.cos.addListener(bl);
295               } catch (Exception e) {
296                   System.err.println("Cannot create compressed stream: " + e.getMessage());
297                   return Error.ERR_CREATE_COMPRESSOR;
298               }
299           } catch (Exception e) {
300               System.err.println("Cannot open output file '" + this.outputName + "' for writing: " + e.getMessage());
301               return Error.ERR_CREATE_FILE;
302           }
303  
304           try {
305               this.is = (this.inputName.equalsIgnoreCase("STDIN")) ? System.in : new FileInputStream(this.inputName); → [FORCE]
306           } catch (Exception e) {
307               System.err.println("Cannot open input file '" + this.inputName + "': " + e.getMessage());
308               return Error.ERR_OPEN_FILE;
309           }
310  
311           // Encode
312           printFlag = this.verbosity > 1; → [FORCE, VERBOSE]
313           printOut("Encoding ...", printFlag);
314           int read = 0;
315           SliceByteArray sa = new SliceByteArray(new byte[DEFAULT_BUFFER_SIZE], 0);
316           int len;
317  
318           if(this.listeners.size() > 0) { → [FORCE]
319               Event evt = new Event(Event.Type.COMPRESSION_START, -1, 0);
320               Listener[] array = this.listeners.toArray(new Listener[this.listeners.size()]);
321               notifyListeners(array, evt);
322           }
323  
324           long before = System.nanoTime();
325  
326           try {
327               while (true) {
328                   try {
329                       len = this.is.read(sa.array, 0, sa.length);
330                   } catch (Exception e) {
331                       System.err.print("Failed to read block from file '" + this.inputName + "': ");
332                       System.err.println(e.getMessage());
333                       return Error.ERR_READ_FILE;
334                   }
335  
336                   if(len <= 0) { → [FORCE]
337                       break;
338                   }
339  
340                   // Just write block to the compressed output stream !
341                   read += len;
342                   this.cos.write(sa.array, 0, len);
343               }
344           } catch (kanzi.io.IOException e) {
345               System.err.println(e.getMessage());
346               return e.getErrorCode();
347           } catch (Exception e) {
348               System.err.println("An unexpected condition happened. Exiting ...");
349               System.err.println(e.getMessage());
350               return Error.ERR_UNKNOWN;
351           } finally {
352               // Close streams to ensure all data are flushed
353               this.dispose();
354  
355               try {
356                   os.close();
357               } catch (IOException e) {
358                   // Ignore
359               }
360           }
361  
362           if(read == 0) { → [FORCE]
363               System.out.println("Empty input file ... nothing to do");
364               return WARN_EMPTY_INPUT;
365           }
366  
367           long after = System.nanoTime();
368           long delta = (after - before) / 1000000L; // convert to ms
369           printOut("", this.verbosity >= 1); → [FORCE, VERBOSE]
370           printOut("Encoding: " + delta + " ms", printFlag);
371           printOut("Input size: " + read, printFlag);
372           printOut("Output size: " + this.cos.getWritten(), printFlag);
373           float f = this.cos.getWritten() / (float) read;
374           printOut("Ratio: " + String.format("%1$.6f", f), printFlag);
375           printOut("Encoding: " + read + " => " + this.cos.getWritten() + → [FORCE, VERBOSE]
376                   " bytes in " + delta + " ms", this.verbosity == 1);
377  
378           if(delta > 0) { → [FORCE]
379               printOut("Throughput (KB/s): " + (((read * 1000L) >> 10) / delta), printFlag);
380           }
381  
382           printOut("", this.verbosity >= 1); → [FORCE, VERBOSE]
383  
384           if(this.listeners.size() > 0) { → [FORCE]
385               Event evt = new Event(Event.Type.COMPRESSION_END, -1, this.cos.getWritten());
386               Listener[] array = this.listeners.toArray(new Listener[this.listeners.size()]);
387               notifyListeners(array, evt);
388           }
389  
390           return 0;
391       }
