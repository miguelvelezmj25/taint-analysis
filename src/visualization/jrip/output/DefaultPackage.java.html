34  /**
35    * A concrete implementation of Package that uses Java properties files/classes
36    * to manage package meta data. Assumes that meta data for individual packages
37    * is stored on the central repository (or possibly in a local cache - both
38    * accessible via http) in properties files that live in a sub-directory with
39    * the same name as the package. Furthermore, each property file is assumed to
40    * be named as the version number of the package in question with a ".props"
41    * extension. A "Latest.props" file should exist for each package and should
42    * always hold meta data on the latest version of a package.
43    *
44    * @author mhall (mhall{[at]}pentaho{[dot]}com).
45    * @version $Revision: 52462 $
46    *
47    */
48  public class DefaultPackage extends Package implements Serializable {

61     /**
62      * Clone this package. Only makes a shallow copy of the meta data map
63      *
64      * @return a copy of this package
65      */
66     @Override
67     public Object clone() {
68  
69       DefaultPackage newP = null;
70  
71       if (m_packageHome != null) {
72         newP =
73           new DefaultPackage(new File(m_packageHome.toString()), m_packageManager);
74       } else {
75         newP = new DefaultPackage(null, m_packageManager);
76       }
77  
78       HashMap<Object, Object> metaData = new HashMap<Object, Object>();
79       Set<?> keys = m_packageMetaData.keySet();
80       Iterator<?> i = keys.iterator();
81  
82       while (i.hasNext()) {
83         Object key = i.next();
84         Object value = m_packageMetaData.get(key);
85         metaData.put(key, value);
86       }
87  
88       newP.setPackageMetaData(metaData);
89  
90       return newP;
91     }
