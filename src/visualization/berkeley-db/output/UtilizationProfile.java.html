55  /**
56    * The UP tracks utilization summary information for all log files.
57    *
58    * <p>Unlike the UtilizationTracker, the UP is not accessed under the log write
59    * latch and is instead synchronized on itself for protecting the cache. It is
60    * not accessed during the primary data access path, except for when flushing
61    * (writing) file summary LNs. This occurs in the following cases:
62    * <ol>
63    * <li>The summary information is flushed at the end of a checkpoint. This
64    * allows tracking to occur in memory in between checkpoints, and replayed
65    * during recovery.</li>
66    * <li>When committing the truncateDatabase and removeDatabase operations, the
67    * summary information is flushed because detail tracking for those operations
68    * is not replayed during recovery</li>
69    * <li>The evictor will ask the UtilizationTracker to flush the largest summary
70    * if the memory taken by the tracker exeeds its budget.</li>
71    * </ol>
72    *
73    * <p>The cache is populated by the RecoveryManager just before performing the
74    * initial checkpoint. The UP must be open and populated in order to respond
75    * to requests to flush summaries and to evict tracked detail, even if the
76    * cleaner is disabled.</p>
77    *
78    * <p>WARNING: While synchronized on this object, eviction is not permitted.
79    * If it were, this could cause deadlocks because the order of locking would be
80    * the UP object and then the evictor. During normal eviction the order is to
81    * first lock the evictor and then the UP, when evicting tracked detail.</p>
82    *
83    * <p>The methods in this class synchronize to protect the cached summary
84    * information. Some methods also access the UP database. However, because
85    * eviction must not occur while synchronized, UP database access is not
86    * performed while synchronized except in one case: when inserting a new
87    * summary record. In that case we disallow eviction during the database
88    * operation.</p>
89    */

476       /**
477        * Updates and stores the FileSummary for a given tracked file. This
478        * method is synchronized and may not perform eviction.
479        */

481           throws DatabaseException {
482  
483           if (env.isReadOnly()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
484               throw EnvironmentFailureException.unexpectedState
485                   ("Cannot write file summary in a read-only environment");
486           }
487  
488           if (tfs.isEmpty()) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
489               return null; // no delta
490           }
491  
492           if (!cachePopulated) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
493               /* Db does not exist and this is a read-only environment. */
494               return null;
495           }
496  
497           long fileNum = tfs.getFileNumber();
498           Long fileNumLong = fileNum;
499  
500           /* Get existing file summary or create an empty one. */
501           FileSummary summary = fileSummaryMap.get(fileNumLong);
502           if (summary == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
503  
504               /*
505                * An obsolete node may have been counted after its file was
506                * deleted, for example, when compressing a BIN. Do not insert a
507                * new profile record if no corresponding log file exists. But if
508                * the file number is greater than the last known file, this is a
509                * new file that has been buffered but not yet flushed to disk; in
510                * that case we should insert a new profile record.
511                */
512               if (!fileSummaryMap.isEmpty() && → [ALLOWCREATE, TRANSACTIONAL, READONLY]
513                   fileNum < fileSummaryMap.lastKey() && → [ALLOWCREATE, TRANSACTIONAL, READONLY]
514                   !env.getFileManager().isFileValid(fileNum)) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
515  
516                   /*
517                    * File was deleted by the cleaner. Remove it from the
518                    * UtilizationTracker and return. Note that a file is normally
519                    * removed from the tracker by FileSummaryLN.writeToLog method
520                    * when it is called via insertFileSummary below. [#15512]
521                    */
522                   env.getLogManager().removeTrackedFile(tfs);
523                   return null;
524               }
525  
526               summary = new FileSummary();
527           }
528  
529           /*
530            * The key discriminator is a sequence that must be increasing over the
531            * life of the file. We use the sum of all entries counted. We must
532            * add the tracked and current summaries here to calculate the key.
533            */
534           FileSummary tmp = new FileSummary();
535           tmp.add(summary);
536           tmp.add(tfs);
537           int sequence = tmp.getEntriesCounted();
538  
539           /* Insert an LN with the existing and tracked summary info. */
540           FileSummaryLN ln = new FileSummaryLN(summary);
541           ln.setTrackedSummary(tfs);
542           insertFileSummary(ln, fileNum, sequence);
543  
544           /* Cache the updated summary object. */
545           summary = ln.getBaseSummary();
546  
547           if (fileSummaryMap.put(fileNumLong, summary) == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
548               MemoryBudget mb = env.getMemoryBudget();
549               mb.updateAdminMemoryUsage(MemoryBudget.UTILIZATION_PROFILE_ENTRY);
550           }
551  
552           return ln.getObsoleteOffsets();
553       }
