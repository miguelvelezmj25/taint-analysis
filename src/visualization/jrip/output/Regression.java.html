275  
276         /** For serialization */
277         private static final long serialVersionUID = 5493100145890252757L;
278  
279         /** The coefficient for this predictor term */
280         protected double m_coefficient = 1.0;
281  
282         /** the indexes of the terms to be multiplied */
283         protected int[] m_indexes;
284  
285         /** The names of the terms (attributes) to be multiplied */
286         protected String[] m_fieldNames;
287  
288         /**
289          * Construct a new PredictorTerm.
290          *
291          * @param predictorTerm the <code>Element</code> describing the predictor term
292          * @param miningSchema the mining schema as an Instances object
293          * @throws Exception if something goes wrong while constructing this
294          * predictor term
295          */
296         protected PredictorTerm(Element predictorTerm,
297                                 Instances miningSchema) throws Exception {
298  
299           String coeff = predictorTerm.getAttribute("coefficient");
300           if (coeff != null && coeff.length() > 0) {
301             try {
302               m_coefficient = Double.parseDouble(coeff);
303             } catch (IllegalArgumentException ex) {
304               throw new Exception("[PredictorTerm] unable to parse coefficient");
305             }
306           }
307          
308           NodeList fields = predictorTerm.getElementsByTagName("FieldRef");
309           if (fields.getLength() > 0) {
310             m_indexes = new int[fields.getLength()];
311             m_fieldNames = new String[fields.getLength()];
312  
313             for (int i = 0; i < fields.getLength(); i++) {
314               Node fieldRef = fields.item(i);
315               if (fieldRef.getNodeType() == Node.ELEMENT_NODE) {
316                 String fieldName = ((Element)fieldRef).getAttribute("field");
317                 if (fieldName != null && fieldName.length() > 0) {
318                   boolean found = false;
319                   // look for this field in the mining schema
320                   for (int j = 0; j < miningSchema.numAttributes(); j++) {
321                     if (miningSchema.attribute(j).name().equals(fieldName)) {
322                      
323                       // all referenced fields MUST be numeric
324                       if (!miningSchema.attribute(j).isNumeric()) {
325                         throw new Exception("[PredictorTerm] field is not continuous: "
326                                             + fieldName);
327                       }
328                       found = true;
329                       m_indexes[i] = j;
330                       m_fieldNames[i] = fieldName;
331                       break;
332                     }
333                   }
334                   if (!found) {
335                     throw new Exception("[PredictorTerm] Unable to find field "
336                                         + fieldName + " in mining schema!");
337                   }
338                 }
339               }
340             }
341           }
342         }

344         /**
345          * Return a textual description of this predictor term.
346          */
347         public String toString() {
348           StringBuffer result = new StringBuffer();
349           result.append("(" + Utils.doubleToString(m_coefficient, 12, 4));
350           for (int i = 0; i < m_fieldNames.length; i++) {
351             result.append(" * " + m_fieldNames[i]);
352           }
353           result.append(")");
354           return result.toString();
355         }

357         /**
358          * Adds this predictor term into the sum for the
359          * current prediction.
360          *
361          * @param preds the prediction computed so far. For regression, it is a
362          * single element array; for classification it is a multi-element array
363          * @param input the input instance's values
364          */
365         public void add(double[] preds, double[] input) {
366           int indx = 0;
367           if (m_targetCategory != -1) {
368             indx = m_targetCategory;
369           }
370  
371           double result = m_coefficient;
372           for (int i = 0; i < m_indexes.length; i++) {
373             result *= input[m_indexes[i]];
374           }
375           preds[indx] += result;
376         }
377       }

436       /**
437        * Construct a regression table from an <code>Element</code>
438        *
439        * @param table the table to encapsulate
440        * @param functionType the type of function
441        * (regression or classification)
442        * to use
443        * @param mSchema the mining schema
444        * @throws Exception if there is a problem while constructing
445        * this regression table
446        */
447       protected RegressionTable(Element table,
448                              int functionType,
449                              MiningSchema mSchema) throws Exception {
450  
451         m_miningSchema = mSchema;
452         m_functionType = functionType;
453  
454         Instances miningSchema = m_miningSchema.getFieldsAsInstances();
455  
456         // get the intercept
457         String intercept = table.getAttribute("intercept");
458         if (intercept.length() > 0) {
459           m_intercept = Double.parseDouble(intercept);
460         }
461  
462         // get the target category (if classification)
463         if (m_functionType == CLASSIFICATION) {
464           // target category MUST be defined
465           String targetCat = table.getAttribute("targetCategory");
466           if (targetCat.length() > 0) {
467             Attribute classA = miningSchema.classAttribute();
468             for (int i = 0; i < classA.numValues(); i++) {
469               if (classA.value(i).equals(targetCat)) {
470                 m_targetCategory = i;
471               }
472             }
473           }
474           if (m_targetCategory == -1) {
475             throw new Exception("[RegressionTable] No target categories defined for classification");
476           }
477         }
478  
479         // read all the numeric predictors
480         NodeList numericPs = table.getElementsByTagName("NumericPredictor");
481         for (int i = 0; i < numericPs.getLength(); i++) {
482           Node nP = numericPs.item(i);
483           if (nP.getNodeType() == Node.ELEMENT_NODE) {
484             NumericPredictor numP = new NumericPredictor((Element)nP, miningSchema);
485             m_predictors.add(numP);
486           }
487         }
488  
489         // read all the categorical predictors
490         NodeList categoricalPs = table.getElementsByTagName("CategoricalPredictor");
491         for (int i = 0; i < categoricalPs.getLength(); i++) {
492           Node cP = categoricalPs.item(i);
493           if (cP.getNodeType() == Node.ELEMENT_NODE) {
494             CategoricalPredictor catP = new CategoricalPredictor((Element)cP, miningSchema);
495             m_predictors.add(catP);
496           }
497         }
498  
499         // read all the PredictorTerms
500         NodeList predictorTerms = table.getElementsByTagName("PredictorTerm");
501         for (int i = 0; i < predictorTerms.getLength(); i++) {
502           Node pT = predictorTerms.item(i);
503           PredictorTerm predT = new PredictorTerm((Element)pT, miningSchema);
504           m_predictorTerms.add(predT);
505         }
506       }
507  
508       public void predict(double[] preds, double[] input) {
509         if (m_targetCategory == -1) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
510           preds[0] = m_intercept;
511         } else {
512           preds[m_targetCategory] = m_intercept;
513         }
514        
515         // add the predictors
516         for (int i = 0; i < m_predictors.size(); i++) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
517           Predictor p = m_predictors.get(i);
518           p.add(preds, input);
519         }
520  
521         // add the PredictorTerms
522         for (int i = 0; i < m_predictorTerms.size(); i++) { → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
523           PredictorTerm pt = m_predictorTerms.get(i);
524           pt.add(preds, input);
525         }
526       }
527     }

616     /**
617      * Return the type of normalization used for this regression
618      *
619      * @param model the <code>Element</code> holding the model
620      * @return the normalization used in this regression
621      */
622     private static Normalization determineNormalization(Element model) {
623      
624       Normalization normMethod = Normalization.NONE;
625  
626       String normName = model.getAttribute("normalizationMethod");
627       if (normName.equals("simplemax")) {
628         normMethod = Normalization.SIMPLEMAX;
629       } else if (normName.equals("softmax")) {
630         normMethod = Normalization.SOFTMAX;
