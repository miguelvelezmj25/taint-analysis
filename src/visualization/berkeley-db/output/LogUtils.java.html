479       /**
480        * The byte[]'s in Xid's are known to be 255 or less in length. So instead
481        * of using read/writeByteArray(), we can save 6 bytes per record by making
482        * the byte[] length be 1 byte instead of 4.
483        */
484       public static int getXidSize(Xid xid) {
485           byte[] gid = xid.getGlobalTransactionId();
486           byte[] bqual = xid.getBranchQualifier();
487           return
488               INT_BYTES + // FormatId
489               1 + // gxid length byte
490               1 + // bqual length byte
491               (gid == null ? 0 : gid.length) + // gid bytes
492               (bqual == null ? 0 : bqual.length); // bqual bytes
493       }
494  
495       /*
496        * Xid.gid[] and bqual[] can't be longer than 64 bytes so we can get away
497        * with writing the length in one byte, rather than 4.
498        */
499       public static void writeXid(ByteBuffer logBuf, Xid xid) {
500           byte[] gid = xid.getGlobalTransactionId();
501           byte[] bqual = xid.getBranchQualifier();
502  
503           writeInt(logBuf, xid.getFormatId());
504  
505           if (gid == null) {
506               logBuf.put((byte) -1);
507           } else {
508               logBuf.put((byte) (gid.length));
509               logBuf.put(gid);
510           }
511  
512           if (bqual == null) {
513               logBuf.put((byte) -1);
514           } else {
515               logBuf.put((byte) (bqual.length));
516               logBuf.put(bqual);
517           }
518       }
519  
520       /*
521        * Xid.gid[] and bqual[] can't be longer than 64 bytes so we can get away
522        * with writing the length in one byte, rather than 4.
523        */
524       public static Xid readXid(ByteBuffer logBuf) {
525           int formatId = readInt(logBuf);
526  
527           int gidLen = logBuf.get();
528           byte[] gid = null;
529           if (gidLen >= 0) {
530               gid = new byte[gidLen];
531               logBuf.get(gid);
532           }
533  
534           int bqualLen = logBuf.get();
535           byte[] bqual = null;
536           if (bqualLen >= 0) {
537               bqual = new byte[bqualLen];
538               logBuf.get(bqual);
539           }
540  
541           return new XidImpl(formatId, gid, bqual);
542       }

568           @Override

570               if (!(o instanceof XidImpl)) {
571                   return false;
572               }
573  
574               XidImpl xid = (XidImpl) o;
575               if (xid.getFormatId() != formatId) {
576                   return false;
577               }
578               if (compareByteArrays(xid.getGlobalTransactionId(), gid) &&
579                   compareByteArrays(xid.getBranchQualifier(), bqual)) {
580                   return true;
581               }
582  
583               return false;
584           }

587           public int hashCode() {
588               int code = formatId;
589               if (gid != null) {
590                   for (int i = 0; i < gid.length; i++) {
591                       code += gid[i];
592                   }
593               }
594               if (bqual != null) {
595                   for (int i = 0; i < bqual.length; i++) {
596                       code += bqual[i];
597                   }
598               }
599               return code;
600           }
601  
602           private boolean compareByteArrays(byte[] b1, byte[] b2) {
603               if (b1 == null ||
604                   b2 == null) {
605                   return b1 == b2;
606               }
607  
608               if (b1.length != b2.length) {
609                   return false;
610               }
611  
612               for (int i = 0; i < b1.length; i++) {
613                   if (b1[i] != b2[i]) {
614                       return false;
615                   }
616               }
617  
618               return true;
619           }
