24       private final static double INV_LOG_2 = 1.4426950408889 * 0x10000L; /* 1.0 / log(2.0) */
25       final long[] lLiterals = new long[288];
26       final long[] lLengths = new long[259];

31       void getFreqs(LzStore store) {
32           int[] sLitLens = this.litLens;
33           int[] sDists = this.dists;
34           System.arraycopy(Cookie.intZeroes, 0, sLitLens, 0, 288);
35           System.arraycopy(Cookie.intZeroes, 0, sDists, 0, 32);
36  
37           int size = store.size;
38           char[] litLens = store.litLens;
39           char[] dists = store.dists;
40           int[] lengthSymbol = Util.LENGTH_SYMBOL;
41           int[] cachedDistSymbol = Util.CACHED_DIST_SYMBOL;
42           for(int i = 0; i < size; i++) { → [COMPRESSOR, ITERATIONS]
43               int d = dists[i];
44               int l = litLens[i];
45               if(d == 0) { → [COMPRESSOR, ITERATIONS]
46                   sLitLens[l]++;
47               }
48               else {
49                   sLitLens[lengthSymbol[l]]++;
50                   sDists[cachedDistSymbol[d]]++;
51               }
52           }
53           sLitLens[256] = 1;
54           calculate();
55       }

70       private void calculateLens() {
71           int sum = 0;
72           int[] litLens = this.litLens;
73           for(int i = 0; i < 288; ++i) { → [COMPRESSOR, ITERATIONS]
74               sum += litLens[i];
75           }
76           double log2sum = (sum == 0 ? Math.log(288) : Math.log(sum)) * INV_LOG_2; → [COMPRESSOR, ITERATIONS]
77           long[] lLiterals = this.lLiterals;
78           for(int i = 0; i < 288; ++i) { → [COMPRESSOR, ITERATIONS]
79               if(litLens[i] == 0) { → [COMPRESSOR, ITERATIONS]
80                   lLiterals[i] = (long) log2sum;
81               }
82               else {
83                   lLiterals[i] = (long) (log2sum - Math.log(litLens[i]) * INV_LOG_2);
84               }
85               if(lLiterals[i] < 0) { → [COMPRESSOR, ITERATIONS]
86                   lLiterals[i] = 0;
87               }
88           }
89           long[] lLengths = this.lLengths;
90           int[] lengthSymbol = Util.LENGTH_SYMBOL;
91           int[] lengthExtraBits = Util.LENGTH_EXTRA_BITS;
92           for(int i = 0; i < 259; ++i) { → [COMPRESSOR, ITERATIONS]
93               lLengths[i] = lLiterals[lengthSymbol[i]] + (lengthExtraBits[i] * 0x10000L);
94           }
95       }

97       private void calculateDists() {
98           int sum = 0;
99           int[] dists = this.dists;
100           for(int i = 0; i < 32; ++i) { → [COMPRESSOR, ITERATIONS]
101               sum += dists[i];
102           }
103           double log2sum = (sum == 0 ? Math.log(32) : Math.log(sum)) * INV_LOG_2; → [COMPRESSOR, ITERATIONS]
104           long[] dSymbols = this.dSymbols;
105           for(int i = 0; i < 32; ++i) { → [COMPRESSOR, ITERATIONS]
106               if(dists[i] == 0) { → [COMPRESSOR, ITERATIONS]
107                   dSymbols[i] = (long) log2sum;
108               }
109               else {
110                   dSymbols[i] = (long) (log2sum - Math.log(dists[i]) * INV_LOG_2);
111               }
112               if(dSymbols[i] < 0) { → [COMPRESSOR, ITERATIONS]
113                   dSymbols[i] = 0;
114               }
115           }
116           for(int i = 4; i < 30; ++i) { → [COMPRESSOR, ITERATIONS]
117               dSymbols[i] += 0x10000L * ((i / 2) - 1);
118           }
119       }

123           for(int i = 0; i < 288; i++) { → [COMPRESSOR, ITERATIONS]
124               litLens[i] += ligandLitLens[i] / 2;

129           for(int i = 0; i < 32; i++) { → [COMPRESSOR, ITERATIONS]
130               dists[i] += ligandDists[i] / 2;
