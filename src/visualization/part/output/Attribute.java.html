28  /**
29    * Class for handling an attribute. Once an attribute has been created, it can't
30    * be changed.
31    * <p>
32    * <p>
33    * The following attribute types are supported:
34    * <ul>
35    * <li>numeric: <br/>
36    * This type of attribute represents a floating-point number.</li>
37    * <li>nominal: <br/>
38    * This type of attribute represents a fixed set of nominal values.</li>
39    * <li>string: <br/>
40    * This type of attribute represents a dynamically expanding set of nominal
41    * values. Usually used in text classification.</li>
42    * <li>date: <br/>
43    * This type of attribute represents a date, internally represented as
44    * floating-point number storing the milliseconds since January 1, 1970,
45    * 00:00:00 GMT. The string representation of the date must be <a
46    * href="http://www.iso.org/iso/en/prods-services/popstds/datesandtime.html"
47    * target="_blank"> ISO-8601</a> compliant, the default is
48    * <code>yyyy-MM-dd'T'HH:mm:ss</code>.</li>
49    * <li>relational: <br/>
50    * This type of attribute can contain other attributes and is, e.g., used for
51    * representing Multi-Instance data. (Multi-Instance data consists of a nominal
52    * attribute containing the bag-id, then a relational attribute with all the
53    * attributes of the bag, and finally the class attribute.)</li>
54    * </ul>
55    * <p>
56    * Typical usage (code from the main() method of this class):
57    * <p>
58    * <p>
59    * <code>
60    * ... <br>
61    * <p>
62    * // Create numeric attributes "length" and "weight" <br>
63    * Attribute length = new Attribute("length"); <br>
64    * Attribute weight = new Attribute("weight"); <br><br>
65    * <p>
66    * // Create list to hold nominal values "first", "second", "third" <br>
67    * List<String> my_nominal_values = new ArrayList<String>(3); <br>
68    * my_nominal_values.add("first"); <br>
69    * my_nominal_values.add("second"); <br>
70    * my_nominal_values.add("third"); <br><br>
71    * <p>
72    * // Create nominal attribute "position" <br>
73    * Attribute position = new Attribute("position", my_nominal_values);<br>
74    * <p>
75    * ... <br>
76    * </code>
77    * <p>
78    *
79    * @author Eibe Frank (eibe@cs.waikato.ac.nz)
80    * @version $Revision: 12880 $
81    */
