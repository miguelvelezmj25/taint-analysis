3  import java.io.File;
4  import java.io.FileFilter;
5  import java.util.Date;
6  
7  /**
8    * File filter based on last modified, last accessed or created time of the
9    * file.
10    */
11  class TimeFilter implements FileFilter {
12       private final Date time;
13       private final TimeType timeType;
14       private final TimeComparator comparator;
15  
16       public TimeFilter(Date time, FindArguments arguments) {
17           this(time, TimeFilter.getTimeType(arguments), TimeFilter.getTimeComparator(arguments));
18       }
19  
20       public TimeFilter(Date time, TimeType timeType, TimeComparator comparator) {
21           this.time = time;
22           this.timeType = timeType;
23           this.comparator = comparator;
24       }
25  
26       private static TimeType getTimeType(FindArguments arguments) {
27           TimeType timeType = new TimeType.Modified();
28  
29           if(arguments.isTimeAccess()) { → [ISTIMEACCESS]
30               timeType = new TimeType.Access();
31           }
32           else if(arguments.isTimeCreate()) { → [ISTIMEACCESS, ISTIMECREATE]
33               timeType = new TimeType.Create();
34           }
35  
36           return timeType;
37       }
38  
39       private static TimeComparator getTimeComparator(FindArguments arguments) {
40           TimeComparator timeComparator = new TimeComparator.Newer();
41  
42           if(arguments.isTimeOlder()) { → [ISTIMEOLDER]
43               timeComparator = new TimeComparator.Older();
44           }
45  
46           return timeComparator;
47       }
48  
49       @Override
50       public boolean accept(File file) {
51           return comparator.accept(file, timeType, time);
52       }
53  
54       public static abstract class TimeType {
55  
56           abstract public long getTime(File file);
57  
58           public static class Create extends TimeType {
59               @Override
60               public long getTime(File file) {
61                   return FileAttributes.INSTANCE.getCreationTime(file);
62               }
63           }
64  
65           public static class Access extends TimeType {
66               @Override
67               public long getTime(File file) {
68                   return FileAttributes.INSTANCE.getLastAccessedTime(file);
69               }
70           }
71  
72           public static class Modified extends TimeType {
73               @Override
74               public long getTime(File file) {
75                   return FileAttributes.INSTANCE.getLastModifiedTime(file);
76               }
77           }
78       }

119               @Override

121                   return timeType.getTime(file) <= time.getTime();
122               }
123           }

126               @Override
