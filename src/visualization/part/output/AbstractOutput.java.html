41  /**
42    * A superclass for outputting the classifications of a classifier.
43    * <p/>
44    * Basic use with a classifier and a test set:
45    *
46    * <pre>
47    * Classifier classifier = ... // trained classifier
48    * Instances testset = ... // the test set to output the predictions for
49    * StringBuffer buffer = ... // the string buffer to add the output to
50    * AbstractOutput output = new FunkyOutput();
51    * output.setHeader(...);
52    * output.printClassifications(classifier, testset);
53    * </pre>
54    *
55    * Basic use with a classifier and a data source:
56    *
57    * <pre>
58    * Classifier classifier = ... // trained classifier
59    * DataSource testset = ... // the data source to obtain the test set from to output the predictions for
60    * StringBuffer buffer = ... // the string buffer to add the output to
61    * AbstractOutput output = new FunkyOutput();
62    * output.setHeader(...);
63    * output.printClassifications(classifier, testset);
64    * </pre>
65    *
66    * In order to make the output generation easily integrate into GUI components,
67    * one can output the header, classifications and footer separately:
68    *
69    * <pre>
70    * Classifier classifier = ... // trained classifier
71    * Instances testset = ... // the test set to output the predictions for
72    * StringBuffer buffer = ... // the string buffer to add the output to
73    * AbstractOutput output = new FunkyOutput();
74    * output.setHeader(...);
75    * // print the header
76    * output.printHeader();
77    * // print the classifications one-by-one
78    * for (int i = 0; i < testset.numInstances(); i++) {
79    * output.printClassification(classifier, testset.instance(i), i);
80    * // output progress information
81    * if ((i+1) % 100 == 0)
82    * System.out.println((i+1) + "/" + testset.numInstances());
83    * }
84    * // print the footer
85    * output.printFooter();
86    * </pre>
87    *
88    * @author fracpete (fracpete at waikato dot ac dot nz)
89    * @version $Revision: 11958 $
90    */
