93       /**

95        * the supplied XML element
96        *
97        * @param svmMachineModelElement the XML element containing the kernel
98        * @param log the logging object to use
99        * @return a new Kernel
100        * @throws Exception if something goes wrong
101        */
102       public static Kernel getKernel(Element svmMachineModelElement,
103           Logger log) throws Exception {
104  
105        
106         NodeList kList = svmMachineModelElement.getElementsByTagName("LinearKernelType");
107         if (kList.getLength() > 0) {
108           return new LinearKernel(log);
109         }
110        
111         kList = svmMachineModelElement.getElementsByTagName("PolynomialKernelType");
112         if (kList.getLength() > 0) {
113           return new PolynomialKernel((Element)kList.item(0), log);
114         }
115        
116         kList = svmMachineModelElement.getElementsByTagName("RadialBasisKernelType");
117         if (kList.getLength() > 0) {
118           return new RadialBasisKernel((Element)kList.item(0), log);
119         }
120        
121         kList = svmMachineModelElement.getElementsByTagName("SigmoidKernelType");
122         if (kList.getLength() > 0) {
123           return new SigmoidKernel((Element)kList.item(0), log);
124         }
125        
126         throw new Exception("[Kernel] Can't find a kernel that I recognize!");
127       }
128     }

415        * @param x the first vector instance

417        * @return the result of the kernel evaluation
418        * @throws Exception if something goes wrong
419        */
420       public double evaluate(VectorInstance x, VectorInstance y)
421         throws Exception {
422        
423         double dotProd = x.dotProduct(y);
424         double z = m_gamma * dotProd + m_coef0;
425         double a = Math.exp(z);
426         double b = Math.exp(-z);
427         return ((a - b) / (a + b));
428       }

459     /**
460      * Inner class implementing a single binary (classification) SVM
461      */
462     static class SupportVectorMachine implements Serializable {
463      
464       /** For serialization */
465       private static final long serialVersionUID = -7650496802836815608L;
466  
467       /** The target label for classification problems */
468       protected String m_targetCategory;
469      
470       /** The index of the global alternate target label for classification */
471       protected int m_globalAlternateTargetCategoryIndex = -1;
472      
473       /** The index of the target label for classification problems */
474       protected int m_targetCategoryIndex = -1;
475          
476       /** PMML 4.0 - index of the alternate category for one-vs-one */
477       protected int m_localAlternateTargetCategoryIndex = -1;
478      
479       /** PMML 4.0 - local threshold (overrides the global one, if set) */
480       protected double m_localThreshold = Double.MAX_VALUE; // default - not set
481          
482       /** The mining schema */
483       protected MiningSchema m_miningSchema;
484      
485       /** The log object to use (if supplied) */
486       protected Logger m_log;
487      
488       /**
489        * True if this is a linear machine expressed in terms of the original
490        * attributes
491        */
492       protected boolean m_coeffsOnly = false;
493      
494       /** The support vectors used by this machine (if not linear with coeffs only */
495       protected List<VectorInstance> m_supportVectors =
496         new ArrayList<VectorInstance>();
497      
498       /** The constant term - b */
499       protected double m_intercept = 0;
500      
501       /** The coefficients for the vectors */
502       protected double[] m_coefficients;
503      
504       /**
505        * Computes the prediction from this support vector machine. For classification,
506        * it fills in the appropriate element in the preds array with either a 0 or a 1.
507        * If the output of the machine is < 0, then a 1 is entered into the array
508        * element corresponding to this machine's targetCategory, otherwise a 0 is entered.
509        * Note that this is different to the scoring procedure in the 3.2 spec (see the comments
510        * in the source code of this method for more information about this).
511        *
512        * @param input the test instance to predict
513        * @param kernel the kernel to use
514        * @param vecDict the vector dictionary (null if the machine is linear and expressed
515        * in terms of attribute weights)
516        * @param preds the prediction array to fill in
517        * @param cMethod the classification method to use (for classification problems)
518        * @param globalThreshold the global threshold (used if there is no local threshold
519        * for this machine)
520        * @throws Exception if something goes wrong
521        */
522       public void distributionForInstance(double[] input, Kernel kernel,
523           VectorDictionary vecDict, double[] preds, classificationMethod cMethod,
524           double globalThreshold)
525         throws Exception {
526         int targetIndex = 0;
527        
528         if (!m_coeffsOnly) {
529           // get an array that only holds the values that are referenced
530           // by the support vectors
531           input = vecDict.incomingInstanceToVectorFieldVals(input);
532         }
533        
534         if (m_miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
535           targetIndex = m_targetCategoryIndex;
536         }
537        
538         double result = 0;
539         for (int i = 0; i < m_coefficients.length; i++) {
540      // System.err.println("X : " + m_supportVectors.get(i).getValues());
541           double val = 0;
542           if (!m_coeffsOnly) {
543             val = kernel.evaluate(m_supportVectors.get(i), input);
544           } else {
545             val = input[i];
546           }
547           val *= m_coefficients[i];
548  // System.err.println("val " + val);
549           result += val;
550         }
551         result += m_intercept;
552        
553  /* if (result < 0 && m_miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
554             System.err.println("[SupportVectorMachine] result ("
555               + result + ") is less than zero" +
556           " for a nominal class value!");
557           result = 0;
558         } else if (result > 1 && m_miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
559           System.err.println("[SupportVectorMachine] result ("
560               + result + ") is greater than one" +
561                           " for a nominal class value!");
562           result = 1;
563         }
564         System.err.println("result " + result); */
565        
566         // TODO revisit this when I actually find out what is going on with
567         // the Zementis model (the only easily available SVM model out there
568         // at this time).
569        
570         if (cMethod == classificationMethod.NONE ||
571             m_miningSchema.getFieldsAsInstances().classAttribute().isNumeric()) {
572           // ----------------------------------------------------------------------
573           // The PMML 3.2 spec states that the output of the machine should lie
574           // between 0 and 1 for a binary classification case with 1 corresponding
575           // to the machine's targetCategory and 0 corresponding to the
576           // alternateBinaryTargetCategory (implying a threshold of 0.5). This seems kind of
577           // non standard, and indeed, in the 4.0 spec, it has changed to output < threshold
578           // corresponding to the machine's targetCategory and >= threshold corresponding
579           // to the alternateBinaryTargetCategory. What has been implemented here is
580           // the later with a default threshold of 0 (since the 3.2 spec doesn't have
581           // a way to specify a threshold). The example SVM PMML model from Zementis,
582           // which is PMML version 3.2, produces output between -1 and 1 (give or take).
583           // Implementing the 3.2 scoring mechanism as described in the spec (and truncating
584           // output at 0 and 1) results in all the predicted labels getting flipped on the data
585           // used to construct the Zementis model!
586           //
587           // April 2010 - the Zementis guys have emailed me to say that their model
588           // has been prepared for PMML 4.0, even though it states it is 3.2 in the
589           // XML.
590           // ----------------------------------------------------------------------
591  
592           if (m_miningSchema.getFieldsAsInstances().classAttribute().isNominal()) {
593             if (result < 0) {
594               preds[targetIndex] = 1;
595             } else {
596               preds[targetIndex] = 0;
597             }
598           } else {
599             preds[targetIndex] = result;
600           }
601         } else {
602           // PMML 4.0
603           if (cMethod == classificationMethod.ONE_AGAINST_ALL) {
604             // smallest value output by a machine is the predicted class
605             preds[targetIndex] = result;
606           } else {
607             // one-vs-one
608             double threshold = (m_localThreshold < Double.MAX_VALUE)
609               ? m_localThreshold
610               : globalThreshold;
611            
612             // vote
613             if (result < threshold) {
614               preds[targetIndex]++;
615             } else {
616               int altCat = (m_localAlternateTargetCategoryIndex != -1)
617                 ? m_localAlternateTargetCategoryIndex
618                 : m_globalAlternateTargetCategoryIndex;
619              
620               preds[altCat]++;
621             }
622           }
623         }
624  // preds[targetIndex] = result;
625       }
626          
627       /**
628        * Constructs a new SupportVectorMachine from the supplied XML element
629        *
630        * @param machineElement the XML element containing the SVM
631        * @param miningSchema the mining schema for the PMML model
632        * @param dictionary the VectorDictionary from which to look up the support
633        * vectors used by this machine (may be null if the machine is linear and
634        * expressed in terms of attribute weights)
635        * @param svmRep the representation of this SVM (uses support vectors or is linear
636        * an uses attribute weights)
637        * @param altCategoryInd the index of the global alternateBinaryTarget (if classification)
638        * @param log the log object to use
639        * @throws Exception if something goes wrong
