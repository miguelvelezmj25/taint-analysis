36       /**
37        * Reads a packed integer at the given buffer offset and returns it.
38        *
39        * @param buf the buffer to read from.
40        *
41        * @param off the offset in the buffer at which to start reading.
42        *
43        * @return the integer that was read.
44        */
45       public static int readInt(byte[] buf, int off) {
46  
47           boolean negative;
48           int byteLen;
49  
50           int b1 = buf[off++];
51           if (b1 < -119) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
52               negative = true;
53               byteLen = -b1 - 119;
54           } else if (b1 > 119) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
55               negative = false;
56               byteLen = b1 - 119;
57           } else {
58               return b1;
59           }
60  
61           int value = buf[off++] & 0xFF;
62           if (byteLen > 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
63               value |= (buf[off++] & 0xFF) << 8;
64               if (byteLen > 2) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
65                   value |= (buf[off++] & 0xFF) << 16;
66                   if (byteLen > 3) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
67                       value |= (buf[off++] & 0xFF) << 24;
68                   }
69               }
70           }
71  
72           return negative ? (-value - 119) : (value + 119); → [ALLOWCREATE, TRANSACTIONAL, READONLY]
73       }

75       /**
76        * Reads a packed long integer at the given buffer offset and returns it.
77        *
78        * @param buf the buffer to read from.
79        *
80        * @param off the offset in the buffer at which to start reading.
81        *
82        * @return the long integer that was read.
83        */
84       public static long readLong(byte[] buf, int off) {
85  
86           boolean negative;
87           int byteLen;
88  
89           int b1 = buf[off++];

91               negative = true;
92               byteLen = -b1 - 119;
93           } else if (b1 > 119) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
94               negative = false;
95               byteLen = b1 - 119;
96           } else {
97               return b1;
98           }
99  
100           long value = buf[off++] & 0xFFL;
101           if (byteLen > 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
102               value |= (buf[off++] & 0xFFL) << 8;
103               if (byteLen > 2) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
104                   value |= (buf[off++] & 0xFFL) << 16;
105                   if (byteLen > 3) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
106                       value |= (buf[off++] & 0xFFL) << 24;
107                       if (byteLen > 4) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
108                           value |= (buf[off++] & 0xFFL) << 32;
109                           if (byteLen > 5) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
110                               value |= (buf[off++] & 0xFFL) << 40;
111                               if (byteLen > 6) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
112                                   value |= (buf[off++] & 0xFFL) << 48;
113                                   if (byteLen > 7) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
114                                       value |= (buf[off++] & 0xFFL) << 56;
115                                   }
116                               }
117                           }
118                       }
119                   }
120               }
121           }
122  
123           return negative ? (-value - 119) : (value + 119); → [ALLOWCREATE, TRANSACTIONAL, READONLY]
124       }

175        *
176        * @param offset the offset in the buffer at which to start writing.

178        * @param value the integer to be written.
179        *
180        * @return the offset past the bytes written.
181        */
182       public static int writeInt(byte[] buf, int offset, int value) {
183  
184           int byte1Off = offset;
185           boolean negative;
186  
187           if (value < -119) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
188               negative = true;
189               value = -value - 119;
190           } else if (value > 119) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
191               negative = false;
192               value = value - 119;
193           } else {
194               buf[offset++] = (byte) value;
195               return offset;
196           }
197           offset++;
198  
199           buf[offset++] = (byte) value;
200           if ((value & 0xFFFFFF00) == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
201               buf[byte1Off] = negative ? (byte) -120 : (byte) 120; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
202               return offset;
203           }
204  
205           buf[offset++] = (byte) (value >>> 8);
206           if ((value & 0xFFFF0000) == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
207               buf[byte1Off] = negative ? (byte) -121 : (byte) 121; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
208               return offset;
209           }
210  
211           buf[offset++] = (byte) (value >>> 16);
212           if ((value & 0xFF000000) == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
213               buf[byte1Off] = negative ? (byte) -122 : (byte) 122; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
214               return offset;
215           }
216  
217           buf[offset++] = (byte) (value >>> 24);
218           buf[byte1Off] = negative ? (byte) -123 : (byte) 123; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
219           return offset;
220       }

370       /**
371        * Reads a sorted packed integer at the given buffer offset and returns it.
372        *
373        * @param buf the buffer to read from.
374        *
375        * @param off the offset in the buffer at which to start reading.
376        *
377        * @return the integer that was read.
378        */
379       public static int readSortedInt(byte[] buf, int off) {
380      
381           int byteLen;
382           boolean negative;
383          
384           /* The first byte of the buf stores the length of the value part. */
385           int b1 = buf[off++] & 0xff;
386           /* Adjust the byteLen to the real length of the value part. */
387           if (b1 < 0x08) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
388               byteLen = 0x08 - b1;
389               negative = true;
390           } else if (b1 > 0xf7) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
391               byteLen = b1 - 0xf7;
392               negative = false;
393           } else {
394               return b1 - 127;
395           }
396          
397           /*
398            * The following bytes on the buf store the value as a big endian
399            * integer. We extract the significant bytes from the buf and put them
400            * into the value in big endian order.
401            */
402           int value;
403           if (negative) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
404               value = 0xFFFFFFFF;
405           } else {
406               value = 0;
407           }
408           if (byteLen > 3) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
409               value = (value << 8) | (buf[off++] & 0xFF);
410           }
411           if (byteLen > 2) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
412               value = (value << 8) | (buf[off++] & 0xFF);
413           }
414           if (byteLen > 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
415               value = (value << 8) | (buf[off++] & 0xFF);
416           }
417           value = (value << 8) | (buf[off++] & 0xFF);
418          
419           /*
420            * After get the adjusted value, we have to adjust it back to the
421            * original value.
422            */
423           if (negative) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
424               value -= 119;
425           } else {
426               value += 121;
427           }
428           return value;
429       }

552       /**
553        * Writes a packed sorted integer starting at the given buffer offset and
554        * returns the next offset to be written.
555        *
556        * @param buf the buffer to write to.
557        *
558        * @param offset the offset in the buffer at which to start writing.
559        *
560        * @param value the integer to be written.
561        *
562        * @return the offset past the bytes written.
563        */
564       public static int writeSortedInt(byte[] buf, int offset, int value) {
565          
566           /*
567            * Values in the inclusive range [-119,120] are stored in a single
568            * byte. For values outside that range, the first byte stores the
569            * number of additional bytes. The additional bytes store
570            * (value + 119 for negative and value - 121 for positive) as an
571            * unsigned big endian integer.
572            */
573           int byte1Off = offset;
574           offset++;
575           if (value < -119) {
576          
577               /*
578                * If the value < -119, then first adjust the value by adding 119.
579                * Then the adjusted value is stored as an unsigned big endian
580                * integer.
581                */
582               value += 119;
583              
584               /*
