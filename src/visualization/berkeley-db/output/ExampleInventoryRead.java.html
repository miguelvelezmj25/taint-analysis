112       // Shows all the inventory items that exist for a given
113       // inventory name.
114       private void showItem() throws DatabaseException {
115  
116           // Use the inventory name secondary key to retrieve
117           // these objects.
118           EntityCursor<Inventory> items =
119                   da.inventoryByName.subIndex(locateItem).entities();
120           try {
121               for(Inventory item : items) {
122                   displayInventoryRecord(item);
123               }
124           } finally {
125               items.close();
126           }
127       }
