50  /**
51    * <!-- globalinfo-start --> Class for generating a PART decision list. Uses
52    * separate-and-conquer. Builds a partial C4.5 decision tree in each iteration
53    * and makes the "best" leaf into a rule.<br/>
54    * <br/>
55    * For more information, see:<br/>
56    * <br/>
57    * Eibe Frank, Ian H. Witten: Generating Accurate Rule Sets Without Global
58    * Optimization. In: Fifteenth International Conference on Machine Learning,
59    * 144-151, 1998.
60    * <p/>
61    * <!-- globalinfo-end -->
62    *
63    * <!-- technical-bibtex-start --> BibTeX:
64    *
65    * <pre>
66    * @inproceedings{Frank1998,
67    * author = {Eibe Frank and Ian H. Witten},
68    * booktitle = {Fifteenth International Conference on Machine Learning},
69    * editor = {J. Shavlik},
70    * pages = {144-151},
71    * publisher = {Morgan Kaufmann},
72    * title = {Generating Accurate Rule Sets Without Global Optimization},
73    * year = {1998},
74    * PS = {http://www.cs.waikato.ac.nz/\~eibe/pubs/ML98-57.ps.gz}
75    * }
76    * </pre>
77    * <p/>
78    * <!-- technical-bibtex-end -->
79    *
80    * <!-- options-start --> Valid options are:
81    * <p/>
82    *
83    * <pre>
84    * -C <pruning confidence>
85    * Set confidence threshold for pruning.
86    * (default 0.25)
87    * </pre>
88    *
89    * <pre>
90    * -M <minimum number of objects>
91    * Set minimum number of objects per leaf.
92    * (default 2)
93    * </pre>
94    *
95    * <pre>
96    * -R
97    * Use reduced error pruning.
98    * </pre>
99    *
100    * <pre>
101    * -N <number of folds>
102    * Set number of folds for reduced error
103    * pruning. One fold is used as pruning set.
104    * (default 3)
105    * </pre>
106    *
107    * <pre>
108    * -B
109    * Use binary splits only.
110    * </pre>
111    *
112    * <pre>
113    * -U
114    * Generate unpruned decision list.
115    * </pre>
116    *
117    * <pre>
118    * -J
119    * Do not use MDL correction for info gain on numeric attributes.
120    * </pre>
121    *
122    * <pre>
123    * -Q <seed>
124    * Seed for random data shuffling (default 1).
125    * </pre>
126    *
127    * <pre>
128    * -doNotMakeSplitPointActualValue
129    * Do not make split point actual value.
130    * </pre>
131    *
132    * <!-- options-end -->
133    *
134    * @author Eibe Frank (eibe@cs.waikato.ac.nz)
135    * @version $Revision: 11004 $
136    */

242     /**
243      * Generates the classifier.
244      *
245      * @param instances the data to train with
246      * @throws Exception if classifier can't be built successfully
247      */
248     @Override
249     public void buildClassifier(Instances instances) throws Exception {
250  
251       // can classifier handle the data?
252       getCapabilities().testWithFail(instances);
253  
254       // remove instances with missing class
255       instances = new Instances(instances);
256       instances.deleteWithMissingClass();
257  
258       ModelSelection modSelection;
259  
260       if (m_binarySplits) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
261         modSelection = new BinC45ModelSelection(m_minNumObj, instances,
262           m_useMDLcorrection, m_doNotMakeSplitPointActualValue);
263       } else {
264         modSelection = new C45ModelSelection(m_minNumObj, instances,
265           m_useMDLcorrection, m_doNotMakeSplitPointActualValue);
266       }
267       if (m_unpruned) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
268         m_root = new MakeDecList(modSelection, m_minNumObj);
269       } else if (m_reducedErrorPruning) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
270         m_root = new MakeDecList(modSelection, m_numFolds, m_minNumObj, m_Seed);
271       } else {
272         m_root = new MakeDecList(modSelection, m_CF, m_minNumObj);
273       }
274       m_root.buildClassifier(instances);
275       if (m_binarySplits) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
276         ((BinC45ModelSelection) modSelection).cleanup();
277       } else {
278         ((C45ModelSelection) modSelection).cleanup();
279       }
280     }
