25  /**
26    * A database cursor for a secondary database. Cursors are not thread safe and
27    * the application is responsible for coordinating any multithreaded access to
28    * a single cursor object.
29    *
30    * <p>Secondary cursors are returned by {@link SecondaryDatabase#openCursor
31    * SecondaryDatabase.openCursor} and {@link
32    * SecondaryDatabase#openSecondaryCursor
33    * SecondaryDatabase.openSecondaryCursor}. The distinguishing characteristics
34    * of a secondary cursor are:</p>
35    *
36    * <ul> <li>Direct calls to <code>put()</code> methods on a secondary cursor
37    * are prohibited.
38    *
39    * <li>The {@link #delete} method of a secondary cursor will delete the primary
40    * record and as well as all its associated secondary records.
41    *
42    * <li>Calls to all get methods will return the data from the associated
43    * primary database.
44    *
45    * <li>Additional get method signatures are provided to return the primary key
46    * in an additional pKey parameter.
47    *
48    * <li>Calls to {@link #dup} will return a {@link SecondaryCursor}.
49    *
50    * </ul>
51    *
52    * <p>To obtain a secondary cursor with default attributes:</p>
53    *
54    * <blockquote><pre>
55    * SecondaryCursor cursor = myDb.openSecondaryCursor(txn, null);
56    * </pre></blockquote>
57    *
58    * <p>To customize the attributes of a cursor, use a CursorConfig object.</p>
59    *
60    * <blockquote><pre>
61    * CursorConfig config = new CursorConfig();
62    * config.setReadUncommitted(true);
63    * SecondaryCursor cursor = myDb.openSecondaryCursor(txn, config);
64    * </pre></blockquote>
65    */

166       /**
167        * Delete the record to which the cursor refers from the primary database
168        * and all secondary indices.
169        *
170        * <p>This method behaves as if {@link Database#delete(Transaction,
171        * DatabaseEntry, WriteOptions)} were called for the primary database,
172        * using the primary key associated with this cursor position.</p>
173        *
174        * <p>The cursor position is unchanged after a delete, and subsequent calls
175        * to cursor functions expecting the cursor to refer to an existing record
176        * will fail.</p>
177        *
178        * <p>WARNING: Unlike read operations using a SecondaryCursor, write
179        * operations like this one are deadlock-prone.</p>
180        *
181        * <!-- inherit other javadoc from overridden method -->
182        */
183       @Override
184       public OperationResult delete(final WriteOptions options) {
185  
186           checkOpenAndState(true);
187  
188           trace(Level.FINEST, "SecondaryCursor.delete: ", null);
189  
190           final CacheMode cacheMode =
191               options != null ? options.getCacheMode() : null;
192  
193           /* Read the primary key (the data of a secondary). */
194           final DatabaseEntry key = new DatabaseEntry();
195           final DatabaseEntry pKey = new DatabaseEntry();
196  
197           /*
198            * Currently we write-lock the secondary before deleting the primary,
199            * which reverses the normal locking order and is deadlock-prone.
200            *
201            * FUTURE: To avoid deadlocks we could use dirty-read-all here, and
202            * then perform a special delete-if-has-secondary-key operation on the
203            * primary. We must be careful not to delete the primary record if,
204            * after locking it, it does not reference this secondary key.
205            */
206           final OperationResult secResult =
207               getCurrentInternal(key, pKey, LockMode.RMW, cacheMode);
208  
209           if (secResult == null) {
210               return null;
211           }
212  
213           final Locker locker = cursorImpl.getLocker();
214           final Database primaryDb = secondaryDb.getPrimary(pKey);
215  
216           if (primaryDb == null) {
217               /* Primary was removed from the association. */
218               deleteNoNotify(cacheMode, getDatabaseImpl().getRepContext());
219               return secResult;
220           }
221  
222           /* Delete the primary and all secondaries (including this one). */
223           final OperationResult priResult =
224               primaryDb.deleteInternal(locker, pKey, cacheMode);
225  
226           if (priResult != null) {
227               return priResult;
228           }
229  
230           /* The primary record may have expired after locking the secondary. */
231           if (cursorImpl.isProbablyExpired()) {
232               return null;
233           }
234  
235           throw secondaryDb.secondaryRefersToMissingPrimaryKey(
236               locker, key, pKey, secResult.getExpirationTime());
237       }

584                   }
585               }
