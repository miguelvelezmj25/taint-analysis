27  /**
28    * Class for handling an instance. All values (numeric, date, nominal, string or
29    * relational) are internally stored as floating-point numbers. If an attribute
30    * is nominal (or a string or relational), the stored value is the index of the
31    * corresponding nominal (or string or relational) value in the attribute's
32    * definition. We have chosen this approach in favor of a more elegant
33    * object-oriented approach because it is much faster.
34    * <p>
35    * <p>
36    * Typical usage (code from the main() method of this class):
37    * <p>
38    * <p>
39    * <code>
40    * ... <br>
41    * <p>
42    * // Create empty instance with three attribute values <br>
43    * Instance inst = new DenseInstance(3); <br><br>
44    * <p>
45    * // Set instance's values for the attributes "length", "weight", and "position"<br>
46    * inst.setValue(length, 5.3); <br>
47    * inst.setValue(weight, 300); <br>
48    * inst.setValue(position, "first"); <br><br>
49    * <p>
50    * // Set instance's dataset to be the dataset "race" <br>
51    * inst.setDataset(race); <br><br>
52    * <p>
53    * // Print the instance <br>
54    * System.out.println("The instance: " + inst); <br>
55    * <p>
56    * ... <br>
57    * </code>
58    * <p>
59    * <p>
60    * All methods that change an instance's attribute values are safe, ie. a change
61    * of an instance's attribute values does not affect any other instances. All
62    * methods that change an instance's attribute values clone the attribute value
63    * vector before it is changed. If your application heavily modifies instance
64    * values, it may be faster to create a new instance from scratch.
65    *
66    * @author Eibe Frank (eibe@cs.waikato.ac.nz)
67    * @version $Revision: 12472 $
68    */

76       /**
77        * Constructor that copies the attribute values and the weight from the given
78        * instance. It does NOT perform a deep copy of the attribute values if the
79        * instance provided is also of type DenseInstance (it simply copies the
80        * reference to the array of values), otherwise it does. Reference to the
81        * dataset is set to null. (ie. the instance doesn't have access to
82        * information about the attribute types)
83        *
84        * @param instance the instance from which the attribute values and the weight
85        * are to be copied
86        */
87       // @ ensures m_Dataset == null;
88       public DenseInstance(/* @non_null@ */Instance instance) {
89  
90           if(instance instanceof DenseInstance) { → [USEMDL, CONFIDENCEFACTOR, REDUCEDERRORPRUNING, BATCHSIZE, BINARYSPLITS, DONOTMAKESPLITPOINTACTUALVALUE, NUMDECIMAL, SEED, MINNUM, UNPRUNED, FOLDS, NOCHECKCAPABILITIES]
91               m_AttValues = ((DenseInstance) instance).m_AttValues;
92           }
93           else {
94               m_AttValues = instance.toDoubleArray();
95           }
96           m_Weight = instance.weight();
97           m_Dataset = null;
98       }

103        * access to information about the attribute types)
104        *
105        * @param weight the instance's weight
106        * @param attValues a vector of attribute values
107        */
108       // @ ensures m_Dataset == null;
109       public DenseInstance(double weight, /* @non_null@ */double[] attValues) {
110  
111           m_AttValues = attValues;
112           m_Weight = weight;
113           m_Dataset = null;
114       }

116       /**
117        * Constructor of an instance that sets weight to one, all values to be
118        * missing, and the reference to the dataset to null. (ie. the instance
119        * doesn't have access to information about the attribute types)
120        *
121        * @param numAttributes the size of the instance
122        */
123       // @ requires numAttributes > 0; // Or maybe == 0 is okay too?
124       // @ ensures m_Dataset == null;
125       public DenseInstance(int numAttributes) {
126  
127           m_AttValues = new double[numAttributes];
128           for(int i = 0; i < m_AttValues.length; i++) {
129               m_AttValues[i] = Utils.missingValue();
130           }
131           m_Weight = 1;
132           m_Dataset = null;
133       }

325       /**
326        * Merges this instance with the given instance and returns the result.
327        * Dataset is set to null. The returned instance is of the same type as this
328        * instance.
329        *
330        * @param inst the instance to be merged with this one
331        * @return the merged instances
332        */
333       @Override
334       public Instance mergeInstance(Instance inst) {
335  
336           int m = 0;
337           double[] newVals = new double[numAttributes() + inst.numAttributes()];
338           for(int j = 0; j < numAttributes(); j++, m++) {
339               newVals[m] = value(j);
340           }
341           for(int j = 0; j < inst.numAttributes(); j++, m++) {
342               newVals[m] = inst.value(j);
343           }
344           return new DenseInstance(1.0, newVals);
345       }

371       /**
372        * Replaces all missing values in the instance with the values contained in
373        * the given array. A deep copy of the vector of attribute values is performed
374        * before the values are replaced.
375        *
376        * @param array containing the means and modes
377        * @throws IllegalArgumentException if numbers of attributes are unequal
378        */
379       @Override
380       public void replaceMissingValues(double[] array) {
381  
382           if((array == null) || (array.length != m_AttValues.length)) {
383               throw new IllegalArgumentException("Unequal number of attributes!");
384           }
385           freshAttributeVector();
386           for(int i = 0; i < m_AttValues.length; i++) {
387               if(isMissing(i)) {
388                   m_AttValues[i] = array[i];
389               }
390           }
391       }
