21  /*
22    * Given a locker T and a record R, a WriteLockInfo stores the info needed to
23    * undo the write ops (insertions, deletions, or updates) performed by T on
24    * R, if T aborts. Specifically, it stores the info needed to restore R to its
25    * "abort" version, i.e., the version of R that existed at the time when T
26    * locked R for the 1st time (and before T actually performed its 1st write
27    * op on R). Given that only transactional lockers may abort, WriteLockInfo
28    * are used by Txn lockers only.
29    *
30    * Notice that, conceptually, to write R, T locks R in exclusive mode and
31    * retains that lock until it terminates (commits or aborts). In other words,
32    * T locks R only once. However, because locks are acquired on LSNs and LSNs
33    * change every time a record is written, T may actually lock R multiple times,
34    * Everytime T locks a new LSN of R, it creates a new WriteLockIno. However,
35    * care is taken so that all WriteLockInfos for the same logical record and
36    * the same txn store the same info about the abort version (for details see
37    * CursorImpl.LockStanding, CursorImpl.lockLN(), and LN.logInternals()). In
38    * fact, if T writes R multiple times and then aborts, only one of the logrecs
39    * generated by T for R will be undone, restoring R to its abort version; the
40    * rest will be skipped.
41    *
42    * Info about the aort version is needed during commit as well. Specifically,
43    * its LSN and on-disk size are needed to count the abort version obsolete
44    * when T commits.
45    */
46  public class WriteLockInfo {
47  
48       /*
49        * The LSN of the record's abort version. This is stored persistently in
50        * each logrec renerated by T on R. May be null if R was created by T
51        * (will definitely be null if the txn did not reuse an existing slot for
52        * the new record).
53        */
54       private long abortLsn;
55  
56       /*
57        * Whether the record's abort version is a deletion version or not.
58        * It is stored persistently in each logrec renerated by T on R.
59        */
60       private boolean abortKnownDeleted;
61  
62       /*
63        * See comment for abortData field below.
64        */
65       private byte[] abortKey;
66  
67       /*
68        * If the record's abort version was embedded in a BIN, the associated
69        * logrec that contains that version may have been cleaned away by the
70        * time the txn aborts. So, we must save in each logrec the data portion
71        * of the abort version. abortdata serves this purpose. If key updates
72        * are allowed in the containing DB, the key of the abort version is
73        * saved in abortKey as well. Finally, if VLSN caching in BINs is enabled,
74        * the VLSN of the abort version is saved in abortVLSN as well.
75        */
76       private byte[] abortData;
77  
78       /*
79        * See comment for abortData field above.
80        */
81       private long abortVLSN = VLSN.NULL_VLSN_SEQUENCE;
82  
83       /*
84        * The on-disk size of the abort version, or zero if abortLsn is NULL_LSN
85        * or if the size is not known. Used for obsolete counting during commit.
86        * Not stored persistently.
87        */
88       private int abortLogSize;
89  
90       /* Abort expiration time. Is negative if in hours, positive if in days. */
91       private int abortExpiration;
92  
93       /*
94        * The containing database, or null if abortLsn is NULL_LSN. Used for
95        * obsolete counting during a commit.
96        */
97       private DatabaseImpl db;
98  
99       /*
100        * True if the LSN has never been locked before by this Txn. Used so we
101        * can determine when to set abortLsn.
102        */
103       private boolean neverLocked;
104  
105       static final WriteLockInfo basicWriteLockInfo = new WriteLockInfo();
106  
107       // public for Sizeof
108       public WriteLockInfo() {
109           abortLsn = DbLsn.NULL_LSN;
110           abortKnownDeleted = false;
111           neverLocked = true;
112       }
113  
114       public boolean getAbortKnownDeleted() {
115           return abortKnownDeleted;
116       }
117  
118       public void setAbortKnownDeleted(boolean v) {
119           abortKnownDeleted = v;
120       }
121  
122       public long getAbortLsn() {
123           return abortLsn;
124       }
125  
126       public void setAbortLsn(long abortLsn) {
127           this.abortLsn = abortLsn;
128       }
129  
130       public byte[] getAbortKey() {
131           return abortKey;
132       }
133  
134       public void setAbortKey(byte[] v) {
135           abortKey = v;
136       }
137  
138       public byte[] getAbortData() {
139           return abortData;
140       }
141  
142       public void setAbortData(byte[] v) {
143           abortData = v;
144       }
145  
146      public long getAbortVLSN() {
147           return abortVLSN;
148       }
149  
150      public void setAbortVLSN(long v) {
151           abortVLSN = v;
152       }
153  
154       public int getAbortLogSize() {
155           return abortLogSize;
156       }
157  
158       public void setAbortLogSize(int logSize) {
159           abortLogSize = logSize;
160       }
161  
162       public void setAbortExpiration(int expiration, boolean expirationInHours) {
163           abortExpiration = expirationInHours ? (-expiration) : expiration; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
164       }
165  
166       public int getAbortExpiration() {
167           return Math.abs(abortExpiration);
168       }
169  
170       public boolean isAbortExpirationInHours() {
171           return abortExpiration < 0; → [ALLOWCREATE, TRANSACTIONAL, READONLY]
172       }
173  
174       public DatabaseImpl getDb() {
175           return db;
176       }
177  
178       public void setDb(DatabaseImpl db) {
179           this.db = db;
180       }
181  
182       public boolean getNeverLocked() {
183           return neverLocked;
184       }
185  
186       public void setNeverLocked(boolean neverLocked) {
187           this.neverLocked = neverLocked;
188       }
189  
190       /*
191        * Copy all the information needed to create a clone of the lock.
192        */
193       public void copyAllInfo(WriteLockInfo source) {
194           abortLsn = source.abortLsn;
195           abortKnownDeleted = source.abortKnownDeleted;
196           abortKey = source.abortKey;
197           abortData = source.abortData;
198           abortVLSN = source.abortVLSN;
199           abortLogSize = source.abortLogSize;
200           abortExpiration = source.abortExpiration;
201           db = source.db;
202           neverLocked = source.neverLocked;
203       }
204  
205       @Override
206       public String toString() {
207           return "abortLsn=" +
208               DbLsn.getNoFormatString(abortLsn) +
209               " abortKnownDeleted=" + abortKnownDeleted +
210               " abortKey=" + Key.getNoFormatString(abortKey) +
211               " abortData=" + Key.getNoFormatString(abortData) +
212               " abortLogSize=" + abortLogSize +
213               " abortVLSN=" + String.format("%,d", abortVLSN) +
214               " abortExpiration=" + getAbortExpiration() +
215               " abortExpirationInHours=" + isAbortExpirationInHours() +
216               " neverLocked=" + neverLocked;
217       }
