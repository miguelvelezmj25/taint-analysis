20  /**
21    * Holds an array of non-negative long values, one for each slot in an IN.
22    *
23    * Zero is the default value and is returned when no value has been set.
24    *
25    * The EMPTY_REP is used at first, and is mutated as necessary as values are
26    * set. A minimum number of bytes per value is used, based on the largest
27    * value passed to set().
28    *
29    * Optionally, a sparse rep is used when a value is set for EMPTY_REP. Up to 4
30    * values are stored along with their indexes. When the 5th values is set, the
31    * rep is mutated to the default rep.
32    *
33    * This object calls IN.updateMemorySize to track the memory it uses.
34    * EMPTY_REP uses no memory because it is a singleton.
35    */
36  public abstract class INLongRep {
37  
38       public abstract long get(int idx);
39       public abstract INLongRep set(int idx, long val, IN parent);
40       public abstract INLongRep compact(IN parent, EmptyRep emptyRep);
41       public abstract INLongRep clear(IN parent, EmptyRep emptyRep);
42       public abstract boolean isEmpty();
43       public abstract INLongRep copy(int from, int to, int n, IN parent);
44       public abstract INLongRep resize(int capacity);
45       public abstract long getMemorySize();
46  
47       /**
48        * Initially empty (all values are zero) but will mutate as needed when
49        * non-zero values are passed to set().
50        */
51       public static class EmptyRep extends INLongRep {
52  
53           final int minLength;
54           final boolean allowSparseRep;
55  
56           public EmptyRep(int minLength, boolean allowSparseRep) {
57               this.minLength = minLength;
58               this.allowSparseRep = allowSparseRep;
59           }
60  
61           @Override
62           public INLongRep resize(int capacity) {
63               return this;
64           }
65  
66           @Override
67           public long get(int idx) {
68               return 0;
69           }
70  
71           /**
72            * When adding to the cache the EMPTY_REP is mutated into a
73            * DefaultRep.
74            */
75           @Override
76           public INLongRep set(int idx, long val, IN parent) {
77  
78               if (val == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
79                   return this;
80               }
81  
82               final INLongRep newCache;
83  
84               if (false /*TODO*/ && allowSparseRep) {
85                   newCache = new SparseRep(minLength);
86               } else {
87                   newCache = new DefaultRep(parent.getMaxEntries(), minLength);
88               }
89  
90               parent.updateMemorySize(getMemorySize(), newCache.getMemorySize());
91  
92               return newCache.set(idx, val, parent);
93           }
94  
95           @Override
96           public INLongRep compact(IN parent, EmptyRep emptyRep) {
97               return this;
98           }
99  
100           @Override
101           public INLongRep clear(IN parent, EmptyRep emptyRep) {
102               return this;
103           }
104  
105           @Override
106           public boolean isEmpty() {
107               return true;
108           }
109  
110           @Override
111           public INLongRep copy(int from, int to, int n, IN parent) {
112               return this;
113           }

115           /**
116            * An EMPTY_REP has no JE cache memory overhead because there is only
117            * one global instance.
118            */
119           @Override
120           public long getMemorySize() {
121               return 0;
122           }
123       }

125       public static class DefaultRep extends INLongRep {
126  
127           /** Maximum value indexed by number of bytes. */
128           private static long[] MAX_VALUE = {
129               0x0L,
130               0xFFL,
131               0xFFFFL,
132               0xFFFFFFL,
133               0xFFFFFFFFL,
134               0xFFFFFFFFFFL,
135               0xFFFFFFFFFFFFL,
136               0xFFFFFFFFFFFFFFL,
137               0x7FFFFFFFFFFFFFFFL,
138           };
139  
140           private final byte[] byteArray;
141           final int bytesPerValue;
142  
143           public DefaultRep(int capacity, int nBytes) {
144               assert capacity >= 1; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
145               assert nBytes >= 1; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
146               assert nBytes <= 8; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
147  
148               bytesPerValue = nBytes;
149               byteArray = new byte[capacity * bytesPerValue];
150           }
151  
152           /* Only for use by the Sizeof utility. */
153           public DefaultRep(@SuppressWarnings("unused") SizeofMarker marker) {
154               bytesPerValue = 0;
155               byteArray = null;
156           }
157  
158           private DefaultRep(byte[] byteArray, int bytesPerValue) {
159               this.byteArray = byteArray;
160               this.bytesPerValue = bytesPerValue;
161           }
162  
163           @Override
164           public DefaultRep resize(int capacity) {
165               return new DefaultRep(
166                   Arrays.copyOfRange(byteArray, 0, capacity * bytesPerValue),
167                   bytesPerValue);
168           }
169  
170           @Override
171           public long get(int idx) {
172  
173               int i = idx * bytesPerValue;
174               final int end = i + bytesPerValue;
175  
176               long val = (byteArray[i] & 0xFF);
177  
178               for (i += 1; i < end; i += 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
179                   val <<= 8;
180                   val |= (byteArray[i] & 0xFF);
181               }
182  
183               return val;
184           }
185  
186           /**
187            * Mutates to a DefaultRep with a larger number of bytes if necessary
188            * to hold the given value.
189            */
190           @Override
191           public INLongRep set(int idx, long val, IN parent) {
192  
193               assert idx >= 0; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
194               assert idx < byteArray.length / bytesPerValue; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
195               assert val >= 0; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
196  
197               /*
198                * If the value can't be represented using bytesPerValue, mutate
199                * to a cache with a larger number of bytes.
200                */
201               if (val > MAX_VALUE[bytesPerValue]) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
202  
203                   final int capacity = byteArray.length / bytesPerValue;
204  
205                   INLongRep newRep;
206  
207                   if (getClass() == SparseRep.class) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
208                       newRep = new SparseRep(bytesPerValue + 1);
209                   } else {
210                       newRep = new DefaultRep(capacity, bytesPerValue + 1);
211                   }
212  
213                   parent.updateMemorySize(
214                       getMemorySize(), newRep.getMemorySize());
215  
216                   /*
217                    * Set new value in new cache, and copy other values from old
218                    * cache.
219                    */
220                   newRep = newRep.set(idx, val, parent);
221  
222                   for (int i = 0; i < capacity; i += 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
223                       if (i != idx) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
224                           newRep = newRep.set(i, get(i), parent);
225                       }
226                   }
227  
228                   return newRep;
229               }
230  
231               /* Set value in this cache. */
232               int i = ((idx + 1) * bytesPerValue) - 1;
233               final int end = i - bytesPerValue;
234  
235               byteArray[i] = (byte) (val & 0xFF);
236  
237               for (i -= 1; i > end; i -= 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
238                   val >>= 8;
239                   byteArray[i] = (byte) (val & 0xFF);
240               }
241  
242               assert ((val & 0xFFFFFFFFFFFFFF00L) == 0) : val; → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
243  
244               return this;
245           }
246  
247           @Override
248           public INLongRep compact(IN parent, EmptyRep emptyRep) {
249  
250               if (isEmpty()) { → [ALLOWCREATE]
251                   return clear(parent, emptyRep);
252               }
253  
254               return this;
255           }
256  
257           @Override
258           public INLongRep clear(IN parent, EmptyRep emptyRep) {
259  
260               parent.updateMemorySize(
261                   getMemorySize(), emptyRep.getMemorySize());
262  
263               return emptyRep;
264           }
265  
266           @Override
267           public boolean isEmpty() {
268  
269               for (byte b : byteArray) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
270                   if (b != 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
271                       return false;
272                   }
273               }
274  
275               return true;
276           }
277  
278           @Override
279           public INLongRep copy(int from, int to, int n, IN parent) {
280               System.arraycopy(byteArray,
281                                from * bytesPerValue,
282                                byteArray,
283                                to * bytesPerValue,
284                                n * bytesPerValue);
285               return this;
286           }
287  
288           @Override
289           public long getMemorySize() {
290               return MemoryBudget.DEFAULT_LONG_REP_OVERHEAD +
291                      MemoryBudget.byteArraySize(byteArray.length);
292           }
293       }

295       public static class SparseRep extends DefaultRep {
296  
297           private static final int MAX_ENTRIES = 4;
298  
299           private final short[] idxs;
300  
301           public SparseRep(int nBytes) {
302  
303               super(MAX_ENTRIES, nBytes);
304  
305               idxs = new short[MAX_ENTRIES];
306               Arrays.fill(idxs, (short) (-1));
307           }
308  
309           /* Only for use by the Sizeof utility. */
310           public SparseRep(@SuppressWarnings("unused") SizeofMarker marker) {
311               super(marker);
312               idxs = null;
313           }
314  
315           @Override
316           public SparseRep resize(int capacity) {
317               return this;
318           }
319  
320           @Override
321           public long get(int idx) {
322  
323               for (int i = 0; i < idxs.length; i += 1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
324                   if (idxs[i] == idx) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
325                       return super.get(i);
326                   }
327               }
328  
329               return 0;
330           }
331  
332           @Override
333           public INLongRep set(int idx, long val, IN parent) {
334  
335               int slot = -1;
336  
337               for (int i = 0; i < idxs.length; i++) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
338  
339                   if (idxs[i] == idx) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
340                       if (val == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
341                           idxs[i] = -1;
342                       }
343                       return super.set(i, val, parent);
344                   }
345  
346                   if (slot < 0 && idxs[i] == -1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY] & [ALLOWCREATE, TRANSACTIONAL, READONLY]
347                       slot = i;
348                   }
349               }
350  
351               if (val == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
352                   return this;
353               }
354  
355               /* Have a free slot, use it. */
356               if (slot >= 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
357                   idxs[slot] = (short) idx;
358                   return super.set(slot, val, parent);
359               }
360  
361               /* It's full, mutate it. */
362               INLongRep newRep =
363                   new DefaultRep(parent.getMaxEntries(), bytesPerValue);
364  
365               parent.updateMemorySize(getMemorySize(), newRep.getMemorySize());
366  
367               for (int i = 0; i < idxs.length; i++) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
368                   if (idxs[i] != -1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
369                       newRep = newRep.set(idxs[i], super.get(i), parent);
370                   }
371               }
372  
373               return newRep.set(idx, val, parent);
374           }
375  
376           @Override
377           public INLongRep compact(IN parent, EmptyRep emptyRep) {
378  
379               if (isEmpty()) { → [ALLOWCREATE]
380                   return clear(parent, emptyRep);
381               }
382  
383               return this;
384           }
385  
386           @Override
387           public INLongRep clear(IN parent, EmptyRep emptyRep) {
388  
389               parent.updateMemorySize(
390                   getMemorySize(), emptyRep.getMemorySize());
391  
392               return emptyRep;
393           }
394  
395           @Override
396           public boolean isEmpty() {
397  
398               for (short idx : idxs) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
399                   if (idx != -1) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
400                       return false;
401                   }
402               }
403  
404               return true;
405           }
406  
407           @Override
408           public INLongRep copy(int from, int to, int n, IN parent) {
409  
410               INLongRep target = this;
411  
412               if ((to == from) || (n == 0)) { → [ALLOWCREATE, READONLY] & [ALLOWCREATE, READONLY]
413                   /* Nothing to do */
414               } else if (to < from) { → [ALLOWCREATE, READONLY]
415                   /* Copy ascending */
416                   for (int i = 0; i < n; i++) { → [ALLOWCREATE, READONLY]
417                       target = target.set(to++, get(from++), parent);
418                   }
419               } else {
420                   /* to > from. Copy descending */
421                   from += n;
422                   to += n;
423                   for (int i = 0; i < n; i++) { → [ALLOWCREATE, READONLY]
424                       target = target.set(--to, get(--from), parent);
425                   }
426               }
427  
428               return target;
429           }
430  
431           @Override
432           public long getMemorySize() {
433               return super.getMemorySize() +
434                   MemoryBudget.SPARSE_LONG_REP_OVERHEAD -
435                   MemoryBudget.DEFAULT_KEYVALS_OVERHEAD +
436                   MemoryBudget.shortArraySize(idxs.length);
437           }
438       }
