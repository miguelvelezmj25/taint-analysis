54       public static File createAndCheckFolder(String path, boolean dryRun) {
55           File f = new File(path);
56  
57           if (dryRun) {
58               return f;
59           }
60  
61           if (!f.exists()) {
62               f.mkdirs();
63           }
64  
65           if (!f.exists() || !f.isDirectory()) {
66               throw new IllegalStateException("could not create folder: " + path);
67           }

122               File[] files = folder.listFiles();
