134       /**
135        * Splits the given string into lines and returns each line as a separate
136        * string in the result list. The result list will contain at least one
137        * entry unless the string is empty.
138        * <p>
139        * A trailing newline after the last line is ignored, meaning that no empty
140        * string is appended as separate line if the string ends with a newline.
141        * However multiple trailing newlines will still lead to empty line strings
142        * at the end of the list.
143        * <p>
144        * Note that all line ending characters are accepted to split lines, no
145        * matter what operating system this code is hosted on. More precisely, the

147        * line ending characters, either as single character or as a pair
148        * {@code CR+LF} or {@code LF+CR}.
149        *
150        * @param s the string to split
151        * @return a list with the lines found in {@code s}
152        */
153       public static final List<Line> splitLines(String s) {
154           final List<Line> lines = new ArrayList<Line>();
155           int start = 0;
156           int index = 0;
157           while(index < s.length()) {
158               final char ch = s.charAt(index);
159               if(ch == Line.LF || ch == Line.CR) {
160                   final int lineEndingStart = index;
161                   index++;
162                   if(index < s.length()) {
163                       final char ch2 = s.charAt(index);
164                       if(ch2 != ch && (ch2 == Line.LF || ch2 == Line.CR)) {
165                           index++;
166                       }
167                   }
168                   final Line line = new SingleCharSequenceLine(s, start,
169                           lineEndingStart - start, index - lineEndingStart);
170                   lines.add(line);
171                   start = index;
172               }
173               else {
174                   index++;
175               }
176           }
177           if(start < s.length()) {
178               final Line line = new SingleCharSequenceLine(s, start, s.length()
179                       - start, 0);
180               lines.add(line);
181           }
182           return lines;
183       }
