183       /**
184        * Sets the compression options to the given preset.
185        *

186        * The presets 0-3 are fast presets with medium compression.
187        * The presets 4-6 are fairly slow presets with high compression.
188        * The default preset (PRESET_DEFAULT) is 6.
189        *

190        * The presets 7-9 are like the preset 6 but use bigger dictionaries
191        * and have higher compressor and decompressor memory requirements.
192        * Unless the uncompressed size of the file exceeds 8 MiB,
193        * 16 MiB, or 32 MiB, it is waste of memory to use the
194        * presets 7, 8, or 9, respectively.
195        *
196        * @throws UnsupportedOptionsException preset is not supported
197        */
198       public void setPreset(int preset) throws UnsupportedOptionsException {
199           if(preset < 0 || preset > 9) {  → 8
200               throw new UnsupportedOptionsException(
201                       "Unsupported preset: " + preset);
202           }
203  
204           lc = LC_DEFAULT;
205           lp = LP_DEFAULT;
206           pb = PB_DEFAULT;
207           dictSize = presetToDictSize[preset];
208  
209           if(preset <= 3) {
210               mode = MODE_FAST;
211               mf = MF_HC4;
212               niceLen = preset <= 1 ? 128 : NICE_LEN_MAX;  → 9
213               depthLimit = presetToDepthLimit[preset];
214           }
215           else {
216               mode = MODE_NORMAL;
217               mf = MF_BT4;
218               niceLen = (preset == 4) ? 16 : (preset == 5) ? 32 : 64;
219               depthLimit = 0;
220           }
221       }

230       /**
231        * Sets the dictionary size in bytes.
232        *

233        * The dictionary (or history buffer) holds the most recently seen
234        * uncompressed data. Bigger dictionary usually means better compression.
235        * However, using a dictioanary bigger than the size of the uncompressed
236        * data is waste of memory.
237        *

238        * Any value in the range [DICT_SIZE_MIN, DICT_SIZE_MAX] is valid,
239        * but sizes of 2^n and 2^n + 2^(n-1) bytes are somewhat
240        * recommended.
241        *
242        * @throws UnsupportedOptionsException dictSize is not supported
243        */
244       public void setDictSize(int dictSize) throws UnsupportedOptionsException {
245           if(dictSize < DICT_SIZE_MIN) {
246               throw new UnsupportedOptionsException(
247                       "LZMA2 dictionary size must be at least 4 KiB: "
248                               + dictSize + " B");
249           }
250  
251           if(dictSize > DICT_SIZE_MAX) {
252               throw new UnsupportedOptionsException(
253                       "LZMA2 dictionary size must not exceed "
254                               + (DICT_SIZE_MAX >> 20) + " MiB: " + dictSize + " B");
255           }
256  
257           this.dictSize = dictSize;
258       }

305       }

314       /**
315        * Sets the number of literal context bits.
316        *

317        * All bytes that cannot be encoded as matches are encoded as literals.
318        * That is, literals are simply 8-bit bytes that are encoded one at
319        * a time.
320        *

321        * The literal coding makes an assumption that the highest lc
322        * bits of the previous uncompressed byte correlate with the next byte.
323        * For example, in typical English text, an upper-case letter is often
324        * followed by a lower-case letter, and a lower-case letter is usually
325        * followed by another lower-case letter. In the US-ASCII character set,
326        * the highest three bits are 010 for upper-case letters and 011 for
327        * lower-case letters. When lc is at least 3, the literal
328        * coding can take advantage of this property in the uncompressed data.
329        *

330        * The default value (3) is usually good. If you want maximum compression,
331        * try setLc(4). Sometimes it helps a little, and sometimes it
332        * makes compression worse. If it makes it worse, test for example
333        * setLc(2) too.
334        *
335        * @throws UnsupportedOptionsException lc is invalid, or the sum
336        * of lc and lp
337        * exceed LC_LP_MAX
338        */
339       public void setLc(int lc) throws UnsupportedOptionsException {
340           setLcLp(lc, lp);
341       }

372       /**
373        * Sets the number of position bits.
374        *

375        * This affects what kind of alignment in the uncompressed data is
376        * assumed in general. The default (2) means four-byte alignment
377        * (2^pb = 2^2 = 4), which is often a good choice when
378        * there's no better guess.
379        *

380        * When the alignment is known, setting the number of position bits
381        * accordingly may reduce the file size a little. For example with text
382        * files having one-byte alignment (US-ASCII, ISO-8859-*, UTF-8), using
383        * setPb(0) can improve compression slightly. For UTF-16
384        * text, setPb(1) is a good choice. If the alignment is
385        * an odd number like 3 bytes, setPb(0) might be the best
386        * choice.
387        *

388        * Even though the assumed alignment can be adjusted with
389        * setPb and setLp, LZMA2 still slightly favors
390        * 16-byte alignment. It might be worth taking into account when designing
391        * file formats that are likely to be often compressed with LZMA2.
392        *
393        * @throws UnsupportedOptionsException pb is invalid
394        */
395       public void setPb(int pb) throws UnsupportedOptionsException {
396           if(pb < 0 || pb > PB_MAX) {
397               throw new UnsupportedOptionsException(
398                       "pb must not exceed " + PB_MAX + ": " + pb);
399           }
400  
401           this.pb = pb;
402       }

411       /**
412        * Sets the compression mode.
413        *

414        * This specifies the method to analyze the data produced by
415        * a match finder. The default is MODE_FAST for presets
416        * 0-3 and MODE_NORMAL for presets 4-9.

426        * @throws UnsupportedOptionsException mode is not supported
427        */

429           if(mode < MODE_UNCOMPRESSED || mode > MODE_NORMAL) {
430               throw new UnsupportedOptionsException(
431                       "Unsupported compression mode: " + mode);
432           }
433  
434           this.mode = mode;
435       }

541       /**
542        * Gets how much memory the LZMA2 decoder will need to decompress the data
543        * that was encoded with these options and stored in a .xz file.
544        *

545        * The returned value may bigger than the value returned by a direct call
546        * to {@link LZMA2InputStream#getMemoryUsage(int)} if the dictionary size
547        * is not 2^n or 2^n + 2^(n-1) bytes. This is because the .xz
548        * headers store the dictionary size in such a format and other values
549        * are rounded up to the next such value. Such rounding is harmess except
550        * it might waste some memory if an unsual dictionary size is used.
551        *

552        * If you use raw LZMA2 streams and unusual dictioanary size, call
553        * {@link LZMA2InputStream#getMemoryUsage} directly to get raw decoder
554        * memory requirements.
555        */
556       public int getDecoderMemoryUsage() {
557           // Round the dictionary size up to the next 2^n or 2^n + 2^(n-1).
558           int d = dictSize - 1;
559           d |= d >>> 2;
560           d |= d >>> 3;
561           d |= d >>> 4;
562           d |= d >>> 8;
563           d |= d >>> 16;
564           return LZMA2InputStream.getMemoryUsage(d + 1);
565       }
