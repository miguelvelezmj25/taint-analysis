22  /**
23    * Utility functions used by the trigger implementation.
24    *
25    */
26  public class TriggerUtils {
27  
28       /**
29        * Determines the size in bytes used to represent the trigger in the log,
30        * that is, the size of the output generated by
31        * {@link #writeTriggers(ByteBuffer, byte[][])}
32        *
33        * @param triggerBytes the triggers whose size is to be estimated.
34        *
35        * @return the size in bytes
36        */
37       static int logSize(byte[][] triggerBytes) {
38  
39           if (triggerBytes == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
40               return LogUtils.getPackedIntLogSize(0);
41           }
42  
43           /* Add up the individual trigger sizes */
44           int size = LogUtils.getPackedIntLogSize(triggerBytes.length);
45           for (byte[] trigger : triggerBytes) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
46               size += LogUtils.getByteArrayLogSize(trigger);
47           }
48           return size;
49       }
50  
51       /**
52        * Writes the triggers out to the log buffer.
53        *
54        * @param logBuffer the buffer in which the bytes are assembled.
55        *
56        * @param triggerBytes the trigger bytes to be written.
57        */
58       static void writeTriggers(ByteBuffer logBuffer,
59                                 byte[][] triggerBytes) {
60           if (triggerBytes == null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
61               LogUtils.writePackedInt(logBuffer, 0);
62           } else {
63               /* Write out the triggers. */
64               LogUtils.writePackedInt(logBuffer, triggerBytes.length);
65               for (byte[] triggerByte : triggerBytes) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
66                   LogUtils.writeByteArray(logBuffer, triggerByte);
67               }
68           }
69       }
70  
71       /**
72        * Reads the triggers from a log buffer and returns then in their
73        * serialized byte array form.
74        *
75        * @param logBuffer the buffer from which to read the triggers.
76        * @param entryVersion the version associated with the current log
77        *
78        * @return the trigger bytes
79        */
80       static byte[][] readTriggers(ByteBuffer logBuffer,
81                                    int entryVersion) {
82  
83           final int triggerCount = LogUtils.readPackedInt(logBuffer);
84           if (triggerCount == 0) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
85               return null;
86           }
87  
88           byte[][] triggerBytes = new byte[triggerCount][];
89           for (int i = 0; i < triggerBytes.length; i++) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
90               triggerBytes[i] =
91                   LogUtils.readByteArray(logBuffer, false /* unpacked */);
92           }
93           return triggerBytes;
94       }
95  
96       /**
97        * Deserializes the trigger representation to yield the trigger object
98        * instance.
99        *
100        * @param dbName the name to be associated with the de-serialized triggers
101        *
102        * @param triggerBytes the serialized representation of the trigger
103        *
104        * @return the list of trigger instances
105        */
106       static LinkedList<Trigger> unmarshallTriggers(String dbName,
107                                                     byte[][] triggerBytes,
108                                                     ClassLoader loader) {
109  
110           if (triggerBytes == null) { → [ALLOWCREATE, READONLY]
111               return null;
112           }
113  
114           final LinkedList<Trigger> triggers = new LinkedList<Trigger>();
115           for (int i = 0; i < triggerBytes.length; i++) { → [ALLOWCREATE, READONLY]
116               final Trigger trigger =
117                   (Trigger)DatabaseImpl.bytesToObject(triggerBytes[i],
118                                                       "trigger:" + i,
119                                                       loader);
120               trigger.setDatabaseName(dbName);
121               triggers.add(trigger);
122           }
123           return triggers;
124       }
125  
126       /**
127        * Dumps an XML representation of the triggers into the StringBuilder. It
128        * gives preference to the instance representation if it's readily
129        * available.
130        *
131        * @param sb the string buffer that will contain the XML representation
132        * @param triggerBytes the bytes representing the trigger
133        * @param triggers the trigger instances corresponding to triggerBytes
134        */
135       static void dumpTriggers(StringBuilder sb,
136                               byte[][] triggerBytes,
137                               List<Trigger> triggers) {
138  
139           if ((triggerBytes == null) || triggerBytes.length == 0) {
140               return;
141           }
142  
143           /* Use trigger instances if available, otherwise fallback to
144            * using the byte arrays.
145            */
146           if ((triggers != null) && (triggers.size() != 0)) {
147               for (Trigger trigger : triggers) {
148                   sb.append("<trigger name=\"" + trigger.getName() + "\"" +
149                             " database name=\"" +
150                             ((trigger.getDatabaseName() == null) ?
151                             "null" : trigger.getDatabaseName()) + "\"" +
152                             " className=\"" + trigger.getClass().getName() +
153                   "\">");
154               }
155           } else {
156               /* Use the byte array */
157               for (int i=0; i < triggerBytes.length; i++) {
158                   sb.append("<trigger name=\"" + "?" + i + "\"" +
159                             " className=\"" + "?" +
160                   "\">");
161               }
162           }
163       }
