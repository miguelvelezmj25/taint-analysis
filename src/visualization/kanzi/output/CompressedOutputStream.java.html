38  // Implementation of a java.io.OutputStream that encodes a stream
39  // using a 2 step process:
40  // - step 1: a ByteFunction is used to reduce the size of the input data (bytes input & output)
41  // - step 2: an EntropyEncoder is used to entropy code the results of step 1 (bytes input, bits output)
42  public class CompressedOutputStream extends OutputStream {

186       /**
187        * Writes <code>len</code> bytes from the specified byte array
188        * starting at offset <code>off</code> to this output stream.
189        * The general contract for <code>write(array, off, len)</code> is that
190        * some of the bytes in the array <code>array</code> are written to the
191        * output stream in order; element <code>array[off]</code> is the first
192        * byte written and <code>array[off+len-1]</code> is the last byte written
193        * by this operation.
194        * <p>
195        * The <code>write</code> method of <code>OutputStream</code> calls
196        * the write method of one argument on each of the bytes to be
197        * written out. Subclasses are encouraged to override this method and
198        * provide a more efficient implementation.
199        * <p>
200        * If <code>array</code> is <code>null</code>, a
201        * <code>NullPointerException</code> is thrown.
202        * <p>
203        * If <code>off</code> is negative, or <code>len</code> is negative, or
204        * <code>off+len</code> is greater than the length of the array
205        * <code>array</code>, then an <tt>IndexOutOfBoundsException</tt> is thrown.
206        *
207        * @param data the data.
208        * @param off the start offset in the data.
209        * @param len the number of bytes to write.
210        * @throws IOException if an I/O error occurs. In particular,
211        * an <code>IOException</code> is thrown if the output
212        * stream is closed.
213        */
214       @Override
215       public void write(byte[] data, int off, int len) throws IOException {
216           if((off < 0) || (len < 0) || (len + off > data.length)) { → [FORCE] & [FORCE] & [FORCE]
217               throw new IndexOutOfBoundsException();
218           }
219  
220           if(this.closed.get() == true) { → [FORCE]
221               throw new kanzi.io.IOException("Stream closed", Error.ERR_WRITE_FILE);
222           }
223  
224           int remaining = len;
225  
226           while (remaining > 0) { → [FORCE]
227               // Limit to number of available bytes in buffer
228               final int lenChunk = (this.sa.index + remaining < this.sa.length) ? remaining : → [FORCE]
229                       this.sa.length - this.sa.index;
230  
231               if(lenChunk > 0) { → [FORCE]
232                   // Process a chunk of in-buffer data. No access to bitstream required
233                   System.arraycopy(data, off, this.sa.array, this.sa.index, lenChunk);
234                   this.sa.index += lenChunk;
235                   off += lenChunk;
236                   remaining -= lenChunk;
237  
238                   if(remaining == 0) { → [FORCE]
239                       break;
240                   }
241               }
242  
243               // Buffer full, time to encode
244               this.write(data[off]);
245               off++;
246               remaining--;
247           }
248       }

250       /**
251        * Writes the specified byte to this output stream. The general
252        * contract for <code>write</code> is that one byte is written
253        * to the output stream. The byte to be written is the eight
254        * low-order bits of the argument <code>b</code>. The 24
255        * high-order bits of <code>b</code> are ignored.
256        * <p>
257        * Subclasses of <code>OutputStream</code> must provide an
258        * implementation for this method.
259        *
260        * @param b the <code>byte</code>..
261        * @throws java.io.IOException
262        */
263       @Override
264       public void write(int b) throws IOException {
265           try {
266               // If the buffer is full, time to encode
267               if(this.sa.index >= this.sa.length) { → [FORCE]
268                   this.processBlock();
269               }
270  
271               this.sa.array[this.sa.index++] = (byte) b;
272           } catch (BitStreamException e) {
273               throw new kanzi.io.IOException(e.getMessage(), Error.ERR_READ_FILE);
274           } catch (kanzi.io.IOException e) {
275               throw e;
276           } catch (ArrayIndexOutOfBoundsException e) {
277               // Happens only if the stream is closed
278               throw new kanzi.io.IOException("Stream closed", Error.ERR_READ_FILE);
279           } catch (Exception e) {
280               throw new kanzi.io.IOException(e.getMessage(), Error.ERR_UNKNOWN);
281           }
282       }

317               return;

344       private void processBlock() throws IOException {
345           if(this.sa.index == 0) { → [FORCE]
346               return;
347           }
348  
349           if(this.initialized.getAndSet(true) == false) { → [FORCE]
350               this.writeHeader();
351           }
352  
353           try {
354               // Protect against future concurrent modification of the list of block listeners
355               Listener[] blockListeners = this.listeners.toArray(new Listener[this.listeners.size()]);
356               final int dataLength = this.sa.index;
357               this.sa.index = 0;
358               List<Callable<Status>> tasks = new ArrayList<>(this.jobs);
359               int firstBlockId = this.blockId.get();
360  
361               // Create as many tasks as required
362               for(int jobId = 0; jobId < this.jobs; jobId++) { → [FORCE]
363                   final int sz = (this.sa.index + this.blockSize > dataLength) ? → [FORCE]
364                           dataLength - this.sa.index : this.blockSize;
365  
366                   if(sz == 0) { → [FORCE]
367                       break;
368                   }
369  
370                   this.buffers[2 * jobId].index = 0;
371                   this.buffers[2 * jobId + 1].index = 0;
372  
373                   if(this.buffers[2 * jobId].array.length < sz) { → [FORCE]
374                       this.buffers[2 * jobId].array = new byte[sz];
375                       this.buffers[2 * jobId].length = sz;
376                   }
377  
378                   System.arraycopy(this.sa.array, this.sa.index, this.buffers[2 * jobId].array, 0, sz);
379  
380                   Callable<Status> task = new EncodingTask(this.buffers[2 * jobId],
381                           this.buffers[2 * jobId + 1], sz, this.transformType,
382                           this.entropyType, firstBlockId + jobId + 1,
383                           this.obs, this.hasher, this.blockId,
384                           blockListeners, new HashMap<>(this.ctx));
385                   tasks.add(task);
386                   this.sa.index += sz;
387               }
388  
389               if(this.jobs == 1) { → [FORCE]
390                   // Synchronous call
391                   Status status = tasks.get(0).call();
392  
393                   if(status.error != 0) { → [FORCE]
394                       throw new kanzi.io.IOException(status.msg, status.error);
395                   }
396               }
397               else {
398                   // Invoke the tasks concurrently and validate the results
399                   for(Future<Status> result : this.pool.invokeAll(tasks)) { → [FORCE]
400                       // Wait for completion of next task and validate result
401                       Status status = result.get();
402  
403                       if(status.error != 0) { → [FORCE]
404                           throw new kanzi.io.IOException(status.msg, status.error);
405                       }
406                   }
407               }
408  
409               this.sa.index = 0;
410           } catch (kanzi.io.IOException e) {
411               throw e;
412           } catch (Exception e) {
413               int errorCode = (e instanceof BitStreamException) ? ((BitStreamException) e).getErrorCode() : → [FORCE]
414                       Error.ERR_UNKNOWN;
415               throw new kanzi.io.IOException(e.getMessage(), errorCode);
416           }
417       }

466  
467           // Encode mode + transformed entropy coded data
468           // mode: 0b1000xxxx => small block (written as is) + 4 LSB for block size (0-15)
469           // 0x00xxxx00 => transform sequence skip flags (1 means skip)
470           // 0x000000xx => size(size(block))-1
471           private Status encodeBlock(SliceByteArray data, SliceByteArray buffer,
472                                      int blockLength, short typeOfTransform,
473                                      short typeOfEntropy, int currentBlockId) {
474               EntropyEncoder ee = null;
475  
476               try {
477                   byte mode = 0;
478                   int dataSize = 0;
479                   int postTransformLength = blockLength;
480                   int checksum = 0;
481  
482                   // Compute block checksum
483                   if(this.hasher != null) {
484                       checksum = this.hasher.hash(data.array, data.index, blockLength);
485                   }
486  
487                   if(this.listeners.length > 0) {
488                       // Notify before transform
489                       Event evt = new Event(Event.Type.BEFORE_TRANSFORM, currentBlockId,
490                               blockLength, checksum, this.hasher != null);
491  
492                       notifyListeners(this.listeners, evt);
493                   }
494  
495                   if(blockLength <= SMALL_BLOCK_SIZE) {
496                       // Just copy
497                       if(data.array != buffer.array) {
498                           if(buffer.length < blockLength) {
499                               buffer.length = blockLength;
500  
501                               if(buffer.array.length < buffer.length) {
502                                   buffer.array = new byte[buffer.length];
503                               }
504                           }
505  
506                           System.arraycopy(data.array, data.index, buffer.array, 0, blockLength);
507                       }
508  
509                       data.index += blockLength;
510                       buffer.index = blockLength;
511                       mode = (byte) (SMALL_BLOCK_MASK | (blockLength & COPY_LENGTH_MASK));
512                   }
513                   else {
514                       this.ctx.put("size", blockLength);
515                       ByteTransformSequence transform = new ByteFunctionFactory().newFunction(this.ctx, typeOfTransform);
516                       int requiredSize = transform.getMaxEncodedLength(blockLength);
517  
518                       if(buffer.length < requiredSize) {
519                           buffer.length = requiredSize;
520  
521                           if(buffer.array.length < buffer.length) {
522                               buffer.array = new byte[buffer.length];
523                           }
524                       }
525  
526                       // Forward transform (ignore error, encode skipFlags)
527                       buffer.index = 0;
528                       data.length = blockLength;
529                       transform.forward(data, buffer);
530                       mode |= ((transform.getSkipFlags() & ByteTransformSequence.SKIP_MASK) << 2);
531                       postTransformLength = buffer.index;
532  
533                       if(postTransformLength < 0) {
534                           return new Status(currentBlockId, Error.ERR_WRITE_FILE, "Invalid transform size");
535                       }
536  
537                       for(long n = 0xFF; n < postTransformLength; n <<= 8)
538                           dataSize++;
539  
540                       if(dataSize > 3) {
541                           return new Status(currentBlockId, Error.ERR_WRITE_FILE, "Invalid block data length");
542                       }
543  
544                       // Record size of 'block size' - 1 in bytes
545                       mode |= (dataSize & 0x03);
546                       dataSize++;
547                   }
548  
549                   if(this.listeners.length > 0) {
550                       // Notify after transform
551                       Event evt = new Event(Event.Type.AFTER_TRANSFORM, currentBlockId,
552                               postTransformLength, checksum, this.hasher != null);
553  
554                       notifyListeners(this.listeners, evt);
555                   }
556  
557                   // Lock free synchronization
558                   while (this.processedBlockId.get() != currentBlockId - 1) {
559                       // Wait for the concurrent task processing the previous block to complete
560                       // entropy encoding. Entropy encoding must happen sequentially (and
561                       // in the correct block order) in the bitstream.
562                       // Backoff improves performance in heavy contention scenarios
563                       LockSupport.parkNanos(10);
564                   }
565  
566                   // Write block 'header' (mode + compressed length);
567                   final long written = this.obs.written();
568                   this.obs.writeBits(mode, 8);
569  
570                   if(dataSize > 0) {
571                       this.obs.writeBits(postTransformLength, 8 * dataSize);
572                   }
573  
574                   // Write checksum
575                   if(this.hasher != null) {
576                       this.obs.writeBits(checksum, 32);
577                   }
578  
579                   if(this.listeners.length > 0) {
580                       // Notify before entropy
581                       Event evt = new Event(Event.Type.BEFORE_ENTROPY, currentBlockId,
582                               postTransformLength, checksum, this.hasher != null);
583  
584                       notifyListeners(this.listeners, evt);
585                   }
586  
587                   // Each block is encoded separately
588                   // Rebuild the entropy encoder to reset block statistics
589                   ee = new EntropyCodecFactory().newEncoder(this.obs, this.ctx, typeOfEntropy);
590  
591                   // Entropy encode block
592                   if(ee.encode(buffer.array, 0, postTransformLength) != postTransformLength) {
593                       return new Status(currentBlockId, Error.ERR_PROCESS_BLOCK, "Entropy coding failed");
594                   }
595  
596                   // Dispose before displaying statistics. Dispose may write to the bitstream
597                   ee.dispose();
598  
599                   // Force ee to null to avoid double dispose (in the finally section)
600                   ee = null;
601  
602                   final int w = (int) ((this.obs.written() - written) / 8L);
603  
604                   // After completion of the entropy coding, increment the block id.
605                   // It unfreezes the task processing the next block (if any)
606                   this.processedBlockId.incrementAndGet();
607  
608                   if(this.listeners.length > 0) {
609                       // Notify after entropy
610                       Event evt = new Event(Event.Type.AFTER_ENTROPY,
611                               currentBlockId, w, checksum, this.hasher != null);
612  
613                       notifyListeners(this.listeners, evt);
614                   }
615  
616                   return new Status(currentBlockId, 0, "Success");
617               } catch (Exception e) {
618                   return new Status(currentBlockId, Error.ERR_PROCESS_BLOCK, e.getMessage());
619               } finally {
620                   // Make sure to unfreeze next block
621                   if(this.processedBlockId.get() == this.blockId - 1) {
622                       this.processedBlockId.incrementAndGet();
623                   }
624  
625                   if(ee != null) {
626                       ee.dispose();
627                   }
628               }
629           }
630       }

632  
633       static class Status {
634           final int blockId;
635           final int error; // 0 = OK
636           final String msg;
637  
638           Status(int blockId, int error, String msg) {
639               this.blockId = blockId;
640               this.error = error;
641               this.msg = msg;
642           }
643       }
