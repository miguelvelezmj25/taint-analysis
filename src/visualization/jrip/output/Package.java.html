21  package weka.core.packageManagement;
22  
23  import java.io.Serializable;
24  import java.net.URL;
25  import java.util.List;
26  import java.util.Map;
27  
28  /**
29    * Abstract base class for Packages.
30    *
31    * @author mhall (mhall{[at]}pentaho{[dot]}com)
32    * @version $Revision: 52462 $
33    *
34    */
35  public abstract class Package implements Cloneable, Serializable {
36  
37     /**
38      * For serialization
39      */
40     private static final long serialVersionUID = -8193697646938632764L;
41  
42     /**
43      * The meta data for the package
44      */
45     protected Map<?, ?> m_packageMetaData;
46  
47     /**
48      * Set the meta data for this package.
49      *
50      * @param metaData the meta data for this package.
51      */
52     public void setPackageMetaData(Map<?, ?> metaData) {
53       m_packageMetaData = metaData;
54     }
55  
56     /**
57      * Get the meta data for this package.
58      *
59      * @return the meta data for this package
60      */
61     public Map<?, ?> getPackageMetaData() {
62       return m_packageMetaData;
63     }
64  
65     /**
66      * Convenience method to return the name of this package.
67      *
68      * @return the name of this package.
69      */
70     public abstract String getName();
71  
72     /**
73      * Convenience method that returns the URL to the package (i.e the provider's
74      * URL). This information is assumed to be stored in the package meta data.
75      *
76      * @return the URL to the package or null if the URL is not available for some
77      * reason
78      * @throws Exception if the URL can't be retrieved for some reason
79      */
80     public abstract URL getPackageURL() throws Exception;
81  
82     /**
83      * Compare the supplied package to this package. Simply does an equals()
84      * comparison between the two. Concrete subclasses should override if they
85      * wan't to do comparison based on specific package meta data elements.
86      *
87      * @param toCompare the package to compare against.
88      * @return true if the supplied package is equal to this one.
89      */
90     public boolean equals(Package toCompare) {
91       return m_packageMetaData.equals(toCompare.getPackageMetaData());
92     }
93  
94     /**
95      * Get the list of packages that this package depends on.
96      *
97      * @return the list of packages that this package depends on.
98      * @throws Exception if a problem occurs while getting the list of
99      * dependencies.
100      */
101     public abstract List<Dependency> getDependencies() throws Exception;
102  
103     /**
104      * Returns true if this package is already installed
105      *
106      * @return true if this package is installed
107      */
108     public abstract boolean isInstalled();
109  
110     /**
111      * Install this package.
112      *
113      * @throws Exception if something goes wrong during installation.
114      */
115     public abstract void install() throws Exception;
116  
117     /**
118      * Returns true if this package is compatible with the currently installed
119      * version of the base system.
120      *
121      * @return true if this package is compatible with the main software system.
122      * @throws Exception if a problem occurs while checking compatibility.
123      */
124     public abstract boolean isCompatibleBaseSystem() throws Exception;
125  
126     /**
127      * Returns true if this package is good to go with the current version of the
128      * software and versions of any installed packages that it depends on.
129      *
130      * @return true if this package is good to go.
131      * @throws Exception if a problem occurs while checking compatibility.
132      */
133     /*
134      * public abstract boolean isCompatible() throws Exception;
135      */
136  
137     /**
138      * Gets the dependency on the base system that this package requires.
139      *
140      * @return the base system dependency(s) for this package
141      * @throws Exception if the base system dependency can't be determined for
142      * some reason.
143      */
144     public abstract List<Dependency> getBaseSystemDependency() throws Exception;
145  
146     /**
147      * Gets a list of packages that this package depends on that are not currently
148      * installed.
149      *
150      * @return a list of missing packages that this package depends on.
151      */
152     public abstract List<Dependency> getMissingDependencies() throws Exception;
153  
154     /**
155      * Gets a list of packages that this package depends on that are not in the
156      * supplied list of packages.
157      *
158      * @param packages a list of packages to compare this package's dependencies
159      * against.
160      * @return those packages that this package depends on that aren't in the
161      * supplied list.
162      * @throws Exception if the list of missing depenencies can't be determined
163      * for some reason.
164      */
165     public abstract List<Dependency>
166       getMissingDependencies(List<Package> packages) throws Exception;
167  
168     /**
169      * Gets a list of installed packages that this package depends on that are
170      * currently incompatible with this package.
171      *
172      * @return a list of incompatible installed packages that this package depends
173      * on.
174      */
175     public abstract List<Dependency> getIncompatibleDependencies()
176       throws Exception;
177  
178     /**
179      * Compares this package's precluded list (if any) against the list of
180      * supplied packages. Any packages in the supplied list that match (by name
181      * and version constraints) any in the precluded list are returned
182      *
183      * @param packages a list of packages to compare against those in this
184      * package's precluded list
185      * @return a list of packages that are covered by those in the precluded list
186      * @throws Exception if a problem occurs
187      */
188     public abstract List<Package> getPrecludedPackages(List<Package> packages)
189       throws Exception;
190  
191     /**
192      * Gets those packages from the supplied list that this package depends on and
193      * are currently incompatible with this package.
194      *
195      * @param packages a list of packages to compare this package's dependencies
196      * against
197      * @return those packages from the supplied list that are incompatible with
198      * respect to this package's dependencies
199      * @throws Exception if the list of incompatible dependencies can't be
200      * generated for some reason.
201      */
202     public abstract List<Dependency> getIncompatibleDependencies(
203       List<Package> packages) throws Exception;
204  
205     /**
206      * Gets the package meta data element associated with the supplied key.
207      *
208      * @param key the key to use to look up a value in the meta data
209      * @return the meta data value or null if the key does not exist.
210      */
211     public Object getPackageMetaDataElement(Object key) {
212       if (m_packageMetaData == null) { → [OPTIMIZATIONS, PRUNE, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
213         return null;
214       }
215  
216       return m_packageMetaData.get(key);
217     }
218  
219     /**
220      * Adds a key, value pair to the meta data map.
221      *
222      * @param key the key
223      * @param value the value to add
224      * @throws Exception if there is no meta data map to add to.
225      */
226     public abstract void setPackageMetaDataElement(Object key, Object value)
227       throws Exception;
228  
229     @Override
230     public abstract Object clone();
231  }
