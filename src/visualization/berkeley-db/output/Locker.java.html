38    * MasterTxn and ReadonlyTxn. Locker instances are in fact only a transaction

40    * semantics.

566       /**
567        * Called when an LN is logged by an operation that will not hold the lock
568        * such as eviction/checkpoint deferred-write logging or cleaner LN
569        * migration. We must acquire a lock on the new LSN on behalf of every
570        * locker that currently holds a lock on the old LSN.
571        *
572        * Lock is non-blocking because no contention is possible on the new LSN.
573        *
574        * Because this locker is being used by multiple threads, this method may
575        * be called for a locker that has been closed or for which the lock on the
576        * old LSN has been released. Unlike other locking methods, in this case
577        * we simply return rather than report an error.
578        */
579       public synchronized void lockAfterLsnChange(long oldLsn,
580                                                   long newLsn,
581                                                   DatabaseImpl dbImpl) {
582           if (!isValid()) {
583               /* Locker was recently closed, made abort-only, etc. */
584               return;
585           }
586  
587           final LockType lockType = lockManager.getOwnedLockType(oldLsn, this);
588           if (lockType == null) {
589               /* Lock was recently released. */
590               return;
591           }
592  
593           final LockResult lockResult = nonBlockingLock
594               (newLsn, lockType, false /*jumpAheadOfWaiters*/, dbImpl);
595  
596           if (lockResult.getLockGrant() == LockGrantType.DENIED) {
597               throw EnvironmentFailureException.unexpectedState
598                   ("No contention is possible on new LSN: " +
599                    DbLsn.getNoFormatString(newLsn) +
600                    " old LSN: " + DbLsn.getNoFormatString(oldLsn) +
601                    " LockType: " + lockType);
602           }
603       }
