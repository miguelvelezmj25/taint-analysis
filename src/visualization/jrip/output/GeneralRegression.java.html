160      * Enumerated type for the link function (general linear and
161      * generalized linear model types only).
162      */
163     enum LinkFunction {
164       NONE ("none") {
165         double eval(double value, double offset, double trials,
166                     double distParam, double linkParam) {
167           return Double.NaN; // no evaluation defined in this case!
168         }
169       },
170       CLOGLOG ("cloglog") {
171         double eval(double value, double offset, double trials,
172                     double distParam, double linkParam) {
173           return (1.0 - Math.exp(-Math.exp(value + offset))) * trials;
174         }
175       },
176       IDENTITY ("identity") {
177         double eval(double value, double offset, double trials,
178                     double distParam, double linkParam) {
179           return (value + offset) * trials;
180         }
181       },
182       LOG ("log") {
183         double eval(double value, double offset, double trials,
184                     double distParam, double linkParam) {
185           return Math.exp(value + offset) * trials;
186         }
187       },
188       LOGC ("logc") {
189         double eval(double value, double offset, double trials,
190                     double distParam, double linkParam) {
191           return (1.0 - Math.exp(value + offset)) * trials;
192         }
193       },
194       LOGIT ("logit") {
195         double eval(double value, double offset, double trials,
196                     double distParam, double linkParam) {
197           return (1.0 / (1.0 + Math.exp(-(value + offset)))) * trials;
198         }
199       },
200       LOGLOG ("loglog") {
201         double eval(double value, double offset, double trials,
202                     double distParam, double linkParam) {
203           return Math.exp(-Math.exp(-(value + offset))) * trials;
204         }
205       },
206       NEGBIN ("negbin") {
207         double eval(double value, double offset, double trials,
208                     double distParam, double linkParam) {
209           return (1.0 / (distParam * (Math.exp(-(value + offset)) - 1.0))) * trials;
210         }
211       },
212       ODDSPOWER ("oddspower") {
213         double eval(double value, double offset, double trials,
214                     double distParam, double linkParam) {
215           return (linkParam < 0.0 || linkParam > 0.0) → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES] & [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
216           ? (1.0 / (1.0 + Math.pow(1.0 + linkParam * (value + offset), (-1.0 / linkParam)))) * trials
217           : (1.0 / (1.0 + Math.exp(-(value + offset)))) * trials;
218         }
219       },
220       POWER ("power") {
221         double eval(double value, double offset, double trials,
222                     double distParam, double linkParam) {
223           return (linkParam < 0.0 || linkParam > 0.0) → [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES] & [OPTIMIZATIONS, BATCHSIZE, NUMDECIMAL, PRUNE, SEED, CHECKERROR, MINNO, DEBUG, FOLDS, NOCHECKCAPABILITIES]
224           ? Math.pow(value + offset, (1.0 / linkParam)) * trials
225               : Math.exp(value + offset) * trials;
226         }
227       },
228       PROBIT ("probit") {
229         double eval(double value, double offset, double trials,
230                     double distParam, double linkParam) {
231           return weka.core.matrix.Maths.pnorm(value + offset) * trials;
232         }
233       };

406     /**
407      * Constructs a GeneralRegression classifier.
408      *
409      * @param model the Element that holds the model definition
410      * @param dataDictionary the data dictionary as a set of Instances
411      * @param miningSchema the mining schema
412      * @throws Exception if there is a problem constructing the general regression
413      * object from the PMML.
414      */
415     public GeneralRegression(Element model, Instances dataDictionary,
416                              MiningSchema miningSchema) throws Exception {
417  
418       super(dataDictionary, miningSchema);
419   
420       // get the model type
421       String mType = model.getAttribute("modelType");
422       boolean found = false;
423       for (ModelType m : ModelType.values()) {
424         if (m.toString().equals(mType)) {
425           m_modelType = m;
426           found = true;
427           break;
428         }
429       }
430       if (!found) {
431         throw new Exception("[GeneralRegression] unknown model type: " + mType);
432       }
433  
434       if (m_modelType == ModelType.ORDINALMULTINOMIAL) {
435         // get the cumulative link function
436         String cLink = model.getAttribute("cumulativeLink");
437         found = false;
438         for (CumulativeLinkFunction c : CumulativeLinkFunction.values()) {
439           if (c.toString().equals(cLink)) {
440             m_cumulativeLinkFunction = c;
441             found = true;
442             break;
443           }
444         }
445         if (!found) {
446           throw new Exception("[GeneralRegression] cumulative link function " + cLink);
447         }
448       } else if (m_modelType == ModelType.GENERALIZEDLINEAR ||
449                   m_modelType == ModelType.GENERALLINEAR) {
450         // get the link function
451         String link = model.getAttribute("linkFunction");
452         found = false;
453         for (LinkFunction l : LinkFunction.values()) {
454           if (l.toString().equals(link)) {
455             m_linkFunction = l;
456             found = true;
457             break;
458           }
459         }
460         if (!found) {
461           throw new Exception("[GeneralRegression] unknown link function " + link);
462         }
463  
464         // get the link parameter
465         String linkP = model.getAttribute("linkParameter");
466         if (linkP != null && linkP.length() > 0) {
467           try {
468             m_linkParameter = Double.parseDouble(linkP);
469           } catch (IllegalArgumentException ex) {
470             throw new Exception("[GeneralRegression] unable to parse the link parameter");
471           }
472         }
473  
474         // get the trials variable
475         String trials = model.getAttribute("trialsVariable");
476         if (trials != null && trials.length() > 0) {
477           m_trialsVariable = trials;
478         }
479  
480         // get the trials value
481         String trialsV = model.getAttribute("trialsValue");
482         if (trialsV != null && trialsV.length() > 0) {
483           try {
484             m_trialsValue = Double.parseDouble(trialsV);
485           } catch (IllegalArgumentException ex) {
486             throw new Exception("[GeneralRegression] unable to parse the trials value");
487           }
488         }
489       }
490    
491       String mName = model.getAttribute("modelName");
492       if (mName != null && mName.length() > 0) {
493         m_modelName = mName;
494       }
495  
496       String fName = model.getAttribute("functionName");
497       if (fName.equals("classification")) {
498         m_functionType = Regression.RegressionTable.CLASSIFICATION;
499       }
500  
501       String algName = model.getAttribute("algorithmName");
502       if (algName != null && algName.length() > 0) {
503         m_algorithmName = algName;
504       }
505  
506       String distribution = model.getAttribute("distribution");
507       if (distribution != null && distribution.length() > 0) {
508         found = false;
509         for (Distribution d : Distribution.values()) {
510           if (d.toString().equals(distribution)) {
511             m_distribution = d;
512             found = true;
513             break;
514           }
515         }
516         if (!found) {
517           throw new Exception("[GeneralRegression] unknown distribution type " + distribution);
518         }
519       }
520  
521       String distP = model.getAttribute("distParameter");
522       if (distP != null && distP.length() > 0) {
523         try {
524           m_distParameter = Double.parseDouble(distP);
525         } catch (IllegalArgumentException ex) {
526           throw new Exception("[GeneralRegression] unable to parse the distribution parameter");
527         }
528       }
529  
530       String offsetV = model.getAttribute("offsetVariable");
531       if (offsetV != null && offsetV.length() > 0) {
532          m_offsetVariable = offsetV;
533       }
534  
535       String offsetVal = model.getAttribute("offsetValue");
536       if (offsetVal != null && offsetVal.length() > 0) {
537         try {
538           m_offsetValue = Double.parseDouble(offsetVal);
539         } catch (IllegalArgumentException ex) {
540           throw new Exception("[GeneralRegression] unable to parse the offset value");
541         }
542       }
543  
544       // get the parameter list
545       readParameterList(model);
546      
547       // get the factors and covariates
548       readFactorsAndCovariates(model, "FactorList");
549       readFactorsAndCovariates(model, "CovariateList");
550  
551       // read the PPMatrix
552       readPPMatrix(model);
553  
554       // read the parameter estimates
555       readParamMatrix(model);
556     }

558     /**
559      * Read the list of parameters.
560      *
561      * @param model the Element that contains the model
562      * @throws Exception if there is some problem with extracting the
563      * parameters.
564      */
565     protected void readParameterList(Element model) throws Exception {
566       NodeList paramL = model.getElementsByTagName("ParameterList");
567  
568       // should be just one parameter list
569       if (paramL.getLength() == 1) {
570         Node paramN = paramL.item(0);
571         if (paramN.getNodeType() == Node.ELEMENT_NODE) {
572           NodeList parameterList = ((Element)paramN).getElementsByTagName("Parameter");
573           for (int i = 0; i < parameterList.getLength(); i++) {
574             Node parameter = parameterList.item(i);
575             if (parameter.getNodeType() == Node.ELEMENT_NODE) {
576               Parameter p = new Parameter();
577               p.m_name = ((Element)parameter).getAttribute("name");
578               String label = ((Element)parameter).getAttribute("label");
579               if (label != null && label.length() > 0) {
580                 p.m_label = label;
581               }
582               m_parameterList.add(p);
583             }
584           }
585         }
586       } else {
587         throw new Exception("[GeneralRegression] more than one parameter list!");
588       }
589     }

591     /**
592      * Read the lists of factors and covariates.
593      *
594      * @param model the Element that contains the model
595      * @param factorOrCovariate holds the String "FactorList" or
596      * "CovariateList"
597      * @throws Exception if there is a factor or covariate listed
598      * that isn't in the mining schema
599      */
600     protected void readFactorsAndCovariates(Element model,
601                                             String factorOrCovariate)
602       throws Exception {
603       Instances miningSchemaI = m_miningSchema.getFieldsAsInstances();
604  
605       NodeList factorL = model.getElementsByTagName(factorOrCovariate);
606       if (factorL.getLength() == 1) { // should be 0 or 1 FactorList element
607         Node factor = factorL.item(0);
608         if (factor.getNodeType() == Node.ELEMENT_NODE) {
609           NodeList predL = ((Element)factor).getElementsByTagName("Predictor");
610           for (int i = 0; i < predL.getLength(); i++) {
611             Node pred = predL.item(i);
612             if (pred.getNodeType() == Node.ELEMENT_NODE) {
613               Predictor p = new Predictor();
614               p.m_name = ((Element)pred).getAttribute("name");
615               // find the index of this predictor in the mining schema
616               boolean found = false;
617               for (int j = 0; j < miningSchemaI.numAttributes(); j++) {
618                 if (miningSchemaI.attribute(j).name().equals(p.m_name)) {
619                   found = true;
620                   p.m_miningSchemaIndex = j;
621                   break;
622                 }
623               }
624               if (found) {
625                 if (factorOrCovariate.equals("FactorList")) {
626                   m_factorList.add(p);
627                 } else {
628                   m_covariateList.add(p);
629                 }
630               } else {
631                 throw new Exception("[GeneralRegression] reading factors and covariates - "
632                                     + "unable to find predictor " +
633                                     p.m_name + " in the mining schema");
634               }
635             }
636           }
637         }
638       } else if (factorL.getLength() > 1){
639         throw new Exception("[GeneralRegression] more than one " + factorOrCovariate
640                             + "! ");
641       }
642     }

644     /**
645      * Read the PPMatrix from the xml. Does not handle multiple PPMatrixes yet.
646      *
647      * @param model the Element that contains the model
648      * @throws Exception if there is a problem parsing cell values.
649      */
650     protected void readPPMatrix(Element model) throws Exception {
651       Instances miningSchemaI = m_miningSchema.getFieldsAsInstances();
652

859                                   + "target category: " + targetCatName
860                                   + " but class attribute is numeric in "
861                                   + "mining schema");
862             }
