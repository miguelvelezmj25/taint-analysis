35  /**
36    * Generates points illustrating prediction tradeoffs that can be obtained by
37    * varying the threshold value between classes. For example, the typical
38    * threshold value of 0.5 means the predicted probability of "positive" must be
39    * higher than 0.5 for the instance to be predicted as "positive". The resulting
40    * dataset can be used to visualize precision/recall tradeoff, or for ROC curve
41    * analysis (true positive rate vs false positive rate). Weka just varies the
42    * threshold on the class probability estimates in each case. The Mann Whitney
43    * statistic is used to calculate the AUC.
44    *
45    * @author Len Trigg (len@reeltwo.com)
46    * @version $Revision: 10153 $
47    */
48  public class ThresholdCurve implements RevisionHandler {

215     /**
216      * Calculates the n point precision result, which is the precision averaged
217      * over n evenly spaced (w.r.t recall) samples of the curve.
218      *
219      * @param tcurve a previously extracted threshold curve Instances.
220      * @param n the number of points to average over.
221      * @return the n-point precision.
222      */
223     public static double getNPointPrecision(Instances tcurve, int n) {
224  
225       if (!RELATION_NAME.equals(tcurve.relationName())
226         || (tcurve.numInstances() == 0)) {
227         return Double.NaN;
228       }
229       int recallInd = tcurve.attribute(RECALL_NAME).index();
230       int precisInd = tcurve.attribute(PRECISION_NAME).index();

232       int[] sorted = Utils.sort(recallVals);
233       double isize = 1.0 / (n - 1);
234       double psum = 0;
235       for (int i = 0; i < n; i++) {
236         int pos = binarySearch(sorted, recallVals, i * isize);
237         double recall = recallVals[sorted[pos]];
238         double precis = tcurve.instance(sorted[pos]).value(precisInd);
239         /*
240          * System.err.println("Point " + (i + 1) + ": i=" + pos + " r=" + (i *
241          * isize) + " p'=" + precis + " r'=" + recall);

243         // interpolate figures for non-endpoints

250             double offset = precis - recall * slope;
251             precis = isize * i * slope + offset;
252             /*

254              * * isize) + " p'=" + precis2 + " r'=" + recall2 + " p''=" + precis);
255              */
256             break;
257           }
258         }

260       }
261       return psum / n;

362     /**
363      * performs a binary search
364      *
365      * @param index the indices
366      * @param vals the values
367      * @param target the target to look for
368      * @return the index of the target
369      */
370     private static int binarySearch(int[] index, double[] vals, double target) {
371  
372       int lo = 0, hi = index.length - 1;
373       while (hi - lo > 1) {
374         int mid = lo + (hi - lo) / 2;
375         double midval = vals[index[mid]];
376         if (target > midval) {
377           lo = mid;
378         } else if (target < midval) {
379           hi = mid;
380         } else {
381           while ((mid > 0) && (vals[index[mid - 1]] == target)) {
382             mid--;
383           }
384           return mid;
385         }
386       }
387       return lo;
388     }
