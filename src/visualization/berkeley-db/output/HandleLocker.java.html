14  package berkeley.com.sleepycat.je.txn;
15  
16  import berkeley.com.sleepycat.je.DatabaseException;
17  import berkeley.com.sleepycat.je.dbi.EnvironmentImpl;
18  
19  /**
20    * Extends BasicLocker to share locks with another Locker that is being used to
21    * open a database. HandleLocker holds a read lock on the NameLN for the
22    * Database object, to prevent rename, removal or truncation of a database
23    * while it is open. The HandleLocker and the Locker used to open the database
24    * both hold a NameLN lock at the same time, so they must share locks to avoid
25    * conflicts.
26    *
27    * Accounts for the fact that the Txn may end before this locker ends by only
28    * keeping the Txn ID rather than a reference to the Txn object. A reference
29    * to a non-tnxl Locker is OK, OTOH, because it is short lived.
30    *
31    * Handle Locking Overview
32    * ------------------------
33    * Environment.openDatabase calls Environment.setupDatabase, which calls
34    * Database.initHandleLocker to create the HandleLocker. setupDatabase ensures
35    * that the HandleLocker is passed to DbTree.getDb and createDb. These latter
36    * methods acquire a read lock on the NameLN for the HandleLocker, in addition
37    * to acquiring a read or write lock for the openDatabase locker.
38    *
39    * If setupDatabase is not successful, it ensures that locks are released via
40    * HandleLocker.endOperation. If setupDatabase is successful, the handle is
41    * returned by openDatabase, and Database.close must be called to release the
42    * lock. The handle lock is released by calling HandleLocker.endOperation.
43    *
44    * A special case is when a user txn is passed to openDatabase. If the txn
45    * aborts, the Database handle must be invalidated. When setupDatabase
46    * succeeds it passes the handle to Txn.addOpenedDatabase, which remembers the
47    * handle. Txn.abort invalidates the handle.
48    *
49    * NameLN Migration and LSN Changes [#20617]
50    * -----------------------------------------
51    * When the log cleaner migrates a NameLN, its LSN changes and the new LSN is
52    * locked on behalf of all existing lockers by CursorImpl.lockAfterLsnChange.
53    * lockAfterLsnChange is also used when a dirty deferred-write LN is logged by
54    * BIN.logDirtyLN, as part of flushing a BIN during a checkpoint or eviction.
55    *
56    * Because handle lockers are legitimately very long lived, it is important
57    * that lockAfterLsnChange releases the locks on the old LSN, to avoid a steady
58    * accumulation of locks in a HandleLocker. Therefore, lockAfterLsnChange will
59    * release the lock on the old LSN, for HandleLockers only. Although it may be
60    * desirable to release the old LSN lock on other long lived lockers, it is too
61    * risky. In an experiment, this caused problems with demotion and upgrade,
62    * when a lock being demoted or upgraded was released.
63    *
64    * Because LSNs can change, it is also important that we don't rely on a single
65    * NameLN locker ID (LSN) as a data structure key for handle locks. This was
66    * acceptable when a stable Node ID was used as a lock ID, but is no longer
67    * appropriate now that mutable LSNs are used as lock IDs.
68    */
69  public class HandleLocker extends BasicLocker {
70  
71       private final long shareWithTxnId;
72       private final Locker shareWithNonTxnlLocker;
73  
74       /**
75        * Creates a HandleLocker.
76        */
77       protected HandleLocker(EnvironmentImpl env, Locker buddy) {
78           super(env);
79           shareWithTxnId =
80               buddy.isTransactional() ? buddy.getId() : TxnManager.NULL_TXN_ID;
81           shareWithNonTxnlLocker =
82               buddy.isTransactional() ? null : buddy;
83       }
84  
85       public static HandleLocker createHandleLocker(EnvironmentImpl env,
86                                                     Locker buddy)
87           throws DatabaseException {
88  
89           return new HandleLocker(env, buddy);
90       }
91  
92       /**
93        * Returns whether this locker can share locks with the given locker.
94        */
95       @Override
96       public boolean sharesLocksWith(Locker other) {
97  
98           if (super.sharesLocksWith(other)) { → [READONLY]
99               return true;
100           }
101           if (shareWithTxnId != TxnManager.NULL_TXN_ID && → [READONLY]
102               shareWithTxnId == other.getId()) { → [READONLY]
103               return true;
104           }
105           if (shareWithNonTxnlLocker != null && → [READONLY] & [READONLY]
106               shareWithNonTxnlLocker == other) {
107               return true;
108           }
109           return false;
110       }
111  
112       /**
113        * Because handle lockers are legitimately very long lived, it is important
114        * that lockAfterLsnChange releases the locks on the old LSN, to avoid a
115        * steady accumulation of locks in a HandleLocker
116        */
117       @Override
118       public boolean allowReleaseLockAfterLsnChange() {
119           return true;
120       }
121  }
