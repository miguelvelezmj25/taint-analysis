17    * Execute a test hook if set. This wrapper is used so that test hook execution
18    * can be packaged into a single statement that can be done within an assert
19    * statement.
20    */
21  public class TestHookExecute {
22  
23       public static boolean doHookSetupIfSet(TestHook<?> testHook) {
24           if (testHook != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
25               testHook.hookSetup();
26           }
27           return true;
28       }
29  
30       public static boolean doHookIfSet(TestHook<?> testHook) {
31           if (testHook != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
32               testHook.doHook();
33           }
34           return true;
35       }
36  
37       public static <T> boolean doHookIfSet(TestHook<T> testHook, T obj) {
38           if (testHook != null) { → [ALLOWCREATE, TRANSACTIONAL, READONLY]
39               testHook.doHook(obj);
40           }
41           return true;
42       }
43  }
